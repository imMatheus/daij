// 🥁 "Ritual Machine" — Polyrhythmic Percussion
// At 96 BPM: intro(8) + grooveA(16) + build(16) + grooveB(16) + peak(16) + outro(8) = 80 bars ≈ 3:20

setcpm(96 / 4)

$: arrange(
  [8,  s("bd ~ ~ ~, ~ ~ ~ cp").bank("RolandTR808").gain(".7 .5 .4 .5").room(.4).swing(3)],
  [16, s("bd ~ [~ bd] ~, ~ cp ~ ~").bank("RolandTR808").gain(".92 .85").room(.3).swing(3)],
  [16, s("bd [~ bd] ~ bd, ~ cp ~ [cp cp]").bank("RolandTR808").gain(".95 .88").room(.25).swing(3).every(4, x => x.fast(2).gain(.5))],
  [16, s("bd ~ [~ bd] ~, ~ [cp sd:1] ~ ~").bank("RolandTR909").gain(".9 .82").room(.35).swing(4)],
  [16, s("bd [~ bd] bd ~, ~ [cp,sd:1] ~ [cp cp], bd ~ ~ bd").bank("RolandTR909").gain(".95 .9").room(.25).swing(3).every(3, x => x.press())],
  [8,  s("bd ~ ~ ~").bank("RolandTR808").gain(saw.slow(8).range(.8, .1)).room(.6)]
)

$: arrange(
  [8,  s("hh*8").bank("RolandTR808").gain(perlin.range(.1, .35)).pan(sine.slow(4)).cut(1)],
  [16, s("[hh*4] [hh hh oh hh]").bank("RolandTR808").gain(perlin.range(.2, .55)).cut(1).swing(3).pan(sine.slow(6))],
  [16, s("hh*16").bank("RolandTR808").gain(perlin.range(.15, .6)).cut(1).every(4, x => x.struct("x ~ x x ~ x x ~")).pan(sine.slow(5))],
  [16, s("[hh*4] [hh hh oh hh]").bank("RolandTR909").gain(perlin.range(.2, .5)).cut(1).swing(4).pan(sine.slow(7))],
  [16, s("[hh oh hh hh]*4").bank("RolandTR909").gain(perlin.range(.2, .65)).cut(1).swing(3).every(3, x => x.rev()).pan(sine.slow(4))],
  [8,  s("hh*4").bank("RolandTR808").gain(saw.slow(8).range(.4, .05)).cut(1).room(.5)]
)

$: arrange(
  [8,  silence],
  [16, s("<[lt ~ ~ mt] [~ ht ~ ~] [~ ~ lt ~] [mt ~ ~ ~]>").bank("RolandTR808").gain(.55).room(.45).delay(".2:.15:.25").pan("0.3 0.7")],
  [16, s("<[lt mt ht ~] [~ lt ~ mt] [ht ~ lt ~] [~ mt ht lt]>").bank("RolandTR808").gain(.6).room(.4).delay(".15:.1:.2").pan("0.2 0.8 0.4 0.6")],
  [16, s("<[~ lt ~ ~] [mt ~ ~ ht] [~ ~ lt ~] [~ mt ~ ~]>").bank("RolandTR909").gain(.5).room(.5).delay(".25:.2:.3").pan("0.3 0.7")],
  [16, s("<[lt mt ht ~] [~ lt ~ mt] [ht ~ lt ~] [~ mt ht lt]>").bank("RolandTR909").gain(.62).room(.35).delay(".15:.1:.2").pan("0.2 0.8 0.4 0.6").every(4, x => x.fast(2).gain(.35))],
  [8,  s("lt ~ ~ mt ~ ~ ht ~").bank("RolandTR808").gain(saw.slow(8).range(.5, .1)).room(.7)]
)

$: arrange(
  [8,  silence],
  [16, s("~ cb ~ ~, rim ~ ~ rim").bank("RolandTR808").gain(".4 .35").room(.3).pan(.7)],
  [16, s("cb(3,8), rim(5,16)").bank("RolandTR808").gain(".45 .38").room(.25).pan(.65).every(4, x => x.gain(.2))],
  [16, s("~ ~ cb ~, rim ~ ~ rim").bank("RolandTR909").gain(".42 .36").room(.35).pan(.3)],
  [16, s("cb(5,16), rim(7,16)").bank("RolandTR909").gain(".48 .4").room(.2).pan(.35).every(3, x => x.rev())],
  [8,  s("~ cb ~ ~").bank("RolandTR808").gain(saw.slow(8).range(.35, .05)).room(.5)]
)

$: arrange(
  [8,  silence],
  [16, silence],
  [16, s("~ ~ ~ cp, ~ cp ~ ~").bank("RolandTR909").gain(".5 .4").room(.5).delay(".3:.2:.35").hpf(800)],
  [16, silence],
  [16, s("~ [cp ~] ~ cp, cp ~ ~ [~ cp]").bank("RolandTR909").gain(".55 .45").room(.45).delay(".25:.15:.3").hpf(900).every(5, x => x.fast(2).gain(.3))],
  [8,  silence]
