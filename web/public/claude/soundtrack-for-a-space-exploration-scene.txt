// 🚀 "Beyond the Heliosphere" — Cinematic Space Ambient / Downtempo

$: s("vinyl:2").gain(.12).room(1).loop(1).lpf(2200)

   .attack(3).decay(1).sustain(.8).release(3)
   .gain(.2).room(.7).roomsize(8)
   .lpf(180)
