// 🎵 "Rhythmic Diffusion" — Industrial Minimalist Texture
// At 105 BPM: intro(8) + phase1(16) + phase2(16) + break(8) + phase3(24) + phase4(12) + outro(8) = 92 bars ≈ 3:30

setcpm(105 / 4)

$: s("vinyl:2").gain(.18).room(1).loop(1).lpf(2200).pan(sine.slow(10))

$: s("noise").gain(perlin.range(.02, .1)).lpf(sine.range(400, 4000).slow(16)).room(2).pan(saw.slow(20))

$: arrange(
  [8,  s("bd ~ ~ ~").bank("RolandTR909").gain(.6).lpf(400)],
  [16, s("bd(3,8)").bank("RolandTR909").gain(.8).room(.2)],
  [16, s("bd(3,8), ~ [sd:2 ~ sd:2]").bank("RolandTR909").gain(".85 .6").room(.3).distort(.1)],
  [8,  s("bd(3,8)").bank("RolandTR909").gain(.5).hpf(800).room(.8)],
  [24, s("bd(5,8), ~ [sd:2 ~ sd:3] ~ [sd:2]").bank("RolandTR909").gain(".9 .7").room(.2).every(4, x=>x.fast(2))],
  [12, s("bd ~ ~ ~").bank("RolandTR909").gain(.7).lpf(600)],
  [8,  silence]
)

$: arrange(
  [8,  silence],
  [16, s("hh*8").bank("RolandTR808").gain(perlin.range(.1, .4)).cut(1).pan(sine.slow(4))],
  [16, s("[hh*4 oh]*2").bank("RolandTR808").gain(perlin.range(.2, .6)).cut(1).room(.2).swing(4)],
  [8,  s("hh*16").bank("RolandTR808").gain(perlin.range(.05, .2)).hpf(2000)],
  [24, s("[hh*8, oh(3,8)]").bank("RolandTR808").gain(perlin.range(.3, .7)).cut(1).room(.1).every(3, x=>x.press())],
  [12, s("hh*4").bank("RolandTR808").gain(.3).cut(1)],
  [8,  silence]
)

$: arrange(
  [8,  s("can:1(3,8)").gain(.4).room(.8).delay(".4:.3:.5").pan(-0.5)],
  [16, s("can:1(3,8), metal:2(5,16)").gain(.45).room(.6).delay(".2:.4:.3").pan(sine.slow(8))],
  [16, s("can:1(3,8), metal:2(5,16), wood:3(7,16)").gain(.5).room(.5).delay(".1:.2:.4").crush(12)],
  [8,  s("metal:2(1,4)").gain(.3).room(1).delay(".8:.6:.7")],
  [24, s("can:1(3,8), metal:2(5,16), wood:3(11,16)").gain(.55).room(.4).delay(".15:.1:.3").every(8, x=>x.rev())],
  [12, s("can:1(1,4)").gain(.3).room(.7)],
  [8,  s("can:1(1,8)").gain(saw.slow(8).range(.3, 0)).room(1)]
)

$: arrange(
  [24, silence],
  [16, s("glitch:3*4").gain(perlin.range(.1, .3)).room(.8).delay(".3:.5:.6").pan(saw.slow(4))],
  [8,  silence],
  [24, s("glitch:2(7,16), glitch:5(3,8)").gain(.35).room(.6).delay(".2:.3:.4").every(4, x=>x.fast(1.5))],
  [20, silence]
)

$: arrange(
  [40, silence],
  [8,  s("sine").gain(perlin.range(.1, .2)).lpf(100).room(1)],
  [24, s("sine").gain(perlin.range(.1, .25)).lpf(sine.range(80, 200).slow(4)).room(.5).distort(.4)],
  [20, silence]
)

$: s("click*16").struct("x(5,8,1)").gain(perlin.range(0, .2)).pan(sine.slow(2)).cut(2)
