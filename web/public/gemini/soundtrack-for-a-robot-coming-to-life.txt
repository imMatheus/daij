// 🤖 "Sentient Boot Sequence" — Glitch Electronica / Cinematic Sci-Fi
// At 110 BPM: intro(16) + boot_sequence(16) + stabilization(16) + awareness(16) + overload(16) + clarity(16) + shutdown(8) = 104 bars ≈ 3:46

setcpm(110 / 4)

// ⚡ Texture: Electrical hum and data static
$: s("vinyl:2").gain(.2).lpf(sine.range(1000, 3000).slow(16)).loop(1)
$: arrange(
  [16, s("noise").gain(perlin.range(0.05, 0.15)).lpf(1500).cut(1)],
  [88, s("noise").gain(perlin.range(0.02, 0.08)).lpf(800).cut(1)]
)

// ⚙️ Mechanical Rhythm: Clicks, ticks, and servos
$: arrange(
  [16, s("hh*4").bank("RolandTR808").gain(perlin.range(.1, .3)).pan(sine.slow(4)).rate(sine.range(0.5, 1.5).slow(8))],
  [16, s("hh*8").bank("RolandTR808").gain(perlin.range(.2, .5)).rate(2).cut(1).crush(8)],
  [16, s("[hh,oh]*8").bank("RolandTR909").gain(.4).cut(1).every(4, x => x.fast(2))],
  [16, s("hh*16").bank("RolandTR909").gain(perlin.range(.3, .6)).cut(1).distort(.2).pan(sine.fast(2))],
  [16, s("hh*16").bank("RolandTR909").gain(.5).cut(1).crush(4).speed(sine.range(0.8, 1.2).fast(4))],
  [16, s("[hh,oh]*8").bank("RolandTR909").gain(.4).cut(1)],
  [8,  s("hh*4").bank("RolandTR808").gain(.2).rate(.5)]
)

// 🥁 Kick & Snare: The Power Source
$: arrange(
  [16, s("bd(1,8)").bank("RolandTR909").gain(.8).lpf(300).room(.8)],
  [16, s("bd(1,4)").bank("RolandTR909").gain(.9).lpf(600).room(.6)],
  [16, s("bd [~ bd] bd sd").bank("RolandTR909").gain(1).room(.4)],
  [16, s("bd(3,8) [sd,cp]").bank("RolandTR909").gain(1).distort(.1)],
  [16, s("bd*4").bank("RolandTR909").gain(1).crush(6).clip(1)],
  [16, s("bd(3,8) [sd,cp]").bank("RolandTR909").gain(1).room(.5)],
  [8,  s("bd(1,4)").bank("RolandTR909").gain(.8).lpf(300).decay(.5)]
)

// 🎸 Bass: Energy Flow
$: arrange(
  [32, silence],
  [16, note("c2").s("sawtooth").lpf(sine.range(200, 600).slow(4)).gain(.6).decay(.2).sustain(.8)],
  [16, note("<c2 eb2 f2 g2>").s("gm_synth_bass_1").lpf(1200).gain(.8).cut(1)],
  [16, note("<c2 eb2 f2 g2>").s("gm_synth_bass_2").distort(.5).gain(.7).cut(1).vib(10)],
  [16, note("<c2 eb2 f2 g2>").s("gm_synth_bass_1").lpf(1500).gain(.8).cut(1)],
  [8,  note("c2").s("sawtooth").lpf(200).decay(2).gain(.5)]
)

// 🎹 Arpeggio: Processing Data
$: arrange(
  [48, silence],
  [16, n("0 3 7 10".fast(4)).scale("C3:minor").s("square").gain(.35).lpf(sine.range(500, 2000).slow(8)).delay(".25:.5:.3")],
  [16, n("0 3 7 10 12 15".fast(8)).scale("C3:minor").s("square").gain(.4).delay(".125:.4:.4").pan(sine.slow(2))],
  [16, n("0 3 7 10".fast(4)).scale("C3:minor").s("square").gain(.35).lpf(1000).delay(".25:.5:.3")],
  [8,  silence]
)

// 🎻 Lead Synth: The "Ghost" in the Machine
$: arrange(
  [64, silence],
  [16, n("<c5 eb5 g5 c6>").s("gm_lead_8_bass_lead").gain(.5).portamento(.1).vib(5).vibmod(.3).room(.7).delay(".5:.3:.4")],
  [16, n("<c5 ~ eb5 ~>").s("gm_lead_8_bass_lead").gain(.4).lpf(sine.range(800, 2000).slow(4)).room(.8)],
  [8,  silence]
)

// 🔉 FX: System Alerts
$: arrange(
  [32, silence],
  [16, s("~ ~ ~ glitch").bank("RolandTR808").rate(4).gain(.3).delay(".1:.5:.5").rarely(x=>x.speed(2))],
  [32, s("~ glitch ~ ~").bank("RolandTR808").rate(8).gain(.4).crush(4).sometimes(x=>x.rev())],
  [24, silence]
)
