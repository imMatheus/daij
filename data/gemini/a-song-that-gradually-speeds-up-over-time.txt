// 🎵 "Kinetic Shift" — An IDM track that gradually accelerates from a deep crawl to a frantic finish

setcps(saw.slow(93).range(0.4, 2.5))
$: stack(
  s("bd:5(3,8)").lpf(400).room(0.5).mask("<1!3 0!90>".slow(93)),
  stack(
    s("bd:5*4, [~ cp]*2, hh*16"),
    n("0!8 -1!8").scale("C2:minor").s("saw").lpf(1200).lpq(12),
    n("0 2 4 6").scale("C4:minor").s("superpiano").decay(0.1)
  ).mask("<0!3 1!20 0!70>".slow(93)),
  stack(
    s("hh*32").degradeBy(0.4),
    n("3!16").scale("C2:minor").s("sine").lpf(300)
  ).mask("<0!23 1!10 0!60>".slow(93)),
  stack(
    s("bd:5*4, [~ cp:2]*2, hh*16"),
    n("0!16").scale("C2:minor").s("saw").lpf(saw.range(800, 6000)).lpq(15),
    n("0 1 2 4").scale("C4:minor").s("superpiano").decay(0.05).echo(3, 0.125, 0.6)
  ).mask("<0!33 1!60>".slow(93))
).jux(rev).delay(0.2).room(0.3)
