setcpm(90 / 4)

$: arrange(
  [8,  s("bd ~ ~ ~").bank("RolandTR909").gain(.8)],
  [16, s("bd ~ sd ~").bank("RolandTR909").gain(.8)],
  [16, s("bd ~ sd ~, hh*4").bank("RolandTR909").gain(".8 .7")],
  [16, s("bd ~ sd ~, hh*8").bank("RolandTR909").gain(".8 .6")],
  [16, s("bd bd sd sd, [hh oh]*4").bank("RolandTR909").cut(1).gain(".9 .8")],
  [16, s("bd*4, [~ sd]*2, [hh oh]*8").bank("RolandTR909").cut(1).gain(".9 .85")],
  [8,  s("bd ~ ~ ~").bank("RolandTR909").gain(saw.slow(8).range(.8, 0))]
).fast(saw.range(1, 2.5).slow(96))
