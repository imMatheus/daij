// 🎵 "Grand Genesis" — Orchestral crescendo via additive layers
setcps(0.5)
stack(
  s("heavy:2").struct("1(3,8)").gain("<1!32>").lpf(300).room(.6),
  s("superpiano").note("<c2!4 f1!4 g1!4 c2!20>").gain("<0!4 1!28>").velocity(.5),
  s("superstrings").note("<[c3,g3,c4]!8 [f3,c4,f4]!4 [g3,d4,g4]!4 [c4,g4,c5]!16>").gain("<0!8 .7!24>").room(.8).delay(.4),
  s("superhorn").note("<c4 e4 g4>").gain("<0!12 .6!20>").detune(sine.slow(8).range(0,10)),
  s("superflute").note("<g5 c6 e6 g6>").gain("<0!16 .5!16>").vib(6).room(1),
  s("superglock").note("c7(5,8,2)").gain("<0!20 .4!12>").delay(.6),
  s("snare:3").struct("~ 1 ~ 1").gain("<0!24 .8!8>").room(.5),
  s("metal:4").struct("1!4").gain("<0!28 1!4>").hpf(2000).room(1)
).reset(32).room(.3)
