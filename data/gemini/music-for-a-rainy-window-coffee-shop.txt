// 🎵 "Rainy Window Latte" — Lo-fi Jazz Ambient for a cozy coffee shop atmosphere

setcps(0.45);

samples('github:tidalcycles/dirt-samples');

stack(
  // Ambient Rain & Atmosphere
  s("pink")
    .gain(perlin.range(0.04, 0.12))
    .lpf(1200)
    .lpq(1)
    .room(0.9)
    .sz(2),

  // Lo-fi Drum Kit (Intro -> Chorus -> Bridge -> Chorus)
  s("<[~!4] [bd:4 rim:1 bd:4 rim:1]*2 [[bd:4 rim:1](3,8)] [bd:4 rim:1 bd:4 rim:1]*2>")
    .gain(0.75)
    .lpf(1500)
    .delay(0.25)
    .dt(1/6)
    .dfb(0.6)
    .postgain(1.2),

  s("<[~!4] hh*16 [hh*8] hh*16>")
    .gain(rand.range(0.15, 0.35))
    .velocity(rand.range(0.4, 0.9))
    .pan(perlin.slow(2)),

  // Mellow Electric Piano
  chord("<[~!4] [Dm9 G13 Cmaj9 Am9] [Fmaj7 Em7 Dm7 G13] [Dm9 G13 Cmaj9 Am9]>")
    .voicing()
    .s("rhodes")
    .gain(0.6)
    .room(0.8)
    .lpf(sine.range(800, 1600).slow(8))
    .jux(rev)
    .delay(0.3)
    .dt(1/3),

  // Deep Walking Sub Bass
  note("<[~!4] [d2 g1 c2 a1] [f1 e1 d1 g1] [d2 g1 c2 a1]>")
    .s("sine")
    .gain(0.8)
    .lpf(250)
    .adsr("0.1:0.4:0.5:0.6")
    .velocity(0.7),

  // Occasional Coffee Shop Foley
  s("<~!8 [~ cup:2]!4 ~!4 [~ glass:1]!4>")
    .gain(0.35)
    .room(0.7)
    .pan(rand.range(0.2, 0.8))
)
.reset("<x!32>")
.clip(0.95)
