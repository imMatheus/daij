// "Celestial Sigh" — a slow-drifting ambient soundscape with evolving filters and sparse crystalline glimmers

setcps(0.08);

stack(
  // Foundation: Deep sub drone with slow filter movement
  note("<f1 c1 d1 bb0>")
    .s("sine")
    .lpf(sine.range(60, 120).slow(8))
    .gain(0.45)
    .room(0.8)
    .slow(8),

  // Heart: Shifting harmonic pads using triangle waves and heavy resonance
  note("<[f3, a3, c4, e4] [d3, f3, a3, c4] [bb2, d3, f3, a3] [c3, e3, g3, b3]>")
    .s("triangle")
    .lpf(perlin.range(200, 1600).slow(6))
    .gain(0.22)
    .room(1)
    .delay(0.6)
    .rev()
    .slow(4),

  // Atmosphere: Generative melodic glimmers with stochastic masking
  note("f5 a5 c6 e6 g6".scramble().struct("1(3,8)"))
    .s("sine")
    .gain(perlin.range(0.03, 0.1).slow(4))
    .room(1)
    .delay(0.9)
    .lpf(sine.range(1500, 5000).slow(9))
    .slow(2)
    .mask("<1 [1 0] 1 [0 1]>"),

  // Texture: Breath-like noise floor
  s("noise")
    .gain(perlin.range(0.01, 0.04).slow(12))
    .lpf(sine.range(100, 500).slow(10))
    .hpf(80)
    .slow(16)
).reset(16)
