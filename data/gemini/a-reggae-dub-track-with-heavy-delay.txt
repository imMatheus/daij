// "Dub Pressure" — Deep sub-bass and modulated feedback loops
setcps(145/240);
samples('github:tidalcycles/dirt-samples');
stack(
  s("bd:4").struct("t f t f").gain(1.3),
  s("~ <rim:1 rim:3 snare:2>").delay(0.7).delayfb(0.6).delaytime(0.25).room(0.6),
  s("hh*16").gain(sine.range(0.3, 0.7).slow(8)).pan(perlin.slow(4)),
  s("superpiano")
    .note("<[c3,eb3,g3,c4] [f3,ab3,c4,f4] [g3,b3,d4,g4] [eb3,g3,bb3,eb4]>")
    .struct("f t f t")
    .delay(0.8).delayfb(0.75).delaytime(0.375)
    .lpf(perlin.range(300, 3000).slow(16))
    .room(0.8),
  s("bass2")
    .note("<c2 f1 g1 eb1>")
    .struct("t [~ t] t [t ~]")
    .lpf(perlin.range(150, 500).slow(12))
    .gain(1.6),
  s("can")
    .n(irand(12))
    .struct("~ ~ ~ <t f>/4")
    .delay(0.95).delayfb(0.9).delaytime(0.5)
    .room(1.2)
    .rev()
    .gain(0.7),
  s("glitch")
    .n(irand(8))
    .struct("~ t/8 ~ ~")
    .delay(0.8).delayfb(0.8).delaytime(0.125)
    .pan(1)
).reset(16)
