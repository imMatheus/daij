// 🎵 "Midnight Shift" — Soulful Alt-R&B in the style of SZA
setcps(0.4);
stack(
  s("superpiano")
    .n("<[d3'maj9, g3'maj9, a3'dom7, b3'min11]>")
    .lpf(1100)
    .room(.8)
    .delay(.4)
    .gain(.7)
    .velocity(perlin.range(.3, .6)),
  s("kick:4 [~ snare:2] [kick:4*2] [snare:2 <~ kick:4>]")
    .bank("drum")
    .gain(1.3)
    .lpf(2000),
  s("hh*16")
    .bank("drum")
    .gain(perlin.range(.2, .5))
    .pan(sine.range(.1, .9)),
  s("sine")
    .n("<d1 g1 a1 b1>")
    .lpf(150)
    .gain(1.2)
    .sustain(.8),
  s("superpiano")
    .n("<[~ a4 g4 f#4] [~ e4 d4 b3]>")
    .room(.9)
    .gain(.3)
    .lpf(sine.range(500, 2500).slow(8))
    .vib(8)
).reset(16)
