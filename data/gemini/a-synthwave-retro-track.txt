// 🎵 "Neon Night Drive" — A retro synthwave track with pulsing bass and shimmering leads

setcps(0.5);

stack(
  // Percussion: LinnDrum foundation with varying intensity
  s("bd [~ sd] bd [sd <~ sd>]").bank("linn").gain(1.1)
    .mask("<0@2 1@14 0.4@6 1@14>"),
  s("hh*16").bank("linn").gain(saw.range(0.2, 0.6).slow(2))
    .mask("<0@2 1@14 0.3@6 1@14>"),

  // Bass: The pulsing heart of the track
  note("<a1*8 f1*8 c2*8 g1*8>").s("saw").lpf(500).resonance(20).gain(0.9),

  // Chords: Ethereal 80s pads with sweeping filters
  note("<[a3,c4,e4,g4] [f3,a3,c4,e4] [c3,e3,g3,b3] [g3,b3,d4,f4]>").s("superpad")
    .lpf(sine.range(400, 1800).slow(8)).room(0.7).gain(0.6)
    .mask("<0.4@2 1@14 0.7@6 1@14>"),

  // Lead: Arpeggiated neon dreams entering in the second half of choruses
  note("a4 c5 e5 g5").fast(4).s("superchip").decay(0.06).sustain(0.1)
    .room(0.5).delay(0.3).gain(0.7)
    .mask("<0@8 1@8 0@6 1@14>").jux(rev)
).reset(36)
