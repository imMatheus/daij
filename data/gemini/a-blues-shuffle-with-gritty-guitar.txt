// "Gritty Delta Shuffle" — A distorted blues stomp with a swing feel and overdriven guitar textures

setcps(0.8);

samples('github:tidalcycles/dirt-samples');

stack(
  // The Shuffle Beat
  s("bd [~ bd] bd ~")
    .gain(1.2)
    .room(0.2),
  
  s("~ [cp,hc] ~ [cp,hc]")
    .gain(0.8)
    .delay(0.2)
    .delayfb(0.3),
  
  s("hh*8")
    .gain(sine.range(0.4, 0.7).slow(4))
    .n("<[0 2] [3 1]>")
    .velocity(0.5 + Math.random() * 0.2),

  // Gritty Guitar Chords (Shuffle Rhythm)
  note("e2*2 g2 a2 a#2".off(1/8, n => n.add(12)))
    .s("gtr")
    .gain(0.9)
    .crush(4)
    .lpf(perlin.range(800, 3000).slow(8))
    .room(0.5),

  // Lead Licks
  note("<e3 [g3 a3] b3 [d4 e4]>")
    .s("gtr")
    .struct("1(3,8, <0 2>)")
    .gain(0.8)
    .distort(0.4)
    .pan(sine.slow(10))
    .delay(0.4)
    .delayfeedback(0.6)
    .room(0.8),

  // Walking Bass
  note("e1 g1 a1 a#1 b1 d2 e2 g2".slow(2))
    .s("bass")
    .gain(1.1)
    .lpf(400)
)
.reset(8)
.swing(0.15)
