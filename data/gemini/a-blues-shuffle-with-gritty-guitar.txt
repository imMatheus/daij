// 🎵 "Rusty Fender Shuffle" — Gritty blues shuffle with overdriven guitar and walking bass
setcps(0.4);
stack(
  s("[bd!2, [~ sd]!2, [hh!4, ~ oh]]")
    .swing(4).gain(1.2).room(0.3).lpf(8000),
  note("0 3 4 5".scale("E2:blues"))
    .transpose("<0!4 5!2 0!2 7 5 0 7>")
    .s("sine").lpf(300).gain(1.1).adsr("0.02:0.2:0.4:0.1"),
  note("[0, 3, 7, 10]".scale("E3:blues"))
    .transpose("<0!4 5!2 0!2 7 5 0 7>")
    .s("sawtooth").distort("7:0.5:diode")
    .lpf(perlin.range(1000, 3000)).lpq(2).gain(0.6)
    .adsr("0.04:0.1:0.6:0.2").jux(rev)
    .mask("<0!1 1!12 0.2!4 1!12>"),
  note("0 3 5 6 7 10 12".shuffle(16).scale("E4:blues"))
    .s("pulse").pw(sine.range(0.2, 0.8).slow(4))
    .distort("9:0.4:fold").lpf(2500).vib(6).gain(0.6)
    .mask("<0!13 1!4 0!12>").delay(0.4).dt(1/6).dfb(0.6)
).reset("<x!1 x!12 x!4 x!12>")
