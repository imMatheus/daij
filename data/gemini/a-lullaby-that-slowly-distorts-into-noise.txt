// 🎵 "Fractured Sleep" — Algorithmic lullaby with progressive noise distortion

setcps(0.125)
stack(
  note("<[c5,e5,g5] [f4,a4,c5] [g4,b4,d5] [c5,e5,g5]>")
    .s("sine")
    .adsr("2.5:0.5:0.4:6")
    .gain("<0.5 [0.6!3] 0.2 [0.8!4]>")
    .pan(sine.slow(2))
    .lpf(saw.slow(9).range(4000, 800)),
  s("pink")
    .gain(saw.slow(9).range(0, 0.8))
    .distort(saw.slow(9).range(0, 12))
    .crush(saw.slow(9).range(16, 1.1))
    .lpf(saw.slow(9).range(15000, 400)),
  note("<c2 f2 g2 c2>").s("triangle")
    .gain(saw.slow(9).range(0.3, 0.1))
    .lpf(100),
  s("bd:4*4").gain(saw.slow(9).range(0, 0.35))
    .distort(20)
    .mask("<0!2 1!7>")
)
.room(0.95)
.roomsize(0.8)
.delay(0.5)
.delaytime(0.75)
.distort(saw.slow(9).range(0, 5))
.postgain(0.7)
