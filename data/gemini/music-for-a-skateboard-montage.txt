// 🎵 "Concrete Shred" — Gritty, high-energy punk-breakbeat for skateboard montages

setcps(0.9);

stack(
  // MAIN DRUMS: Punchy breakbeat with occasional jitter
  s("bd [sd bd] [bd bd] <sd [sd! sd]>").bank("tr909").gain(1.2).jux(rev)
    .reset("<x!4 x(5,8)!16 ~!8 x(7,16)!16>")
    .when("<1!4 0!16 1!8 0!16>", x => x.lpf(500).lpq(10)),

  // HI-HATS: Rapid, perlin-modulated velocity for a "rushing" feel
  s("hh*16").gain(perlin.range(0.4, 0.9)).pan(sine.slow(8))
    .mask("<0.5!4 1!16 0.2!8 1!16>"),

  // BASS: Distorted, driving pulse bass
  note("<e1 g1 a1 b1>(3,8)").s("pulse").pw(0.4).distort(5)
    .lpf(perlin.range(300, 1500).slow(2))
    .gain(0.85).reset("<x!4 x!16 x!8 x!16>")
    .when("<1!4 0!16 1!8 0!16>", x => x.lpf(200).gain(0.6)),

  // CHORDS: Crunchy punk power chords
  note("<e3 g3 a3 b3>*2").chord("m").voicing().s("sawtooth")
    .distort(1.5).gain(0.55).room(0.2)
    .mask("<0!4 1!16 0!8 1!16>"),

  // "SKATE" TEXTURE: Wood and rim hits echoing like board impact on pavement
  s("github:tidalcycles/dirt-samples/wood:[0|1|2|3]*8")
    .gain(0.75).echo(4, 0.125, 0.65).jux(rev)
    .mask("<0!20 1!8 0!16>") // Featured during the bridge
    .lpf(sine.range(1000, 8000).slow(1)),

  // BRIDGE TRANSITION: High-pass noise sweep
  s("white*16").gain(0.3).hpf(sine.range(2000, 12000).slow(4))
    .mask("<0!19 1!1 0!23 1!1>")
)
.reset("<x!4 x!16 x!8 x!16>")
.room(0.3).delay(0.2).delayfeedback(0.6).delaytime(1/3)
