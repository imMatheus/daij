// 🎵 "Mutant Flesh" — Glitch / Avant-Pop / Deconstructed Club
// At 108 BPM: intro(16) + verse(16) + build(16) + drop(16) + breakdown(16) + outro(8) = 88 bars ≈ 3:15

setcpm(108 / 4)

// 1. Texture / Atmosphere Layer
$: s("vinyl:2").gain(.15).room(1).loop(1).hpf(200)

// 2. The "Deconstructed" Beat
// Uses speed modulation to create that elastic, stretching drum sound signature to Arca
$: arrange(
  [16, silence],
  [16, s("bd ~ [sd,cp] ~").bank("RolandTR909").distort(.4).speed(sine.range(.8, 1.2).slow(4)).room(.4)],
  [16, s("bd(3,8) [sd,cp]*2 bd [sd,cp]").bank("RolandTR909").distort(.6).speed(perlin.range(.5, 1.5).fast(2)).cut(1).room(.3)],
  [16, s("bd(5,8) [sd,cp]*4 [bd,hh]*2 [sd,cp]").bank("RolandTR909").crush(4).distort(.8).speed(saw.range(.2, 2).fast(4)).cut(1).room(.2).chop(4)],
  [16, silence],
  [8,  s("bd*8").bank("RolandTR909").distort(1).speed(sine.range(0.1, 0.5).slow(8)).gain(saw.slow(8).range(1, 0))]
)

// 3. Metallic / Industrial Percussion
// Adds the clicky, ASMR-like textures
$: arrange(
  [16, s("hh*8").bank("RolandTR909").gain(perlin.range(0, .3)).hpf(5000).pan(sine.fast(4))],
  [16, s("hh*16").bank("RolandTR909").gain(perlin.range(.1, .4)).hpf(4000).speed(sine.range(1, 2).slow(2)).pan(sine.fast(8))],
  [16, s("rim*8").bank("RolandTR909").gain(perlin.range(.3, .6)).distort(.2).speed(perlin.range(0.5, 3)).pan(sine.fast(16))],
  [16, s("rim*16").bank("RolandTR909").gain(perlin.range(.4, .8)).distort(.8).speed(perlin.range(0.1, 4)).chop(8).pan(sine.fast(32))],
  [16, s("hh*4").bank("RolandTR909").gain(perlin.range(0, .2)).hpf(6000).room(1)],
  [8,  silence]
)

// 4. Distorted Sub Bass
// Deep, blown-out sine waves
$: arrange(
  [16, silence],
  [16, note("<c1 ~ eb1 ~>").s("triangle").lpf(400).distort(.6).gain(.7).sustain(2)],
  [16, note("<c1 g0 ab0 f0>").s("triangle").lpf(600).distort(.8).gain(.8).sustain(1).port(0.2)],
  [16, note("<c1 g0 ab0 f0>").s("sine").lpf(800).distort(1.2).gain(.9).sustain(1).port(0.5).vib(10).vibmod(1)],
  [16, note("<c1 ~ ~ ~>").s("sine").lpf(300).distort(.4).gain(.6).sustain(4).room(1)],
  [8,  silence]
)

// 5. Ethereal Choir / Pads
// The emotional, operatic element contrasting the harsh noise
$: arrange(
  [16, chord("<Cm9 Abmaj7>").s("gm_choir_aahs").attack(1).release(2).room(2).hpf(300).gain(.5).vib(2).vibmod(.2)],
  [16, chord("<Cm9 Abmaj7 Fm9 G7b13>").s("gm_choir_aahs").attack(.5).release(1).room(1.5).hpf(300).gain(.6).vib(4).vibmod(.3)],
  [16, chord("<Cm9 Abmaj7 Fm9 G7b13>").s("gm_choir_aahs").attack(.2).release(0.5).room(1).hpf(400).gain(.5).vib(6).vibmod(.5).chop(2)],
  [16, chord("<Cm9 Abmaj7 Fm9 G7b13>").s("gm_choir_aahs").attack(.1).release(0.1).room(.5).hpf(500).gain(.4).vib(10).vibmod(1).chop(8)],
  [16, chord("<Cm9 Abmaj7>").s("gm_choir_aahs").attack(2).release(4).room(3).hpf(200).gain(.6).vib(1).vibmod(.1)],
  [8,  chord("Cm9").s("gm_choir_aahs").attack(4).release(4).room(4).gain(saw.slow(8).range(.5, 0))]
)

// 6. Crystalline Plucks / Glitch Melody
// Fast, random, insect-like sounds
$: arrange(
  [16, note("c5 eb5 g5 c6").s("gm_glockenspiel").rarely(x=>x.rev()).speed(sine.range(0.99, 1.01).slow(2)).gain(.3).room(1)],
  [16, note("c5 eb5 g5 c6").s("gm_glockenspiel").fast(2).sometimes(x=>x.speed(2)).gain(.4).room(.8).delay(".25:.5:.5")],
  [16, note("c5 eb5 g5 c6").s("gm_glockenspiel").fast(4).struct("x(5,8)").gain(.5).distort(.2).speed(perlin.range(0.5, 1.5)).cut(1)],
  [16, note("c5 eb5 g5 c6").s("gm_glockenspiel").fast(8).struct("x(3,8)").gain(.6).distort(.8).speed(perlin.range(0.1, 2)).chop(4).cut(1)],
  [16, note("c5").s("gm_glockenspiel").slow(2).gain(.3).room(2).delay(".5:.5:.8")],
  [8,  silence]
)

// 7. Experimental Noise Swells
$: arrange(
  [16, s("wind").gain(sine.slow(16).range(0, .3)).lpf(500)],
  [16, s("wind").gain(sine.slow(8).range(.2, .5)).lpf(sine.slow(4).range(500, 2000))],
  [16, s("noise").gain(saw.fast(4).range(0, .2)).hpf(1000).pan(sine.slow(2))],
  [16, s("noise").gain(saw.fast(8).range(0, .4)).hpf(500).pan(sine.fast(4)).chop(16)],
  [16, s("wind").gain(sine.slow(16).range(.4, 0)).lpf(400)],
  [8,  silence]
)
