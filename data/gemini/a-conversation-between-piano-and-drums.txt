// 🎵 "Dialogue in Felt and Skin" — A jazz-inflected conversation between an expressive piano and a dynamic drum kit
setcps(0.5)
stack(
  note("<[c3,e3,g3,b3] [f3,a3,c4,e4] [d3,f3,a3,c4] [g2,b2,d3,f3]>")
    .s("superpiano")
    .velocity(perlin.slow(4).range(0.3, 0.7))
    .room(0.8).size(0.9)
    .struct("<t(3,8) [~ t(3,8)] t(5,8,1) [~ t]>")
    .lpf(sine.slow(16).range(400, 4000)),
  note("<[~ c5,e5,g5] [~ a4,c5,e5] [~ d5,f5,a5] [~ g4,b4,d5]>".add("[0, 12]"))
    .s("superpiano")
    .velocity(0.5)
    .delay(0.3).delayfb(0.6)
    .struct("<[~ t] t(3,8,2) [~ t(2,4)] t(5,8)>")
    .pan(perlin.slow(4)),
  s("<[bd ~ sd ~] [bd [~ bd] sd [~ sd]] [~ bd ~ sd] [[bd sd]*2]>")
    .bank("drum")
    .gain(perlin.slow(10).range(0.3, 0.9))
    .room(0.2)
    .hpf(sine.slow(14).range(100, 1200))
).reset(16)
