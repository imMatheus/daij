// 🎵 "Recursive Echoes" — Ambient Polyphonic Canon
samples('github:tidalcycles/dirt-samples')
setcps(0.45)
stack(
  s("bd:3 [~ bd:3] bd:3 [~ bd:3]").gain(0.8).lpf(800),
  s("~ hc:2 ~ hc:2").gain(0.4).room(0.5).pan(sine.slow(8)),
  note("<[c3 e3 g3 b3] [f3 a3 c4 e4] [g3 b3 d4 f4] [e3 g3 b3 d4]>")
    .s("superpiano")
    .off(1, x => x.add(7).gain(0.7).pan(0.2))
    .off(2, x => x.add(12).gain(0.5).pan(0.8))
    .room(2).sz(0.9)
    .delay(0.6).delaytime(0.375).delayfeedback(0.7)
    .lpf(sine.range(500, 3500).slow(16))
    .reset(sine.slow(32))
)
