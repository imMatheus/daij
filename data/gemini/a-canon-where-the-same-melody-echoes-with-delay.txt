// 🎵 "Recursive Echoes" — Algorithmic Canon in C Minor with Cascading Delays

setcps(0.8);
stack(
  // The Subject: A melodic line that evolves through harmonic layers
  n("<[0 3 7 10] [7 5 3 2] [3 7 12 14] [15 12 7 3]>")
    .scale("C:minor")
    .s("sawtooth")
    .adsr("0.1:0.3:0.4:1.2")
    .lpf(sine.range(400, 2800).slow(12))
    // The Followers: Creating the Canon effect with offsets and transpositions
    .layer(
      x => x.late(1).transpose(7).gain(0.5).s("triangle").lpf(1800).jux(rev),
      x => x.late(2).transpose(-12).gain(0.3).s("sine").lpf(600).jux(iter(4))
    )
    // Structure: Intro (0-4), Chorus (5-24), Bridge (25-34), Chorus (35-56)
    .mask("<1!4 1!20 0.5!10 1!22>")
    .reset("<0!4 [1(3,8)]!20 0!10 [1(5,8)]!22>")
    .room(0.8)
    .delay(0.4)
    .delaytime(0.33)
    .delayfeedback(0.7),

  // Rhythmic Foundation: Minimalist Techno-inspired percussive layers
  s("bd:2(3,8), [~ rim:4], hh*8")
    .bank("tr909")
    .gain("<0!4 0.8!20 0!10 0.9!22>")
    .lpf(perlin.range(300, 1200).slow(4))
    .room(0.4)
    .delay(0.2)
)
