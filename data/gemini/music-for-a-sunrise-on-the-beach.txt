// "Glistening Shore" — an ambient sunrise piece with warm pads and rolling waves

setcps(0.45)

stack(
  // Warm horizon pads
  note("C3:maj9*2, F3:maj9, G3:11, E3:m9")
    .s("superpiano")
    .room(0.8).size(0.9)
    .lpf(sine.range(300, 1500).slow(8))
    .gain(0.7),

  // Melodic light reflections
  note("<[c4 e4 g4 b4] [g4 b4 d5 f#5] [a4 c5 e5 g5] [f4 a4 c5 e5]>")
    .s("supervibe")
    .struct("1(3,8)")
    .delay(0.6).delayfb(0.5)
    .rev()
    .gain(sine.range(0.2, 0.5).slow(4)),

  // Rolling ocean waves
  s("noise")
    .lpf(sine.range(100, 800).slow(12))
    .hpf(150)
    .gain(sine.range(0.1, 0.4).slow(12))
    .pan(sine.slow(10)),

  // Gentle morning pulse
  s("bd:4(3,8), [~ hh]*2")
    .gain(0.5)
    .lpf(1000)
    .room(0.3),

  // Distant birds/glitch textures
  note("c6(5,16,12)")
    .s("superchip")
    .release(0.1)
    .gain(perlin.range(0, 0.3).slow(5))
    .pan(perlin.slow(3))
)
.reset(16)
.slow(2)
