// 🎵 "Midnight Haze" — A smoky, downtempo trip hop beat

setcps(0.65);

stack(
  // Dusty Trip Hop Beat
  s("bd:4 <[~ bd:4] bd:4> sd:2 <~ [~ bd:4]>")
    .bank("dirt")
    .speed(0.8)
    .mask("<[0*2] [1*16] [0*8] [1*20]>".slow(46))
    .swing(0.12)
    .room(0.2),

  s("hh*16")
    .gain(perlin.range(0.2, 0.5).fast(4))
    .mask("<[0*1] [1*45]>".slow(46))
    .speed(0.9)
    .hpf(2500),

  // Deep Sinister Bass
  note("e1 <e1 [e1 g1]> a0 <a0 [a0 b0]>")
    .s("fm")
    .lpf(250)
    .gain(0.8)
    .distort(0.1)
    .mask("<[0*2] [1*44]>".slow(46)),

  // Melancholic E-Piano
  note("<e2:min11 a2:min9>")
    .s("superpiano")
    .room(0.8)
    .sz(0.9)
    .lpf(sine.range(400, 1500).slow(8))
    .gain(0.7)
    .vowel("<a o u>".slow(20)),

  // Ghostly Lead
  note("~ [g3 b3] [~ d4] ~")
    .s("supervibe")
    .gain(0.4)
    .room(0.9)
    .mask("<[0*18] [1*28]>".slow(46))
    .delay(0.5)
    .delayfb(0.4)
    .rev(),

  // Vinyl & Atmosphere
  s("noise")
    .gain(sine.range(0.04, 0.07).slow(12))
    .hpf(3500)
    .crush(4),

  // Sampled Vocal Fragments
  s("~ <numbers:1 numbers:3> ~ ~")
    .speed(0.45)
    .gain(0.25)
    .room(1)
    .sz(0.9)
    .mask("<[0*18] [1*4] [0*24]>".slow(46))
).reset(46)
