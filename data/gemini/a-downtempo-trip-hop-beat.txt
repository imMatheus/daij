// "Midnight Dust" — a hazy downtempo trip hop beat with dusty drums and a deep bassline

setcps(84 / 60 / 4);

samples('github:tidalcycles/dirt-samples');

stack(
  s("noise").gain(0.15).lpf(sine.range(400, 1200).slow(16)).hpf(200).room(0.4),
  s("bd:4 [~ bd:4] <bd:4 [~ bd:4]> [~ bd:4]").gain(1.2).room(0.1).lpf(800),
  s("~ <sn:2 [~ sn:2]> ~ <sn:2 [sn:2*2]>").gain(0.7).room(0.6).delay(0.2).delayfb(0.3).pan(0.6),
  s("hh*16").gain(perlin.range(0.2, 0.6)).pan(sine.range(0.3, 0.7).slow(8)).cutoff(5000),
  note("<c2 f1 g1 eb1>").s("bass").lpf(300).gain(0.8).room(0.2),
  note("<[c3,eb3,g3,bb3] [f3,ab3,c4,eb4] [g3,bb3,d4,f4] [ab3,c4,eb4,g4]>")
    .s("supervibe")
    .gain(0.5)
    .room(0.9)
    .lpf(sine.range(500, 1800).slow(12))
    .delay(0.4)
    .delayfb(0.5)
    .pan(perlin.range(0.2, 0.8).slow(4))
).reset(16)
