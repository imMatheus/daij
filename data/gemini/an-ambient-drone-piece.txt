// 🎵 "Eternal Resonances" — A deep, evolving ambient drone piece featuring layers of modulated synthesis and shimmering resonances.
setcps(0.25);
stack(
  arrange([10, note("c2")], [2, note("f1")], [6, note("c2")])
    .s("sine").gain(sine.range(0.35, 0.55).slow(4)).lpf(300).room(4),
  arrange([2, silence], [8, note("c3,g3,bb3,d4")], [2, note("f2,c3,eb3,g3")], [6, note("c3,g3,bb3,d4")])
    .s("supersaw").detune(perlin.range(0.3, 0.9).slow(6)).lpf(sine.range(400, 1800).slow(10)).gain(0.22).room(5).delay(0.8),
  arrange([2, silence], [8, note("c5,g5,d6").struct("x(3,8)")], [2, silence], [6, note("c5,g5,d6").struct("x(5,8)")])
    .s("sine").gain(0.12).room(10).delay(0.95).vib(4, 0.2),
  s("pink").gain(perlin.range(0.02, 0.05).slow(12)).hpf(4000).room(2)
).reset("<x!18>")
