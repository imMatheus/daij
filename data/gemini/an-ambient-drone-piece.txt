// "Void Echoes" — a deep, evolving ambient drone with shifting spectral textures

setcps(0.25)
  .stack(
    note("c1 [g1 <c2 eb2>]").s("saw")
      .lpf(perlin.mul(1200).add(100))
      .lpq(8)
      .gain(0.6)
      .room(0.9).sz(0.9),
    note("<c3 eb3 g3 bb3>").s("sine")
      .slow(8)
      .vowel("<a i o u>")
      .gain(0.3)
      .hpf(perlin.mul(2000))
      .delay(0.7).delayfb(0.8),
    note("g2 [c3 <f3 g3>]").s("triangle")
      .slow(4)
      .distort(sine.mul(0.2))
      .lpf(400)
      .gain(0.2)
      .pan(perlin),
    s("hh*16")
      .gain(perlin.mul(0.1))
      .crush(2)
      .lpf(sine.mul(5000))
      .room(0.5)
  )
  .reset(16)
  .slow(2)
  .room(0.8)
  .gain(0.8)
