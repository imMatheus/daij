// 🎵 "Peachy State" — Smooth R&B Pop with a Justin Bieber feel

setcps(100/60/4);
samples('github:tidalcycles/dirt-samples');

stack(
  // 🥁 Drums: Pop-Trap blend
  s("bd:4 [~ bd:4] bd:4 <~ bd:4>").gain(1.2),
  s("~ cp:2 ~ cp:2").gain(0.9).room(0.3),
  s("hh:8*16").gain(sine.range(0.4,0.8).slow(4)).pan(perlin),
  s("~ [~ hc:3] ~ [hc:3 ~]").gain(0.6),

  // 🎹 Chords: Soft airy keys
  note("<[eb3,g3,bb3,d4] [c3,eb3,g3,bb3] [ab2,c3,eb3,g3] [bb2,d3,f3,ab3]>")
    .s("superpiano")
    .velocity(0.6)
    .room(0.6)
    .jux(rev)
    .lpf(sine.range(1200, 2500).slow(8)),

  // 🎸 Bass: Warm syncopated sub
  note("<eb2 c2 ab1 bb1>")
    .s("superbass")
    .lpf(400)
    .gain(1.1)
    .struct("1(3,8)"),

  // 🎤 Lead: Dreamy pluck melody
  note("<g4 f4 eb4 d4>(3,8,<0 2>)")
    .s("supersaw")
    .gain(0.25)
    .room(0.8)
    .delay(0.5)
    .lpf(sine.range(1000, 3000).slow(8))
    .decay(0.05)
    .release(0.4)
).reset(8)
