setcps(1);

stack(
  note("<c1!3 [c1,eb1,g1,bb1]!30 [f1,ab1,c2,eb2]!12 [c1,eb1,g1,bb1]!30>".scale("C:minor"))
    .s("sawtooth")
    .lpf(sine.slow(20).range(80, 500))
    .lpq(3)
    .gain(0.5)
    .room(0.8).size(0.95),

  note("<~!3 [c5 g5 eb6 f5 g5 c6]!30 [eb5 f5 g5 ab5]!12 [c5 g5 eb6 f5 g5 c6]!30>".scale("C:minor").sometimes(rev))
    .s("sine")
    .gain("<0!3 0.3!30 0.1!12 0.35!30>")
    .delay(0.7).delaytime(0.35).delayfeedback(0.7)
    .room(0.9).size(0.8)
    .pan(sine.slow(15)),

  s("<~!3 [bd:4 ~ bd:4 ~ ~ ~ ~ ~]!30 ~!12 [bd:4 ~]!30>")
    .gain(0.4)
    .lpf(sine.slow(12).range(100, 300))
    .room(0.6),

  s("noise").gain(perlin.slow(30).range(0.04, 0.12))
    .hpf(perlin.slow(10).range(3000, 6000))
    .pan(cosine.slow(25))
).reset("<x!75>")
