// "Submerged Echoes" — a sparse, haunting James Blake-inspired soundscape with deep sub and fragmented vocals

setcps(0.65)

stack(
  // Fragile Piano Chords
  note("~ [g2,bb2,d3,f3] ~ <[eb2,g2,bb2,d3] [c2,eb2,g2,bb2]>")
    .s("superpiano")
    .velocity(0.4)
    .room(0.8).size(0.9)
    .lpf(sine.range(400, 1200).slow(4)),

  // Deep Sub Bass
  note("<g1 eb1 c1 d1>")
    .s("supersaw")
    .lpf(150)
    .resonance(0.2)
    .gain(0.8)
    .adsr(0.2, 0.1, 0.8, 0.5),

  // Sparse, Glitchy Percussion
  s("bd:4 [~ sd:2] ~ <[~ bd:4] ~>")
    .gain(0.6)
    .delay(0.4).delayfb(0.3)
    .room(0.2),

  s("~ [hh*3] ~ [hh*2]")
    .gain(sine.range(0.2, 0.5).slow(3))
    .pan(perlin.slow(2)),

  // Ghostly Vocal Fragments
  n("<7 5 3 0>/2")
    .s("vocals")
    .speed("<0.5 0.75 1 0.5>")
    .gain(0.4)
    .room(1).size(0.95)
    .delay(0.6).delayfb(0.7)
    .cut(1)
    .pan(sine.slow(5))
)
.reset(8)
.lpf(sine.range(1000, 3000).slow(12))
