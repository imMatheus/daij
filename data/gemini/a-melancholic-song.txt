// 🎵 "Echoes of a Long Gone Autumn" — A melancholic ambient piano and slow lo-fi beat

setcps(0.8);

stack(
  // Evolving Ambient Bed
  s("sine")
    .note("<c3 eb3 f3 g3>")
    .gain(0.35)
    .attack(2)
    .release(4)
    .room(0.9)
    .lpf(sine.range(200, 900).slow(8))
    .vibrato(3, 0.15),

  // Melancholic Piano (Chorus Sections)
  s("piano")
    .note(scale("c:minor", "[0 2 4 7] ~ [5 4 2 0] ~").add("<0!4 -2!4>"))
    .gain(0.55)
    .velocity(perlin.range(0.3, 0.6))
    .room(0.7)
    .delay(0.4)
    .delayfeedback(0.6)
    .mask("<0!2 1!20 0!10 1!20 0!4>"),

  // Somber Low End
  s("saw")
    .note("<c2 eb2 f2 g2>")
    .lpf(300)
    .gain(0.45)
    .attack(1)
    .release(2)
    .mask("<0!2 1!50 0!4>"),

  // Sparse, Dust-Covered Drums (Fades in after intro)
  s("bd:4 [~ rim:3] bd:4 <~ [rim:3 rim:3]>")
    .gain(0.4)
    .lpf(800)
    .room(0.3)
    .hpf(200)
    .mask("<0!6 1!16 0!10 1!20 0!4>"),

  // Distant, Drifting Lead (Bridge only)
  s("triangle")
    .note(scale("c:minor", "[12 11 9 7]").add(12))
    .gain(0.15)
    .attack(1.5)
    .release(3)
    .delay(0.6)
    .delayfeedback(0.7)
    .mask("<0!22 1!10 0!24>"),

  // Texture: Rain/Vinyl Crackle
  s("noise")
    .gain(0.1)
    .lpf(sine.range(1000, 3000).slow(10))
    .hpf(500)
    .room(0.5)
).reset(56)
