// 🎵 "Crown of the Victor" — Triumphant Orchestral Hybrid Anthem
setcps(135/120);
stack(
  // --- Heroic Brass Section ---
  s("superbrass")
    .note("<[eb3,g3,bb3] [bb2,d3,f3] [c3,eb3,g3] [ab2,c3,eb3]>")
    .sustain(1).lpf(sine.range(800, 2400).slow(16))
    .fast(38).gain("[0.4 0.6 0.8!15 0.5!6 0.9!15]/38"),
  
  // --- Lead Trumpet Melody ---
  s("superlead")
    .note("<[eb4 ~ g4 ~] [bb4 ~ eb5 ~] [d5 ~ c5 ~] [bb4 ~ ~ ~]>")
    .sustain(0.2).vibrato(5).detune(4)
    .jux(p => p.rev().transpose(12).gain(0.4))
    .fast(38).gain("[0!2 0.8!15 0!6 1.0!15]/38"),

  // --- Marching Foundations ---
  s("bd:3")
    .struct("[1 [~ 1] 1 [~ 1]]*2")
    .fast(38).gain("[0.7!2 1.2!15 0.6!6 1.3!15]/38"),
  
  s("sd:2")
    .struct("~ 1 ~ 1")
    .crush(5).lpf(3000)
    .fast(38).gain("[0!2 0.9!15 0!6 1.1!15]/38"),

  s("hh:27")
    .struct("1*16")
    .hpf(6000)
    .fast(38).gain("[0!2 0.5!15 0.3!6 0.6!15]/38"),

  // --- Shimmering Textures ---
  s("supermandolin")
    .note("<eb5 g5 bb5 c6>")
    .struct("1(5,8)")
    .delay(0.4).delayfeedback(0.6).room(0.5)
    .fast(38).gain("[0!17 0.6!6 0.8!15]/38"),

  // --- Orchestral Accents ---
  s("industrial:4")
    .struct("1 ~ ~ ~")
    .lpf(sine.range(500, 1500).slow(4)).decay(0.6)
    .fast(38).gain("[0!2 0.8!15 0.6!6 1.0!15]/38")

).slow(38).room(0.7).size(0.9)
