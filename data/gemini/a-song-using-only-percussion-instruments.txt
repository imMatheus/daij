// 🥁 "Rhythm Protocol" — Tribal Percussion Ensemble
// At 110 BPM: intro(8) + groveA(16) + break(8) + groveB(24) + climax(16) + outro(8) = 80 bars ≈ 2:54

setcpm(110 / 4)

$: arrange(
  [8,  s("gm_taiko_drum").struct("x ~ ~ ~").gain(.8).room(.4)],
  [16, s("gm_taiko_drum(3,8)").struct("x ~ x ~ x ~ x ~").gain(.9).room(.3)],
  [8,  s("gm_taiko_drum").struct("x*4").gain(saw.slow(8).range(.6,.2)).room(.5)],
  [24, s("gm_taiko_drum(5,8)").gain(.95).room(.3).sometimes(x=>x.fast(2))],
  [16, s("gm_taiko_drum(3,8), bd*2").bank("RolandTR909").gain(1).room(.2)],
  [8,  s("gm_taiko_drum").struct("x ~ ~ ~").gain(saw.slow(8).range(.7,0)).room(.6)]
)

$: arrange(
  [8,  silence],
  [16, s("~ gm_conga_high ~ gm_conga_low").gain(.85).room(.3).swing(4)],
  [8,  s("gm_snare_drum(3,8)").gain(.7).room(.6).lpf(1200)],
  [24, s("~ [gm_conga_high,sd:1] ~ [gm_conga_low,sd:1]").bank("RolandTR808").gain(.9).room(.4).every(4, x=>x.press())],
  [16, s("sd*4").bank("RolandTR909").gain(perlin.range(.7,1)).room(.3).every(2, x=>x.fast(2))],
  [8,  silence]
)

$: arrange(
  [8,  s("gm_maracas*8").gain(perlin.range(.4,.7)).pan(sine.slow(4))],
  [16, s("[hh hh oh hh]*2").bank("RolandTR808").cut(1).gain(.8).room(.2)],
  [8,  s("gm_cabasa*16").gain(perlin.range(.3,.6)).hpf(2000)],
  [24, s("[hh hh oh hh]*2, gm_maracas*16").bank("RolandTR808").cut(1).gain(.85).room(.2)],
  [16, s("hh*16").bank("RolandTR909").gain(1).room(.1).every(4, x=>x.fast(1.5))],
  [8,  s("gm_maracas*8").gain(saw.slow(8).range(.6,0))]
)

$: arrange(
  [8,  silence],
  [16, silence],
  [8,  s("gm_agogo(5,8)").n("<0 2 4 7>").gain(.75).room(.5).delay(".2:.1:.3")],
  [24, s("gm_agogo(5,8), gm_cowbell(3,8)").n("<0 2 4 7>").gain(.8).room(.4).delay(".15:.1:.2")],
  [16, s("gm_agogo(7,8), gm_tinkle_bell(1,4)").n("<7 9 12 7>").gain(.85).room(.3).delay(".1:.05:.15")],
  [8,  silence]
)

$: arrange(
  [8,  s("gm_woodblock(3,8)").n("<0 5 0 7>").gain(.7).room(.4).pan(-0.5)],
  [16, s("gm_woodblock(3,8)").n("<0 5 0 7>").gain(.75).room(.3).pan(sine.slow(8))],
  [8,  silence],
  [24, s("gm_woodblock(5,8), gm_castanets(2,8)").n("<0 2 5 7>").gain(.8).room(.4).press()],
  [16, s("gm_woodblock*8").n("<12 7 5 0>").gain(.85).room(.2).sometimes(x=>x.rev())],
  [8,  s("gm_woodblock(1,4)").gain(.6).room(.6)]
)

$: arrange(
  [8,  silence],
  [16, n("<c1 eb1 ab0 g0>").s("gm_timpani").gain(.8).room(.5).lpf(800)],
  [8,  n("<c2 eb2 ab1 g1>").s("gm_tubular_bells").gain(.6).room(.8).delay(".4:.3:.5")],
  [24, n("<c1 eb1 ab0 g0>").s("gm_timpani").gain(.85).room(.4).lpf(1000)],
  [16, n("<c1 eb1 f1 g1>").s("gm_timpani").struct("x(3,8)").gain(.9).room(.3)],
  [8,  n("c0").s("gm_timpani").gain(saw.slow(8).range(.7,0)).room(.7)]
)
