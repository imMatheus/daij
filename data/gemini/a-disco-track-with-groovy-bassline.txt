// "Electric Disco" — A driving disco floor-filler with a syncopated bassline and phased chords

setcps(126/120);

stack(
  s("bd:4*4").gain(1.3),
  s("[~ cp:3]*2").gain(0.9).room(0.3),
  s("hh*16").gain(sine.range(0.3, 0.8).slow(2)).pan(perlin.slow(4)),
  note("0 [~ 12] 7 10 <0 12> [~ 3] 5 0".scale("minor").add("<0 0 5 -2>"))
    .s("fm")
    .gain(1.1)
    .cutoff(perlin.range(200, 1500).slow(4))
    .resonance(0.2),
  note("<[0,3,7,10] [5,8,12,15] [0,3,7,10] [-2,2,5,9]>".scale("minor"))
    .s("supervibe")
    .gain(0.5)
    .attack(0.01).release(0.1)
    .lpf(sine.range(800, 3000).slow(12))
    .room(0.6)
    .jux(rev),
  s("gretsch:4(5,8)").gain(0.6).hpf(3000).room(0.5)
).reset(16)
