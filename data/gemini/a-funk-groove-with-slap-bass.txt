// 🎵 "Thump & Pop" — A gritty funk groove featuring slap bass and syncopated drums
samples('github:tidalcycles/dirt-samples');
setcps(112/60/4);
const chorus = stack(
  s("bd:3 [~ bd:3] bd:3 [~ bd:3], [~ sd:2]*2, [hh*8, ~ hh:4*2]").gain(1.1).pan(perlin.range(-0.05, 0.05)),
  s("bass3").n("e1 [~ e1] <g1 a1> [b1 [~ d2]]").velocity(perlin.range(0.9, 1.4)).lpf(950).crush(3).room(0.05),
  s("bass3").n("e2 [~ e2] <g2 a2> [b2 [~ d3]]").velocity(1.1).hpf(2600).gain(0.75).delay(0.07).release(0.12),
  s("superp5").chord("Em9").struct("~ 1 [~ 1] 1").lpf(sine.range(800, 2600).slow(4)).gain(0.22)
);
const bridge = stack(
  s("bd:3 [~ bd:3], [~ sd:2], hh*4").gain(0.85),
  s("bass3").n("<a1*4 b1*4>").velocity(0.95).lpf(800).room(0.1),
  s("superp5").chord("<Am11*2 Bm11*2>").struct("1*4").gain(0.35).lpf(1100)
);
const intro = stack(s("bd:3 [~ bd:3]"), s("hh*16").gain(0.45));
cat(intro, ...Array(14).fill(chorus), ...Array(4).fill(bridge), ...Array(14).fill(chorus))
