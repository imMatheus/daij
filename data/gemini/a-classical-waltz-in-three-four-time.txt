// "Grand Vienna Waltz" — A classical 3/4 waltz with mechanical precision
setcps(180/240)

stack(
  s("superpiano")
    .note("<c2 f1 g1 c2>")
    .struct("1 0 0")
    .gain(1.3)
    .room(.4),

  s("superpiano")
    .note("<[e3,g3,c4] [f3,a3,c4] [g3,b3,d4] [e3,g3,c4]>")
    .struct("0 1 1")
    .velocity(0.6)
    .room(.5),

  s("superpiano")
    .note("<[c4 e4 g4] [f4 a4 c5] [g4 b4 d5] [c5 g4 [e4 f4]]>")
    .velocity(perlin.range(0.4, 0.9))
    .room(0.8)
    .delay(0.3)
    .delayfeedback(0.4),

  s("supervibe")
    .note("<c5 b4 a4 g4>".slow(4))
    .struct("1*3")
    .gain(0.3)
    .room(1.5)
    .lpf(sine.range(400, 2200).slow(8))
).reset(16)
