// 🎵 "Viennese Moonlight" — A classical waltz in three-four time for piano and strings

setcps(165/60/3);

stack(
  // Main Waltz Piano (Oom-pah-pah)
  s("superpiano")
    .note("<[c2 [e3 g3] [e3 g3]] [f2 [f3 a3] [f3 a3]] [g2 [g3 b3] [g3 b3]] [c2 [e3 g3] [e3 g3]]>")
    .velocity(sine.range(0.4, 0.7).slow(4))
    .gain(0.95),

  // Orchestral Strings (Chorus Sections)
  s("superreese")
    .lpf(sine.range(1200, 2500).slow(8))
    .attack(0.3).release(1.2)
    .note("<[c5 e5 g5] [f5 a5 c6] [g5 f5 d5] [e5 d5 b4]>".scale("C:major"))
    .mask("<0*3 1*24 0*10 1*27>".slow(64))
    .room(0.8).size(0.9).gain(0.45)
    .jux(rev),

  // Cello Counter-melody (Minor Bridge)
  s("superreese")
    .lpf(600)
    .attack(0.4).release(1.5)
    .note("<c3 g3 eb3 f3 c3 d3 g2 c3>".scale("C:minor"))
    .struct("1*3")
    .mask("<0*27 1*10 0*27>".slow(64))
    .gain(0.6).room(0.6).pan(-0.3),

  // High Woodwind Ornaments
  s("supervibe")
    .lpf(4000)
    .attack(0.1).release(0.4)
    .note("<[~ ~ c6] [e6 d6 c6] [~ ~ g6] [f6 e6 d6]>".scale("C:major"))
    .mask("<0*10 1*17 0*10 1*27>".slow(64))
    .gain(0.35).pan(perlin.slow(4))
    .delay(0.4).delayfb(0.3)

).reset(64)
