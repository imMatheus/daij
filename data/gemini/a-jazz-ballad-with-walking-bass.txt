// 🎵 "Velvet Dusk" — A smoky jazz ballad featuring a fluid walking bass and lush piano voicings

samples('github:tidalcycles/dirt-samples');

setcps(0.17);

stack(
  // Rhythmic Foundation: Brushed Ride, subtle Kick, and ghosted Snare
  arrange(
    [2, s("jazz:4*4").gain(0.25)],
    [4, s("jazz:4*4, [~ jazz:1?0.2]*2, [jazz:0?0.5]!4").gain(0.5).swing(4)],
    [2, s("jazz:4*4, [jazz:1*2]?0.3, jazz:0(3,8)").gain(0.4).swing(4)],
    [4, s("jazz:4*4, [~ jazz:1?0.2]*2, [jazz:0?0.5]!4").gain(0.5).swing(4)]
  ).room(0.5).delay(0.2),

  // Walking Bass: Quarter note melodic movement with chromatic chord connections
  arrange(
    [2, n("f1 a1 c2 d2").s("bass").gain(0.7)],
    [4, n("<[f1 a1 c2 e2] [d1 f1 a1 c2] [g1 bb1 d2 f2] [c1 e1 g1 bb1]>").s("bass").gain(0.9)],
    [2, n("<[bb0 d1 f1 a1] [a0 c1 e1 g1] [ab0 c1 eb1 gb1] [g0 b0 d1 f1]>").s("bass").gain(0.8)],
    [4, n("<[f1 a1 c2 e2] [d1 f1 a1 c2] [g1 bb1 d2 f2] [c1 e1 g1 bb1]>").s("bass").gain(0.9)]
  ).lpf(350).release(0.15),

  // Harmonic Layer: Lush jazz piano extensions with organic velocity variations
  arrange(
    [2, silence],
    [4, chord("<Fmaj9 Dm11 Gm9 C13>").voicing().s("piano").gain(0.4)],
    [2, chord("<Bbmaj7 A7alt Abmaj9 G7alt>").voicing().s("piano").gain(0.35)],
    [4, chord("<Fmaj9 Dm11 Gm9 C13>").voicing().s("piano").gain(0.4)]
  ).attack(0.1).release(2.5).room(0.8).velocity(perlin.range(0.35, 0.65)),

  // Lead Voice: Melancholic saxophone phrases with deep spatial effects
  arrange(
    [2, silence],
    [4, n("<[~ a4 ~ g4] [~ f4 ~ e4] [g4 ~ a4 ~] [~ bb4 ~ a4]>").s("sax").gain(0.5)],
    [2, n("<d5 ~ c5 ~ [b4 a4] ~ [g4 f4]>").s("sax").gain(0.45)],
    [4, n("<[~ a4 ~ g4] [~ f4 ~ e4] [g4 ~ a4 ~] [~ g4 ~ f4]>").s("sax").gain(0.5)]
  ).attack(0.25).release(2).room(0.7).delay(0.6).velocity(perlin.range(0.4, 0.75)).vib(4.5).vibmod(0.25)
).reset(12)
