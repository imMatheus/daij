// "Fractured Breaks" — high energy breakbeat with chopped vocal fragments and gritty textures

setcps(172/120/4)

.stack(
  // Chopped Breakbeat
  s("~ [~ amit] <[~ amit] [~ amit*2]> [amit ~]")
    .bank("amit")
    .speed(1.1)
    .cut(1)
    .room(0.2).sz(0.5)
    .gain(0.9),

  // Amen-style layering
  s("amencorn:3*8")
    .slice(8, "0 [3 2] [4 1] 7 [2 5] [6 3] 1 [0 4]")
    .speed(1.2)
    .gain(0.8)
    .lpf(sine.range(1000, 5000).segment(8))
    .pan(perlin.range(0.2, 0.8)),

  // Deep Sine Bass
  note("c1 <c1 g0 f1 c1>")
    .s("sine")
    .decay(0.2).sustain(0.1)
    .lpf(400)
    .gain(1.2),

  // Chopped Vocal Samples
  s("voodoo:4*16")
    .slice(16, "<[0 4 2 8] [12 1 15 3] [7 10 5 13] [2 2 1 0]>")
    .speed(saw.range(0.8, 1.2).segment(4))
    .cut(2)
    .delay(0.4).dt(1/6).dfb(0.5)
    .room(0.6).sz(0.8)
    .gain(perlin.range(0.4, 0.7).segment(16)),

  // Glitchy Highs
  s("hh*16")
    .euclid(11, 16)
    .crush(sine.range(2, 8).segment(4))
    .hpf(4000)
    .gain(0.5)
)
.reset(4)
