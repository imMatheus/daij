// 🎵 "Golf Radio" — Tyler-inspired wonky soul with gritty textures and quirky synth patterns

setcps(0.7);

samples('github:tidalcycles/dirt-samples');

stack(
  // Wonky Rhodes Chords with slight pitch modulation for a vintage feel
  note("<[f3,a3,c4,e4,g4] [e3,g3,b3,d4,f#4] [d3,f3,a3,c4,e4] [g2,b2,d3,f3,a3]>")
    .s("superpiano")
    .gain(0.8)
    .lpf(perlin.range(500, 1500).slow(8))
    .lpq(15)
    .room(0.5)
    .vib(5, 0.1)
    .delay(0.2).delayfb(0.3),

  // Crunchy, off-kilter drums with heavy kick and layered percussion
  s("[bd:6 ~ [~ bd:6] ~] [sn:2 ~ ~ ~] [bd:6 bd:6 ~ ~] [sn:2 ~ ~ ~]")
    .gain(1.1)
    .stack(
      s("hc*16").gain(sine.range(0.2, 0.6).slow(4))
    )
    .stack(
      s("~ rim:2").struct("0 1 0 1").gain(0.7)
    ),

  // Quirky, high-pitched lead using superchip for that 8-bit flavor Tyler often blends
  note("<[c5 e5 g5 b5]*2 [b4 d5 f#5 a5]*2>")
    .s("superchip")
    .decay(0.05)
    .sustain(0.02)
    .gain(0.5)
    .pan(perlin.slow(2))
    .lpf(2000)
    .room(0.3),

  // Deep, distorted sub bass to ground the track
  note("<f2 e2 d2 g1>")
    .s("supervibe")
    .lpf(150)
    .gain(1.2)
    .distort(0.2),

  // Atmospheric vinyl-like noise for texture
  s("noise").gain(0.12).lpf(300).hpf(100)
).reset(16)
