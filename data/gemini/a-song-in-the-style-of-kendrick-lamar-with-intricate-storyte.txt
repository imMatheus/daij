// "Compton Narrative" — Shifting jazz textures and syncopated beats inspired by K.Dot's storytelling.

setcps(1.15)
samples('github:tidalcycles/dirt-samples')

stack(
  s("bd(5,8), <[~ sn:2]!3 [sn:2 sn:2]>, [hh*8, <[~ hc]*2>]")
    .gain(1.1)
    .room(.1),
  note("<c2!4 eb2!4 f2!4 [g2!2 bb1!2]>")
    .s("bass")
    .lpf(400)
    .gain(.9),
  note("<[c3,eb3,g3,bb3]!2 [eb3,g3,bb3,d4] [d3,f3,a3,c4]>")
    .s("superpiano")
    .struct("x(3,8,1)")
    .room(.5)
    .gain(.6)
    .clip(sine.range(0,0.2).slow(4)),
  note("c4 eb4 g4 bb4 c5".segment(32).scramble().slow(2))
    .s("supervibe")
    .gain(perlin.range(0,0.4).slow(8))
    .delay(.3)
    .room(.9),
  s("feel:3")
    .slow(8)
    .gain(.3)
    .room(1)
    .delay(.5)
    .rev()
).reset(32).lpf(perlin.range(1500, 10000).slow(24))
