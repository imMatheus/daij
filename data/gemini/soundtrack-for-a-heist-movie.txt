// 🎵 "The Vault Breach" — Tense Cinematic Heist
setcps(0.5);
samples('github:tidalcycles/dirt-samples');

stack(
  s("bd:4 [~ sd:2] bd:4 [sd:2 <~ [~ sd:2]>]")
    .gain("<0!2 0.8!2 1.1!11 0!5 1.2!15>")
    .room("<0.1!15 0.4!5 0.2!15>"),
  s("hh*16")
    .gain(perlin.range(0.3, 0.6).fast(4))
    .mask("<0!1 1!14 1!5 1!15>")
    .jux(rev),
  s("rim(5,8)")
    .gain("<0!15 0.7!5 0.8!15>")
    .delay(0.5).dt(0.125),
  note("<c2(3,8) [c2!3 eb2] [c2!3 gb2]>")
    .s("sawtooth")
    .lpf("<400!2 650!13 350!5 950!15>")
    .lpq(4).lpenv(2).lpa(0.01)
    .gain("<0.6!2 1!13 0.8!5 1.2!15>"),
  note("~ <[c4,eb4,gb4] [eb4,gb4,bb4] [d4,f4,ab4]>")
    .s("pulse").pw(0.5)
    .adsr("0.01:0.08:0:0.1")
    .gain("<0!8 0.6!7 0.2!5 0.7!15>")
    .room(0.8).delay(0.5).dt(0.375),
  note("<[c3,eb3,g3] [ab2,c3,eb3] [g2,b2,d3]>")
    .s("sine")
    .gain("<0!15 0.4!5 0.1!15>")
    .lpf(sine.range(300, 1000).slow(4))
    .room(1).slow(2)
)
.reset("<x!4 [~ x]!11 x!5 [x*2]!15>")
.slow(35)
