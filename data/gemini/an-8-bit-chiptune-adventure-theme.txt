// 🎵 "Pixel Quest: The Final Level" — 8-bit Chiptune Adventure
// At 135 BPM: intro(8) + verse1(16) + chorus1(16) + verse2(16) + chorus2(16) + bridge(16) + outro(8) = 96 bars ≈ 2:50

setcpm(135 / 4)

$: arrange(
  [8,  s("bd [~ cp] bd [~ cp]").bank("RolandTR808").crush(1).gain(.7)],
  [16, s("bd [~ sd:1] bd [~ sd:1]").bank("RolandTR808").crush(1).gain(.8)],
  [16, s("bd sd:1 bd [sd:1 sd:1], [hh*4]").bank("RolandTR808").crush(2).gain(.9).cut(1)],
  [16, s("bd [~ sd:1] bd [~ sd:1]").bank("RolandTR808").crush(1).gain(.8)],
  [16, s("bd sd:1 bd [sd:1 sd:1], [hh*4]").bank("RolandTR808").crush(2).gain(.9).cut(1)],
  [16, s("bd ~ bd ~, [~ sd:1]*2").bank("RolandTR808").crush(3).gain(.7)],
  [8,  s("bd ~ ~ ~").bank("RolandTR808").crush(1).gain(saw.slow(8).range(.7, 0))]
)

  [8,  s("hh*4").bank("RolandTR808").crush(8).gain(saw.slow(8).range(.4, 0))]
