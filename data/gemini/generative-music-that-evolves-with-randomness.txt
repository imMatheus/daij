// 🎵 "The Stochastic Drift" — Generative Glitch Electronica

samples('github:tidalcycles/dirt-samples')
setcps(0.5)

stack(
  // Percussion: Shifting Euclidean rhythms with randomized accents
  s("bd(<3 5 4>, 8), ~ [sd:4|cp]", "hh*<8 16 32 64>/4")
    .sometimesBy(0.12, x => x.ply("<2 3>"))
    .gain(perlin.range(0.65, 0.95))
    .pan(sine.slow(8))
    .mask("<0!3 1!12 0.4!5 1!15>"),

  // Generative Bass: Random Phrygian degrees with fluctuating LPF
  n(irand(7).segment("<4 8 2 16>"))
    .scale("C2:phrygian")
    .s("sawtooth")
    .lpf(perlin.range(150, 1800).slow(10))
    .lpq(rand.range(2, 12))
    .gain(0.35)
    .lfo({c: 'lpf', s: 4, dr: 0.3})
    .mask("<1!3 1!12 0.6!5 1!15>"),

  // Chaotic Lead: Shuffled fragments in higher octaves with glitchy distortion
  n(run(16).shuffle(8).segment(16))
    .scale("C:phrygian")
    .transpose("<12 19 24 31>")
    .s("sine")
    .jux(x => x.rev().transpose(0.1))
    .distort(rand.range(0, 6))
    .delay(0.5).dt(1/8).dfb(0.65)
    .room(0.8)
    .gain(rarely(x => x.gain(0.75), 0.35))
    .mask("<0!5 0.5!10 0.2!5 0.6!15>"),

  // Ambient Layer: Washed out random pulses
  n(irand(12).segment(1))
    .scale("C5:phrygian")
    .s("triangle")
    .attack(0.5).release(2)
    .room(1).sz(2)
    .gain(0.15)
    .mask("<0!1 1!34>")
)
.reset(8)
.pianoroll({ labels: 1, smear: 1 })
