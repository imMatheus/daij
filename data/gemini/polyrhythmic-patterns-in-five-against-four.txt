// 🎵 "Quintessential Pulse" — Polyrhythmic IDM with 5:4 layering

samples('github:tidalcycles/dirt-samples');
setcps(0.45);
stack(
  s("bd*4, [~ sd]*2")
    .gain("<0.6!1 0.9!13 0.3!5 0.9!13>")
    .room(0.2).delay(0.1),
  s("hh*5, [~ hc]*5")
    .gain("<0.3!1 0.7!13 0.1!5 0.7!13>")
    .pan(perlin.range(0.1, 0.9).slow(4))
    .velocity(sine.range(0.5, 0.9).slow(2)),
  note("c2*4").s("saw")
    .gain("<0!1 0.6!13 0.3!5 0.6!13>")
    .lpf(sine.range(200, 1000).slow(8))
    .resonance(0.2),
  note("c4 d4 eb4 g4 bb4")
    .s("superfm")
    .gain("<0.2!1 0.5!13 0.8!5 0.5!13>")
    .jux(rev)
    .delay(0.4).delayfeedback(0.6)
    .room(0.5)
    .reset(4)
    .lpf(sine.range(1000, 5000).slow(12))
)
