// 🎵 "Neon Hemlock" — Yves Tumor style psychedelic experimental rock

setcps(0.45)

stack(
  // Heavy, saturated drums
  s("bd [~ sd:1] bd [~ <sd:1 cp>]").bank("gretsch").distort(0.2).gain(1.1).room(0.3),
  s("hh*8").bank("gretsch").gain(perlin.range(0.3, 0.7)).pan(sine.slow(4)),
  s("~ [~ <sd:2!3 [sd:2*2]>]").bank("gretsch").gain(0.8).delay(0.2).delayfb(0.4),

  // Fuzzy, driving bassline
  note("e1 [e1 <g1 b0>] a1 [g1 <d2 b1>]").s("bass8")
    .lpf(sine.range(400, 1200).slow(8))
    .distort(0.4)
    .gain(0.9)
    .resonance(0.2),

  // Psychedelic, detuned pads
  note("e3:min9 [a3:min11 b3:min9]").s("superchip")
    .detune(perlin.slow(2).range(-40, 40))
    .room(0.9).size(0.9)
    .gain(0.5)
    .lpf(sine.range(1000, 3000).slow(12)),

  // Ethereal, gliding lead
  note("<e4 g4 b4 a4> <b4 a4 g4 f#4>").s("supervibe")
    .slow(2)
    .vowel("<a e i o>".slow(8))
    .delay(0.5).delayfb(0.6)
    .room(1)
    .gain(0.4)
    .pan(perlin.slow(2)),

  // Gritty guitar-like textures
  note("e3*2 [g3 a3] b3*2 [a3 g3]").s("supermandolin")
    .legato(2)
    .distort(0.6)
    .gain(perlin.slow(4).range(0.2, 0.5))
    .lpf(2000)
    .room(0.5)

).reset(16)
