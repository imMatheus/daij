// 🎵 "The Hollow Corridor" — A suspenseful and eerie track with dark drones and metallic echoes

setcps(0.5);
samples('github:tidalcycles/dirt-samples');

stack(
  // Low-end dissonant drone
  s("sine").note("<c1 eb1 [c1 b0] [ab0 g0]>").slow(4).gain(0.8).lpf(400).room(0.6).distort(0.2),
  
  // Ghostly atmospheric texture
  s("cosmicg").n(2).note("<c4 b3 eb4 f#4>").slow(8).gain(0.5).room(0.9).sz(0.9).lpf(perlin.range(300, 1200)),
  
  // Tense heartbeat kick
  s("bd:4").struct("1(3,8,2)").gain(0.9).room(0.4).mask("[0*2 1*33]"),
  
  // Metallic scraping echoes
  s("alphabet").n(12).struct("1(5,8)").speed("<0.4 0.35 0.3>").gain(0.5).room(1).mask("[0*6 1*29]"),
  
  // High-pitched unsettling melody
  note("<c6 [b5 g5] [f#5 eb5] [g5 ab5]>").s("triangle").decay(0.1).room(1).delay(0.7).gain(0.25).mask("[0*10 1*25]"),
  
  // Bridge tension: creeping filter saw
  note("c3*4").s("saw").decay(0.05).lpf(sine.range(400, 2000)).resonance(15).gain(0.6).mask("[0*14 1*6 0*15]"),
  
  // Bridge tension: white noise swell
  s("noise").gain(perlin.range(0, 0.4)).lpf(perlin.range(200, 5000)).mask("[0*14 1*6 0*15]"),
  
  // Random metallic clinks
  s("metal:2").struct("~ 1").room(1).sz(0.9).gain(0.4).mask("[0*4 1*31]")
).reset("[0*14 1 0*20]")
