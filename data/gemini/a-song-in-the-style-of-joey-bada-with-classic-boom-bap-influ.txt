// 🎵 "Pro Era Soul" — Joey Bada$ style boom bap with jazzy Rhodes and dusty rhythms
setcps(90/60/4);
stack(
  s("bd:4 [~ bd:4] ~ [~ bd:4]").gain(1.2).delay(0.01),
  s("~ sd:2 ~ sd:2").gain(1.1).room(0.2).delay(0.02),
  s("hh*8").gain(sine.range(0.4, 0.7).slow(4)).pan(perlin),
  s("[~ hh:3]*2").gain(0.5).hpf(2000),
  note("<[c3,eb3,g3,bb3,d4] [f2,ab2,c3,eb3,g3] [bb2,d3,f3,a3,c4] [g2,bb2,d3,f3]>")
    .s("rhodes")
    .gain(0.6)
    .lpf(sine.range(800, 1500).slow(16))
    .room(0.5)
    .delay(0.2).delayfeedback(0.4),
  note("<c2 [~ c2] f1 [g1 g1]>")
    .s("bass2")
    .lpf(250)
    .gain(1.0),
  s("~ ~ ~ sax:1").gain(0.2).room(0.9).lpf(1200).hpf(400),
  s("crackle*4").gain(0.2).pan(sine)
).reset()
