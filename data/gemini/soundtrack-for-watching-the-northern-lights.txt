// 🎵 "Emerald Shimmer" — Ethereal Polar Ambient Soundscape

setcps(0.08);
stack(
  s("superpwm")
    .note("<[c2, g2, c3] [ab1, eb2, ab2] [f2, c3, f3] [g2, d3, g3]>")
    .lpf(perlin.range(200, 1500).slow(12))
    .lpq(0.3)
    .room(1)
    .sz(0.98)
    .gain(0.55)
    .slow(8),
  s("supervibe")
    .note("<[g4, c5, eb5, bb5] [f4, bb4, d5, a5]>".scramble())
    .struct("1(5,12, <0 2 4>)")
    .room(1)
    .sz(0.95)
    .delay(0.8)
    .delayfb(0.8)
    .gain(perlin.range(0.1, 0.35).slow(6))
    .hpf(sine.range(1000, 5000).slow(18))
    .pan(sine.slow(20)),
  s("superhex")
    .note("<c6 eb6 g6 bb6>*[4 | 8 | 16]")
    .gain(perlin.range(0.02, 0.15).slow(9))
    .room(1)
    .hpf(perlin.range(3000, 8000).slow(10))
    .pan(perlin.slow(14)),
  s("supernoise")
    .lpf(sine.range(40, 500).slow(32))
    .gain(perlin.range(0.1, 0.4).slow(28))
    .room(1)
    .sz(1)
    .pan(sine.slow(40))
).reset(32)
