// 🎵 "Velvet Whispers" — A romantic and tender melody with soft electric piano and ethereal synth swells

samples('github:tidalcycles/dirt-samples');
setcps(0.3);

stack(
  // Main Harmonic Progression - Gentle Electric Piano (Sine-based)
  note("<[c3,g3,c4,e4] [a2,e3,a3,c4] [f2,c3,f3,a3] [g2,d3,g3,b3]>".slow(4))
    .s('sine')
    .adsr(0.1, 0.2, 0.6, 2.5)
    .velocity(perlin.range(0.4, 0.7))
    .room(0.9)
    .lpf(sine.range(800, 1400).slow(8)),

  // Tender Lead Melody - Soft Triangle Synth
  note("<[~ c5] [~ g5] [~ e5] [~ b4] [~ d5] [~ a4] [~ c5] [~ g4]>".slow(4))
    .s('triangle')
    .gain("<0@1 0.5@8 0.2@4 0.6@8 0@1>".slow(22))
    .adsr(0.2, 0.1, 0.5, 1.8)
    .delay(0.7)
    .room(1)
    .pan(sine.slow(12)),

  // Heartbeat Percussion - Soft Kick
  s("bd(3,8, <0 2>)")
    .gain("<0@1 0.4@8 0.1@4 0.5@8 0@1>".slow(22))
    .lpf(300)
    .shape(0.4)
    .room(0.4),

  // Delicate Rhythm - Ghosted High Hats
  s("hc*16?")
    .gain("<0@1 0.2@8 0@4 0.3@8 0@1>".slow(22))
    .velocity(perlin.range(0.1, 0.3))
    .hpf(7000)
    .room(0.6),

  // Ethereal Bridge Pad - Soft Saw with Vowel Filter
  note("<d3 e3 f3 g3>".slow(4))
    .s('saw')
    .lpf(sine.range(300, 700).slow(6))
    .gain("<0@9 0.4@4 0@9>".slow(22))
    .adsr(3, 1, 0.6, 4)
    .vowel('o')
    .pan(sine.slow(5))
)
.reset(22)
