// "Faded Polaroids" — A nostalgic lo-fi track with dusty piano and tape-warped textures
setcps(82/120);
stack(
  s("noise*8").gain(perlin.range(0.02, 0.05)).lpf(1200).hpf(300),
  n("<[f3,a3,c4,e4] [d3,f3,a3,c4] [g3,b3,d4,f4] [c3,e3,g3,b3]>").s("superpiano")
    .gain(0.4).velocity(0.3).lpf(sine.range(400, 900).slow(10))
    .detune(perlin.range(-15, 15).slow(4)).room(0.6),
  n("<f2 d2 g2 c2>").s("superchip").gain(0.35).lpf(120),
  n("~ <c5 [~ a4]> <g4 [~ e4]> <f4 [~ d4]>").s("superpiano")
    .gain(0.2).room(0.9).delay(0.8).mask("<0!2 1!8 0.2!4 1!11>"),
  s("bd [~ sd:2] bd sd:2").bank("gretsch").gain(0.55).lpf(700)
    .mask("<0!3 1!7 0!4 1!11>")
).reset(25)
