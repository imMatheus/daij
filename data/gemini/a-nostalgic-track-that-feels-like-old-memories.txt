  arrange(

  .vib(perlin.range(1, 4)).vibmod(0.2)
  .delay(0.5).dt(1/3).dfb(0.6)

  arrange(

  .vib(perlin.range(4, 8)).vibmod(0.15)
  .lpf(1600)

  .gain(0.6)
