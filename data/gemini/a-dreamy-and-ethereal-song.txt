// 🎵 "Luminous Drift" — Ethereal ambient soundscape with shifting filters and sparkling textures

setcps(0.25);

stack(
  s("supervibe")
    .note("<[f3,a3,c4,e4] [g3,b3,d4,f4] [d3,f3,a3,c4] [e3,g3,b3,d4]>")
    .slow(4)
    .sustain(12)
    .gain(0.6)
    .lpf(sine.range(400, 1200).slow(16)),

  s("superpiano")
    .note("<c5 e5 g5 b5>*8".shuffle())
    .mask("<t(5,8) t(3,8) t(7,16)>")
    .iter(8)
    .gain(perlin.range(0.2, 0.5).slow(12))
    .jux(rev)
    .delay(0.75)
    .delayfb(0.8),

  s("superfork")
    .note("<c2 f2 d2 e2>".slow(4))
    .gain(0.4)
    .sustain(4)
    .lpf(300),

  s("supernoise")
    .gain(perlin.range(0.05, 0.12).slow(7))
    .hpf(sine.range(2000, 5000).slow(20))
    .room(1)
)
.room(0.9)
.size(0.8)
.reset(16)
.lpf(sine.range(1000, 6000).slow(32))
