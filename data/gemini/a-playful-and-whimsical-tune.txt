// "The Juggling Jester" — A bouncy, shifting melody with toy-like textures and unpredictable rhythms

setcps(0.6)

stack(
  s("bd:5 [~ cp:2] [~ bd:5] <cp:2 [cp:2*2]>")
    .gain(0.9)
    .room(0.2),
  s("hc*8").gain(sine.range(0.4, 0.8)),
  s("gl(3,8)").n(irand(4)).gain(0.6).delay(0.3).delayfb(0.4).room(0.5),
  note("<c3 g3> <f3 d3> <e3 a2> <g2 b2>")
    .s("superchip")
    .decay(0.05)
    .gain(0.7)
    .lpf(sine.range(400, 3000)),
  note("c4(3,8) e4(3,8,1) g4(3,8,2) b4(3,8,3)")
    .s("supervibe")
    .struct("x(5,8)")
    .iter(4)
    .room(0.6)
    .size(0.8)
    .pan(perlin),
  note("<[c5 d5 e5 f5] [g5 a5 b5 c6]>")
    .s("supermando")
    .gain(0.5)
    .jux(rev)
    .delay(0.2).delayfb(0.5)
)
.reset(8)
