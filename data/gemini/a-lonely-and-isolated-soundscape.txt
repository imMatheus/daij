// 🎵 "Desolate Horizon" — A lonely and isolated soundscape

setcps(0.125);

stack(
  s("sine")
    .note("<d1 f1 g1 a1>")
    .gain(0.4)
    .lpf(sine.range(100, 600).slow(16))
    .room(0.9)
    .size(0.95),
    
  s("triangle")
    .note("<[~ a5] [~ ~ d6] [~ f5] [~ ~ c6]>")
    .struct("1(3,8)")
    .gain(perlin.range(0.05, 0.2).slow(8))
    .room(1)
    .delay(0.9)
    .delayfb(0.7)
    .delayt(0.375)
    .pan(sine.slow(12)),
    
  s("noise")
    .gain(perlin.range(0.02, 0.1).slow(14))
    .hpf(sine.range(400, 3000).slow(20))
    .room(0.8)
    .pan(perlin.slow(15)),
    
  s("saw")
    .note("<d2 a2 c2 e2>")
    .lpf(sine.range(200, 800).slow(10))
    .gain(0.15)
    .room(0.9)
    .slow(2)
    .vowel("<a e i u>")
).reset(64)
