// 🎵 "The Heat Death of a Groove" — A rhythmic industrial loop that disintegrates into a granular void
samples('github:tidalcycles/dirt-samples');
setcps(125/60/4);
stack(
  s("bd [~ sd] [bd*2] [~ sd]").gain(0.8),
  s("hh*16").gain(0.4).pan(sine.slow(8)),
  s("saw").note("0 [3 7] 5 [10 12]").struct("1*16").lpf(3000).lpq(15).gain(0.6),
  s("superpiano").note("12 15 17 19").struct("1(5,8)").room(0.2).delay(0.3)
)
.slow("1 1 2 4 8 16 32 64".slow(32))
.chop("1 1 16 64 128 256 512 1024".slow(32))
.delay("0 0 0.5 0.7 0.8 0.9 0.95 0.98".slow(32)).delayfb(0.8)
.room("0.1 0.1 0.5 0.7 0.9 0.95 0.98 1".slow(32))
.lpf("5000 5000 2000 1000 500 200 100 50".slow(32))
.gain("1 1 0.9 0.8 0.7 0.6 0.4 0.2".slow(32))
