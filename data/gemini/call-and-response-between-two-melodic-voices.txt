// 🎵 "Spectral Dialogue" — A rhythmic call and response between two distinct algorithmic synth voices
setcps(1.1);
stack(
  s("bd").struct("1(5,8)").bank("drum").gain(0.8),
  s("sd").struct("~ 1").room(0.4).gain(0.6),
  s("hh*16").gain(range(0.2, 0.6, saw.slow(4))),
  n("<[c5,g5]!2 ~ ~>").s("saw")
    .add(irand(7).slow(8))
    .struct("1(3,8,2)")
    .decay(0.05).sustain(0.05).room(0.8).gain(0.5).pan(0.2)
    .lpf(sine.range(1000, 4000).slow(4)),
  n("<~ ~ [f4,c5]!2>").s("triangle")
    .sub(irand(7).slow(4))
    .struct("~ ~ 1(5,8,1)")
    .decay(0.3).sustain(0.2).delay(0.5).delayfb(0.7).gain(0.6).pan(0.8)
    .lpf(2000),
  n("<c2 g1 f1 d2>").s("sine").lpf(120).gain(0.9)
).reset(16)
