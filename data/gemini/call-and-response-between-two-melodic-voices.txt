// 🎵 "Binary Dialogue" — A melodic IDM piece featuring a call-and-response between a warm piano and a glitchy digital synth

setcps(105/60/4)
stack(
  // Percussive Foundation: Kick, Snare and ghost notes
  s("bd [~ sd] [bd bd] <sd [sd*2]>")
    .mask("<0!2 1!12 0!4 1!12>")
    .room(0.2)
    .gain(0.8),
  // Shimmering Hi-hats: Constant rhythmic movement
  s("hh*16")
    .gain(perlin.range(0.2, 0.5).slow(4))
    .mask("<1!30>"),
  // Voice A (The Call): Lower-register piano chords
  note("<[g2,b2,d3,f#3] [a2,c3,e3,g3]> ~")
    .s("superpiano")
    .velocity(perlin.range(0.5, 0.8).slow(2))
    .pan(0.2)
    .mask("<1!14 0.4!4 1!12>")
    .lpf("<4000!14 800!4 4000!12>"),
  // Voice B (The Response): High-register digital chime
  note("~ <[c5,a4,f4,e4] [d5,b4,g4,f#4]>")
    .s("superchip")
    .decay(0.4)
    .pan(0.8)
    .mask("<0!2 1!12 1!4 1!12>")
    .delay(0.3)
    .room(0.5)
    .lpf("<4000!14 1200!4 4000!12>")
)
.reset(30)
.clip(0.9)
