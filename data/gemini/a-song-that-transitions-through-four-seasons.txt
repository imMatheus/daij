// 🎵 "The Turning Year" — Ambient/Glitch transition through four seasons

samples('github:tidalcycles/dirt-samples');

setcps(0.8);

stack(
  // Percussive Lifecycle
  s(cat(
    "hh*8, [bd(3,8), ~]", // Spring: Germination
    "[bd*2, [~ sd], hh*16, [~ cp/4]]", // Summer: Full Bloom
    "[[bskick*2, ~], [~ cp], hh(5,8)]", // Autumn: Decay
    "[~ [glitch:5, glitch:2?], ~]" // Winter: Stasis
  ).slow(16))
  .gain(cat(0.5, 0.8, 0.6, 0.3).slow(16).smooth(0.8))
  .pan(perlin.slow(4)),

  // Harmonic Atmosphere
  note(cat(
    "c3'maj9", // Spring
    "f3'maj11", // Summer
    "d3'min9", // Autumn
    "eb2'maj7#11" // Winter
  ).slow(16).struct("1*4"))
  .s("superfm")
  .lpf(cat(1500, 6000, 1000, 350).slow(16).smooth(0.9))
  .room(0.9)
  .size(0.8)
  .gain(0.4),

  // Melodic Fragments
  note(cat(
    "[c4 d4 e4 g4]*2",
    "[f4 g4 a4 b4 c5 e5]*2",
    "[d4 f4 g4 a4]*2",
    "[eb4 f4 g4 bb4]"
  ).slow(16).struct("<1(3,8) 1(5,8)>").off(0.125, x => x.add(12)))
  .s("superpiano")
  .velocity(sine.slow(8).range(0.3, 0.7))
  .crush(cat(0, 0, 4, 16).slow(16))
  .delay(0.4)
  .delayfb(0.6)
  .delaytime(cat(0.2, 0.1, 0.3, 0.5).slow(16))
).reset(64)
