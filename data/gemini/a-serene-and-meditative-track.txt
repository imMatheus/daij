// 🎵 "Serene Silence" — A meditative ambient track

setcps(0.2);

stack(
  // Harmonic foundation
  note("<[c3,e3,g3,b3] [f2,c3,e3,a3] [g2,d3,f3,b3] [e2,b2,d3,g3]>")
    .s("sine")
    .sustain(8)
    .gain(0.4)
    .lpf(sine.range(300, 800).slow(16))
    .room(0.9)
    .size(0.9),

  // Drifting melody
  note("g4 e4 b4 c5 d5".slow(3).scramble())
    .s("triangle")
    .gain(0.25)
    .sustain(2)
    .room(0.8)
    .delay(0.7)
    .dfb(0.5)
    .pan(perlin.slow(12)),

  // Deep resonant pulse
  note("c2").s("sine")
    .gain(0.3)
    .lpf(150)
    .sustain(16)
    .room(1)
    .slow(4),

  // Ethereal texture
  s("noise")
    .gain(sine.range(0.02, 0.08).slow(20))
    .lpf(sine.range(100, 500).slow(24))
    .room(0.9)
    .pan(sine.slow(15))
).reset(64)
