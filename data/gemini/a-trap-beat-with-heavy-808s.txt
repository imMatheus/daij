// 🎵 "Obsidian Core" — Dark Trap with heavy distorted 808s and aggressive percussion
setcps(140 / 60 / 4);
samples('github:tidalcycles/dirt-samples');
stack(
  // 808 BASS - Heavy distortion, pitch envelope for thump, and low pass filtering
  arrange(
    [2, silence],
    [16, note("<c1!3 f1 g1!3 b0>(3,8,2)").s("sine").adsr(".02:.6:.2:.1").penv(36).pdec(.04).distort(8).lpf(400).lpq(4)],
    [6, note("c1").s("sine").lpf(100).gain(0.5).adsr(".5:1:.5:1")],
    [16, note("<c1!3 f1 g1!3 b0>(3,8,2)").s("sine").adsr(".02:.6:.2:.1").penv(48).pdec(.05).distort(12).lpf(350).lpq(5).gain(1.1)]
  ),
  // DRUMS - Fast hi-hat rolls, standard trap clap, and transient-boosted kicks
  arrange(
    [2, s("hh*16").gain(perlin.range(.4, .7))],
    [16, stack(
      s("hh*[16 [32 16] 16 [16 32]]").gain(perlin.range(.6, 1)).sometimes(x => x.ply(2)),
      s("~ cp").bank("tr909").gain(0.8).room(0.2),
      s("bd(3,8,2)").bank("tr808").gain(1.3).distort(0.5)
    )],
    [6, s("hh*8").gain(.4).lpf(2000).delay(0.2)],
    [16, stack(
      s("hh*[16 [64 16] 16 [32 16 32]]").gain(perlin.range(.7, 1.1)).often(x => x.ply("<2 4>")),
      s("~ cp").bank("tr909").gain(0.9).room(0.4).delay(0.1),
      s("bd(3,8,2)").bank("tr808").gain(1.5).distort(0.8)
    )]
  ),
  // MELODY - Dark Phrygian synth motif with movement and texture
  arrange(
    [2, note("c3 eb3 g3 ab3".scale("c:phrygian").iter(4)).s("saw").lpf(1200).lpq(2).gain(.5)],
    [16, note("c3 eb3 g3 ab3".scale("c:phrygian").iter(4)).s("saw").lpf(1500).lpq(3).gain(.4).jux(rev).delay(.3).room(.2)],
    [6, note("c3 eb3 g3 ab3".scale("c:phrygian").iter(2)).s("saw").lpf(sine.range(400, 1000).slow(2)).lpq(6).gain(.6).room(2)],
    [16, note("c3 eb3 g3 ab3".scale("c:phrygian").iter(4)).s("saw").lpf(1800).lpq(4).gain(.4).jux(rev).delay(.4).room(.3).distort(0.3)]
  )
).reset("<x(1,16)>")
