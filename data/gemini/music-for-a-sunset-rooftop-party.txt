// 🎵 "Rooftop Solstice" — Nu-Jazz Deep House with lush Rhodes and sunset warmth
setcps(1.95);

stack(
  // Foundation: Solid 909 pulse with evolving hats
  s("bd [~ bd] bd <~ bd>").bank("rolandtr909").gain(1.2),
  s("~ <cp!3 [cp*2]>").bank("rolandtr909").gain(0.7).room(0.3).delay(0.2),
  s("hh*16").gain(perlin.range(0.3, 0.7).slow(4)).bank("rolandtr909").pan(sine.slow(4)),
  s("~ <ho!7 [ho ho ho]>").bank("rolandtr909").gain(0.5).hpf(2000),

  // Bass: Deep and melodic
  note("<g1!2 c2!2 f1!2 d1!2>").s("sine").gain(0.9).lpf(400).decay(0.1),

  // Chords: Jazzy electric piano textures
  note("<[g2,b2,d3,f3] [c3,e3,g3,b3] [f2,a2,c3,e3] [d2,f2,a2,c3]>")
    .s("triangle").struct("1(3,8)")
    .velocity(0.4).lpf(sine.range(600, 1500).slow(8))
    .room(0.6).delay(0.4).delaytime(1/3).delayfb(0.4),

  // Atmosphere: Shimmering sunset reflections
  note("g4(5,8,<0 2 4 6>)").s("saw").gain(sine.range(0, 0.2).slow(16))
    .lpf(2000).room(0.8).rev().jux(rev)
).reset(16)
