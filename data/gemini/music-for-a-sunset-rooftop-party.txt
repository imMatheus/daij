// 🎵 "Golden Hour Chill" — Melodic Deep House for a rooftop sunset session
setcps(115 / 60 / 4)

stack(
  // 🎹 Atmospheric Rhodes Chords
  n("<[c4,e4,g4,b4] [a3,c4,e4,g4] [f3,a3,c4,e4] [g3,b3,d4,f4]>")
    .s("sawtooth").gain(0.25).lpf(sine.slow(16).range(600, 1400))
    .adsr("0.2:0.4:0.6:1").room(0.9).distort(0.05).slow(2),

  // 🎸 Smooth Bouncy Bass
  n("<0 [0 3] 5 0 7 [~ 5] 3 2>".scale("C:major").sub(24))
    .s("sine").gain(0.85).lpf(160).decay(0.18).sustain(0)
    .struct("x(5,8,1)")
    .mask("<0!2 1!12 0!5 1!15>"),

  // 🌴 Tropical Pluck Lead
  n("<7 5 12 7 [9 7] 5 3 2>".scale("C:major").add(12))
    .s("triangle").gain(0.45).decay(0.12).sustain(0).room(0.7).delay(0.4)
    .struct("x(3,8,2)")
    .jux(rev)
    .mask("<0!2 1!12 0!5 1!15>"),

  // 🥁 Rooftop Rhythm Section
  s("bd:4*4").gain("<0!2 1.1!12 0!5 1.1!15>").distort(0.1),
  s("[~ cp:2]*2").gain("<0!2 0.8!12 0!5 0.8!15>").room(0.4),
  s("hh*16").gain(perlin.range(0.2, 0.45)).pan(sine.slow(4))
    .mask("<1!34>")
)
.room(0.4)
.reset("<x(1,34)>")
