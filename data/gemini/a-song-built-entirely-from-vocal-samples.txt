// 🎵 "Oral Logic" — Algorithmic vocal-only experimental pop

setcps(137/60/4);
stack(
  // Percussive Mouth Noises (Kick, Snare, Hat approximations)
  s("mouth:0(3,8), mouth:2, [~ mouth:1]*4")
    .gain("<1!14 0.6!8 1!14>".slow(40).mask("<0!2 1!36 0!2>".slow(40)))
    .room(0.2),
  // Deep Vocal Bass - Filtered for sub-heavy texture
  s("vocal:0")
    .note("<c2 eb2 f2 g1>".slow(4))
    .lpf(sine.range(200, 600).slow(8))
    .gain("<0!2 1!14 0.5!8 1!14 0!2>".slow(40)),
  // Breathing Vowel Pad - Long ethereal chords
  s("vocal:4")
    .note("<[c4,g4,bb4] [f4,a4,c5] [g4,bb4,d5] [eb4,g4,bb4]>".slow(4))
    .room(1)
    .delay(0.5)
    .delayfb(0.5)
    .legato(1)
    .gain("<0!4 0.7!12 0!8 0.7!14 0!2>".slow(40)),
  // Glitchy Alphabet Bridge - Granular vocal texture
  s("alphabet:12 alphabet:0 alphabet:1 alphabet:8")
    .speed("<0.5 1.5 0.75 1.25>")
    .chop(16)
    .room(0.6)
    .gain("<0!16 0.9!8 0!16>".slow(40))
    .pan(perlin.slow(4)),
  // Vowel Lead - Shifting formants and melodies
  s("vocal:2")
    .note("<c5*2 [eb5 g5] bb5 f5>".slow(2))
    .vowel("<a e i o>".slow(8))
    .room(0.5)
    .gain("<0!6 0.8!10 0.2!8 0.8!14 0!2>".slow(40))
).reset(40)
