// "Rainy Windowpane" — A warm, lo-fi study for a cozy afternoon
setcps(0.45);
stack(
  s("noise*32")
    .gain(perlin.range(0.02, 0.12))
    .lpf(perlin.range(200, 1800).slow(4))
    .room(0.95).sz(0.9),
  note("<[f3'maj9, a2'm11] [bb2'maj9, c3'9sus4]>")
    .s("superpiano")
    .velocity(perlin.range(0.3, 0.7))
    .lpf(600)
    .room(0.8)
    .delay(0.75).delayfb(0.4)
    .jux(rev),
  s("[bd <[~ sd] [~ [~ sd]]>, [~ hh]*4]")
    .gain(0.55)
    .lpf(1200)
    .room(0.3)
    .distort(0.1),
  s("hh*16")
    .gain(perlin.range(0.1, 0.45))
    .hpf(6000)
    .pan(sine.slow(8)),
  note("<f2 [g2 c3]>")
    .s("supervibe")
    .gain(0.5)
    .lpf(450)
    .room(0.6)
    .slow(2)
).reset(16)
