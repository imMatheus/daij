stack(
  s("superpiano")
    .note("<[c3,e3,g3,b3] [a2,c3,e3,g3] [f2,a2,c3,e3] [g2,b2,d3,f3]>")
    .struct("1(3,8)")
    .velocity(perlin.range(0.3, 0.4).slow(4))
    .lpf(sine.range(400, 1000).slow(12))
    .room(0.8).size(0.9),
  s("superpiano")
    .note("~ <g4 a4 f4 e4>")
    .velocity(0.15)
    .delay(0.6).delaytime(1/3).delayfb(0.5),
  s("superpiano")
    .note("<c2 a1 f1 g1>")
    .velocity(0.3)
    .lpf(200),
  s("bd:5([3 5],8)")
    .gain(0.6)
    .lpf(500),
  s("sn:2")
    .struct("~ 1 ~ 1")
    .gain(0.4)
    .room(0.2)
    .lpf(2000),
  s("hh*[8 12]")
    .velocity(perlin.range(0.1, 0.3).slow(2))
    .gain(0.25)
    .hpf(3000),
