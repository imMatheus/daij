// 🎵 "Midnight Ink" — Mellow lo-fi study beats with jazzy chords and dusty percussion
setcps(82/60/4).stack(
  // Vinyl crackle and ambient floor
  s("crackle*16").gain(0.12).pan(rand).lpf(1800),
  // Jazzy Rhodes-style keys (Dm9 - G13 - Cmaj9 - Fmaj7)
  note("<[d3,f3,a3,c4,e4] [g2,b2,f3,a3,e4] [c3,e3,g3,b3,d4] [f2,a2,e3,g3,c4]>")
    .s("piano").gain(0.65).adsr("0.4:1.5:0.4:2.5")
    .lpf(perlin.range(600,1200).slow(8)).room(0.85).delay(0.5).delayfb(0.4).jux(rev),
  // Lo-fi drum pattern: Intro(1), Chorus(10), Bridge(3), Chorus(10)
  s("<~ [bd ~ <rim!3 rim:2> [~ bd]]!10 [~ rim!2 ~]!3 [bd ~ <rim!3 rim:2> [~ bd]]!10>")
    .gain(0.6).lpf(1000).room(0.3).swing(6).velocity("0.8 1 0.7 0.9"),
  // Shuffling hi-hats
  s("<~ [hh*16]!10 [hh*8]!3 [hh*16]!10>")
    .gain(rand.range(0.2, 0.4)).pan(sine.slow(8)).hpf(4000),
  // Warm sub bass
  note("<d2 g1 c2 f1>").s("sine").gain("<0 [0.8]!10 [0.4]!3 [0.8]!10>")
    .lpf(150).adsr("0.1:0.5:0.7:0.5").velocity(0.8)
).reset("<x!4 ~!2 x!4 ~!2>")
