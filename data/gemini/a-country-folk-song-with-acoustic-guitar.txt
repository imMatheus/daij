// 🎵 "Autumn Valley" — A warm country folk tune with acoustic guitar strums and a steady walking bass

setcps(0.45)

$: stack(
  // Acoustic Guitar (Synthesized)
  arrange(
    [2, note("[g3,b3,d4]!2").arp("<0 2 1 2>")], // Intro
    [12, note("<[g3,b3,d4]!2 [c3,e3,g3] [d3,f#3,a3]>").arp("<0 1 2 3 1 2 0 2>*2")], // Chorus
    [6, note("<[a3,c4,e4] [d3,f#3,a3]>").arp("<0 2 1 2>*2")], // Bridge
    [12, note("<[g3,b3,d4]!2 [c3,e3,g3] [d3,f#3,a3]>").arp("<0 1 2 3 1 2 0 2>*2")] // Chorus
  )
    .s("sawtooth").gain(0.75).adsr("0.01:0.1:0.05:0.15")
    .lpf(perlin.range(1200, 2800)).lpq(1.2).room(0.5).jux(rev)
  ,
  // Walking Acoustic Bass
  arrange(
    [2, silence],
    [12, note("<g2!2 c2 d2>").struct("x ~ x x")],
    [6, note("<a2 d2>").struct("x ~ x x")],
    [12, note("<g2!2 c2 d2>").struct("x ~ x x")]
  )
    .s("sine").gain(0.9).lpf(300).decay(0.4)
  ,
  // Rustic Percussion
  arrange(
    [2, silence],
    [12, s("[bd:4,hh:8] [hh:2,sd:2] [bd:4,hh:8] [hh:2,sd:2]")],
    [6, s("[bd:4,hh:8] [hh:2,sd:2]")],
    [12, s("[bd:4,hh:8] [hh:2,sd:2] [bd:4,hh:8] [hh:2,sd:2]")]
  )
    .gain(0.65).room(0.1).delay(0.2)
)
.slow(32).reset("<x!32>")
