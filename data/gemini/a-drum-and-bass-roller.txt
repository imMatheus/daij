// "Iron Current" — Gritty drum and bass roller with morphing reese bass and industrial textures

samples('github:tidalcycles/dirt-samples')

setcps(174/60/4)

stack(
  s("bd:3 [~ sd:2] [~ bd:3] [~ sd:2]")
    .gain(1.2)
    .every(8, x => x.fast(2).gain(0.9)),
  s("hh*16")
    .gain(perlin.range(0.4, 0.8))
    .pan(sine.slow(4)),
  s("hc(11,16)")
    .n("<0 1 2 3>")
    .gain(0.5)
    .hpf(4000)
    .room(0.2),
  note("<c1 [c1*2] f0 [g0 <bb0 c1>]>")
    .s("saw")
    .detune(sine.slow(2).range(10, 50))
    .lpf(perlin.slow(4).range(150, 1600))
    .lpq(25)
    .gain(0.8)
    .distort(0.4)
    .shape(0.3)
    .every(12, x => x.lpf(5000).gain(0.6)),
  note("<c4 eb4 g4 bb4>(3,8)")
    .s("sine")
    .gain(sine.slow(16).range(0.1, 0.4))
    .room(1)
    .delay(0.5)
    .delayfb(0.5),
  s("feel:3(3,8)")
    .gain(0.4)
    .room(0.8)
    .rev()
    .pan(perlin.slow(3))
).reset(64)
