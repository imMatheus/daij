// 🎸 "Blown Speaker" — Garage Rock
// At 132 BPM: intro(8) + verse(16) + chorus(16) + verse2(16) + chorus2(16) + bridge(8) + outro(8) = 88 bars ≈ 2:40

setcpm(132 / 4)

// gritty room noise
$: s("vinyl:2").gain(.1).room(1).loop(1).lpf(3200)

// drums — loose, punchy, trashy
$: arrange(
  [8,  s("bd ~ ~ ~, ~ ~ ~ ~, hh*8")
       .bank("RolandTR909").gain(".75 .6").room(.4).swing(2)],
  [16, s("bd ~ bd ~, ~ sd ~ sd, hh*8, ~ ~ ~ oh")
       .bank("RolandTR909").cut(1).swing(2)
       .gain(perlin.range(.8,.98))
       .room(.35)
       .every(4, x => x.fast(2).gain(.6))],
  [16, s("bd bd ~ bd, ~ sd ~ [sd sd], [hh oh]*4, ~ ~ cp ~")
       .bank("RolandTR909").cut(1).swing(2)
       .gain(".95 .88 .92 .88")
       .room(.3).distort(.15)
       .every(3, x => x.press())],
  [16, s("bd ~ bd ~, ~ sd ~ sd, hh*8, ~ ~ ~ oh")
       .bank("RolandTR909").cut(1).swing(2)
       .gain(perlin.range(.8,.98))
       .room(.35)
       .sometimes(x => x.gain(.5))],
  [16, s("bd bd ~ bd, ~ sd ~ [sd sd], [hh oh]*4, ~ ~ cp ~")
       .bank("RolandTR909").cut(1).swing(2)
       .gain(".95 .88 .92 .88")
       .room(.3).distort(.15)
       .every(4, x => x.fast(2).gain(.55))],
  [8,  s("bd ~ sd ~, ~ ~ ~ cp, [oh ~ hh ~]*2")
       .bank("RolandTR909").cut(1).swing(3)
       .gain(".85 .78").room(.45).crush(10)],
  [8,  s("bd ~ ~ ~, ~ sd ~ ~")
       .bank("RolandTR909").swing(2)
       .gain(saw.slow(8).range(.8,.1)).room(.5)]
)

// bass — raw, driving root notes
$: arrange(
  [8,  silence],
  [16, note("<[e2 e2 ~ e2] [e2 ~ e2 ~] [a1 a1 ~ a1] [b1 ~ b1 ~]>")
       .s("sawtooth")
       .lpf(900).decay(.2).sustain(.6)
       .distort(.25).gain(.72).room(.2)],
  [16, note("<[a1 a1 a1 ~] [a1 ~ a1 a1] [d2 d2 ~ d2] [e2 ~ e2 ~]>")
       .s("sawtooth")
       .lpf(1100).decay(.15).sustain(.65)
       .distort(.3).gain(.78).room(.2)],
  [16, note("<[e2 e2 ~ e2] [e2 ~ e2 ~] [a1 a1 ~ a1] [b1 ~ b1 ~]>")
       .s("sawtooth")
       .lpf(900).decay(.2).sustain(.6)
       .distort(.25).gain(.72).room(.2)],
  [16, note("<[a1 a1 a1 ~] [a1 ~ a1 a1] [d2 d2 ~ d2] [e2 ~ e2 ~]>")
       .s("sawtooth")
       .lpf(1100).decay(.15).sustain(.65)
       .distort(.3).gain(.78).room(.2)],
  [8,  note("<e2 ~ a1 ~ d2 ~ b1 ~>/2")
       .s("sawtooth")
       .lpf(800).decay(.4).sustain(.4)
       .distort(.2).gain(.6).room(.3)],
  [8,  note("<e2 ~ ~ ~ e2 ~ ~ ~>/2")
       .s("sawtooth")
       .lpf(600).decay(.5).sustain(.3)
       .gain(saw.slow(8).range(.6,.1)).room(.3)]
)

// rhythm guitar — crunchy power chords
$: arrange(
  [8,  chord("<Em>").dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("x ~ ~ ~")
       .lpf(2200).room(.5).gain(.35)
       .distort(.2).crush(12)],
  [16, chord("<Em Em Am Bm>").dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("[x ~ x x] [~ x ~ x]")
       .lpf(3500).room(.35)
       .distort(.35).crush(11)
       .gain(.55).speed(.98)],
  [16, chord("<Am Am Dm Em>").dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("[x x ~ x] [x ~ x x]")
       .lpf(4000).room(.3)
       .distort(.4).crush(10)
       .gain(.6).speed(.99)
       .every(4, x => x.fast(2).gain(.35))],
  [16, chord("<Em Em Am Bm>").dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("[x ~ x x] [~ x ~ x]")
       .lpf(3500).room(.35)
       .distort(.35).crush(11)
       .gain(.55).speed(.98)],
  [16, chord("<Am Am Dm Em>").dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("[x x ~ x] [x ~ x x]")
       .lpf(4000).room(.3)
       .distort(.4).crush(10)
       .gain(.6).speed(.99)
       .every(3, x => x.press())],
  [8,  chord("<Am Em Bm Em>/2").dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("x ~ ~ x")
       .lpf(2800).room(.5)
       .distort(.3).crush(12)
       .gain(.48).delay(".2:.15:.25")],
  [8,  chord("<Em>").dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("x ~ ~ ~")
       .lpf(sine.range(1000,2500).slow(8)).room(.6)
       .distort(.25).gain(saw.slow(8).range(.45,.05))]
)

// lead guitar — wild, loose melody
$: arrange(
  [8,  silence],
  [16, n("<[~ 0 ~ 2] [4 ~ 2 ~] [~ 0 ~ 4] [2 ~ ~ 0]>")
       .scale("E4:minor:pentatonic")
       .s("gm_electric_guitar_clean")
       .lpf(sine.range(1500,4500).slow(10))
       .room(.5).delay(".2:.15:.3")
       .distort(.3).gain(.4)
       .rarely(x => x.add(note(12)))],
  [16, n("<[0 2 4 ~] [~ 4 7 ~] [4 ~ 2 0] [~ 4 2 ~]>")
       .scale("E5:minor:pentatonic")
       .s("gm_electric_guitar_clean")
       .lpf(sine.range(2000,5000).slow(8))
       .room(.45).delay(".15:.1:.25")
       .distort(.35).gain(.48)
       .sometimes(x => x.add(note(12)))],
  [16, n("<[~ 0 ~ 2] [4 ~ 2 ~] [~ 0 ~ 4] [2 ~ ~ 0]>")
       .scale("E4:minor:pentatonic")
       .s("gm_electric_guitar_clean")
       .lpf(sine.range(1500,4500).slow(10))
       .room(.5).delay(".2:.15:.3")
       .distort(.3).gain(.4)],
  [16, n("<[0 2 4 ~] [~ 4 7 ~] [4 ~ 2 0] [~ 4 2 ~]>")
       .scale("E5:minor:pentatonic")
       .s("gm_electric_guitar_clean")
       .lpf(sine.range(2000,5000).slow(8))
       .room(.45).delay(".15:.1:.25")
       .distort(.35).gain(.48)
       .every(4, x => x.fast(2).gain(.3))],
  [8,  n("<[~ 4 ~ 7] [~ ~ 4 ~] [~ 2 ~ ~] [0 ~ ~ ~]>")
       .scale("E5:minor:pentatonic")
       .s("gm_electric_guitar_clean")
       .room(.7).delay(".5:.35:.6")
       .distort(.25).gain(.35).crush(12)
       .lpf(3000)],
  [8,  silence]
)

// feedback/texture layer — sustained noise
$: arrange(
  [8,  silence],
  [16, silence],
  [16, note("<e4 b4>/4")
       .s("sawtooth")
       .attack(.5).decay(.3).sustain(.8).release(1)
       .lpf(sine.range(600,2000).slow(6))
       .distort(.5).crush(8)
       .room(.7).roomsize(6)
       .gain(.18).pan(sine.slow(4))],
  [16, silence],
  [16, note("<e4 b4>/4")
       .s("sawtooth")
       .attack(.5).decay(.3).sustain(.8).release(1)
       .lpf(sine.range(600,2000).slow(6))
       .distort(.5).crush(8)
       .room(.7).roomsize(6)

  [8,  note("<e3>/4")

       .lpf(sine.range(300,1200).slow(8))
       .distort(.4).room(.8).roomsize(8)
       .gain(saw.slow(8).range(.2,.02))],
  [8,  silence]
)
