// 🔁 "Shifting Glass" — Minimalist Repetition with Subtle Variations
// At 90 BPM: intro(8) + A(16) + B(16) + A'(16) + C(16) + outro(8) = 80 bars ≈ 3:33

setcpm(90 / 4)

// Sub bass drone — constant foundation
$: note("a0").s("sine").gain(.15).room(.2).lpf(100).attack(3).sustain(.8).release(4)

// Core arpeggio — always present, slowly evolving
$: arrange(
  [8,  n("0 2 4 7 4 2 0 4").scale("A3:minor").s("gm_epiano1").gain(.42).room(.55).delay(".2:.15:.2").lpf(1600)],
  [16, n("0 2 4 7 4 2 0 4").scale("A3:minor").s("gm_epiano1").gain(.48).room(.5).delay(".2:.15:.2").lpf(2000).every(4, x => x.add(note(2)))],

  [16, n("0 2 4 7 4 2 0 4").scale("A3:minor").s("gm_epiano1").gain(.48).room(.5).delay(".2:.15:.22").lpf(2000).every(5, x => x.add(note(2)))],
  [16, n("0 2 4 7 4 2 0 4").scale("A3:minor").s("gm_epiano1").gain(.52).room(.46).delay(".15:.1:.15").lpf(3000).every(4, x => x.rev()).every(7, x => x.add(note(5)))],
  [8,  n("0 2 4 7 4 2 0 4").scale("A3:minor").s("gm_epiano1").gain(saw.slow(8).range(.42,.05)).room(.7).delay(".35:.25:.4").lpf(sine.range(600,1800).slow(8))]
)

// Second arpeggio — shifted voicing, enters in B and C
$: arrange(
  [8,  silence],
  [16, silence],
  [16, n("4 7 9 11 9 7 4 7").scale("A3:minor").s("gm_epiano1").gain(.28).room(.55).delay(".25:.2:.3").lpf(1800).pan(.3)],
  [16, silence],
  [16, n("4 7 9 11 9 7 4 7").scale("A3:minor").s("gm_epiano1").gain(.32).room(.55).delay(".25:.2:.3").lpf(2200).pan(.3).every(5, x => x.rev())],
  [8,  silence]
)

// Bass — sustained whole notes, minimal movement
$: arrange(
  [8,  silence],
  [16, note("<a1 a1 c2 c2>").s("gm_acoustic_bass").gain(.68).room(.2).decay(.6).sustain(.6)],
  [16, note("<a1 c2 e2 c2>").s("gm_acoustic_bass").gain(.7).room(.2).decay(.6).sustain(.6)],
  [16, note("<a1 a1 c2 c2>").s("gm_acoustic_bass").gain(.68).room(.2).decay(.6).sustain(.6)],
  [16, note("<a1 c2 e2 g1>").s("gm_acoustic_bass").gain(.72).room(.2).decay(.55).sustain(.6)],
  [8,  note("a1").s("gm_acoustic_bass").gain(saw.slow(8).range(.6,.1)).room(.3).decay(.8).sustain(.4)]
)

// Strings pad — glacial wash, B and C only
$: arrange(
  [8,  silence],
  [16, silence],
  [16, chord("<Am Cmaj7>/4").dict('ireal').voicing().s("gm_synth_strings_1").attack(1.5).sustain(.7).release(2).lpf(1200).room(.7).gain(.22)],
  [16, silence],
  [16, chord("<Am Cmaj7 Em Fmaj7>/4").dict('ireal').voicing().s("gm_synth_strings_1").attack(1.2).sustain(.7).release(2).lpf(1500).room(.7).gain(.28)],
  [8,  chord("Am").dict('ireal').voicing().s("gm_synth_strings_1").attack(2).sustain(.5).release(3).lpf(sine.range(400,1000).slow(8)).room(.8).gain(saw.slow(8).range(.22,.03))]
)

// Percussion — restrained, organic dynamics
$: arrange(
  [8,  silence],
  [16, s("hh*4").bank("RolandTR808").gain(perlin.range(.12,.28)).room(.3).cut(1).pan(sine.slow(4))],
  [16, s("bd ~ ~ ~, hh*4, ~ ~ ~ [cp ~]").bank("RolandTR808").gain(.3).room(.3).cut(1)],
  [16, s("hh*4").bank("RolandTR808").gain(perlin.range(.12,.28)).room(.3).cut(1).pan(sine.slow(4))],
  [16, s("bd ~ ~ ~, hh*8, ~ ~ ~ [cp ~]").bank("RolandTR808").gain(.33).room(.3).cut(1).every(8, x => x.press())],
  [8,  s("hh*4").bank("RolandTR808").gain(saw.slow(8).range(.22,.02)).room(.4).cut(1)]
