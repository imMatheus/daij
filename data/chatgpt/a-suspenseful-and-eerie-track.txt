// "Underfloor Whispers" - suspense pulses and haunted drones
samples('github:tidalcycles/dirt-samples')
setcps(0.46)
$: s("bd(3,8,<2 3>) [~ bd]*2 <~ [cp ~ cp]>")
  .gain("0.7 0.8 0.65 0.9")
  .lpf(perlin.range(220,980).slow(6))
  .room(sine.range(0.25,0.85).slow(12))
  .delay(0.25)
  .stack(
    s("<hh(5,8) [~ hh(3,8)]> [~ oh]*2")
      .gain("0.25 0.35 0.2 0.4")
      .hpf(perlin.range(2000,7000).slow(5))
      .room(0.7)
      .delay(0.125)
      .reset(8),
    n("<0 [0 -1] [2 1] [-3 -2]>*2")
      .s("sawtooth")
      .scale("minor")
      .slow(2)
      .legato(1.4)
      .gain(0.42)
      .lpf(perlin.range(120,520).slow(10))
      .room(0.95)
      .delay(0.375)
      .reset(16),
    n("<[7 9] [8 10] [6 8] [5 7]>")
      .s("supersaw")
      .slow(4)
      .gain(0.18)
      .lpf(sine.range(500,2200).slow(7))
      .room(1)
      .delay(0.5)
      .reset(16),
    s("<~ [cp:3 ~] [~ oh:2] [~ ~ hh:4]>*2")
      .gain("0.22 0.16 0.2 0.14")
      .speed("<0.7 0.8 0.6 0.9>")
      .room(0.9)
      .delay(0.25)
      .lpf(perlin.range(400,2600).slow(11))
      .reset(32),
    n("<~ [12 11] ~ [10 9]>*2")
      .s("sine")
      .slow(2)
      .gain(0.16)
      .lpf(300)
      .room(1)
      .delay(0.75)
      .reset(32)
  )
  .reset(32)
