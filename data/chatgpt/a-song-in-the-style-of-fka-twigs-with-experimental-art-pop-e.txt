// "Glass Veins" — fractured experimental art-pop with breathy glitch textures
samples('github:tidalcycles/dirt-samples')
setcps(0.9)

$: s("[bd(3,8,<1 2>) ~ bd] <~ sd:2 [~ sd]> [~ bd(5,16)]")
  .gain(0.9)
  .stack(
    s("hh*16? [~ hh*8] <cp(5,16) rim(3,8)>")
      .gain("0.22 0.3 0.26 0.34")
      .pan(rand)
      .hpf(sine.slow(6).range(3500,9000)),
    s("<arpy:0 arpy:2 arpy:4 [arpy:7 arpy:9]>")
      .euclid(5,8)
      .speed("<0.5 1 1.5 -1>")
      .gain(0.28)
      .room(0.75)
      .delay(0.375)
      .delayfeedback(0.65)
      .lpf(perlin.slow(4).range(1200,5400)),
    note("<[e4 g4 b4 d5] [d4 fs4 a4 c5] [c4 e4 g4 b4] [b3 d4 fs4 a4]>")
      .slow(2)
      .s("sawtooth")
      .attack(0.02)
      .release(0.7)
      .legato(1.2)
      .gain(0.38)
      .pan(sine.slow(5).range(0.2,0.8))
      .lpf(sine.slow(7).range(700,2600))
      .room(0.82)
      .size(0.9),
    note("<e2 ~ [e2 b1] ~ d2 ~ [d2 a1] ~ c2 ~ [c2 g1] ~ b1 ~ [b1 fs1] ~>")
      .s("sine")
      .lpf(sine.slow(9).range(90,240))
      .gain(0.42),
    s("<[~ ~] [flick:2 ~] [~ glitch:3] [flick:5 glitch:1]>")
      .gain(0.25)
      .speed("<1 0.5 -1 2>")
      .room(0.7)
      .delay(0.5)
      .delayfeedback(0.7)
      .lpf(perlin.slow(3).range(900,7000))
  )
  .shape(0.18)
  .room(0.35)
  .reset(64)
