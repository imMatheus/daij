// 🎵 "Mirrored Lanterns" — raw call-and-response duet over broken pulse
samples('github:tidalcycles/dirt-samples')
setcps(0.8)
note("<[c4 e4 g4 b4] ~ [d4 f4 a4 c5] ~ [e4 g4 b4 d5] ~ [g4 b4 d5 f5] ~>")
  .s("<sawtooth square>")
  .struct("euclid(5,8)")
  .legato(0.92)
  .gain(0.56)
  .pan(sine.slow(7).range(-0.75,-0.2))
  .stack(
    note("<~ [g3 b3 d4 f4] ~ [a3 c4 e4 g4] ~ [f3 a3 c4 e4] ~ [e3 g3 b3 d4]>")
      .s("<triangle sine>")
      .struct("euclid(5,8,3)")
      .legato(0.95)
      .gain(0.58)
      .pan(sine.slow(9).range(0.2,0.75))
  )
  .stack(
    s("bd*<4 3 5> [~ bd] <[~ bd] [bd ~]>")
      .lpf(170)
      .gain(0.9)
  )
  .stack(
    s("<hh*8 [hh*16]>")
      .struct("euclid(11,16)")
      .hpf(5200)
      .gain(0.27)
  )
  .stack(
    s("~ cp ~ <cp rim>")
      .struct("euclid(3,8)")
      .gain(0.33)
  )
  .room(perlin.slow(6).range(0.25,0.72))
  .delay(sine.slow(12).range(0.08,0.33))
  .lpf(perlin.slow(10).range(850,2800))
  .reset(48)
