samples('github:tidalcycles/dirt-samples')
setcps(24/70)

const grind = sine.slow(9).range(900,3200)

const haze = sine.slow(13).range(.16,.36)

    .fit()
    .slice(8,"<0 ~ 2 ~ 4 ~ [6 7] ~>")
    .speed("<1 .94 -1 .88>")
    .begin("<0 .03 .06 .02>")
    .end("<.94 .72 .4 .56>")

    .room(.28).delay(".22:.25:.32").gain(.28).orbit(2)
    .stack(
      note("<~ e4 g4 a4 b4 d5 e5 ~>")

        .vib(4).vibmod(.06).room(.28).delay(".18:.125:.22").gain(".0 .14 .2 .28 .34 .4 .46 .0").orbit(3)
    )
    .stack(
      s("hh*8")
        .begin(".02").end(".05")
        .hpf(air)
        .gain(".0 .0 .01 .02 .01 .03 .02 .04")
    )
