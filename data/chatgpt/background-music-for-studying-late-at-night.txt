setcps(0.5)
$: n("<[0 2 4 7] [0 3 5 8] [2 5 7 10] [0 3 7 9]>/2")
  .s("triangle")
  .slow(2)
  .legato(1.4)
  .lpf(perlin.range(500,2200).slow(6))

  .delay(0.35).delaytime(sine.range(0.18,0.42).slow(5)).delayfeedback(0.45)
  .gain(0.28)
  .stack(
    n("<[12 14 15 19] [10 12 14 17] [9 10 12 15] [7 10 12 14]>/4")
      .s("sawtooth")
      .slow(4)
      .lpf(sine.range(900,3200).slow(7))
      .room(0.5)
      .delay(0.22)
      .gain(0.16)
      .pan(sine.slow(9)),
    s("<bd(3,8) [~ bd] ~ bd(5,16)>")
      .gain(0.55)
      .lpf(160)
      .room(0.2),
    s("[~ hh(5,8) <hh*2 hh(3,8)>] [~ <~ hh>]")
      .gain(0.22)
      .hpf(4500)
      .room(0.35)
      .delay(0.12),
    s("<~ cp(2,8) ~ [cp cp]>")
      .gain(0.25)
      .lpf(2300)
      .room(0.4)
      .delay(0.2),
    s("~ [hh(7,16) <~ hh(5,16,1)>]")
      .gain(0.08)
      .hpf(7000)
      .room(0.8)
      .pan(perlin.slow(3))
  )
