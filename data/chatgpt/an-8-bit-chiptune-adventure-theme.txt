const intro = 1

const finale = 7

const introDrums = s("~ ~ ~ ~, ~ ~ ~ ~, hh*4").gain(.06).begin(.01).end(.05).bpf(7500).bpq(7).room(.5)

const chorusArp = note("<[c6,e6,g6,c7] [g5,b5,d6,g6] [a5,c6,e6,a6] [f5,a5,c6,f6] [c6,e6,g6,c7] [g5,b5,d6,g6] [f5,a5,c6,f6] [g5,b5,d6,g6]>").slow(4).arp("0 [1 2] 3 [2 1] 2 1 [2 3] 1").s("square").orbit(3).attack(.01).decay(.08).sustain(.06).release(.05).lpf(sine.slow(8).range(2200,7600)).delay(".32:.25:.36").room(.18).gain(.22).reset("<x ~ x ~>")

  arrange([intro, silence], [build, buildBass], [chorus, chorusBass], [quest, questBass], [bridge, bridgeBass], [finale, chorusBass]),
  arrange([intro, introArp], [build, buildArp], [chorus, chorusArp], [quest, questArp], [bridge, bridgeArp], [finale, chorusArp]),

  arrange([intro, silence], [build, silence], [chorus, pulse], [quest, pulse], [bridge, silence], [finale, pulse])
