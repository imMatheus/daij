samples('github:tidalcycles/dirt-samples')
setcps(0.8)

    s("~ sn ~ [sn sn:2] <~ sn>").gain(0.92).lpf(sine.range(1200,4200).slow(12)),

    s("<cp ~ [~ cp]> [~ cp cp ~]").gain(0.48).room(0.35),
    s("tabla*8").n("<0 2 5 3 6 1 7 4>").gain(0.33).delay(0.25).lpf(perlin.range(500,2400).slow(5))
  )
  .delay(0.18)
  .room(0.58)
  .size(0.72)
  .reset(64)
