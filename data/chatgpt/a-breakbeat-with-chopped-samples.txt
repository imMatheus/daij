// "Dustline Cut-Up" — chopped breakbeat collage with dirty low-end
samples('github:tidalcycles/dirt-samples')
setcps(24/70)

const low = sine.slow(11).range(110,280)
const cut = perlin.slow(7).range(700,3200)
const air = sine.slow(5).range(7000,12000)
const space = perlin.slow(9).range(.16,.38)

const intro =
  s("breaks125")
    .fit()
    .slice(16, "<[0 1 3 5] [7 6 5 3] [0 2 4 6] [7 7 6 5]>")
    .speed("<1 .98 1.02 -1>")
    .begin("<0 .02 .04 .01>")
    .end("<.95 .7 .8 .6>")
    .lpf(sine.slow(6).range(500,2200))
    .room(.28).delay(".22:.25:.35").gain(.34).orbit(2)
    .stack(
      s("breaks165")
        .fit()
        .splice(8, "<0 [2 3] 4 5 6 [7 0] 1 3>")
        .slow(2)
        .speed(".9 1 .95 1.05")
        .hpf(1800).lpf(5200)
        .room(.42).gain(.16).orbit(3)
        .reset("<x ~ ~ x>")
    )
    .stack(
      note("<d1 ~ ~ c1>")
        .slow(2)
        .s("sawtooth")
        .attack(.02).decay(.25).sustain(.04).release(.08)
        .lpf(300).lpq(4).gain(.24)
    )
    .stack(
      s("hh*8")
        .gain(".01 .02 .02 .03 .01 .02 .02 .04")
        .hpf(air)
    )

const chorus =
  s("breaks165")
    .fit()
    .splice(16, "<0 [1 2] 3 [4 5] 6 [7 0] [8 9] 10 [11 12] 13 [14 15] 12>")
    .speed("<1 1 .98 1.02>")
    .lpf(cut).room(.08).gain(.54).orbit(2)
    .stack(
      s("breaks125")
        .fit()
        .slice(16, "0 [2 3] 4 [6 5] 7 [1 0] 3 [5 7] 4 [2 1] 6 [7 6]")
        .speed("<1 -1 1 .5>")
        .begin("<0 .03 .01 .05>")
        .end("<.9 .35 .8 .45>")
        .hpf(900).lpf(sine.slow(8).range(1800,7600))
        .room(.22).delay(".18:.125:.36").gain(.22).orbit(3)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("bd")
        .struct("x ~ ~ [~ x] ~ ~ ~ ~ x ~ ~ ~ ~ ~ [x x] ~")
        .lpf(800).gain(.78)
    )
    .stack(
      s("sd")
        .struct("~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~ x ~ ~ ~")
        .gain(.36).room(.12)
    )
    .stack(
      s("hh*16")
        .gain(".03 .06 .04 .08 .03 .09 .04 .07 .03 .06 .04 .09 .03 .08 .04 .11")
        .hpf(air)
        .pan(sine.slow(5).range(.35,.65))
        .reset("<x ~ x ~>")
    )
    .stack(
      note("<d1 d1 ~ f1 c1 d1 ~ bb0 d1 d1 ~ f1 c1 d1 bb0 c1>")
        .s("square")
        .attack(.01).decay(.18).sustain(.02).release(.07)
        .lpf(low).lpq(7).drive("1 .8 1.1 .9")
        .gain(.36).orbit(4)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("breaks165")
        .fit()
        .slice(8, "<0 1 [2 3] 4 [5 6] 7 [6 5] 3>")
        .slow(2)
        .speed(".85 1.1 .9 1")
        .hpf(2500).bpf(sine.slow(6).range(2800,8200)).bpq(2)
        .room(space).delay(".24:.25:.42").gain(.12).orbit(3)
    )
    .stack(
      s("bd")
        .struct("x ~ ~ [~ x] ~ ~ ~ ~ x ~ ~ ~ ~ ~ [x x] ~")
        .duckorbit("3:4").duckattack("0.14:0.18").duckdepth("0.45:0.9").postgain(0)
    )

const bridge =
  s("breaks125")
    .fit()
    .slice(8, "<0 ~ 2 ~ 4 ~ [6 7] ~>")
    .slow(2)
    .speed("<1 .75 -1 .5>")
    .begin("<0 .08 .12 .02>")
    .end("<.8 .35 .25 .5>")
    .hpf(500).lpf(sine.slow(5).range(900,2600))
    .room(.38).delay(".3:.375:.52").gain(.28).orbit(2)
    .stack(
      s("breaks165")
        .fit()
        .splice(16, "<0 ~ ~ [3 2] ~ 5 ~ [7 6] ~ 8 ~ [10 11] ~ 12 ~ [15 14]>")
        .slow(2)
        .speed("<1 .85 -1 .65>")
        .hpf(2200).lpf(6800).room(.44).gain(.12).orbit(3)
        .reset("<x ~ x ~>")
    )
    .stack(
      note("<d1 ~ c1 ~ bb0 ~ c1 ~>")
        .slow(2)
        .s("sawtooth")
        .attack(.02).decay(.32).sustain(.03).release(.1)
        .lpf(sine.slow(9).range(140,520)).lpq(6).gain(.28).orbit(4)
    )
    .stack(
      s("hh*8")
        .slow(2)
        .gain(".02 .04 .03 .05 .02 .05 .03 .06")
        .hpf(9000)
    )
    .stack(
      s("white")
        .gain(perlin.slow(8).range(0,.04))
        .bpf(sine.slow(4).range(1800,5400)).bpq(3)
        .room(.48)
    )
    .stack(
      s("bd")
        .struct("x ~ ~ ~ ~ ~ x ~")
        .slow(2)
        .gain(.44)
        .duckorbit("2:3:4").duckattack("0.22:0.28:0.2").duckdepth("0.3:0.5:0.75").postgain(0)
    )

arrange([1, intro], [10, chorus], [3, bridge], [10, chorus]).postgain(.96)
