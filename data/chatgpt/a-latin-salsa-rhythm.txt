// "Barrio Candela" — raw salsa pulse with clave-led street percussion
samples('github:tidalcycles/dirt-samples')
setcps(1.1)
s("bd [~ bd] bd*2 [~ bd]")
  .gain(0.95)
  .stack(
    s("[~ claves] ~ claves ~ ~ claves ~ claves")
      .gain(0.82)
      .pan("<0.35 0.65>")
  )
  .stack(
    s("conga*8")
      .n("<0 2 1 [2 3] 1 2 [3 2] 1>")
      .euclid("<5 7>",8)
      .gain(0.86)
      .pan(perlin.range(0.2,0.8).slow(6))
  )
  .stack(
    s("<bongo cowbell>*8")
      .n("<[0 1] 2 [1 3] 2>")
      .gain(0.63)
      .speed("<1 1 0.96 1.04>")
      .pan("<0.75 0.25>")
  )
  .stack(
    s("shaker*16")
      .gain(0.33)
      .hpf(2500)
      .pan(perlin.range(0.1,0.9).slow(10))
  )
  .stack(
    s("bass*4")
      .n("<0 ~ [0 1] ~ [2 1] ~ 0 ~>")
      .lpf(sine.range(220,920).slow(4))
      .gain(0.76)
  )
  .stack(
    s("piano*8")
      .n("<0 [7 12] 10 [7 12] 0 [8 12] 10 [7 12]>")
      .gain(0.48)
      .room(0.24)
      .lpf(sine.range(800,2600).slow(5))
  )
  .reset(48)
  .room(sine.range(0.14,0.5).slow(8))
  .delay(perlin.range(0.05,0.24).slow(7))
  .lpf(perlin.range(650,4300).slow(11))
