// "Aurora Drift" — glacial ambient pulse for skywatching
samples('github:tidalcycles/dirt-samples')
setcps(0.67)

$: note("<[e3 b3 g3 d4] [e3 a3 fs3 c4] [d3 a3 f3 c4] [c3 g3 eb3 bb3]>")
  .s("sine")
  .slow(2)
  .legato(1.4)
  .gain(0.45)
  .lpf(perlin.range(500,2200).slow(10))
  .room(0.85)
  .size(0.92)
  .delay(0.35)
  .stack(
    note("<e5 [g5 b5] [d6 b5] [g5 e5]>*2")
      .s("triangle")
      .gain(0.2)
      .release(0.35)
      .sometimesBy(0.35, x => x.fast(2))
      .pan(sine.slow(9))
      .lpf(sine.range(900,4200).slow(6))
      .delay(0.45),
    s("<~ [hh(5,16)? hh(3,16)] ~ [hh(7,16)]>")
      .gain(0.12)
      .lpf(7000)
      .pan(perlin.slow(12))
      .room(0.6),
    s("<~ ~ [bd(3,16) cp(2,16)] ~>")
      .gain(0.18)
      .lpf(900)
      .room(0.5)
      .delay(0.2),
    note("<[e6 b6] ~ [g6 d7] ~>")
      .s("sawtooth")
      .gain(0.1)
      .attack(0.02)
      .release(0.5)
      .hpf(1200)
      .pan("<0.2 0.8>")
      .every(6, x => x.rev())
      .room(0.9)
  )
  .reset(40)
