// "Parallel Hearts" — two interwoven songs breathing in one harmonic pulse
samples('github:tidalcycles/dirt-samples')
setcps(0.72)
n("<[0 2 4 7] [2 4 5 9] [4 7 9 11] [5 7 9 12]>/2")
  .scale("d3 minor")
  .s("sawtooth")
  .legato(1.1)
  .gain(0.75)
  .lpf(sine.slow(6).range(500, 2600))
  .room(0.45)
  .delay(0.22)
  .reset(64)
  .stack(
    n("<[7 5 4 2] [9 7 5 4] [11 9 7 5] [12 11 9 7]>/2")
      .scale("d3 minor")
      .s("square")
      .legato(0.85)
      .gain(0.62)
      .lpf(perlin.slow(8).range(700, 3200))
      .room(0.3)
      .delay(0.33),
    s("bd(5,8,<0 1 2>) [~ bd] sd(3,8,2) [~ sd]")
      .gain(0.95)
      .lpf(sine.slow(12).range(200, 1200))
      .room(0.08),
    s("hh*16")
      .gain("0.2 0.13 0.16 0.11")
      .hpf(7000)
      .delay(0.08),
    n("<[0 7 12] [2 9 14] [4 11 16] [5 12 17]>")
      .scale("d3 minor")
      .s("triangle")
      .slow(2)
      .gain(0.28)
      .lpf(sine.slow(10).range(180, 900))
      .room(0.7)
  )
