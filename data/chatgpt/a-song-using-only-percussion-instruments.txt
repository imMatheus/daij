// "Hide and Hammer" — raw percussion ritual with dub space
samples('github:tidalcycles/dirt-samples')
setcps(24/70)

const weight = sine.slow(9).range(500,1700)
const metal = sine.slow(11).range(5200,12000)
const punch = perlin.slow(7).range(1200,3600)
const air = perlin.slow(13).range(.16,.46)

const intro =
  s("[hh rim]*4")
    .gain(".025 .05 .02 .06 .025 .05 .02 .07")
    .end(".05 .08")
    .hpf(metal)
    .room(".28:.82")
    .delay(".14:.125:.2")
    .orbit(2)
    .stack(
      s("~ ~ ~ ~ lt ~ mt [ht mt]")
        .speed(".94 1 .98 1.05")
        .lpf(weight)
        .room(.26)
        .gain(".0 .0 .0 .0 .22 .0 .18 .16")
        .orbit(3)
    )
    .stack(
      s("bd")
        .struct("~ ~ ~ ~ ~ ~ x ~")
        .lpf(900)
        .gain(.7)
    )

const chorus =
  s("bd")
    .struct("x ~ ~ [~ x] ~ ~ x ~ x ~ ~ [~ x] ~ ~ [x x] ~")
    .lpf(980)
    .room(.06)
    .gain(.92)
    .stack(
      s("sd cp")
        .struct("~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~ x ~ ~ ~")
        .lpf(punch)
        .room(.16)
        .gain(".34 .24")
        .orbit(3)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("hh*16")
        .gain(".08 .05 .07 .04 .08 .06 .07 .05 .09 .05 .08 .04 .08 .06 .07 .1")
        .end(".05 .05 .06 .04")
        .hpf(metal)
        .room(.18)
        .delay(".08:.125:.14")
        .pan(sine.slow(7).range(.35,.65))
        .orbit(2)
        .reset("<x ~ x ~>")
    )
    .stack(
      s("~ ~ ~ oh ~ ~ ~ ~ ~ ~ ~ oh ~ oh ~ ~")
        .end(".16 .12 .22")
        .hpf(6200)
        .room(.3)
        .delay(".18:.25:.24")
        .gain(".11 .09 .14")
        .orbit(2)
    )
    .stack(
      s("rim")
        .euclidRot(5,16,3)
        .lpf(2600)
        .room(.22)
        .gain(.15)
        .orbit(4)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("lt mt ht mt")
        .struct("~ ~ ~ ~ x ~ ~ x ~ ~ x ~ ~ ~ x ~")
        .speed(".94 1 1.04 1.08")
        .lpf(weight)
        .room(air)
        .delay(".22:.25:.34")
        .gain(".24 .18 .16 .2")
        .orbit(4)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("bd")
        .struct("x ~ ~ [~ x] ~ ~ x ~ x ~ ~ [~ x] ~ ~ [x x] ~")
        .duckorbit("2:3:4")
        .duckattack("0.14:0.18:0.2")
        .duckdepth(".5:.55:.72")
        .postgain(0)
    )

const bridge =
  s("bd")
    .struct("x ~ ~ ~ ~ ~ x ~ x ~ ~ ~ ~ ~ x ~")
    .lpf(820)
    .room(.1)
    .gain(.72)
    .stack(
      s("lt mt ht mt")
        .euclidRot(5,8,1)
        .slow(2)
        .speed(".9 .98 1.04 1.1")
        .lpf(sine.slow(6).range(420,1500))
        .room(".54:.94")
        .delay(".32:.375:.46")
        .gain(".26 .22 .18 .2")
        .orbit(3)
        .reset("<x ~ x ~>")
    )
    .stack(
      s("[hh ~]*8")
        .gain(".02 .05 .02 .06 .02 .04 .02 .07")
        .end(".04 .06")
        .hpf(9000)
        .room(.4)
        .delay(".12:.125:.18")
        .orbit(2)
    )
    .stack(
      s("~ rim ~ ~ ~ ~ rim ~")
        .slow(2)
        .hpf(2400)
        .room(.5)
        .delay(".26:.25:.38")
        .gain(.12)
        .orbit(4)
        .reset("<x ~ x ~>")
    )
    .stack(
      s("~ ~ oh ~ ~ ~ ~ ~")
        .slow(2)
        .end(.22)
        .hpf(7000)
        .room(".44:.9")
        .delay(".24:.375:.3")
        .gain(.1)
        .orbit(2)
    )
    .stack(
      s("bd")
        .struct("x ~ ~ ~ ~ ~ x ~ x ~ ~ ~ ~ ~ x ~")
        .duckorbit("2:3:4")
        .duckattack("0.22:0.24:0.28")
        .duckdepth(".34:.48:.68")
        .postgain(0)
    )

arrange([1, intro], [9, chorus], [4, bridge], [9, chorus]).postgain(.98)
