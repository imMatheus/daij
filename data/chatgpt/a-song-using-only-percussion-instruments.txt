// "Skin and Steel" — raw polyrhythmic percussion ritual
samples('github:tidalcycles/dirt-samples')
setcps(0.9)

$: s("bd(<5 3>,8) [~ bd(3,8)] <bd*2 bd>")
  .n("<0 0 1 2>")
  .stack(
    s("[~ sn] [sn ~] <sn(3,8) [~ sn]>")
      .n("<0 1 2 1>")
      .gain(0.9),
    s("hh*16")
      .gain("<0.4 0.6 0.5 0.7>")
      .speed("<1 1.2 0.8 1.4>"),
    s("cp(3,8,<1 2 3>) <~ cp> [~ cp]")
      .gain(0.7),
    s("<lt mt ht>*4")
      .euclid(5,16)
      .gain(0.55)
      .speed("<0.9 1 1.1 1.2>"),
    s("rim(7,16)")
      .gain(0.45)
      .pan(sine.slow(6)),
    s("[perc*2 ~ perc] <perc(5,16) perc(3,8)>")
      .gain(0.35)
      .hpf(perlin.range(900,5000).slow(8))
  )
  .reset(32)
  .room(perlin.range(0.08,0.5).slow(5))
  .delay("<0 0.25 0.5 0.375>")
  .delaytime("<0.125 0.25 0.1875 0.3125>")
  .delayfeedback(perlin.range(0.15,0.55).slow(4))
  .lpf(sine.range(700,5000).slow(9))
  .gain(1.05)
