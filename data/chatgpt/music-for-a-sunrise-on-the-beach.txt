setcps(108/60/4)

const intro = 1
const shore = 4

const introPad = note("[d4 fs4 a4 cs5]").s("triangle").orbit(3).attack(.9).decay(.5).sustain(.84).release(2.8).lpf(sine.slow(2).range(500,1800)).room(.92).delay(.42).delaytime(.5).delayfeedback(.58).gain(.22)

stack(
  arrange([intro, wash], [shore, wash.gain(saw.slow(4).range(.08, .14))], [chorus, wash.gain(saw.slow(6).range(.1, .16))], [bridge, wash.gain(sine.slow(2).range(.06, .12))], [chorus, wash.gain(saw.slow(6).range(.1, .16))], [outro, wash.gain(isaw.slow(2).range(.1, .02))]),
  arrange([intro, silence], [shore, shoreDrums], [chorus, chorusDrums], [bridge, bridgeDrums], [chorus, chorusDrums], [outro, shoreDrums.gain(isaw.slow(2).range(.12, .02))]),
  arrange([intro, introPad], [shore, shorePad], [chorus, chorusPad], [bridge, bridgePad], [chorus, chorusPad], [outro, introPad.gain(isaw.slow(2).range(.18, .04))]),
  arrange([intro, silence], [shore, shoreBass], [chorus, chorusBass], [bridge, bridgeBass], [chorus, chorusBass], [outro, bridgeBass.gain(isaw.slow(2).range(.16, .03))]),
  arrange([intro, silence], [shore, shoreArp], [chorus, chorusArp], [bridge, silence], [chorus, chorusArp], [outro, silence]),
  arrange([intro, silence], [shore, silence], [chorus, chorusLead], [bridge, bridgeLead], [chorus, chorusLead], [outro, silence]),
  arrange([intro, silence], [shore, silence], [chorus, sparkle], [bridge, silence], [chorus, sparkle], [outro, silence])
).postgain(1.04)
