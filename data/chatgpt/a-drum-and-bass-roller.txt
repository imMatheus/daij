// "Roller Pressure" — low-slung drum and bass with morphing reese and shuffled breaks
samples('github:tidalcycles/dirt-samples')
setcps(0.74)
$: s("bd*2 [~ bd] sn ~ bd [~ sn] <[~ bd] [bd ~]> sn")
  .gain(0.95)
  .stack(
    s("<hh*8 hh*16> [~ hh]*2 hh(11,16) [hh*2 ~]").gain(0.45).hpf(6000).room(0.12),
    s("cp(3,8,<0 1 2>) ~ [cp ~] ~").gain(0.3).room(0.2),
    s("breaks125:1*2 <breaks125:3 breaks125:6> [~ breaks125:8]").cut("<1 2 1 3>").speed("<1 1 1.02 0.98>").gain(0.45).room(0.25).delay(0.18),
    note("<c1 [c1 c1] [c1 d1] [c1 g0]> [~ c1] <f0 g0> [c1 ~]")
      .s("sawtooth")
      .legato(0.95)
      .gain(0.8)
      .lpf(perlin.range(120,1800).slow(6))
      .shape(0.22),
    note("<[c3 eb3 g3] [d3 f3 a3] [bb2 d3 f3] [g2 bb2 d3]>")
      .s("supersaw")
      .slow(4)
      .gain(0.22)
      .lpf(sine.range(700,2600).slow(8))
      .room(0.6)
      .delay(0.33)
  )
  .lpf(perlin.range(900,9000).slow(10))
  .room(sine.range(0.12,0.45).slow(12))
  .delay(sine.range(0.08,0.28).slow(7))
  .reset(48)
