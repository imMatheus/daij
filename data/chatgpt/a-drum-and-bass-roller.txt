// "Nightbus Roller" — raw drum and bass roller
samples('github:tidalcycles/dirt-samples')
setcps(42/70)

const sub = sine.slow(9).range(80,220)
const growl = perlin.slow(7).range(220,2400)
const air = sine.slow(11).range(5000,12000)
const space = perlin.slow(13).range(.16,.42)

const intro =
  note("<d1 ~ ~ a0 ~ ~ c1 ~>")
    .fast(2)
    .s("sawtooth")
    .attack(.01).decay(.22).sustain(.05).release(.08)
    .lpf(500).lpq(2.8).gain(.34).orbit(2)
    .stack(
      note("<[d4 a4 c5 f5] ~ [c4 g4 bb4 e5] ~>")
        .s("supersaw")
        .attack(.3).release(1.1).sustain(.45)
        .lpf(sine.slow(6).range(300,2200)).lpq(.3)
        .room(space).delay(".25:.25:.45").orbit(3).gain(.24)
    )
    .stack(
      s("hh*8")
        .gain(".01 .02 .02 .03 .01 .02 .02 .04")
        .hpf(air)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("white")
        .gain(perlin.slow(10).range(0,.04))
        .hpf(7000).room(.5)
    )

const bridge =

    .struct("x ~ ~ ~ ~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~")

      s("hh*8")
        .gain(".03 .05 .04 .07 .03 .06 .04 .08")
