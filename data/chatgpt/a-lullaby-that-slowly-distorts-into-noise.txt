// "Quiet Static" — lullaby melody slowly fraying into hiss and broken light
samples('github:tidalcycles/dirt-samples')
setcps(24/70)

const intro = 1
const chorus = 10
const bridge = 3
const finalChorus = 9

const sectionReset = "<x@1 ~@9 x@1 ~@2 x@1 ~@8>"
const rise = saw.slow(23)
const fall = isaw.slow(23)
const bloom = sine.slow(11).range(.18,.46)

const introPad = note("<[a3 c4 e4] [f3 a3 c4]>")
  .slow(2)
  .s("supersaw").orbit(2)
  .attack(.7).decay(.9).sustain(.55).release(2.4)
  .lpf(7000).lpq(.18)
  .room(.84).delay(".42:0.375:0.44").chorus(.18)
  .gain(.18)

const introLead = note("<e5 a5 c6 a5 g5 e5 d5 c5>")
  .slow(2)
  .s("triangle").orbit(3)
  .attack(.01).decay(.2).sustain(.12).release(.9)
  .vib(4).vibmod(.08)
  .lpf(8200)
  .delay(".34:0.25:0.38").room(.56)
  .gain(.13)

const introDrone = note("<a2 e3>")
  .slow(2)
  .s("sine").orbit(6)
  .attack(.2).decay(.6).sustain(.6).release(2.1)
  .lpf(320)
  .gain(.1)

const introHush = s("pink")
  .struct("~ x ~ ~")
  .orbit(4)
  .attack(.02).release(.8)
  .bpf(2200).bpq(4)
  .room(.9)
  .gain(.01)

const chorusPad = note("<[a3 c4 e4] [g3 b3 d4] [f3 a3 c4] [e3 g#3 b3]>")
  .slow(4)
  .struct("[x ~ x [~ x]]")
  .s("supersaw").orbit(2)
  .attack(.04).decay(.48).sustain(.52).release(1.2)
  .lpf(fall.rangex(6800,900)).lpq(.24)
  .chorus(.24).room(bloom).delay(".24:0.25:0.34")
  .distort(rise.range(0,2.8))
  .gain(.26)
  .reset(sectionReset)

const chorusLead = note("<e5 a5 c6 a5 g5 e5 d5 c5 e5 a5 c6 a5 g5 e5 d5 b4>")
  .fast(2)
  .s("triangle").orbit(3)
  .attack(.005).decay(.12).sustain(.14).release(.12)
  .vib("4 5 6 5").vibmod(".04 .08 .14 .1")
  .lpf(fall.rangex(12000,1500))
  .delay(".28:0.375:0.46").room(.26)
  .distort(rise.range(0,4.5))
  .gain(.09)
  .reset(sectionReset)

const chorusBass = note("<a2 ~ c3 ~ f2 ~ e2 ~>")
  .slow(2)
  .s("sine").orbit(6)
  .attack(.005).decay(.28).sustain(.06).release(.1)
  .lpf(fall.rangex(260,120))
  .gain(.26)
  .reset(sectionReset)

const chorusPulse = s("bd")
  .euclidRot(5,8,1)
  .slow(2)
  .orbit(1)
  .begin("<0 .01 .015 .02>")
  .end("<1 .08>")
  .lpf(fall.rangex(1400,220))
  .distort(rise.range(0,1.6))
  .gain(".44 .38 .42 .34")
  .duckorbit("2:3:4").duckattack(".28:.22:.16").duckdepth(".88:.5:.22")
  .reset(sectionReset)

const cradleAir = s("~ hh ~ [~ hh]")
  .slow(2)
  .orbit(5)
  .begin("<0 .01>").end("<1 .08>")
  .hpf(fall.rangex(9000,4200))
  .delay(".16:0.125:0.22").room(.14)
  .gain(rise.range(.02,.08))
  .reset(sectionReset)

const staticBed = s("pink")
  .struct("x ~ ~ ~")
  .orbit(4)
  .attack(.02).release(1.1)
  .bpf(rise.rangex(900,7600)).bpq("1 3 7 14")
  .hpf(rise.rangex(40,2600))
  .delay(".52:0.125:0.78").room(.88)
  .distort(rise.range(0,8))
  .gain(rise.range(.004,.08))
  .reset(sectionReset)

const bridgePad = cat(
  note("<[a3 c4 e4] [f3 a3 c4]>")
    .slow(2)
    .s("supersaw").orbit(2)
    .attack(.2).decay(.4).sustain(.4).release(1.2)
    .lpf(1400).room(.7)
    .gain(.12),
  note("<[bb3 db4 e4] [b3 d4 f4]>")
    .slow(2)
    .s("supersaw").orbit(2)
    .attack(.02).decay(.22).sustain(.3).release(.8)
    .bpf(1800).bpq(4)
    .distort("4:.5:fold")
    .delay(".48:0.125:0.74").room(.7)
    .gain(.1),
  s("pink")
    .struct("x [~ x] x [x x]")
    .orbit(4)
    .attack(.01).release(.6)
    .bpf(3600).bpq(10)
    .distort("9:.45:diode")
    .delay(".62:0.125:0.82").room(.9)
    .gain(.08)
)

const bridgeLead = cat(
  note("<e5 a5 c6 a5>")
    .slow(2)
    .s("triangle").orbit(3)
    .attack(.01).release(.5)
    .hpf(1400)
    .delay(".38:0.125:0.46")
    .gain(.05),
  note("<a5 g#5 g5 f#5 e5 eb5 d5 c#5>")
    .fast(2)
    .s("pulse").orbit(3)
    .attack(.005).decay(.09).sustain(.05).release(.08)
    .hpf(1600)
    .distort("6:.42:sinefold")
    .delay(".54:0.125:0.76").room(.42)
    .gain(.08),
  s("pink")
    .struct("x x [x ~] x")
    .orbit(3)
    .attack(.01).release(.25)
    .bpf(3200).bpq(14)
    .distort("10:.35:chebyshev")
    .gain(.05)
)

const bridgeBass = cat(
  note("<a2 ~ c3 ~>")
    .slow(2)
    .s("sine").orbit(6)
    .attack(.01).decay(.2).sustain(.04).release(.1)
    .lpf(180)
    .gain(.16),
  note("<bb1 b1 c2 d#2>")
    .fast(2)
    .s("sine").orbit(6)
    .attack(.005).decay(.18).sustain(0).release(.1)
    .penv(18).pdecay(.06)
    .distort("3:.6")
    .lpf(160)
    .gain(.1),
  note("a1")
    .s("sine").orbit(6)
    .attack(.005).decay(.24).sustain(0).release(.24)
    .penv(-24).pdecay(.12)
    .lpf(90)
    .gain(.07)
)

const bridgePulse = cat(
  s("bd")
    .euclidRot(3,8,1)
    .slow(2)
    .orbit(1)
    .gain(.28)
    .duckorbit("2:3:4").duckattack(".22:.16:.1").duckdepth(".7:.36:.16"),
  s("[bd bd] ~ [~ bd] bd")
    .fast(2)
    .orbit(1)
    .distort("4:.55:fold")
    .lpf(500)
    .gain(.24)
    .duckorbit("2:3:4").duckattack(".18:.14:.1").duckdepth(".7:.36:.16"),
  s("bd*4")
    .orbit(1)
    .begin(.02).end(.05)
    .hpf(1200)
    .room(.82)
    .gain(.07)
    .duckorbit("2:3:4").duckattack(".3:.22:.14").duckdepth(".9:.46:.2")
)

arrange([intro, introPad], [chorus, chorusPad], [bridge, bridgePad], [finalChorus, chorusPad])
  .stack(arrange([intro, introLead], [chorus, chorusLead], [bridge, bridgeLead], [finalChorus, chorusLead]))
  .stack(arrange([intro, introDrone], [chorus, chorusBass], [bridge, bridgeBass], [finalChorus, chorusBass]))
  .stack(arrange([intro, silence], [chorus, chorusPulse], [bridge, bridgePulse], [finalChorus, chorusPulse]))
  .stack(arrange([intro, introHush], [chorus, staticBed], [bridge, staticBed.fast(2).gain(1.4)], [finalChorus, staticBed.gain(1.35)]))
  .stack(arrange([intro, silence], [chorus, cradleAir], [bridge, silence], [finalChorus, cradleAir]))
  .postgain(.96)
