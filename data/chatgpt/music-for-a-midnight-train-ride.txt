  "[bb3,f4,a4] ~ [c4,g4,bb4] ~",

  "[d4,a4,c5] ~ ~ ~"
).note()

  "[g3,d4,a4] ~ [a3,e4,c5] ~",
  "[bb3,f4,c5] ~ [a3,e4,g4] ~",
  "[c4,g4,d5] ~ [a3,e4,c5] ~"

  "d2 ~ a1 ~ d2 ~ a1 c2",
  "bb1 ~ f2 ~ c2 ~ g1 ~",
  "d2 ~ a1 ~ f2 ~ c2 ~",
  "f1 ~ c2 ~ c2 ~ g1 ~",

  "bb1 ~ f2 ~ c2 ~ g1 ~",
  "d2 ~ a1 ~ f2 ~ a1 ~",
  "bb1 ~ f2 ~ c2 ~ g1 ~",
  "d2 a1 c2 a1 d2 ~ ~ ~"
).note()
let bridgeBass = cat(
  "g1 ~ ~ d2 ~ ~ a1 ~",
  "bb1 ~ ~ f2 ~ ~ a1 ~",
  "c2 ~ ~ g1 ~ ~ a1 ~"
).note()

let introLead = "~ ~ a5 ~".note()
let chorusLead = cat(
  "~ a4 c5 d5 f5 ~ d5 c5",
  "~ a4 c5 f5 a5 ~ g5 f5",
  "~ a4 d5 f5 c6 ~ a5 f5",
  "~ g4 bb4 c5 g5 ~ f5 d5",
  "~ a4 c5 e5 g5 ~ e5 d5",
  "~ f4 a4 c5 f5 ~ d5 c5",
  "~ a4 c5 d5 a5 ~ f5 d5",
  "~ g4 bb4 c5 g5 ~ f5 d5",
  "~ a4 c5 d5 f5 ~ d5 a4"
).note()

  "~ ~ d5 ~ f5 ~ a5 ~",
  "~ e5 ~ g5 ~ f5 ~ d5",
  "~ ~ c5 ~ e5 ~ g5 ~"
).note()

let introSteps = "~ ~ lt ~ ~ ~ ~ ~"
let chorusSteps = cat(
  "bd hh [lt hh] hh [bd cp] hh lt hh",
  "bd hh lt hh bd hh cp hh",
  "[bd lt] hh bd hh cp hh lt hh",
  "bd hh lt [hh hh] bd hh cp hh",
  "bd hh [lt cp] hh bd hh lt hh",
  "bd hh lt hh [bd cp] hh hh hh",
  "[bd lt] hh cp hh bd hh lt hh",
  "bd hh lt hh bd hh cp hh",
  "bd hh [lt hh] hh [bd cp] hh [lt hh] hh"
)
let bridgeSteps = cat(
  "~ hh ~ ~ cp ~ hh ~",
  "~ ~ lt ~ ~ hh ~ ~",
  "~ hh ~ ~ cp ~ ~ ~"
)

arrange(
  [1, introPad],
  [9, chorusPad],
  [3, bridgePad],
  [9, chorusPad],
  [1, silence]
)
  .s("supersaw")
  .attack(0.22)
  .release(1.9)
  .chorus(0.32)
  .lpf(sine.slow(14).range(700,2400))
  .room(0.78)
  .delay(0.38)
  .delaytime(0.375)
  .delayfeedback(0.74)
  .gain(0.2)

      [1, introBass],
      [9, chorusBass],
      [3, bridgeBass],
      [9, chorusBass],
      [1, silence]
    )

      .attack(0.01)
      .release(0.34)
      .lpf(sine.slow(10).range(170,620))
      .room(0.14)
      .gain(0.3)
      .reset("<x ~ ~ ~ x ~ ~ ~>"),
    arrange(
      [1, introSteps.s()],
      [9, chorusSteps.s()],
      [3, bridgeSteps.s()],
      [9, chorusSteps.s()],

      .lpf(sine.slow(11).range(900,3200))
      .room(0.24)

      .reset("<x ~ ~ ~ ~ ~ ~ ~>"),

      [9, s("hh*16")],

      .end(0.95)
      .speed("<1 0.5 1 -0.5>")
      .bpf(sine.slow(9).range(3200,7800))
      .bpq(8)
      .room(0.58)
      .delay(0.14)
      .delaytime(0.125)
      .delayfeedback(0.48)
      .gain(0.08),
    arrange(
      [1, introLead],
      [9, chorusLead],
      [3, bridgeLead],
      [9, chorusLead],
      [1, silence]
    )
      .s("triangle")
      .attack(0.01)
      .release(0.3)
      .vib(4)
      .vibmod(0.18)
      .chorus(0.18)
      .tremolosync(4)
      .tremolodepth(0.16)
      .lpf(sine.slow(12).range(1400,4200))
      .delay(0.42)
      .delaytime(0.25)
      .delayfeedback(0.68)
      .room(0.5)
      .gain(0.16)
      .reset("<x ~ ~ x ~ ~ ~ ~>"),
    arrange(
      [1, s("pink*4")],
      [9, s("pink*8")],
      [3, s("pink*4")],
      [9, s("pink*8")],
      [1, s("pink*2")]
    )
      .speed("<1 0.5 1 -0.5>")
      .bpf(sine.slow(16).range(500,1800))
      .bpq(7)
      .room(0.9)
      .delay(0.2)
      .delaytime(0.5)
      .delayfeedback(0.56)
      .gain(0.04)
  )
  .postgain(1.04)
