// "Pressure Bloom" — submerged downtempo pulse with drifting bubbles
samples('github:tidalcycles/dirt-samples')
setcps(24/70)

const intro = 1
const chorus = 10
const bridge = 2
const form = (i, c, b) => arrange([intro, i], [chorus, c], [bridge, b], [chorus, c])

const tide = perlin.slow(8).range(260, 2200)
const glint = sine.slow(6).range(1400, 5400)
const swell = sine.slow(7).range(.82, .98)

const wash = s("pink")
  .struct("x ~ ~ x")
  .slow(2)
  .attack(.4).release(3.8)
  .bpf(sine.slow(9).range(700, 2200)).bpq(6)
  .room(.99).delay(".42:0.25:0.8").gain(.024)
  .reset("<x ~ ~ x>")

const bubbles = s("hh ~ [hh hh] ~, ~ ~ oh ~")
  .fast(2)
  .orbit(5)
  .begin("<.02 .08 .14>")
  .end("<.04 .09 .18>")
  .speed("<.82 .95 1.04 -1.08>")
  .bpf(sine.slow(11).range(1200, 4200)).bpq(8)
  .room(.94).delay(".34:0.125:0.72").gain(.055)
  .reset("<x ~ x ~>")

const introPad = chord("<Dm7 C>")
  .voicing().anchor("c5")
  .slow(2)
  .s("triangle")
  .attack(1.2).release(4.8).sustain(.84)
  .lpf(tide).lpq(.24)
  .room(.98).delay(".58:0.5:0.84").gain(.18)

const chorusPad = chord("<Dm7 Bbmaj7 Fmaj7 C Dm7>")
  .voicing().anchor("c5")
  .slow(5)
  .struct("[x ~ x x]")
  .s("supersaw")
  .orbit(3)
  .attack(.03).decay(.34).sustain(.64).release(.36)
  .detune(.08)
  .lpf(perlin.slow(7).range(700, 2600)).lpq(.23)
  .room(swell).delay(".28:0.25:0.42").chorus(.24).gain(.26)
  .reset("<x ~ x ~>")

const bridgePad = chord("<Gm7 A7>")
  .voicing().anchor("c5")
  .slow(2)
  .s("triangle")
  .orbit(3)
  .attack(.5).release(3.4).sustain(.52)
  .hpf(sine.slow(3).range(180, 900))
  .lpf(sine.slow(4).range(900, 2400)).lpq(.2)
  .room(.99).delay(".76:0.5:0.9").gain(.14)

const chorusBass = note("<d2 bb1 f1 c2 d2>")
  .slow(5)
  .struct("[x [~ x] x [~ x]]")
  .s("sine")
  .orbit(2)
  .attack(.01).decay(.18).sustain(.18).release(.12)
  .lpf(perlin.slow(6).range(110, 260)).lpenv(2.5).lpa(.02)
  .gain(.46)
  .reset("<x ~ x ~>")

const bridgeBass = note("<g1 a1>")
  .slow(2)
  .struct("[x ~ x ~]")
  .s("sine")
  .orbit(2)
  .attack(.04).release(1.8).sustain(.08)
  .lpf(140)
  .gain(.28)

const chorusKick = s("bd [~ bd] ~ bd")
  .fast(2)
  .orbit(1)
  .begin("<0 .01 .02 .01>")
  .end("<1 .1 .08 .12>")
  .lpf(480)
  .distort("<0 .4 0 .65>")
  .gain(".76 .62 .68 .58")
  .duckorbit("2:3:4:5").duckattack(".28:.18:.12:.1").duckdepth(".7:.28:.18:.16")
  .reset("<x ~ x ~>")

const bridgeKick = s("~ bd ~ ~")
  .fast(2)
  .orbit(1)
  .lpf(260)
  .gain(.3)
  .duckorbit("2:3:4:5").duckattack(".34:.2:.14:.12").duckdepth(".45:.18:.1:.08")

const chorusPerc = s("~ cp ~ cp, hh*8, ~ ~ [~ hh] ~")
  .fast(2)
  .orbit(5)
  .begin("<0 .01>").end("<1 .08>")
  .speed("<1 .98 1.02 1>")
  .hpf(perlin.slow(9).range(3000, 9000))
  .room(.18).delay(".18:0.125:0.32").gain(".44 .18 .14")
  .reset("<x ~ x ~>")

const bridgePerc = s("hh [~ hh] ~ ~, ~ ~ oh ~")
  .fast(2)
  .rev()
  .orbit(5)
  .bpf(sine.slow(4).range(1400, 4200)).bpq(6)
  .room(.56).delay(".44:0.125:0.76").gain(.12)

const chorusLead = n("<0 2 4 6 4 2 0 1 2 4>")
  .scale("D5:minor")
  .slow(5)
  .s("triangle")
  .orbit(4)
  .attack(.01).decay(.14).sustain(.12).release(1.6)
  .vib(6).vibmod(".07 .12 .18 .09")
  .lpf(glint)
  .room(.9).delay(".62:0.25:0.84").gain(.085)
  .reset("<x ~ ~ x>")

const bridgeLead = note("<d6 c6>")
  .slow(2)
  .s("sine")
  .orbit(4)
  .attack(.06).release(2.4).sustain(.1)
  .vib(7).vibmod(.22)
  .hpf(1800).lpf(4200)
  .room(.99).delay(".86:0.5:0.92").gain(.05)

form(wash.gain(.02), wash.gain(.03).stack(bubbles), wash.gain(.035).stack(bubbles.rev().gain(.6)))
  .stack(form(introPad, chorusPad, bridgePad))
  .stack(form(silence, chorusBass, bridgeBass))
  .stack(form(silence, chorusKick, bridgeKick))
  .stack(form(silence, chorusPerc, bridgePerc))
  .stack(form(silence, chorusLead, bridgeLead))
  .postgain(.94)
