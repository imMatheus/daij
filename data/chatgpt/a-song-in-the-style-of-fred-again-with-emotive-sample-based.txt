// "Midnight Voicemail" — emotive chopped-vocal dance pulse
samples('github:tidalcycles/dirt-samples')
setcps(0.86)
s("bd(5,8,<1 2 3>) [~ bd] <cp ~ cp [~ cp]>")
  .gain(1.1)
  .stack(
    s("[~ hh(7,16)] <hh*2 [~ hh] [hh hh ~ hh]>").hpf(7000).gain(.45).pan(sine.slow(6).range(.2,.8)),
    s("<~ [sd ~] ~ [sd ~ sd ~]>").gain(.7).room(.25),
    s("<numbers:2 numbers:5 numbers:8 [numbers:3 numbers:7]>*2")
      .chop("<8 16 12 32>").speed("<1 1.2 .8 -1>")
      .begin(perlin.slow(4).range(0,.25)).end(perlin.slow(5).range(.45,.95))
      .gain(.65).room(sine.slow(16).range(.2,.8)).delay("<0 .125 .25 .375>").delayfeedback(.55),
    s("<arpy:2 arpy:4 arpy:6 [arpy:8 arpy:10]>*4").speed("<0.5 1 1.5>")
      .lpf(perlin.slow(8).range(900,4200)).gain(.32).room(.4),
    n("<[0 2 4 7] [0 3 5 8] [2 5 7 10] [0 4 7 11]>")
      .s("supersaw").legato(.9).slow(2).gain(.32)
      .lpf(sine.slow(12).range(500,2400)).room(.55).delay(.25).delayfeedback(.35)
  )
  .room(sine.slow(24).range(.15,.7))
  .lpf(sine.slow(10).range(600,3500))
  .reset(32)
