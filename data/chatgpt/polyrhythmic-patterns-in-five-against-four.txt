// 🎵 "Five/Four Friction" — gritty polyrhythmic club pressure
setcps(0.75)
samples('github:tidalcycles/dirt-samples')
$: s("bd*4")
  .gain(1.05)
  .lpf(perlin.slow(8).range(120,1400))
  .room(0.18)
  .stack(
    s("cp*5")
      .gain(0.78)
      .hpf(perlin.slow(6).range(500,2200))
      .room(0.35)
      .pan(sine.slow(5)),
    s("<hh(5,8) hh(4,8)>")
      .fast(2)
      .gain(0.42)
      .lpf(sine.slow(9).range(4000,12000))
      .delay(sine.slow(7).range(0,0.35)),
    s("[oh ~ oh] [~ oh] [oh ~ ~ oh] [~ oh ~]")
      .fast("<1 5/4 1 4/5>")
      .gain(0.33)
      .room(0.55)
      .delay(0.25),
    s("<[perc(5,16) ~ perc(5,16)] [perc(4,16) perc(5,16)]>")
      .gain(0.5)
      .pan(perlin.slow(4))
      .lpf(perlin.slow(11).range(900,6000)),
    s("<bd(5,16) [~ bd bd] [bd ~] [~ bd]>")
      .gain(0.55)
      .speed("<1 0.5 1.5 1>")
      .hpf(220)
      .delay(0.18)
  )
  .delay(sine.slow(13).range(0,0.28))
  .room(sine.slow(16).range(0.2,0.7))
  .reset(64)
