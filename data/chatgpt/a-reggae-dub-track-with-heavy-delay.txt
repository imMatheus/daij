// "Kingston Smoke" — reggae dub with cavernous delay
samples('github:tidalcycles/dirt-samples')
setcps(24/70)

const swell = perlin.slow(9).range(.48,.82)
const smoke = sine.slow(7).range(700,2400)
const air = sine.slow(11).range(4200,9200)

const intro =
  chord("Gm7").voicing().anchor("d5").s("triangle")
    .struct("~ x ~ x ~ x ~ x")
    .attack(.01).decay(.18).sustain(.15).release(1.8)
    .lpf(sine.slow(6).range(900,2200)).room(.78).delay(".82:.375:.9").gain(.12)
    .stack(
      s("rim ~ rim ~ rim ~ rim ~")
        .lpf(1800).room(.66).delay(".9:.5:.92").gain(.035)
    )
    .stack(
      note("d5 ~ ~ bb4 ~ g4 ~ d5").s("sine")
        .attack(.02).decay(.12).sustain(.1).release(2.2)
        .vib(5).vibmod(.25).lpf(2600).room(.84).delay(".86:.375:.93").gain(.05)
    )
    .stack(
      s("[~ hh]*4")
        .begin("<.02 .04>").end("<.06 .09>").bpf(air).bpq(6)
        .room(.4).gain("0 .012 0 .018 0 .012 0 .02 0 .014")
    )

const chorus =
  chord("<Gm7 F7 Gm7 Dm7>").voicing().anchor("d5").slow(2).s("triangle")
    .struct("~ x ~ x ~ x ~ x")
    .attack(.01).decay(.16).sustain(.12).release(1.4)
    .lpf(smoke).room(swell).delay(".72:.375:.9 .8:.5:.92").gain(.14)
    .reset("<x ~ ~ x>")
    .stack(
      note("g1 ~ ~ g1 ~ d2 ~ f1 f1 ~ ~ f1 ~ c2 ~ d2").slow(2).s("sine")
        .attack(.01).decay(.24).sustain(.14).release(.55)
        .lpf(sine.slow(8).range(120,260)).gain(.26)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("~ ~ [bd rim] ~ ~ ~ ~ ~")
        .lpf(1100).room(.18).gain(.18)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("~ ~ rim ~ ~ ~ rim ~")
        .lpf(2200).room(.42).delay(".42:.25:.78").gain(.08)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("[~ hh]*4")
        .begin("<.01 .02 .03 .02>").end("<.05 .08 .06 .09>")
        .bpf(air).bpq(6).room(.22)
        .gain("0 .034 0 .026 0 .032 0 .024 0 .036")
        .reset("<x ~ ~ x>")
    )
    .stack(
      note("<d5 f5 g5 bb5 a5 g5 f5 d5>").slow(4).s("sine")
        .struct("~ ~ x ~ ~ x ~ ~")
        .attack(.02).decay(.12).sustain(.1).release(2.4)
        .vib(6).vibmod(.3).lpf(sine.slow(6).range(1800,4200))
        .room(.84).delay(".88:.75:.94").gain(.055)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("cp ~ ~ ~ cp ~ rim ~").slow(2)
        .lpf(perlin.slow(10).range(1000,2800))
        .room(.7).delay(".92:.375:.95").gain(.032)
        .reset("<x ~ ~ x>")
    )

const bridge =
  chord("<Ebmaj7 Dm7 Gm7>").voicing().anchor("c5").slow(3).s("triangle")
    .struct("~ x ~ ~ ~ x ~ ~")
    .attack(.02).decay(.18).sustain(.12).release(2.1)
    .lpf(perlin.slow(7).range(500,1600)).room(.9).delay(".9:.5:.95").gain(.11)
    .reset("<x ~ x ~>")
    .stack(
      note("<eb2 d2 g1>").slow(3).s("sine")
        .struct("x ~ ~ ~ ~ ~ x ~")
        .attack(.01).decay(.28).sustain(.16).release(1.2)
        .lpf(sine.slow(9).range(110,220)).gain(.23)
    )
    .stack(
      note("<bb4 d5 g5>").slow(3).s("sine")
        .struct("~ ~ x ~ ~ ~ x ~")
        .attack(.02).decay(.1).sustain(.08).release(2.6)
        .vib(7).vibmod(.35).lpf(sine.slow(5).range(1500,3400))
        .room(.92).delay(".94:.75:.96").gain(.05)
    )
    .stack(
      s("~ ~ rim ~ ~ ~ ~ ~")
        .lpf(1800).room(.8).delay(".88:.75:.92").gain(.05)
    )
    .stack(
      s("~ hh ~ ~ ~ hh ~ ~")
        .begin(.02).end(.07).bpf(sine.slow(10).range(3500,7000)).bpq(5)
        .room(.5).gain(.018)
    )
    .stack(
      s("cp").euclidRot(3,8,5)
        .lpf(900).room(.92).delay(".96:.375:.97").gain(.026)
    )

arrange([1, intro], [10, chorus], [3, bridge], [10, chorus]).postgain(.98)
