// 🎵 "Flicker Reels" — haunted ragtime pulse for an old silent film
samples('github:tidalcycles/dirt-samples')
setcps(0.9)
note("<[0 3 7 10] [2 5 9 12] [5 8 12 15] [7 10 14 17]>")
  .scale("minor")
  .s("piano:1")
  .slow(2)
  .attack(0.01)
  .release(0.35)
  .gain(0.55)
  .room(sine.slow(8).range(0.2,0.75))
  .delay(perlin.slow(7).range(0,0.35))
  .lpf(perlin.slow(5).range(600,2600))
  .stack(
    s("<~ [bd bd*2] [bd(5,8) ~ bd] [bd*3] [bd(3,8) ~]>")
      .gain(0.8)
      .lpf(perlin.slow(6).range(400,1700))
      .room(0.25),
    s("<[~ sd] [sd ~ sd] [~ sd sd] [sd:2 ~]>")
      .gain(0.45)
      .room(0.4)
      .delay(0.15),
    s("hh(13,16) <~ hh:2> [hh(5,8) ~]")
      .gain(0.22)
      .hpf(perlin.slow(10).range(3500,9000))
      .delay(0.2),
    n("<0 0 -2 -3 [0 2] 3 5 [3 2]>")
      .s("bass:<saw square>")
      .slow(2)
      .gain(0.42)
      .lpf(perlin.slow(4).range(120,520))
      .room(0.18),
    note("<[12 14 15 17] [10 12 14 15] [8 10 12 14] [7 8 10 12]>")
      .s("violin:2")
      .slow(4)
      .legato(1.2)
      .gain(0.28)
      .room(0.82)
      .delay(0.3)
      .lpf(sine.slow(6).range(1200,3400)),
    s("cp*2 [~ cp] [cp cp:2] [~ ~ cp]")
      .gain(0.18)
      .speed("<1 0.8 1.2 1>")
      .room(0.65)
  )
  .reset(32)
