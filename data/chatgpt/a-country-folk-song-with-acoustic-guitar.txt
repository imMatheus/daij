// "Front Porch Dust" — warm country folk picking with acoustic twang
samples('github:tidalcycles/dirt-samples')
setcps(0.62)
s("<gtr gtr:1 [gtr gtr:2] gtr:3>*2")
  .n("<0 2 4 5 7 5 4 2>")
  .speed("<1 0.99 1.01 1>*2")

  .lpf(perlin.slow(6).range(900,2800))

    s("[~ sd] ~ <sd ~ [sd sd]> ~").gain(0.44).room(0.2),
    s("hh*8").gain("0.18 0.12 0.16 0.1 0.2 0.12 0.15 0.09").hpf(5000),
    s("<cp ~> [~ cp] <~ cp cp ~>").gain(0.24).room(0.45).delay(0.18),
    note("<[0,4,7] [2,5,9] [4,7,11] [5,9,12]>")
      .sound("triangle")
      .slow(2)
      .gain(0.16)
      .lpf(sine.slow(8).range(300,1200))
      .room(0.55)
      .delay(0.35)
  )
