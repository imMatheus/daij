// "Fractured Blue" — sparse heartbroken pulse with drowned keys and ghost vox
samples('github:tidalcycles/dirt-samples')
setcps(0.75)
s("~")
  .stack(
    s("[bd(3,8,<0 1>) ~] [~ bd:2] ~ [bd ~ ~ bd:1]")
      .gain("0.8 0.7 0.75 0.65")
      .lpf(perlin.range(200,900).slow(6))
      .room(0.08),
    s("~ [~ sd:3] ~ [~ sd:4]")
      .gain(0.5)
      .shape(0.2)
      .room(0.35)
      .delay(0.12),
    s("hh(5,16,<2 4>)")
      .gain(perlin.range(0.04,0.16).slow(3))
      .lpf(perlin.range(2500,9000).slow(5))
      .room(0.2),
    note("<[d3 a3 c4] [f3 c4 e4] [bb2 f3 a3] [g2 d3 f3]>")
      .s("triangle")
      .slow(2)
      .legato(1.3)
      .gain(0.22)
      .lpf(sine.slow(8).range(350,1400))
      .room(0.85)
      .size(0.9)
      .delay(0.3),
    note("<d5 [f5 a5] ~ [c5 bb4]>")
      .s("sine")
      .slow(4)
      .legato(0.9)
      .gain(perlin.range(0.03,0.11).slow(7))
      .room(0.92)
      .delay(0.45)
      .lpf(perlin.range(900,2600).slow(4)),
    s("<[~ ~ voice:2] [~ voice:4 ~] [~ ~ ~] [voice:1 ~ ~]>")
      .speed("<0.5 0.75 1>")
      .begin("<0 0.1 0.2>")
      .end("<0.25 0.35 0.5>")
      .gain(0.22)
      .hpf(250)
      .room(0.9)
      .delay(0.5)
  )
  .reset(48)
