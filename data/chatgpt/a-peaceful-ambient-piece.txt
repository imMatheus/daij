// "Still Water" — soft ambient drift with breathing harmonics
samples('github:tidalcycles/dirt-samples')
setcps(0.34)
$: n("<[0 2 4 7] [0 3 5 7] [2 5 7 9] [0 4 7 11]>")
  .scale("minor")
  .s("superpiano")
  .slow(2)
  .legato(1.8)
  .gain(.55)
  .lpf(perlin.range(500,2600).slow(6))
  .room(.85).size(.95).delay(.3)
  .stack(
    n("<0 [~ 7] 5 [2 ~]>")
      .scale("minor")
      .s("sine")
      .slow(4)
      .legato(2)
      .gain(.35)
      .lpf(sine.range(200,900).slow(8))
      .room(.9).delay(.45),
    s("<wind:3 wind:5 wind:7 [~ wind:9]>")
      .euclid(5,8)
      .slow(2)
      .gain(.22)
      .lpf(perlin.range(1200,5000).slow(10))
      .room(.98).delay(.52),
    s("<~ [hh:8*2]? [shaker:2*4] ~>")
      .euclid(3,8)
      .gain(.12)
      .lpf(2400)
      .room(.7),
    s("<~ ~ [bd:4 ~] ~>")
      .slow(2)
      .gain(.08)
      .lpf(220)
      .room(.65)
  )
  .reset(16)
