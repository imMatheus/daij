// "Friction Ritual" - no-melody drum architecture built from pulse, hiss, and dub space
samples('github:tidalcycles/dirt-samples')
setcps(11/25)

const throb = perlin.slow(9).range(420,1200)
const body = sine.slow(7).range(700,1800)
const metal = sine.slow(11).range(4200,11000)
const fog = perlin.slow(13).range(.18,.72)

const intro =
  s("bd")
    .struct("x ~ ~ ~ ~ ~ ~ ~ ~ ~ ~ ~ ~ ~ ~ ~")
    .lpf(760)
    .room(.08)
    .gain(.9)
    .stack(
      s("[hh ~]*8")
        .begin(".01 .02")
        .end(".03 .05 .04 .06")
        .hpf(metal)
        .delay(".16:.125:.22")
        .room(".28:.82")
        .gain(".025 .06 .02 .07 .03 .05 .02 .08")
        .orbit(2)
    )
    .stack(
      s("~ ~ ~ ~ ~ ~ ~ oh")
        .speed(-1)
        .end(.18)
        .hpf(5400)
        .room(".62:.96")
        .delay(".3:.5:.44")
        .gain(.11)
        .orbit(3)
    )
    .stack(
      s("rim")
        .struct("~ ~ ~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~ ~ ~")
        .hpf(2400)
        .room(.22)
        .gain(.08)
        .orbit(4)
    )

const chorus =
  s("bd")
    .struct("x ~ ~ [~ x] ~ ~ x ~ x ~ ~ [~ x] ~ ~ [x x] ~")
    .lpf(throb)
    .distort(".6:.95")
    .room(.05)
    .gain(.98)
    .stack(
      s("sd cp")
        .struct("~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~ x ~ [~ x] ~")
        .lpf(body)
        .room(".14:.32")
        .delay(".06:.125:.12")
        .gain(".26 .18")
        .orbit(2)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("hh*16")
        .begin(".01 .015 .02 .01")
        .end(".04 .05 .03 .06")
        .hpf(metal)
        .room(".12:.46")
        .delay(".08:.125:.14")
        .gain(".07 .04 .08 .035 .075 .045 .085 .04 .09 .045 .08 .035 .075 .045 .085 .11")
        .pan(sine.slow(9).range(.28,.72))
        .orbit(3)
        .reset("<x ~ x ~>")
    )
    .stack(
      s("~ ~ oh ~ ~ ~ ~ [~ oh] ~ ~ ~ oh ~ ~ ~ ~")
        .speed("<1 -1 1 -1>")
        .end(".12 .18 .14 .22")
        .hpf(sine.slow(8).range(3800,9200))
        .room(".36:.94")
        .delay(".2:.25:.28 .32:.5:.44")
        .gain(".07 .11 .08 .12")
        .orbit(4)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("rim")
        .euclidRot(5,16,3)
        .hpf(2600)
        .room(.24)
        .delay(".05:.125:.1")
        .gain(.12)
        .orbit(2)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("lt mt ht mt")
        .struct("~ x ~ ~ x ~ ~ x ~ x ~ ~ x ~ ~ ~")
        .speed(".92 1 1.06 1.12")
        .lpf(sine.slow(6).range(500,1800))
        .room(fog)
        .delay(".22:.25:.34")
        .gain(".18 .15 .12 .14")
        .orbit(4)
        .reset("<x ~ x ~>")
    )
    .stack(
      s("bd")
        .struct("x ~ ~ [~ x] ~ ~ x ~ x ~ ~ [~ x] ~ ~ [x x] ~")
        .duckorbit("2:3:4")
        .duckattack("0.16:0.12:0.2")
        .duckdepth(".42:.28:.64")
        .postgain(0)
    )

const bridge =
  s("bd")
    .struct("x ~ ~ ~ ~ ~ x ~ x ~ ~ ~ ~ ~ x ~")
    .lpf(860)
    .room(.12)
    .gain(.78)
    .stack(
      s("lt mt ht rim")
        .euclidRot(7,16,5)
        .slow(2)
        .speed(".88 .96 1.04 1.12")
        .lpf(sine.slow(5).range(420,1500))
        .room(".48:.98")
        .delay(".28:.375:.42")
        .gain(".2 .16 .12 .09")
        .orbit(2)
        .reset("<x ~ x ~>")
    )
    .stack(
      s("[hh ~]*8")
        .begin(".02 .01")
        .end(".05 .04")
        .hpf(sine.slow(10).range(7000,12000))
        .room(".3:.84")
        .delay(".12:.25:.18")
        .gain(".02 .05 .02 .06 .025 .05 .02 .07")
        .orbit(3)
    )
    .stack(
      s("oh")
        .struct("~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~ x ~ ~ ~")
        .speed("<-1 1>")
        .end(".22 .14")
        .hpf(4600)
        .room(".66:.98")
        .delay(".34:.5:.52")
        .gain(".1 .08")
        .orbit(4)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("cp")
        .struct("~ ~ ~ ~ ~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~")
        .hpf(1800)
        .room(.54)
        .delay(".18:.375:.3")
        .gain(.08)
        .orbit(2)
    )
    .stack(
      s("bd")
        .struct("x ~ ~ ~ ~ ~ x ~ x ~ ~ ~ ~ ~ x ~")
        .duckorbit("2:3:4")
        .duckattack("0.22:0.18:0.26")
        .duckdepth(".3:.18:.48")
        .postgain(0)
    )

arrange([1, intro], [13, chorus], [5, bridge], [13, chorus]).postgain(.98)
