const outroBass = note("<e1 b0>").s("sine").release(1.8).lpf(120).gain(.12).orbit(3)

const introDrums = s("~ hh ~ [hh hh]").hpf(9000).gain(.09).delay("0.15:0.12:0.25")
const chorusKick = s("bd*4").gain("1 .92 1 .95").distort("1.4:.82").duckorbit("2:3").duckattack("0.25:0.14").duckdepth("1:.58")
const chorusDrums = stack(chorusKick,s("~ sd ~ [sd cp]").gain(.72).room(.08),s("hh*8").gain(".34 .24 .39 .26 .31 .21 .37 .19").hpf(6800).sometimesBy(.3,x=>x.degradeBy(.2)),s("~ oh ~ ~").cut(1).gain(.24)).swing(4).someCyclesBy(.25,x=>x.brak())

const introAir = s("pink*8").gain(.03).bpf(perlin.slow(2).rangex(900,2600)).bpq(2).room(".5:6").orbit(4)
const chorusAir = s("pink*16").gain(".025 .015 .03 .02").bpf(perlin.slow(5).rangex(1200,4800)).bpq(1.8).pan(rand.segment(16)).delay("0.25:0.125:0.4").orbit(4).sometimesBy(.28,x=>x.degradeBy(.45))

const outroAir = s("pink*8").gain(.01).bpf(2200).bpq(2).room(".6:8").orbit(4)

  arrange([1,introBass],[12,chorusBass],[4,bridgeBass],[12,chorusBass],[3,outroBass]),
  arrange([1,introDrums],[12,chorusDrums],[4,bridgeDrums],[12,chorusDrums],[3,outroDrums]),
