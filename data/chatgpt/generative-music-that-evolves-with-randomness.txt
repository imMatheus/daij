// "Entropy Bloom" — generative polyrhythmic ambient breakbeat
samples('github:tidalcycles/dirt-samples')
setcps(0.82)

$: s("bd(5,8,<0 1 2>) [~ bd]*2")
  .gain(1.05)
  .lpf(sine.range(280,920).slow(7))
  .stack(
    s("<hh(13,16) hh(9,16)> [~ hh]*2")
      .gain(perlin.range(0.18,0.55).slow(4))
      .hpf(sine.range(4500,9800).slow(9))
      .room(0.18),
    s("<cp(3,8,2) [~ cp]>")
      .gain(0.6)
      .delay(0.125)
      .delayfeedback(0.42)
      .room(0.25),
    s("<lt(5,8) mt(3,8) ht(7,16) [lt ht](5,8)>*2")
      .gain(0.35)
      .lpf(perlin.range(500,2200).slow(5))
      .pan(sine.slow(6))
      .sometimesBy(0.35, x => x.rev()),
    s("arpy*4")
      .note("<[0 3 7 10] [2 5 9 12] [3 7 10 14] [5 9 12 15]>")
      .add(irand(3))
      .scale("minor")
      .slow("<2 2 4 1>")
      .gain(0.42)
      .lpf(perlin.range(350,2800).slow(11))
      .delay(0.35)
      .delayfeedback(0.6)
      .room(0.82)
      .pan(perlin.range(0.15,0.85).slow(8)),
    s("<[~ feel] [can can ~ can]>*2")
      .speed("<0.5 1 1.5 -1>")
      .begin(rand.range(0,0.2))
      .end(rand.range(0.8,1))
      .gain(0.25)
      .room(0.7)
      .delay(0.5)
      .lpf(sine.range(700,4200).slow(13))
  )
  .reset(16)
  .room(0.2)

setTimeout(() => hush(), 48000)
