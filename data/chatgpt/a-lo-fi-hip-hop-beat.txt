setcps(0.8)

    s("[hh*8 <hh*4 [~ hh]*4>]").gain(0.28).hpf(5500).delay(0.125).delayfeedback(0.35),

    note("<[c4 d4 eb4 g4] [c4 eb4 g4 bb4] [bb3 d4 f4 a4] [ab3 c4 eb4 g4]>")
      .s("sawtooth")
      .gain(0.24)
      .lpf(sine.range(500,1800).slow(8))
      .room(0.6)
      .delay(0.25)
      .delayfeedback(0.45),
    note("<c2 [~ c2] [bb1 c2] [ab1 ~]>")
      .s("triangle")
      .gain(0.34)
      .lpf(420)
