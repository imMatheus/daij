// "Skybound Sprint" - uplifting energetic synth-house
samples('github:tidalcycles/dirt-samples')
setcps(140/240)

const intro = 1
const lift = 5
const chorus = 10
const verse = 8
const bridge = 4
const outro = 3

const introDrums = s("bd ~ ~ ~, ~ ~ ~ ~, hh*8").gain(".9 .2 .2 .2").lpf(2200).room(.35)
const liftDrums = s("bd*2 [~ hh]*2, ~ cp ~ ~, hh*8, ~ ~ oh ~").gain(.78).lpf(sine.slow(5).range(2400,12000)).room(.18)
const chorusDrums = s("bd*4, ~ cp ~ cp, hh*8, ~ ~ oh ~ ~ ~ oh ~").gain("1 .95 .92 .95").room(.15).duckorbit("2:3").duckattack(.22).duckdepth("1:.7")
const verseDrums = s("bd*4, ~ cp ~ cp, hh*4 [~ hh]*2, ~ ~ oh ~").gain(.88).room(.15).duckorbit("2:3").duckattack(.18).duckdepth("1:.55")
const bridgeDrums = s("bd ~ ~ bd, ~ cp ~ ~, hh*4").gain(.74).room(.22).lpf(6000).duckorbit("2:3").duckattack(.28).duckdepth("1:.45")

const liftBass = note("<d2 a1 b1 g1>").slow(4).struct("[x ~ ~ x]").s("sawtooth").orbit(2).lpf(sine.slow(6).range(180,900)).lpq(.15).attack(.01).decay(.16).sustain(.4).release(.08).gain(.58).reset("<x ~ ~ ~>")
const chorusBass = note("<d2 a1 b1 g1>").slow(4).struct("[x ~ x [~ x]]").s("sawtooth").orbit(2).lpf(sine.slow(6).range(240,1400)).lpq(.2).attack(.01).decay(.18).sustain(.45).release(.08).drive(.15).gain(.72).reset("<x ~ ~ x>")
const bridgeBass = note("<b1 g1 d2 a1>").slow(4).struct("[x ~ ~ x]").s("sawtooth").orbit(2).lpf(sine.slow(4).range(180,700)).attack(.02).decay(.2).sustain(.35).release(.1).gain(.46).reset("<x ~ x ~>")

const introPads = note("[d4 fs4 a4]").s("supersaw").orbit(2).attack(.35).decay(.4).sustain(.65).release(1.3).lpf(sine.slow(1).range(500,2800)).room(.82).delay(".35:.4:.45").gain(.24)
const liftPads = note("<[d4 fs4 a4] [a3 cs4 e4] [b3 d4 fs4] [g3 b3 d4]>").slow(4).struct("[x ~ x ~]").s("supersaw").orbit(2).attack(.03).decay(.35).sustain(.55).release(.28).lpf(sine.slow(5).range(700,4800)).room(.5).delay(".22:.25:.35").gain(.32).reset("<x ~ x ~>")
const chorusPads = note("<[d4 fs4 a4] [a3 cs4 e4] [b3 d4 fs4] [g3 b3 d4]>").slow(4).struct("[x ~ x x]").s("supersaw").orbit(2).attack(.02).decay(.32).sustain(.58).release(.24).lpf(sine.slow(8).range(1800,6200)).lpq(.2).room(.36).delay(".25:.375:.45").gain(.42).reset("<x ~ x ~>")
const versePads = note("<[d4 fs4 a4] [a3 cs4 e4] [b3 d4 fs4] [g3 b3 d4]>").slow(4).struct("[x ~ ~ x]").s("pulse").orbit(2).attack(.01).decay(.18).sustain(.35).release(.14).lpf(sine.slow(8).range(1000,4200)).room(.24).delay(".16:.25:.25").gain(.24).reset("<x ~ ~ x>")
const bridgePads = note("<[b4 d5 fs5] [g4 b4 d5] [d4 fs4 a4] [a4 cs5 e5]>").slow(4).struct("[x ~ x ~]").s("supersaw").orbit(2).attack(.08).decay(.45).sustain(.62).release(.45).lpf(sine.slow(4).range(1200,5000)).room(.7).delay(".3:.5:.5").gain(.28).reset("<x ~ x ~>")

const liftArp = n("<0 2 4 7 9 7 4 2>").scale("D5:major:pentatonic").fast(2).s("triangle").orbit(3).attack(.01).decay(.12).sustain(.28).release(.1).lpf(sine.slow(5).range(1200,4800)).delay(".22:.25:.32").room(.24).gain(.18).reset("<x ~ ~ x>")
const chorusArp = n("<0 2 4 7 9 7 4 2 0 2 4 7 11 9 7 4>").scale("D5:major:pentatonic").fast(2).s("triangle").orbit(3).attack(.01).decay(.14).sustain(.3).release(.12).lpf(sine.slow(7).range(1500,7200)).delay(".3:.25:.4").room(.25).gain(.32).reset("<x ~ ~ x>")
const verseArp = n("<0 ~ 2 ~ 4 ~ 2 ~ 0 ~ 2 ~ 5 ~ 4 ~>").scale("D5:major:pentatonic").fast(2).s("triangle").orbit(3).attack(.01).decay(.12).sustain(.22).release(.1).lpf(sine.slow(8).range(1200,5200)).delay(".2:.25:.28").room(.18).gain(.18).reset("<x ~ ~ ~>")
const bridgeArp = n("<11 9 7 4 2 4 7 9>").scale("D5:major").fast(1).s("triangle").orbit(3).attack(.02).decay(.18).sustain(.32).release(.22).lpf(sine.slow(3).range(1800,5600)).delay(".38:.5:.45").room(.5).gain(.2).reset("<x ~ x ~>")

const chorusLead = n("<7 ~ 9 ~ 11 9 7 4 7 ~ 9 ~ 7 4 2 ~>").scale("D5:major").fast(2).s("supersaw").orbit(3).attack(.02).decay(.18).sustain(.4).release(.15).lpf(sine.slow(5).range(2200,9000)).delay(".35:.25:.5").room(.3).gain(.26).reset("<x ~ x ~>")
const bridgeLead = n("<9 ~ 7 ~ 4 ~ 2 ~>").scale("D5:major").fast(2).s("supersaw").orbit(3).attack(.03).decay(.18).sustain(.36).release(.18).lpf(sine.slow(3).range(1800,6400)).delay(".42:.5:.55").room(.55).gain(.16).reset("<x ~ x ~>")
const sparkle = s("hh*16").begin(sine.slow(4).range(0,.08)).end(.12).hpf(sine.slow(4).range(4500,9500)).room(.42).gain(.12)

$: stack(
  arrange(
    [intro, introDrums],
    [lift, liftDrums],
    [chorus, chorusDrums],
    [verse, verseDrums],
    [bridge, bridgeDrums],
    [chorus, chorusDrums],
    [outro, liftDrums.gain(saw.slow(3).range(.45,.12)).lpf(sine.slow(3).range(1400,5000))]
  ),
  arrange(
    [intro, silence],
    [lift, liftBass],
    [chorus, chorusBass],
    [verse, liftBass],
    [bridge, bridgeBass],
    [chorus, chorusBass],
    [outro, bridgeBass.gain(saw.slow(3).range(.3,.05))]
  ),
  arrange(
    [intro, introPads],
    [lift, liftPads],
    [chorus, chorusPads],
    [verse, versePads],
    [bridge, bridgePads],
    [chorus, chorusPads],
    [outro, introPads.gain(saw.slow(3).range(.22,.04))]
  ),
  arrange(
    [intro, silence],
    [lift, liftArp],
    [chorus, chorusArp],
    [verse, verseArp],
    [bridge, bridgeArp],
    [chorus, chorusArp],
    [outro, silence]
  ),
  arrange(
    [intro, silence],
    [lift, silence],
    [chorus, chorusLead],
    [verse, silence],
    [bridge, bridgeLead],
    [chorus, chorusLead],
    [outro, silence]
  ),
  arrange(
    [intro, silence],
    [lift, silence],
    [chorus, silence],
    [verse, silence],
    [bridge, silence],
    [chorus, sparkle],
    [outro, silence]
  )
).postgain(1.1)
