// "Clockwork Kites" — playful toybox groove
samples('github:tidalcycles/dirt-samples')
setcps(0.72)
note("<[c5 e5 g5 b5] [c5 f5 a5 c6] [b4 g4 e4 d4] [c5 e5 a5 d6]>/2")
  .s("<triangle square>")
  .legato(0.85)
  .attack(0.01)
  .release(0.15)
  .gain(0.8)
  .pan(sine.slow(6).range(0.2,0.8))
  .stack(
    note("<c4 [e4 g4] a4 [g4 e4]>*2")
      .s("sine")
      .octave(4)
      .gain(0.35)
      .legato(0.6)
      .delay(0.25)
      .delaytime(3/16)
      .delayfeedback(0.45),
    s("[bd(3,8,<1 2 3>) ~] <hh*4 [hh*8]> [~ cp] [~ bd]")
      .gain(0.9)
      .lpf(perlin.slow(5).range(700,5000))
      .room(sine.slow(7).range(0.05,0.35)),
    s("<perc*2 [click*3] [rim(5,8)]>")
      .speed("<1 1.5 0.5 2>")
      .gain(0.45)
      .pan(perlin.slow(9).range(0,1))
      .room(0.5),
    note("[[c6 b5 a5] ~ [g5 a5 b5] ~] [~ [d6 c6 b5] ~ [a5 g5 e5]]")
      .s("sawtooth")
      .gain(0.2)
      .hpf(1200)
      .delay(0.4)
      .delaytime(1/8)
  )
  .lpf(perlin.slow(12).range(900,4200))
  .room(0.25)
  .size(0.7)
  .reset(32)
