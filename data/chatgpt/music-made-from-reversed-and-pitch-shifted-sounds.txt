// 🎵 "Backmask Mirage" — reverse-shifted tape ghost groove
samples('github:tidalcycles/dirt-samples')
setcps(0.85)
s("[bd*2 <~ bd> [bd bd*2] ~]")
  .speed("<1 0.9 1.1 1>")
  .lpf(perlin.range(180,1400).slow(8))
  .room(0.25)
  .stack(
    s("hh(11,16) [~ hh(5,8)] <hh*2 [hh hh*3]>")
      .speed("<-1 -0.5 -1.25 -0.75>")
      .gain(0.33)
      .lpf(perlin.range(2500,12000).slow(5))
      .delay(0.35).delaytime(0.125).delayfeedback(0.45)
      .pan(sine.range(0.15,0.85).slow(6)),
    s("<sd:2 ~ [sd:5 sd:7] ~> [~ cp ~ cp]")
      .speed("<-1.5 -1 -0.5 -2>")
      .gain(0.5)
      .room(0.6)
      .size(0.9)
      .lpf(perlin.range(400,3000).slow(7)),
    s("arpy*8 [~ arpy*4] <arpy*2 [arpy arpy*3]>")
      .n("<[0 3 7 10] [12 10 7 3] [5 8 12 15] [3 7 10 14]>")
      .speed("<-2 -1 -1.5 -0.5>")
      .gain(0.42)
      .legato(1.2)
      .room(0.8)
      .delay(0.5).delaytime(0.25).delayfeedback(0.55)
      .lpf(perlin.range(300,4200).slow(9))
      .pan(sine.range(0.1,0.9).slow(12)),
    s("<lt*4 [mt mt*2] ~ ht*3> [~ cp ~ [cp cp]]")
      .speed("<-1 -2 -0.5 -1.25>")
      .gain(0.28)
      .hpf(perlin.range(500,3000).slow(4))
      .room(0.7)
      .crush("<4 6 8 5>")
  )
  .reset(64)
  .room(sine.range(0.2,0.85).slow(16))
  .delay(0.2)
  .delaytime(0.375)
  .delayfeedback(0.35)
