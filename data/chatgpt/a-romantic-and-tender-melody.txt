samples('github:tidalcycles/dirt-samples')
setcps(0.75)
$: note("<[c4 e4 g4] [d4 fs4 a4] [e4 g4 b4] [b3 d4 fs4]>")

  .gain(0.55)
  .lpf(sine.slow(6).range(500,2400))

      .legato(1.2)
      .gain(0.32)
      .delay(0.35)
      .room(0.82)
      .lpf(perlin.slow(5).range(900,3200))
      .reset(8),

      .s("sine")
      .slow(2)
      .legato(1.1)
      .gain(0.28)
      .lpf(900)
      .room(0.6)
      .reset(16),
    s("~ [hh(5,8) hh(3,8)] <~ cp(3,8)> [~ hh] ~")
      .bank("RolandTR808")
      .gain(0.2)
      .lpf(sine.slow(10).range(2500,7000))

      .reset(4),

      .gain(0.18)

      .room(0.18)
