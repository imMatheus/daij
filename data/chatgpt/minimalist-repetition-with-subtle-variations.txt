// 🎵 "Glass Loop Drift" — minimalist repetition with subtle variations
setcps(0.75)
samples('github:tidalcycles/dirt-samples')
$: note("<[0 0 2 0] [0 0 3 0] [0 0 2 5] [0 0 3 7]>")
  .scale("c:minor")
  .s("sine")
  .octave("<4 4 4 5>")
  .legato(0.92)
  .gain(0.22)
  .lpf(perlin.slow(6).range(500,1800))
  .room(sine.slow(12).range(0.18,0.42))
  .delay(0.25)
  .stack(
    s("<bd(3,8) bd(3,8,1)>")
      .gain(0.9)
      .lpf(900)
      .room(0.08),
    s("[~ hh]*2 hh(5,8,<0 1>)")
      .gain(0.22)
      .lpf(perlin.slow(9).range(4000,9000))
      .room(0.2),
    s("<~ cp ~ cp [~ cp]>")
      .gain(0.3)
      .room(0.35)
      .delay(0.125),
    note("<[-12 ~ -12 ~] [-12 ~ -10 ~]>")
      .s("square")
      .slow(4)
      .gain(0.16)
      .lpf(sine.slow(10).range(120,260))
      .room(0.4)
  )
  .reset(48)
