samples('github:tidalcycles/dirt-samples')

  .lpf(perlin.slow(6).range(90,170))

    s("<hh*8 hh*16> [~ hh]*2")
      .gain(0.33)
      .hpf(9000)
      .delay(0.125)
      .room(0.2)

    s("cp(3,8) <~ cp> [cp ~ cp]")
      .gain(0.56)
      .room(0.28)
      .delay(0.25)
      .pan(sine.slow(5))
  )

    s("perc:[2 4 7 9](5,8,<2 3>)")
      .gain(0.42)
      .speed("<1 1.1 0.9>")
      .room(0.35)
      .pan(perlin.slow(7))
  )
  .stack(
    note("<e2 ~ e2 [e2 e2] g2 a2>")
      .sound("square")
      .gain(0.43)
      .lpf(perlin.slow(5).range(120,430))
      .room(0.16)
      .delay(0.125)
      .reset(16)
  )
  .stack(
    note("<[e4 g4 b4 d5] [d4 f#4 a4 c5] [c4 e4 g4 b4] [b3 d4 f#4 a4]>*2")

      .gain(0.25)

      .room(0.54)
      .delay(0.375)
      .pan(perlin.slow(9))
      .mask("<1 1 0 1 1 0 1 1>")

    note("<e5 [g5 b5] ~ [d5 f#5 a5] [c5 e5 g5] ~ [b4 d5 f#5]>")
      .sound("triangle")
      .gain(0.19)
      .delay(0.5)
      .room(0.65)
      .hpf(700)
      .pan(sine.slow(3))
      .reset(16)
  )
  .stack(
    s("<oh ~ [oh oh] ~>*2")
      .gain(0.27)
      .hpf(6500)
      .delay(0.25)
      .mask("<0 1 0 1 1 0 1 1>")
  )

    s("<wind:2 ~ ~ wind:3>*2")
      .gain(0.2)
      .speed("<0.75 0.8>")
      .lpf(1200)
      .room(0.92)
      .reset(32)
