// "Golden Hour Lift" — sunset rooftop house with warm synth swells and glitter hats
samples('github:tidalcycles/dirt-samples')
setcps(.5)

const progression = "<F^7 Am7 Gm7 C7>"

const intro = chord(progression)
  .anchor("c5")
  .voicing()
  .s("supersaw")
  .attack(.05)
  .release(1.4)
  .lpf(1100)
  .room(".7:0.9")
  .delay(".3:0.25:0.45")
  .gain(.26)
  .orbit(2)
  .stack(
    note("f2 ~ a2 ~ g2 ~ c2 ~")
      .s("sine")
      .attack(.02)
      .release(.5)
      .gain(.16)
      .orbit(2)
  )
  .stack(
    s("~ ~ hh ~")
      .gain(.12)
      .lpf(7500)
      .room(.3)
  )

const chorus = chord(progression)
  .anchor("c5")
  .voicing()
  .s("supersaw")
  .attack(.02)
  .release(.85)
  .lpf(perlin.slow(6).range(1000,3200))
  .delay(".38:0.25:0.68")
  .room(".35:0.82")
  .gain(.3)
  .orbit(2)
  .stack(
    note("f2 a2 g2 c2")
      .s("sawtooth")
      .attack(.01)
      .release(.18)
      .lpf(perlin.slow(4).range(420,880))
      .gain(.42)
      .orbit(2)
  )
  .stack(
    n("<0 [2 4] 4 7 6 4 2 [4 6]>*2")
      .scale("F:major")
      .transpose("<0 0 0 2>/4")
      .s("pulse")
      .attack(.01)
      .release(.22)
      .lpf(perlin.slow(5).range(900,2600))
      .chorus(.28)
      .delay(".42:0.125:0.5")
      .room(.24)
      .vib(5)
      .vibmod(.12)
      .pan(sine.slow(8))
      .gain(.22)
      .reset("<x ~ ~ ~ x ~ ~ ~>")
      .orbit(2)
  )
  .stack(
    s("bd*4")
      .gain(.96)
      .duckorbit(2)
      .duckattack(.22)
      .duckdepth(.9)
  )
  .stack(
    s("~ cp ~ cp")
      .gain(.46)
      .room(.18)
  )
  .stack(
    s("hh*8")
      .gain(".15 .12 .17 .13")
      .lpf(perlin.slow(8).range(5000,11000))
      .pan(sine.fast(4))
  )
  .stack(
    s("~ oh ~ oh")
      .gain(.18)
      .end(.3)
      .delay(".24:0.25:0.35")
  )
  .stack(
    s("[~ hh] ~ [hh hh] ~")
      .fast(2)
      .gain(.09)
      .lpf(9000)
      .late(.01)
  )

const bridge = chord("<Dm7 Gm7 C7 Am7>")
  .anchor("c6")
  .voicing()
  .s("sawtooth")
  .attack(.04)
  .release(1.1)
  .hpf(perlin.slow(4).range(500,1800))
  .delay(".55:0.375:0.72")
  .room(".6:0.95")
  .gain(.24)
  .orbit(2)
  .stack(
    n(run(8).slow(2))
      .scale("F:major")
      .transpose("<7 9 5 4>")
      .s("triangle")
      .attack(.02)
      .release(.28)
      .lpf(perlin.slow(3).range(1200,2600))
      .pan(sine.slow(6))
      .gain(.14)
      .reset("<x ~ x ~>")
      .orbit(2)
  )
  .stack(
    note("d2 ~ g2 ~ c2 ~ a2 ~")
      .s("sine")
      .attack(.02)
      .release(.4)
      .gain(.14)
      .orbit(2)
  )
  .stack(
    s("~ bd ~ ~")
      .gain(.6)
      .duckorbit(2)
      .duckattack(.35)
      .duckdepth(.7)
  )
  .stack(
    s("hh*8")
      .gain(.08)
      .hpf(7000)
      .pan(isaw.slow(4))
  )

arrange(
  [1, intro],
  [14, chorus],
  [6, bridge],
  [14, chorus]
)
