// "Runaway Clock" — Accelerating techno spiraling from groove to frenzy
setcps(0.5)
let acc = saw.slow(35).range(1, 3.5)

s("bd [~ bd:3] sd:4 [bd ~ bd:3]").fast(acc)
.bank("RolandTR909").mask("<0!2 1!13 0!5 1!15>").gain(.85)
.stack(
  s("hh*8").fast(acc).bank("RolandTR909")
    .gain(".5!2 .8 .4 .5 .3 .9 .4")
    .lpf(saw.slow(35).range(3000, 16000))
    .pan(sine.range(.3, .7)),
  s("~ cp:3").fast(acc).bank("RolandTR909")
    .mask("<0!2 1!13 0!5 1!15>").gain(.5).room(.2),
  s("oh(3,8)").fast(acc).bank("RolandTR909")
    .mask("<0!15 1!5 0!15>").gain(.2).room(.25),
  note("<c1 c1 f1 g1>").fast(acc).s("sawtooth")
    .lpf(saw.slow(35).range(120, 2500)).lpq(8).lpenv(5)
    .decay(.1).sustain(0).gain(.7),
  n("<0 [0 2] [4 2] [5 4] 7 [5 7] [4 2] [0 ~]>").fast(acc)
    .mask("<0!3 1!12 0!5 1!15>")
    .scale("C4:minor").s("square")
    .gain(.22).lpf(sine.slow(6).range(600, 5000))
    .delay(".2:.12:.4").room(.15),
  chord("<Cm Fm Gm Eb>").voicing().s("sawtooth")
    .gain(saw.slow(35).range(.03, .15))
    .lpf(saw.slow(35).range(200, 4000))
    .room(.5).attack(.2).release(.4)
).room(.08)
