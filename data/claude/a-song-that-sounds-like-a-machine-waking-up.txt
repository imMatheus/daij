// 🎵 "Boot Sequence" — Industrial Ambient / Machine Music
// At 110 BPM: hum(4) + sputter(8) + warmup(8) + running(12) + shutdown(4) = 36 bars ≈ 1:18

setcpm(110 / 4)

$: arrange(
  [4,  note("c1").s("sawtooth").lpf(saw.slow(4).range(40,120)).gain(saw.slow(4).range(.02,.18)).room(.3).decay(2).sustain(.9)],
  [8,  note("c1").s("sawtooth").lpf(sine.range(80,300).slow(8)).gain(.22).room(.25).decay(1.5).sustain(.8)],
  [8,  note("c1").s("sawtooth").lpf(sine.range(150,500).slow(6)).gain(.28).room(.2).decay(1).sustain(.85).distort(.05)],
  [12, note("<c1 c1 c1 [c1 eb1]>").s("sawtooth").lpf(sine.range(200,800).slow(8)).gain(.32).room(.15).decay(.8).sustain(.9).distort(.08)],
  [4,  note("c1").s("sawtooth").lpf(saw.slow(4).range(600,60)).gain(saw.slow(4).range(.28,.02)).room(.3).decay(2).sustain(.8)]
)

$: arrange(
  [4,  s("hh:3(2,16)").bank("RolandTR909").gain(perlin.range(.0,.08)).lpf(1200).room(.6)],
  [8,  s("hh:3(3,8)").bank("RolandTR909").gain(perlin.range(.05,.22)).lpf(sine.range(800,3000).slow(8)).room(.5).pan(perlin)],
  [8,  s("hh:3(5,8), rim(3,8)").bank("RolandTR909").gain(perlin.range(.1,.32)).lpf(sine.range(1200,5000).slow(6)).room(.4).pan(sine.slow(5))],
  [12, s("hh*16, rim(5,8)").bank("RolandTR909").gain(perlin.range(.15,.42)).lpf(sine.range(2000,8000).slow(4)).room(.3).pan(sine.slow(3)).cut(1)],
  [4,  s("hh*8").bank("RolandTR909").gain(saw.slow(4).range(.3,.0)).lpf(saw.slow(4).range(6000,800)).room(.5).cut(1)]
)

$: arrange(
  [4,  silence],
  [8,  s("<[bd ~ ~ ~] [~ ~ bd ~] [bd ~ ~ ~] [~ bd ~ ~]>").bank("RolandTR909").gain(perlin.range(.15,.4)).room(.35).lpf(sine.range(200,800).slow(8))],
  [8,  s("bd ~ ~ bd, ~ ~ bd ~").bank("RolandTR909").gain(.6).room(.3).lpf(sine.range(400,1500).slow(6))],
  [12, s("bd*4").bank("RolandTR909").gain(.82).room(.2).distort(.06)],
  [4,  s("bd*4").bank("RolandTR909").gain(saw.slow(4).range(.7,.0)).room(.3)]
)

$: arrange(
  [4,  silence],
  [8,  silence],
  [8,  s("~ cp ~ ~").bank("RolandTR909").gain(.45).room(.5).delay(".2:.15:.3")],
  [12, s("~ cp ~ ~, ~ ~ ~ [~ cp]").bank("RolandTR909").gain(.65).room(.35).delay(".15:.1:.2").every(4, x=>x.fast(2).gain(.3))],
  [4,  s("~ cp ~ ~").bank("RolandTR909").gain(saw.slow(4).range(.5,.0)).room(.5)]
)

$: arrange(
  [4,  note("c2").s("square").lpf(saw.slow(4).range(60,200)).gain(.08).room(.4).decay(4).sustain(.6)],
  [8,  note("c2").s("square").lpf(sine.range(100,400).slow(8)).gain(.12).room(.35).decay(2).sustain(.7)],
  [8,  note("<c2 g2>/4").s("square").lpf(sine.range(200,700).slow(6)).gain(.15).room(.3).decay(1.5).sustain(.75)],
  [12, note("<c2 g2 c2 [g2 eb2]>").s("square").lpf(sine.range(300,1200).slow(4)).gain(.18).room(.25).decay(1).sustain(.8).distort(.05)],
  [4,  note("c2").s("square").lpf(saw.slow(4).range(800,60)).gain(saw.slow(4).range(.15,.0)).room(.4).decay(3).sustain(.5)]
)

$: arrange(
  [4,  silence],
  [8,  note("<[~ ~ c5 ~] [~ ~ ~ ~] [~ ~ ~ ~] [~ ~ ~ eb5]>/2").s("sine").gain(.1).room(.7).delay(".4:.3:.5").lpf(2000).attack(.01).decay(.15).sustain(.1).release(.3)],
  [8,  note("<[~ c5 ~ ~] [~ ~ eb5 ~] [~ c5 ~ ~] [~ ~ g4 ~]>").s("sine").gain(.16).room(.6).delay(".3:.25:.4").lpf(3000).attack(.01).decay(.2).sustain(.15).release(.3)],
  [12, note("<[c5 ~ eb5 ~] [~ g5 ~ c5] [eb5 ~ c5 ~] [~ g4 ~ eb5]>").s("sine").gain(.22).room(.5).delay(".25:.2:.35").lpf(4000).attack(.005).decay(.15).sustain(.2).release(.25).sometimes(x=>x.add(note(12)).gain(.1))],
  [4,  silence]
)
