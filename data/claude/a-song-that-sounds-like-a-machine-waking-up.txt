// 🤖 "Boot Sequence" — Industrial Ambient / Glitch Electronics
// At 90 BPM: boot(8) + init(16) + process(16) + full-power(16) + overdrive(16) + shutdown(8) = 80 bars ≈ 3:33

setcpm(90 / 4)

// low machine hum that persists throughout
$: note("c1")
   .s("sawtooth")
   .attack(4).decay(2).sustain(.6).release(3)
   .lpf(sine.range(80,300).slow(16))
   .gain(saw.slow(80).range(.05,.25))
   .room(.4).roomsize(6)
   .slow(4)

// mechanical clicks and glitches — gradually more active
$: arrange(
  [8,  s("hh:3*2")
       .bank("RolandTR808")
       .gain(saw.slow(8).range(.02,.25))
       .pan(perlin.range(0,1))
       .room(.8).crush(8).lpf(1200)],
  [16, s("hh:3*4, ~ [cp:2 ~] ~ ~")
       .bank("RolandTR808")
       .gain(perlin.range(.15,.45))
       .pan(sine.slow(3))
       .room(.5).crush(10).lpf(2400)
       .every(4, x => x.fast(2).gain(.2))],
  [16, s("hh:3*8, ~ [cp:2 ~] ~ cp:2, [oh ~ hh ~]*2")
       .bank("RolandTR808").cut(1)
       .gain(perlin.range(.3,.6))
       .pan(sine.slow(4))
       .room(.4).crush(12).lpf(3200)
       .every(3, x => x.press())],
  [16, s("hh:3*16, ~ [cp:2,sd:1] ~ [cp:2,sd:1], [oh hh oh hh]*2")
       .bank("RolandTR808").cut(1)
       .gain(perlin.range(.5,.85))
       .room(.3).lpf(4500)
       .every(4, x => x.fast(2))
       .every(7, x => x.rev())],
  [16, s("hh:3*16, ~ [cp:2,sd:1] ~ [cp:2,sd:1], [oh hh oh hh]*2")
       .bank("RolandTR808").cut(1)
       .gain(perlin.range(.5,.9))
       .room(.3).lpf(5000).distort(.15)
       .every(3, x => x.fast(2))
       .every(5, x => x.rev())],
  [8,  s("hh:3*4, ~ cp:2 ~ ~")
       .bank("RolandTR808")
       .gain(saw.slow(8).range(.5,.02))
       .room(.7).crush(6).lpf(saw.slow(8).range(3000,200))]
)

// engine kick — heartbeat of the machine
$: arrange(
  [8,  s("bd ~ ~ ~")
       .bank("RolandTR909")
       .gain(saw.slow(8).range(.05,.4))
       .room(.6).lpf(400)],
  [16, s("bd ~ bd ~")
       .bank("RolandTR909")
       .gain(".65 .55")
       .room(.4).lpf(800).distort(.05)],
  [16, s("bd(3,8)")
       .bank("RolandTR909")
       .gain(perlin.range(.6,.85))
       .room(.35).lpf(1500).distort(.1)],
  [16, s("bd(5,8)")
       .bank("RolandTR909")
       .gain(".9 .75 .85 .75 .8")
       .room(.3).lpf(3000).distort(.2)
       .every(4, x => x.fast(2).gain(.5))],
  [16, s("bd(7,16)")
       .bank("RolandTR909")
       .gain(perlin.range(.7,.95))
       .room(.25).distort(.25)
       .every(3, x => x.fast(2))],
  [8,  s("bd ~ ~ ~")
       .bank("RolandTR909")
       .gain(saw.slow(8).range(.7,.0))
       .room(.6)]
)

  [16, note("<[c1 ~ c1 ~] [eb1 ~ ~ eb1] [f1 ~ c1 ~] [g1 ~ ~ ~]>")

  [16, note("<[c1 ~ c1 c1] [eb1 ~ eb1 ~] [f1 ~ c1 ~] [g1 ~ g1 ~]>")

       .lpf(sine.range(600,2000).slow(6))

// circuit sparks — arpeggiated synth waking up
$: arrange(
  [8,  silence],
  [16, n("<[~ 0 ~ ~] [~ ~ 2 ~] [~ ~ ~ 3] [~ 0 ~ ~]>")
       .scale("C3:minor")
       .s("square")
       .lpf(sine.range(400,1800).slow(12))
       .decay(.15).sustain(.2).release(.1)
       .gain(.22).room(.6).delay(".3:.2:.3")
       .crush(10)],
  [16, n("<[0 ~ 2 ~] [3 ~ 5 ~] [~ 7 5 ~] [3 ~ 2 0]>")
       .scale("C4:minor")
       .s("square")
       .lpf(sine.range(800,3500).slow(8))
       .decay(.1).sustain(.3).release(.15)
       .gain(.35).room(.5).delay(".25:.15:.35")
       .crush(12)
       .rarely(x => x.add(note(12)))],
  [16, n("<[0 2 3 5] [~ 7 5 3] [2 3 5 7] [5 3 2 0]>")
       .scale("C4:minor")
       .s("square")
       .lpf(sine.range(1500,5000).slow(6))
       .decay(.08).sustain(.35).release(.2)
       .gain(.44).room(.4).delay(".2:.1:.3")
       .sometimes(x => x.add(note(12)))
       .every(4, x => x.fast(2).gain(.25))],
  [16, n("<[0 2 3 5] [7 5 3 2] [3 5 7 10] [7 5 3 0]>")
       .scale("C5:minor")
       .s("square")
       .lpf(sine.range(2000,6000).slow(5))
       .decay(.06).sustain(.4).release(.2)
       .gain(.5).room(.35).delay(".15:.1:.25")
       .sometimes(x => x.add(note(12)))
       .every(3, x => x.fast(2))],
  [8,  n("<[~ 0 ~ ~] [~ ~ ~ 3] [~ ~ 0 ~] [~ ~ ~ ~]>")
       .scale("C4:minor")
       .s("square")
       .lpf(saw.slow(8).range(2500,300))
       .gain(saw.slow(8).range(.35,.0))
       .room(.8).delay(".5:.4:.6")
       .crush(8)]
)
