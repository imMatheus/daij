// "Four Winds" — Ethereal ambient journey through spring, summer, autumn, winter
setcps(0.5)

const chorus = stack(
  // seasonal melody — arpeggios shift character each cycle
  n("<[0 2 4 7] [0 2 4 6] [7 5 3 0] [0 1 3 5]>*2")
    .scale("<D4:major G4:mixolydian E4:dorian A4:minor>")
    .s("sawtooth").lpf(perlin.range(600,3500).slow(4)).lpq(3)
    .attack(.01).decay(.15).sustain(.4).release(.3)
    .gain(.28).room(.3)
    .off(1/8, x => x.add(2).gain(.14).delay("0.3:0.125:0.4")),
  // bass follows seasonal roots
  note("<d2 g2 e2 a2>").s("sawtooth").lpf(280).gain(.42)
    .decay(.2).sustain(.6).release(.1),
  // drums — density shifts with seasons
  s("<[bd [~ bd] sd ~] [bd bd sd [bd sd]] [bd ~ sd ~] [bd ~ ~ sd]>, [hh hh:1]*4")
    .bank("RolandTR909").gain(.42).lpf(perlin.range(2000,8000)),
  // shimmer pad — chords drift through seasons
  note("<[d4,f#4,a4] [g4,b4,d5] [e4,g4,b4] [a4,c5,e5]>")
    .s("sine").gain(.12).attack(.3).release(.8).room(.5)
    .delay("0.35:0.18:0.55")
)

arrange(
  // INTRO — spring awakening, gentle sine melody fading in
  [2, stack(
    note("[d5 a4 f#5 d5]*2").s("sine")
      .gain(saw.slow(2).range(.05,.22)).room(.7)
      .delay("0.5:0.2:0.6").attack(.05).release(.5).lpf(3500),
    note("[a5 d6 f#6 a5]").s("sine").gain(.04).room(.6)
      .delay("0.6:0.3:0.65").attack(.15).release(1).lpf(5000)
  )],
  // CHORUS — full four-season cycle
  [14, chorus],
  // BRIDGE — autumn decay, descending through falling leaves
  [5, stack(
    n("[7 6 5 4 3 2 1 0]").scale("E4:minor")
      .s("triangle").lpf(sine.range(300,1400).slow(2.5)).lpq(2)
      .attack(.08).decay(.4).sustain(.2).release(.6)
      .gain(.22).room(.65).delay("0.5:0.22:0.6")
      .jux(rev),
    note("<e2 [~ d2] c2 [~ b1]>").s("sawtooth").lpf(200).gain(.28),
    s("[~ bd:3] ~ [sd:1 ~] ~, hh*3").bank("RolandTR909")
      .gain(.18).lpf(1200).room(.35)
  )],
  // CHORUS RETURN — seasons cycle again
  [14, chorus]
)
