// "Ritual Grounds" — Afro-Cuban polyrhythmic percussion jam
samples('github:tidalcycles/dirt-samples')
setcps(21/60)

const warmth = perlin.slow(8).range(700,2800)
const bright = sine.slow(10).range(5500,13000)
const space = sine.slow(6).range(.18,.42)

const intro =
  s("rim ~ ~ rim ~ ~ ~ ~")
    .gain(".06 ~ ~ .04")
    .hpf(3200)
    .room(".38:.86")
    .delay(".18:.375:.28")
    .orbit(2)
    .stack(
      s("~ ~ ~ ~ ~ ~ bd ~")
        .lpf(700).gain(.25).room(.1)
    )

const chorus =
  s("bd ~ [~ bd] ~ bd ~ ~ [bd ~] ~ [~ bd] ~ bd ~ ~ bd ~")
    .lpf(1050).gain(.9).room(.06)
    .stack(
      s("sd:3").struct("~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~ x ~ ~ ~")
        .lpf(warmth).room(.18).gain(.32).orbit(3)
    )
    .stack(
      s("hh*16")
        .gain(".08 .04 .07 .04 .09 .05 .07 .04 .08 .04 .07 .05 .09 .05 .07 .04")
        .end(".04 .06 .05 .04")
        .hpf(bright).room(.12)
        .delay(".05:.125:.1")
        .pan(perlin.slow(12).range(.28,.72))
        .orbit(2).reset("<x ~ x x>")
    )
    .stack(
      s("~ oh ~ ~ ~ ~ oh ~ ~ ~ oh ~ ~ oh ~ ~")
        .end(".12 .16 .14")
        .hpf(6500).room(.32)
        .delay(".14:.25:.2")
        .gain(".1 .08 .12").orbit(2)
    )
    .stack(
      s("lt mt ht").euclidRot(5,16,2)
        .speed(".93 1.02 1.07")
        .lpf(warmth).room(space)
        .delay(".18:.25:.28")
        .gain(".2 .16 .14").orbit(4)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("rim").euclidRot(7,16,3)
        .lpf(2900).room(.16).gain(.13).orbit(4)
    )
    .stack(
      s("cp").struct("~ ~ ~ ~ ~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~")
        .room(.28).lpf(3400).gain(.22)
        .delay(".12:.25:.18").orbit(3)
    )
    .stack(
      s("bd").struct("x ~ [~ x] ~ x ~ ~ [x ~] ~ [~ x] ~ x ~ ~ x ~")
        .duckorbit("2:3:4")
        .duckattack("0.12:0.15:0.18")
        .duckdepth(".48:.5:.65").postgain(0)
    )

const bridge =
  s("bd").struct("x ~ ~ ~ ~ ~ ~ x ~ ~ ~ ~ x ~ ~ ~")
    .lpf(800).gain(.68).room(.1)
    .stack(
      s("lt mt ht lt mt").euclidRot(5,8,1).slow(2)
        .speed(".9 .97 1.05 .94 1.1")
        .lpf(sine.slow(4).range(380,1400))
        .room(".52:.9").delay(".26:.375:.4")
        .gain(".22 .18 .14 .2 .16").orbit(3)
    )
    .stack(
      s("[~ hh]*6")
        .gain(".02 .04 .03 .05 .02 .04 .02 .05 .03 .04 .02 .06")
        .end(".04 .05").hpf(9500)
        .room(.42).delay(".1:.125:.16").orbit(2)
    )
    .stack(
      s("~ rim ~ ~ rim ~ ~ ~").slow(2)
        .hpf(2600).room(.5)
        .delay(".28:.25:.36").gain(.1).orbit(4)
    )
    .stack(
      s("~ ~ oh ~ ~ ~ ~ ~").slow(2)
        .end(.18).hpf(7500)
        .room(".48:.9").delay(".2:.375:.3")
        .gain(.08).orbit(2)
    )
    .stack(
      s("bd").struct("x ~ ~ ~ ~ ~ ~ x ~ ~ ~ ~ x ~ ~ ~")
        .duckorbit("2:3:4")
        .duckattack("0.18:0.2:0.24")
        .duckdepth(".32:.42:.58").postgain(0)
    )

arrange([1, intro], [10, chorus], [3, bridge], [10, chorus]).postgain(.95)
