// 🎵 "Hollow Bones" — Polyrhythmic Percussion Ritual
// At 116 BPM: intro(8) + build(16) + main(16) + breakdown(8) + peak(16) + peak2(16) + outro(8) = 88 bars ≈ 3:02

setcpm(116 / 4)

// Kick — the backbone
$: arrange(
  [8,  s("bd(3,8)").bank("RolandTR808").gain(saw.slow(8).range(.3,.88)).room(.25).lpf(sine.range(200,800).slow(8))],
  [16, s("bd*4").bank("RolandTR808").gain(".92 .85 .9 .85").room(.15)],
  [16, s("bd [~ bd] ~ bd, ~ ~ ~ [bd ~]").bank("RolandTR808").gain(.95).room(.12).distort(.04)],
  [8,  s("bd ~ ~ ~, ~ ~ bd ~").bank("RolandTR808").gain(saw.slow(8).range(.85,.4)).room(.4).delay(".3:.2:.4")],
  [16, s("bd [~ bd] ~ bd, ~ ~ ~ [bd ~]").bank("RolandTR808").gain(.95).room(.12)],
  [16, s("bd*4, ~ [bd ~] ~ bd").bank("RolandTR808").gain(.97).room(.1).distort(.06).every(4, x => x.fast(2).gain(.5))],
  [8,  s("bd(3,8)").bank("RolandTR808").gain(saw.slow(8).range(.8,.0)).room(.35)]
)

// Hi-hats — texture and drive
$: arrange(
  [8,  s("hh*8").bank("RolandTR909").gain(perlin.range(.08,.25)).pan(sine.slow(5)).cut(1)],
  [16, s("hh*16").bank("RolandTR909").gain(perlin.range(.15,.42)).pan(sine.slow(4)).cut(1).lpf(sine.range(4000,9000).slow(6))],
  [16, s("hh*16, ~ ~ ~ oh").bank("RolandTR909").gain(perlin.range(.18,.48)).pan(sine.slow(6)).cut(1).every(3, x => x.rev())],
  [8,  s("[hh oh]*2").bank("RolandTR909").gain(perlin.range(.1,.28)).pan(sine.slow(3)).cut(1).room(.5).delay(".35:.25:.5")],
  [16, s("hh*16, ~ ~ ~ oh").bank("RolandTR909").gain(perlin.range(.2,.5)).pan(sine.slow(5)).cut(1)],
  [16, s("hh*16, [~ oh]*2").bank("RolandTR909").gain(perlin.range(.22,.52)).pan(sine.slow(4)).cut(1).every(4, x => x.fast(2).gain(.25))],
  [8,  s("hh*8").bank("RolandTR909").gain(saw.slow(8).range(.3,.0)).pan(sine.slow(3)).cut(1)]
)

// Clap/snare — backbeat and ghost notes
$: arrange(
  [8,  silence],
  [16, s("~ cp ~ ~").bank("RolandTR909").gain(.75).room(.35).delay(".12:.1:.15").swing(2)],
  [16, s("~ cp ~ ~, ~ ~ ~ [~ cp]").bank("RolandTR909").gain(.8).room(.3).delay(".1:.08:.12")],
  [8,  s("~ ~ cp ~").bank("RolandTR909").gain(.55).room(.6).delay(".4:.3:.55")],
  [16, s("~ cp ~ ~, ~ ~ ~ [~ cp]").bank("RolandTR909").gain(.8).room(.3).delay(".1:.08:.12")],
  [16, s("~ cp ~ [~ cp], [~ sd] ~ cp ~").bank("RolandTR909").gain(.82).room(.25).every(4, x => x.fast(2).gain(.4))],
  [8,  s("~ cp ~ ~").bank("RolandTR909").gain(saw.slow(8).range(.65,.0)).room(.5)]
)

// Rimshot — polyrhythmic layer
$: arrange(
  [8,  s("rim(3,8)").bank("RolandTR808").gain(.3).room(.45).pan(sine.slow(7))],
  [16, s("rim(5,8)").bank("RolandTR808").gain(.38).room(.35).pan(sine.slow(5)).delay(".15:.1:.2")],
  [16, s("rim(5,8,1)").bank("RolandTR808").gain(.42).room(.3).pan(sine.slow(4))],
  [8,  s("rim(3,8)").bank("RolandTR808").gain(.28).room(.6).delay(".45:.35:.55").pan(sine.slow(6))],
  [16, s("rim(7,16)").bank("RolandTR808").gain(.4).room(.3).pan(sine.slow(5))],
  [16, s("rim(7,16,2)").bank("RolandTR808").gain(.44).room(.25).pan(sine.slow(3)).every(5, x => x.rev())],
  [8,  silence]
)

// Toms — melodic percussion
$: arrange(
  [8,  silence],
  [16, s("<lt ~ ~ mt> <~ ~ ht ~>").bank("RolandTR808").gain(.5).room(.3).lpf(sine.range(800,3000).slow(8)).pan(sine.slow(6))],
  [16, s("<lt mt ~ ht> <~ lt ~ mt>, ~ ~ [lt ht] ~").bank("RolandTR808").gain(.55).room(.25).lpf(sine.range(1000,3500).slow(6)).pan(sine.slow(5))],
  [8,  s("lt ~ ~ ~, ~ ~ ~ mt").bank("RolandTR808").gain(.4).room(.55).delay(".35:.3:.5")],
  [16, s("<lt mt ~ ht> <~ lt ~ mt>, ~ ~ [lt ht] ~").bank("RolandTR808").gain(.55).room(.25).lpf(sine.range(1000,3500).slow(6))],
  [16, s("<[lt mt] ~ ht ~> <~ [lt lt] ~ [mt ht]>").bank("RolandTR808").gain(.58).room(.2).lpf(sine.range(1200,4000).slow(5)).every(3, x => x.fast(2).gain(.3))],
  [8,  s("<lt ~ ~ ~>/1").bank("RolandTR808").gain(saw.slow(8).range(.45,.0)).room(.5)]
)

// Cowbell + claves — high-end polyrhythm
$: arrange(
  [8,  silence],
  [16, silence],
  [16, s("cb(3,8)").bank("RolandTR808").gain(.28).room(.4).pan(sine.slow(8)).delay(".2:.15:.25").lpf(sine.range(3000,7000).slow(10))],
  [8,  s("cb ~ ~ ~").bank("RolandTR808").gain(.2).room(.65).delay(".5:.4:.6")],
  [16, s("cb(3,8), cl(5,16)").bank("RolandTR808").gain(.3).room(.35).pan(sine.slow(6)).delay(".15:.1:.2")],
  [16, s("cb(5,8), cl(7,16)").bank("RolandTR808").gain(.34).room(.3).pan(sine.slow(5)).every(4, x => x.rev()).sometimes(x => x.speed(1.5).gain(.18))],
  [8,  s("cb(2,8)").bank("RolandTR808").gain(saw.slow(8).range(.25,.0)).room(.5).delay(".4:.3:.5")]
)

// Shaker/maracas — constant texture
$: arrange(
  [8,  s("ma*8").bank("RolandTR808").gain(perlin.range(.04,.15)).pan(sine.slow(3))],
  [16, s("ma*16").bank("RolandTR808").gain(perlin.range(.08,.22)).pan(sine.slow(4)).lpf(sine.range(5000,12000).slow(8))],
  [16, s("ma*16").bank("RolandTR808").gain(perlin.range(.1,.25)).pan(sine.slow(5)).lpf(sine.range(6000,14000).slow(6))],
  [8,  s("ma*8").bank("RolandTR808").gain(perlin.range(.05,.15)).pan(sine.slow(3)).room(.5)],
  [16, s("ma*16").bank("RolandTR808").gain(perlin.range(.1,.25)).pan(sine.slow(5)).lpf(sine.range(6000,14000).slow(6))],
  [16, s("ma*16").bank("RolandTR808").gain(perlin.range(.12,.28)).pan(sine.slow(4)).lpf(sine.range(7000,15000).slow(5))],
  [8,  s("ma*8").bank("RolandTR808").gain(saw.slow(8).range(.15,.0)).pan(sine.slow(3))]
)
