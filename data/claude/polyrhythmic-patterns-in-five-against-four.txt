// "Five Over Four" — Polyrhythmic minimalist groove
samples('github:tidalcycles/dirt-samples')
setcps(35/70)

const sweep = sine.slow(9).range(220, 580)
const glow = perlin.slow(6).range(900, 3800)
const shimmer = sine.slow(11).range(5800, 11000)

const intro = stack(
  note("d1").s("sine").gain(.35).lpf(90)
    .attack(.1).release(1.5),
  s("hh*5").gain(".06 .04 .05 .04 .08")
    .hpf(shimmer).pan(.75).room(.35),
  n("0 2 4 7 5").scale("D5:minor").s("triangle")
    .gain(".06 .08 .07 .09 .07")
    .decay(.35).sustain(0).lpf(1400)
    .delay(".3:.2:.5").room(".4:.9")
)

const chorus = stack(
  s("bd*4").gain(".88 .58 .78 .58").lpf(200)
    .every(8, x => x.ply(2))
    .duckorbit("2:3").duckattack("0.22:0.16").duckdepth(".85:.4"),
  note("<[d2 ~ f2 ~] [d2 ~ a1 ~] [d2 ~ g2 ~] [d2 ~ e2 ~]>")
    .s("sawtooth").lpf(sweep).lpenv(3).lpd(.15)
    .decay(.12).sustain(0).gain(.55).orbit(2),
  s("hh*4").gain(".28 .16 .24 .16").hpf(6500).pan(.2),
  s("~ cp ~ cp cp").gain(".4 .3 .45 .3 .5")
    .speed(1.2).room(.14),
  n("<[0 2 4 7 5] [0 3 5 7 9] [0 2 4 7 5] [-2 0 2 5 4]>")
    .scale("D4:minor").s("triangle").gain(.32)
    .lpf(glow).decay(.2).sustain(0)
    .lastOf(4, rev).echo(2, 1/5, .45)
    .room(.18).pan(sine.slow(5).range(.3, .7))
    .orbit(3),
  s("hh*5").gain(".2 .12 .18 .12 .22")
    .hpf(shimmer).pan(.8).delay(".15:.2:.25"),
  chord("<Dm Am Gm Dm>").voicing()
    .s("sawtooth").gain(.12)
    .lpf(sine.slow(12).range(400, 1200))
    .attack(.5).release(.8)
    .room(".32:.88").delay(.08).slow(2),
  note("d1").s("sine").gain(.42).lpf(95)
)

const bridge = stack(
  s("hh*5").gain(".18 .12 .16 .12 .2")
    .hpf(shimmer).pan(.75)
    .delay(".25:.2:.4").room(.4),
  n("<[0 2 4 7 5] [-2 0 2 5 4]>")
    .scale("D4:minor").s("triangle").gain(.18)
    .lpf(sine.slow(5).range(600, 1800))
    .decay(.28).sustain(0)
    .echo(3, 1/5, .38)
    .room(".5:.95").delay(".3:.2:.45"),
  chord("<Dm Gm>").voicing()
    .s("sawtooth").gain(.09)
    .lpf(sine.slow(6).range(300, 900))
    .attack(.7).release(1.2)
    .room(".55:.96").slow(2),
  note("d1").s("sine").gain(.38).lpf(85)
    .attack(.05).release(1.2)
)

arrange(
  [2, intro],
  [13, chorus],
  [5, bridge],
  [15, chorus]
).postgain(.92)
