// "Last Light" — Bittersweet ambient ballad about endings
setcps(20/70)
arrange(
  // INTRO — lone chord fading in
  [1, note("[f3,ab3,c4]")
    .s("supersaw").detune(.15).att(.8).rel(1.5).lpf(600).gain(.22)
    .room(.85).sz(2).delay(".3:.2:.6")],
  // CHORUS — full bittersweet arrangement
  [8, note("<[f3,ab3,c4] [db3,f3,ab3] [ab3,c4,eb4] [eb3,g3,bb3]>")
    .s("supersaw").detune(.2).att(.15).rel(.7)
    .lpf(perlin.range(400,1600).slow(3)).gain(.28).room(.6).delay(".2:.18:.5")
    .stack(
      note("<[c5 ~ ab4 f4 ~ ab4 c5 ~] [db5 ~ c5 ab4 ~ ~ f4 ~] [eb5 ~ c5 ~ ab4 ~ c5 eb5] [bb4 ~ g4 ~ eb4 ~ ~ ~]>")
        .s("triangle").att(.03).rel(.35).gain(.22).delay(".3:.12:.4").room(.4).lpf(3000),
      note("<f2 db2 ab2 eb2>").s("sawtooth").lpf(200).gain(.32).rel(.25),
      s("~ hh ~ hh ~ hh ~ hh").gain(".18 .09").lpf(4500).room(.15)
    )],
  // BRIDGE — stripped back, C major for fleeting hope
  [3, note("<[bb3,db4,f4] [c3,e3,g3] [f3,ab3,c4]>")
    .s("triangle").att(.5).rel(1.2).gain(.18).room(.9).sz(3).delay(".4:.2:.6")
    .stack(
      note("<[bb4 ~ db5 ~] [c5 ~ e5 ~] [f4 ~ ab4 ~]>")
        .s("sine").att(.1).rel(.7).gain(.13).delay(".45:.1:.55").room(.6)
    )],
  // FINAL CHORUS — same but slowly dissolving
  [8, note("<[f3,ab3,c4] [db3,f3,ab3] [ab3,c4,eb4] [eb3,g3,bb3]>")
    .s("supersaw").detune(.25).att(.2).rel(.9)
    .lpf(perlin.range(300,1400).slow(4)).gain(.26).room(.7).delay(".25:.2:.55")
    .stack(
      note("<[c5 ~ ab4 f4 ~ ab4 c5 ~] [db5 ~ c5 ab4 ~ ~ f4 ~] [eb5 ~ c5 ~ ab4 ~ c5 eb5] [bb4 ~ g4 ~ eb4 ~ ~ ~]>")
        .s("triangle").att(.05).rel(.5).gain(.2).delay(".35:.15:.5").room(.5).lpf(2500),
      note("<f2 db2 ab2 eb2>").s("sawtooth").lpf(180).gain(.3).rel(.3),
      s("~ hh ~ hh ~ hh ~ hh").gain(".15 .08").lpf(3500).room(.2)
    )]
)
