// "Solar Rise" — Uplifting energetic electronic
setcps(0.55)

stack(
  // Intro riser
  s("hh*16").bank("RolandTR909")
    .lpf(saw.slow(2).range(400, 13000))
    .gain(saw.slow(2).range(0, .55))
    .mask("<1!2 0!38>"),
  // Kick — four on the floor
  s("bd*4").bank("RolandTR909")
    .gain("<0 0 [.95 .65 .85 .65]!16 [.5 ~ .5 ~]!6 [.95 .65 .85 .65]!16>"),
  // Clap + hats + open hat
  s("~ cp ~ <cp [cp cp:1]>, hh(7,8), ~ ~ ~ oh")
    .bank("RolandTR909")
    .gain(".8, .4 .7 .5 .85 .4 .7 .5 1, .3")
    .swingBy(1/8, 4)
    .mask("<0!2 1!16 0!6 1!16>"),
  // Bridge minimal perc
  s("~ cp ~ ~, hh*4").bank("RolandTR909")
    .gain(".35, .25").mask("<0!18 1!6 0!16>"),
  // Sub bass
  note("<d1!2 b0!2 g1!2 a1!2>").s("sine")
    .struct("x(3,8)").gain(.7).lpf(130)
    .mask("<0!2 1!38>"),
  // Synth bass
  note("<d2!2 b1!2 g1!2 a1!2>").s("sawtooth")
    .struct("x(5,8)")
    .lpf(sine.range(250, 1100).slow(4)).lpq(3)
    .decay(.06).sustain(.5).release(.04)
    .gain("<0!2 .5!16 .3!6 .55!16>"),
  // Chord stabs
  chord("<D Bm G A>").voicing().s("supersaw")
    .lpf(perlin.range(900, 5500).slow(6))
    .attack(.005).decay(.2).sustain(.45).release(.3)
    .room(.2).delay(.12).delaytime(3/8).delayfeedback(.25)
    .gain("<0!2 .35!16 0!6 .38!16>"),
  // Bridge pad
  chord("<Bm G D A>").voicing().s("triangle")
    .attack(.4).sustain(.5).release(.8)
    .lpf(sine.range(600, 2500).slow(3))
    .room(.6).delay(.3).delaytime(1/3).delayfeedback(.4)
    .gain("<0!18 .4!6 0!16>"),
  // Lead arp
  n("<[0 2 4 7]*2 [7 4 2 0]*2 [0 3 5 7]*2 [2 4 7 9]*2>")
    .scale("D4:major").s("square")
    .decay(.1).sustain(.25).release(.1)
    .lpf(sine.range(1500, 7000).slow(8))
    .delay(.3).delaytime(3/16).delayfeedback(.4)
    .room(.1).gain("<0!2 .4!16 0!6 .45!16>"),
  // High sparkle
  n("{0 2 4 7 4 2}%6").scale("D5:major").s("sine")
    .decay(.03).sustain(0)
    .delay(.5).delaytime(1/6).delayfeedback(.5)
    .pan(sine.range(.2, .8).slow(3))
    .gain("<0!2 .1!16 .05!6 .12!16>")
)
