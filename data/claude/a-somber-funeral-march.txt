// ⚰️ "Procession of Ash" — Somber Funeral March
// At 56 BPM: intro(4) + march(6) + lament(4) + outro(3) = 17 bars ≈ 73s

setcpm(56 / 4)

// Sub drone — the weight of earth
$: note("c1")
   .s("sine")
   .attack(2).decay(1).sustain(.9).release(3)
   .gain(.2).room(.5).lpf(100)

// Tolling bell
$: arrange(
  [4,  note("c3").s("gm_tubular_bells")
       .struct("x ~ ~ ~")
       .gain(.2).room(.9).delay(".5:.3:.6")],
  [6,  note("c3").s("gm_tubular_bells")
       .struct("x ~ ~ ~")
       .gain(.25).room(.85).delay(".4:.25:.5")],
  [4,  note("c3").s("gm_tubular_bells")
       .struct("x ~ ~ ~")
       .gain(.2).room(.9).delay(".5:.3:.6")],
  [3,  note("c3").s("gm_tubular_bells")
       .struct("x ~ ~ ~")
       .gain(saw.slow(3).range(.22,.03)).room(.95).delay(".6:.4:.7")]
)

// Funeral drums — slow, deliberate
$: arrange(
  [4,  s("bd ~ ~ ~")
       .bank("RolandTR808").gain(.5).room(.7).decay(.9)],
  [6,  s("bd ~ [~ bd:2] ~, ~ ~ ~ [~ cp:3]")
       .bank("RolandTR808").gain(".65 .45 .55 .35").room(.6).decay(.5)],
  [4,  s("bd ~ ~ ~, ~ ~ [~ cp:3] ~")
       .bank("RolandTR808").gain(perlin.range(.35,.6)).room(.75).decay(.7)],
  [3,  s("bd ~ ~ ~")
       .bank("RolandTR808").gain(saw.slow(3).range(.5,.05)).room(.85).decay(.9)]
)

// Bass — mournful roots
$: arrange(
  [4,  note("<c2 ~ ~ g1>/2")
       .s("gm_acoustic_bass").decay(.5).sustain(.4).room(.3).gain(.5)],
  [6,  note("<[c2 ~ ~ g1] [ab1 ~ ~ eb2] [f1 ~ ~ c2] [g1 ~ ~ g1] [ab1 ~ ~ eb2] [bb1 ~ ~ f1]>")
       .s("gm_acoustic_bass").decay(.4).sustain(.5).room(.25).gain(.62)],
  [4,  note("<[c2 ~ ~ eb2] [ab1 ~ ~ bb1] [f1 ~ ~ g1] [ab1 ~ ~ g1]>")
       .s("gm_acoustic_bass").decay(.45).sustain(.45).room(.3).gain(.55)],
  [3,  note("<c2 ~ ~ ~>/2")
       .s("gm_acoustic_bass").decay(.6).sustain(.3).room(.4).gain(saw.slow(3).range(.5,.08))]
)

// Church organ — dark sustained harmony
$: arrange(
  [4,  chord("<Cm Gm>/2").dict('ireal').voicing()
       .s("gm_church_organ")
       .attack(1).sustain(.7).release(2)
       .lpf(800).room(.8).gain(.25)],
  [6,  chord("<Cm Ab Fm G7>").dict('ireal').voicing()
       .s("gm_church_organ")
       .attack(.6).sustain(.75).release(1.5)
       .lpf(sine.range(600,1400).slow(12))
       .room(.7).gain(.35)],
  [4,  chord("<Cm Ab Fm Gm>").dict('ireal').voicing()
       .s("gm_church_organ")
       .attack(.8).sustain(.7).release(2)
       .lpf(sine.range(500,1100).slow(8))
       .room(.75).gain(.3).delay(".3:.2:.4")],
  [3,  chord("<Cm>/1").dict('ireal').voicing()
       .s("gm_church_organ")
       .attack(1.5).sustain(.6).release(3)
       .lpf(sine.range(300,700).slow(6))
       .room(.9).gain(saw.slow(3).range(.3,.05))]
)

// Oboe melody — descending, mournful phrases
$: arrange(
  [4,  silence],
  [6,  n("<[~ 4 ~ 3] [2 ~ 1 ~] [~ 3 ~ 2] [0 ~ ~ ~] [~ 4 ~ 3] [1 ~ 0 ~]>")
       .scale("C4:minor")
       .s("gm_oboe")
       .room(.65).delay(".35:.25:.45")
       .gain(.3).lpf(2800)
       .attack(.08).decay(.3).sustain(.6).release(.5)
       .vib(5).vibmod(.4)],
  [4,  n("<[~ 4 ~ ~] [~ ~ 2 ~] [~ 0 ~ ~] [~ ~ ~ ~]>")
       .scale("C4:minor")
       .s("gm_oboe")
       .room(.8).delay(".5:.35:.6")
       .gain(.25).lpf(2200)
       .attack(.1).decay(.35).sustain(.5).release(.7)
       .vib(4).vibmod(.5)],
  [3,  silence]
)

// Strings — distant grief
$: arrange(
  [4,  chord("<Cm>/1").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(1.5).sustain(.5).release(2.5)
       .lpf(600).room(.85).gain(.18)],
  [6,  chord("<Cm Ab Fm G7>").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(1).sustain(.6).release(2)
       .lpf(sine.range(400,1200).slow(12))
       .room(.75).gain(.25)],
  [4,  chord("<Cm Ab Fm Gm>").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(1.2).sustain(.55).release(2.5)
       .lpf(sine.range(350,900).slow(8))
       .room(.8).gain(.22)],
  [3,  chord("<Cm>/1").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(2).sustain(.4).release(3)
       .lpf(sine.range(200,600).slow(6))
       .room(.9).gain(saw.slow(3).range(.2,.03))]
)
