// "Requiem in Grey" — Somber funeral march in C minor
setcps(0.2)

const bell = note("c5 ~@3").s("sine")
  .fm(1.5).fmh(4.01).fmdecay(1.2).fmsustain(0)
  .decay(3).sustain(0).release(2).gain(.25)
  .room(.9).size(4).delay(".3:.15:.5")

const march = s("[bd@3 bd] ~ sd [~@3 bd]")
  .gain("[.8@3 .5] ~ .55 [~@3 .35]")
  .lpf(400).room(.4).size(2)

const bass = note("<[c2@3 c2] [c2 ~ c2 ~] [ab1@3 ~] [g1@2 g1 ~]>")
  .s("sawtooth").lpf(180).lpq(2)
  .att(.01).dec(.4).sus(.5).rel(.7)
  .gain(.6).room(.25)

const melody = note("<[c4@3 eb4] [d4 ~@2 c4] [bb3@2 ab3 ~] [g3@4]>")
  .s("triangle").lpf(perlin.range(400,1000).slow(8))
  .att(.06).dec(.5).sus(.3).rel(1.2)
  .gain(.35).room(.5).delay(".15:.08:.4")

const bridgeMel = note("<[eb4 ~ f4 ~] [g4@2 ~ f4] [eb4@2 d4 ~] [c4@4]>")
  .s("triangle").lpf(800)
  .att(.08).dec(.6).sus(.25).rel(1.5)
  .gain(.3).room(.6).delay(".2:.1:.5")

const pad = note("<[c3,eb3,g3] [f3,ab3,c4] [bb2,d3,f3] [g2,bb2,d3]>")
  .s("sawtooth").lpf(sine.range(400,700).slow(12))
  .att(1).dec(1.5).sus(.3).rel(2)
  .gain(.18).room(.7).size(3)

arrange(
  [1, stack(bell, pad.gain(.08))],
  [6, stack(bell, march, bass, melody, pad)],
  [2, stack(bell, bass.lpf(140).gain(.4), bridgeMel, pad.gain(.22))],
  [5, stack(bell, march, bass, melody, pad)]
)
