// "Midnight Confessions" — Drake-style introspective rap with melodic hooks
// At 76 BPM, 24 cycles ~ 75 seconds
// intro(2) + verse(6) + hook(6) + verse2(6) + hook2(4) = 24 cycles
// Key: Bb minor — moody pads, 808s, melodic piano hooks

setcpm(76 / 4)

// -- ATMOSPHERE — dark ambient pad, always present
$: note("<bb2,db3,f3>/1").s("sawtooth")
   .attack(2).decay(1.5).sustain(.4).release(3)
   .lpf(sine.range(300, 1100).slow(16))
   .room(.7).gain(.11).pan(sine.slow(8))

// -- DRUMS — 808 trap-influenced, soft intro then full
$: arrange(
  [2,  s("bd ~ ~ ~").bank("RolandTR808").gain(.55).room(.35)],
  [6,  s("bd ~ ~ bd, ~ [cp,sd:1] ~ ~, [~ hh]*4")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(".88 .78 .28").room(.25)],
  [6,  s("bd ~ bd ~, ~ [cp,sd:1] ~ [~ cp], [hh oh hh hh]*2")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(".94 .84 .36").room(.2)
       .every(4, x => x.fast(2).gain(.3))],
  [6,  s("bd ~ ~ bd, ~ [cp,sd:1] ~ ~, [~ hh]*4")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(".85 .76 .26").room(.25)],
  [4,  s("bd ~ bd ~, ~ [cp,sd:1] ~ [~ cp], [hh oh hh hh]*2, ~ ~ ~ [bd ~]")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(".96 .86 .38 .6").room(.18)
       .every(3, x => x.fast(2).gain(.28))]
)

// -- 808 BASS — sub-heavy, enters verse
$: arrange(
  [2,  silence],
  [6,  note("<[bb1 ~ ~ bb1] [gb1 ~ ~ ~] [db2 ~ ~ db2] [ab1 ~ ~ ~]>")
       .s("sine").lpf(140)
       .decay(.2).sustain(.6).release(.08)
       .gain(.32).room(.1)],
  [6,  note("<[bb1 ~ bb1 ~] [gb1 ~ ~ gb1] [db2 ~ db2 ~] [ab1 ~ ~ ab1]>")
       .s("sine").lpf(150)
       .decay(.18).sustain(.65).release(.08)
       .gain(.36).room(.08)],
  [6,  note("<[bb1 ~ ~ bb1] [gb1 ~ ~ ~] [db2 ~ ~ db2] [ab1 ~ ~ ~]>")
       .s("sine").lpf(140)
       .decay(.2).sustain(.6).release(.08)
       .gain(.32).room(.1)],
  [4,  note("<[bb1 ~ bb1 ~] [gb1 ~ ~ gb1] [db2 ~ db2 ~] [ab1 ~ ~ ab1]>")
       .s("sine").lpf(155)
       .decay(.15).sustain(.68).release(.08)
       .gain(.38).room(.08)]
)

// -- KEYS — moody piano stabs, Drake-style Rhodes
$: arrange(
  [2,  chord("<Bbm9>/1").dict('ireal').voicing()
       .s("gm_epiano1").struct("x ~ ~ ~")
       .lpf(900).room(.65).delay(".35:.25:.45")
       .gain(.25).attack(.15)],
  [6,  chord("<Bbm7 Gbmaj7 Db Ab>").dict('ireal').voicing()
       .s("gm_epiano1").struct("~ [x ~] ~ [~ x]")
       .lpf(1600).room(.45).delay(".22:.15:.25")
       .gain(.42).speed(.98)],
  [6,  chord("<Bbm9 Gbmaj7 Dbmaj7 Ab>").dict('ireal').voicing()
       .s("gm_epiano1").struct("[~ x]*2")
       .lpf(2200).room(.4).delay(".18:.12:.2")
       .gain(.48)],
  [6,  chord("<Bbm7 Gbmaj7 Db Ab>").dict('ireal').voicing()
       .s("gm_epiano1").struct("~ [x ~] ~ [~ x]")
       .lpf(1600).room(.45).delay(".22:.15:.25")
       .gain(.42).speed(.98)],
  [4,  chord("<Bbm9 Gbmaj7 Dbmaj7 Ab>").dict('ireal').voicing()
       .s("gm_epiano1").struct("[~ x]*2")
       .lpf(2400).room(.35).delay(".15:.1:.18")
       .gain(.5)]
)

// -- MELODIC HOOK — floaty, reverb-drenched, enters at hook sections
$: arrange(
  [2,  silence],
  [6,  silence],
  [6,  n("<[0 ~ 2 4] [~ 5 ~ 4] [6 ~ 4 ~] [2 ~ 0 ~]>")
       .scale("Bb4:minor:pentatonic").s("gm_epiano1")
       .lpf(sine.range(1000, 3200).slow(8))
       .room(.55).delay(".3:.2:.38").gain(.36)
       .sometimes(x => x.add(note(12)))],
  [6,  n("<[~ 0 ~ 2] [4 ~ ~ ~] [~ 2 ~ 0] [~ ~ ~ ~]>")
       .scale("Bb4:minor:pentatonic").s("gm_epiano1")
       .lpf(1400).room(.5).delay(".25:.18:.3").gain(.28)],
  [4,  n("<[0 ~ 2 4] [~ 5 ~ 4] [6 ~ 4 ~] [2 ~ 0 ~]>")
       .scale("Bb4:minor:pentatonic").s("gm_epiano1")
       .lpf(sine.range(1200, 3600).slow(6))
       .room(.5).delay(".25:.18:.32").gain(.4)
       .rarely(x => x.add(note(12)))]
)

// -- HI-HAT TEXTURE — perlin-modulated, always rolling
$: s("hh*16").bank("RolandTR808")
   .gain(perlin.range(.06, .32))
   .pan(sine.slow(5))
   .lpf(sine.range(3000, 8000).slow(12))
   .cut(1)
