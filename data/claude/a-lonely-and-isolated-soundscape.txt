.s("gm_epiano2").gain(perlin.range(.18,.28)).room(.95).roomsize(8).delay(".6:.4:.7")
.lpf(sine.range(600,2000).slow(14)).attack(.04).decay(.8).sustain(.15).release(3.5)
.stack(
  note("b1").s("sine").attack(5).sustain(.65).release(6).gain(.1).room(.85).roomsize(8).lpf(100),

  chord("<Bm9 ~ ~ ~ F#m7 ~ ~ ~>/4").dict('ireal').voicing()
  .s("gm_synth_strings_1").attack(5).sustain(.3).release(7)
  .lpf(sine.range(180,600).slow(18)).room(.95).roomsize(8).gain(.08)
  .pan(sine.slow(22).range(.2,.8)),
  n("<~ ~ ~ ~ ~ ~ [~ 3 ~ ~] ~ ~ ~ ~ [5 ~ ~ ~]>/3")
  .scale("B5:minor:pentatonic").s("gm_flute").gain(.1).room(.95).roomsize(8)
  .delay(".75:.5:.85").lpf(2200).vib(4).vibmod(.6).attack(.15).sustain(.25).release(2.5),
  s("vinyl:2").gain(.05).room(1).loop(1).lpf(1200)
