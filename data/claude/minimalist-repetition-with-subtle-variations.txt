// "Phase Study No. 1" — Minimalist process music with phasing patterns
setcps(29/70)

const cell = n("0 2 3 5 7 5 3 2").scale("C3:minor:pentatonic")
const lpmod = sine.range(500, 3200).slow(16)

const chorus = cell.s("triangle").release(.35).gain(.4)
  .lpf(lpmod).lpq(perlin.range(.2, 2).slow(10))
  .iter(8).off(1/8, x => x.add(5).gain(.2).pan(.7))
  .juxBy(.4, x => x.late(1/64))
  .delay(".18:.33:.4").room(.3).roomsize(2)
  .stack(
    note("<c2!3 [c2 eb2]>").s("sine").gain(.5).lpf(250).release(.7),
    n("0 ~ 4 ~ 7 ~ 4 ~").scale("C5:minor:pentatonic").s("sine").gain(.07)
      .release(1.2).room(.5).delay(".35:.5:.5").rarely(x => x.add(2)),
    s("~ ~ ~ hh").gain(.07).lpf(perlin.range(3000, 8000).slow(8))
      .room(.15).sometimesBy(.15, x => x.speed(.75))
  )

const bridge = cell.s("sine").release(.5).gain(.3)
  .lpf(sine.range(400, 1800).slow(6))
  .iter(8).rev().off(1/4, x => x.add(3).gain(.12).pan(.3))
  .delay(".3:.5:.55").room(.5).roomsize(3)
  .stack(note("<c2 ~ eb2 ~>").s("sine").gain(.4).lpf(200).release(.9))

arrange(
  [1, cell.s("triangle").release(.5).gain(.2).lpf(600).iter(8).room(.6).delay(".2:.5:.4")],
  [12, chorus],
  [4, bridge],
  [12, chorus]
)
