// 🎵 "Still Turning" — Minimalist Repetition with Subtle Variations
// At 80 BPM: intro(4) + A(8) + B(8) + fade(4) = 24 bars ≈ 72s

setcpm(80 / 4)

// Sub drone — constant low hum
$: note("d1").s("sine").attack(3).sustain(.9).release(4).gain(.15).room(.5).lpf(130)

// Core cell — repeating 6-note arpeggio, filter and harmonics shift gradually
$: arrange(
  [4, n("0 2 4 7 4 2").scale("D3:minor").s("gm_epiano1").gain(.38).room(.5).delay(".12:.08:.12").lpf(1200)],
  [8, n("0 2 4 7 4 2").scale("D3:minor").s("gm_epiano1").gain(.46).room(.46).delay(".15:.1:.15").lpf(sine.range(1300,2200).slow(8)).every(7, x => x.add(note(2)))],
  [8, n("0 2 4 7 4 2").scale("D3:minor").s("gm_epiano1").gain(.48).room(.44).delay(".18:.12:.2").lpf(sine.range(1600,2800).slow(6)).every(5, x => x.rev()).rarely(x => x.add(note(7)))],
  [4, n("0 2 4 7 4 2").scale("D3:minor").s("gm_epiano1").gain(saw.slow(4).range(.38,.04)).room(.6).delay(".25:.18:.3").lpf(sine.range(500,1200).slow(4))]
)

// Mirror voice — same cell displaced one step, an octave up
$: arrange(
  [12, silence],
  [8, n("~ 2 4 7 4 2").scale("D4:minor").s("gm_epiano1").gain(.22).room(.55).delay(".22:.18:.3").lpf(1700).pan(.32)],
  [4, silence]
)

// Bass pedal — sustained, glacially slow
$: arrange(
  [4, silence],
  [8, note("<d2 d2 f2 d2>").s("gm_acoustic_bass").gain(.58).room(.2).decay(.7).sustain(.5)],
  [8, note("<d2 f2 a2 f2>").s("gm_acoustic_bass").gain(.6).room(.2).decay(.65).sustain(.5)],
  [4, note("d2").s("gm_acoustic_bass").gain(saw.slow(4).range(.5,.06)).room(.3).decay(1).sustain(.3)]
)

// Strings — gossamer wash, B section only
$: arrange(
  [12, silence],
  [8, chord("<Dm Fmaj7>/4").dict('ireal').voicing().s("gm_synth_strings_1").attack(1.5).sustain(.6).release(2).lpf(sine.range(600,1200).slow(8)).room(.7).gain(.18)],
  [4, silence]
)

// Hi-hats — whisper dynamics
$: arrange(
  [4, silence],
  [8, s("hh*4").bank("RolandTR808").gain(perlin.range(.06,.16)).room(.35).cut(1).pan(sine.slow(5))],
  [8, s("hh*4").bank("RolandTR808").gain(perlin.range(.08,.2)).room(.3).cut(1).pan(sine.slow(4)).every(8, x => x.fast(2).gain(.08))],
  [4, s("hh*4").bank("RolandTR808").gain(saw.slow(4).range(.12,.01)).room(.4).cut(1)]
)
