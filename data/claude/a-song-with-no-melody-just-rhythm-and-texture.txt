$: arrange(
  [8,  s("bd(3,8), ~ ~ ~ cp")
       .bank("RolandTR909").swing(3)
       .gain(perlin.range(.3,.6)).room(.5)
       .lpf(sine.range(400,2000).slow(4))],
  [16, s("bd [~ bd] ~ [bd ~], ~ cp ~ [cp ~], [hh hh oh hh]*2, rim(5,8)")
       .bank("RolandTR909").cut(1).swing(2)
       .gain(perlin.range(.7,.95)).room(.3)
       .every(4, x => x.fast(2).gain(.5))
       .every(7, x => x.rev())],
  [16, s("[bd bd] ~ [~ bd] bd, ~ [cp sd:2] ~ cp, [oh hh hh oh hh]*2, [~ rim ~ rim]*2, sd:1(3,8,2)")
       .bank("RolandTR909").cut(1).swing(3)
       .gain(".95 .8 .9 .85").room(.25)
       .every(3, x => x.press())
       .every(5, x => x.fast(2).gain(.55))
       .sometimes(x => x.crush(8))],
  [8,  s("bd ~ [~ bd] ~, ~ cp ~ ~, hh(3,8)")
       .bank("RolandTR909").swing(5)
       .gain(perlin.range(.5,.8)).room(.5)
       .every(2, x => x.rev())],
  [16, s("[bd bd] [~ bd] [bd ~] bd, [cp sd:2 cp]*2, [oh hh]*5, rim(7,16), bd(3,8,1)")
       .bank("RolandTR909").cut(1).swing(2)
       .gain(perlin.range(.6,1)).room(.2)
       .every(2, x => x.fast(2).gain(.6))
       .every(3, x => x.rev())
       .sometimes(x => x.crush(5))],
  [8,  s("bd(3,16), ~ ~ cp ~")
       .bank("RolandTR909").swing(4)
       .gain(saw.slow(8).range(.7,.05)).room(.7)]
)
