// "Pulse & Grain" — Pure percussive rhythm and texture, no melody
samples('github:tidalcycles/dirt-samples')
setcps(0.55)
let kit = s("bd:4(3,8), [~ sd:3]*2, [hh hh:1]*4")
  .bank("RolandTR909").gain("1.1 .8 .5")
  .swingBy(1/6, 4)
let chopped = s("breaks165")
  .slice(8, "0 1 <2 2*2> 3 [4 0] 5 6 7".every(3, rev))
  .slow(0.75).gain(0.35).hpf(800)
let grit = s("<pink brown>").gain(0.18)
  .lpf(perlin.range(300, 4000).slow(4))
  .decay(0.4).sustain(0)
let metal = s("metal:2(5,16)").gain(0.22)
  .speed("<1 1.5 .7>").delay("0.25:0.125:0.4").pan(rand)
arrange(
  [2, s("bd:4 ~!3, ~!3 hh:1").bank("RolandTR909")
    .lpf(600).room(0.5).rsize(3).gain(0.65)
    .stack(
      s("pink").gain(0.2).lpf(perlin.range(200, 1200).slow(2))
        .decay(0.8).sustain(0)
    )],
  [14, kit.lpf(sine.range(3000, 12000).slow(8))
    .stack(
      chopped.distort("1.5:0.4"),
      grit,
      metal,
      s("cp:3(3,8,2)").bank("RolandTR909").gain(0.4).room(0.35).hpf(600)
    ).room(0.2).rsize(2)],
  [6, s("bd:4 ~ ~ bd:4, ~ cp ~ ~").bank("RolandTR909")
    .lpf(sine.range(400, 2000).slow(3)).gain(0.85)
    .stack(
      s("brown").gain(0.28).bpf(perlin.range(200, 2000).slow(2))
        .bpq(8).decay(0.6).sustain(0).room(0.6).rsize(4),
      s("[hh?]*16").bank("RolandTR909").gain(0.25)
        .pan(rand).lpf(4000).delay("0.15:0.0625:0.55")
    )],
  [14, kit.lpf(sine.range(3000, 14000).slow(6))
    .jux(x => x.late(0.02).lpf(4000))
    .stack(
      s("breaks165").slice(8, "<0 4> 1 [2 3] <7 5> [0 6] 5 3 7")
        .slow(0.75).gain(0.35).hpf(800).distort("2:0.35"),
      s("<pink white>").gain(0.16)
        .lpf(perlin.range(500, 5000).slow(3)).decay(0.4).sustain(0),
      s("metal:2(7,16)").gain(0.2).speed("<.75 1.2 1>")
        .delay("0.3:0.125:0.5").pan(rand),
      s("cp:3(5,8)").bank("RolandTR909").gain(0.35)
        .room(0.3).hpf(700).sometimes(x => x.speed(1.5))
    ).room(0.25).rsize(2)],
  [2, s("bd:4 ~!3").bank("RolandTR909").lpf(400)
    .room(0.7).rsize(5).gain(0.55)
    .stack(
      s("brown").gain(0.22).lpf(perlin.range(100, 600).slow(2))
        .decay(1).sustain(0).room(0.7)
    )]
)
