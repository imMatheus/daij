setcps(108/60/4)

stack(
  s("bd ~ sd ~, ~ [hh hh] ~ [hh oh]")
    .bank("RolandTR808").gain(perlin.range(.5, .78))
    .room(.3).swing(3).lpf(6000)
    .sometimes(x => x.press()),

  note("<[d2 ~ ~ d2] [g1 ~ ~ g1] [a1 ~ ~ a1] [d2 ~ ~ ~]>")
    .s("gm_acoustic_bass").decay(.3).sustain(.5)
    .room(.25).gain(.72).swing(3),

  chord("<D G A7 D>").dict('ireal').voicing()
    .s("gm_acoustic_guitar_nylon")
    .struct("<[x ~ x ~] [~ x ~ x] [x x ~ x] [~ x x ~]>")
    .lpf(sine.range(2800, 4800).slow(16))
    .room(.45).delay(".12:.08:.18").gain(.48)
    .swing(3).attack(.01),

  chord("<D G A7 D>").dict('ireal').voicing()
    .s("gm_acoustic_guitar_steel")
    .struct("~ [~ x] ~ [~ x]")
    .lpf(4000).room(.4).gain(.33).swing(3)
    .every(3, x => x.struct("[x x x x]").gain(.22)),

  n("<[~ 0 ~ 2] [4 ~ 2 ~] [~ 0 ~ 4] [2 ~ ~ 0]>")
    .scale("D5:major:pentatonic")
    .s("gm_violin")
    .lpf(sine.range(2000, 5000).slow(8))
    .room(.5).delay(".2:.12:.28").gain(.3)
    .vib(4).vibmod(.4)
    .sometimes(x => x.add(note(12))),

  n("<[~ ~ 3 ~] [~ 4 ~ ~] [2 ~ ~ ~] [~ ~ ~ 0]>")
    .scale("D4:major")
    .s("gm_harmonica")
    .lpf(3200).room(.55).delay(".3:.2:.35").gain(.2)
    .vib(5).vibmod(.5).attack(.04).sustain(.6),

  note("<d1 g1 a1 d1>/4")
    .s("gm_string_ensemble_1")
    .attack(2).decay(1).sustain(.7).release(2)
    .gain(.14).room(.65).lpf(650)
)
