// "Raindrop Reverie" — Lo-fi jazz for a rainy window coffee shop
setcps(0.5)

stack(
  // warm piano chords — slow filter drift like condensation on glass
  chord("<C^7 Am7 F^7 Fm7>".slow(2))
    .voicing().s("piano")
    .gain("<.2@2 .48@14 .32@6 .48@14>")
    .lpf(perlin.range(800, 2400).slow(4))
    .attack(.02).release(1.2)
    .room(.55).rsize(2).delay(".28:0.375:0.45")
    .pan(sine.range(.35,.65).slow(7))
    .orbit(1),
  // lo-fi beat — muffled and swung
  s("[bd:4 ~]*2, ~ cp ~ [~ cp], [hh:1 [~ hh]]*2")
    .gain("<0@2 .5@14 0@6 .5@14>")
    .swingBy(1/6, 4)
    .lpf(1800).room(.2)
    .sometimesBy(.3, x => x.speed(".96"))
    .orbit(2),
  // bass — warm triangle following chord roots
  "<C^7 Am7 F^7 Fm7>".slow(2).rootNotes(2).note()
    .s("triangle").lpf(420).lpq(1.5)
    .gain("<0@2 .38@14 .15@6 .38@14>")
    .decay(.55).sustain(.12)
    .orbit(3),
  // rain on glass — filtered pink noise
  s("pink").gain(.06)
    .lpf(sine.range(600, 1500).slow(9))
    .hpf(200).room(.85).rsize(4)
    .orbit(4),
  // sparse melody — echoed piano fragments
  n("<[~ 4] [2 ~] [~ 5] [3 ~] [~ 6] [4 ~] [~ 2] [5 ~]>")
    .scale("C4:major").s("piano")
    .gain("<.1@2 .3@14 .22@6 .3@14>")
    .off(1/8, x => x.add(2).gain(.16))
    .lpf(3200).room(.55)
    .delay(".3:0.5:0.5")
    .pan(perlin.range(.25,.75))
    .orbit(5)
)
