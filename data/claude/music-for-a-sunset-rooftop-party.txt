// "Amber Skyline" — Warm deep house groove for a sunset rooftop party
setcps(.5)

const progression = "<AbM7 Fm7 Bbm7 Eb7>"

const intro = chord(progression)
  .anchor("c5")
  .voicing()
  .s("supersaw")
  .attack(.08)
  .release(1.2)
  .lpf(sine.range(400, 900).slow(2))
  .room(".6:0.85")
  .delay(".25:0.3:0.4")
  .detune(.15)
  .gain(.2)
  .orbit(2)
  .stack(
    s("bd*4").gain(.4).lpf(600).orbit(2),
    s("~ ~ hh ~").gain(.12).lpf(2000).room(.2)
  )

const chorus = chord(progression)
  .anchor("c5")
  .voicing()
  .s("supersaw")
  .attack(.02)
  .release(.6)
  .lpf(perlin.slow(5).range(1200, 3800))
  .delay(".2:0.25:0.55")
  .room(".3:0.75")
  .detune(.12)
  .gain(.24)
  .orbit(2)
  .stack(
    note("<ab2 f2 bb1 eb2>")
      .s("sawtooth")
      .lpf(480).lpenv(3).lpd(.12)
      .gain(.48).decay(.12).sustain(.5)
      .orbit(2),
    n("<0 [2 4] 4 [7 5] ~ 4 2 [4 7]>*2")
      .scale("Ab4:major")
      .s("triangle")
      .attack(.01).release(.18)
      .lpf(perlin.slow(4).range(1500, 3200))
      .delay(".3:0.125:0.45")
      .room(.2)
      .vib(4).vibmod(.1)
      .pan(sine.slow(6))
      .gain(.16)
      .reset("<x ~ ~ ~ x ~ ~ ~>")
      .orbit(2),
    s("bd*4")
      .gain(.85)
      .duckorbit(2)
      .duckattack(.2)
      .duckdepth(.85),
    s("~ cp ~ cp")
      .gain(.48).room(.15),
    s("hh*8")
      .gain(".14 .11 .16 .12")
      .lpf(perlin.slow(7).range(4000, 9500))
      .pan(sine.fast(3))
      .swingBy(1/6, 4),
    s("~ ~ ~ oh")
      .gain(.2).end(.28)
      .delay(".2:0.25:0.3"),
    s("[~ hh] ~ [hh hh] ~")
      .fast(2).gain(.07).lpf(8000).late(.01)
  )

const bridge = chord("<Db^7 Eb7 Cm7 Fm7>")
  .anchor("c6")
  .voicing()
  .s("sawtooth")
  .attack(.05)
  .release(1)
  .lpf(sine.range(600, 2400).slow(3))
  .hpf(perlin.slow(4).range(300, 1200))
  .delay(".5:0.375:0.65")
  .room(".55:0.9")
  .gain(.2)
  .orbit(2)
  .stack(
    n(run(8).slow(2))
      .scale("Ab4:major")
      .transpose("<5 7 3 0>")
      .s("triangle")
      .attack(.02).release(.25)
      .lpf(perlin.slow(3).range(1000, 2800))
      .pan(sine.slow(5))
      .gain(.12)
      .reset("<x ~ x ~>")
      .orbit(2),
    note("<db3 eb3 c3 f3>")
      .s("sine")
      .attack(.02).release(.5)
      .gain(.12).orbit(2),
    s("bd ~ bd ~")
      .gain(.5)
      .duckorbit(2)
      .duckattack(.3)
      .duckdepth(.65),
    s("hh*8")
      .gain(.07).hpf(6000)
      .pan(isaw.slow(3))
  )

arrange(
  [1, intro],
  [14, chorus],
  [6, bridge],
  [14, chorus]
)
