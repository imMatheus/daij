// "Skull Crusher" — relentless industrial aggression in C minor
setcps(150/60/4)

s("bd*4, [~ [sd sd] ~ sd]*2, [hh hh oh hh]*4, cp(5,8)")
.bank("RolandTR909").cut(1)
.gain(".98 .88 .95 .9")
.room(.15).distort(.35)
.every(3, x => x.fast(2).gain(.6))
.every(7, x => x.rev())
.stack(
  note("<[c1 c1 ~ c1] [eb1 ~ c1 ~] [f1 f1 ~ f1] [eb1 ~ c1 c1]>")
  .s("sawtooth").lpf(sine.range(200,1600).slow(8))
  .gain(.8).distort(.5).decay(.1).sustain(.15).room(.1),

  note("<[c3,eb3,g3] [eb3,g3,bb3] [f3,ab3,c4] [eb3,g3,bb3]>")
  .s("square").struct("[x ~ x ~]*2")
  .lpf(sine.range(600,3200).slow(6))
  .gain(.42).distort(.4).room(.2).crush(7)
  .every(4, x => x.fast(2).gain(.25)),

  n("<[0 3 ~ 7] [~ 10 7 ~] [3 5 ~ 7] [10 ~ 5 3]>")
  .scale("C5:minor:pentatonic").s("sawtooth")
  .lpf(sine.range(1200,4500).slow(5))
  .gain(.3).distort(.3).room(.35).delay(".2:.15:.25")
  .decay(.12).sustain(.2)
  .every(3, x => x.rev()),

  note("c2*16").s("sine")
  .gain(perlin.range(.08,.25))
  .lpf(sine.range(60,400).slow(4))
  .distort(.65).room(.15),

  note("<c1 eb1 f1 c1>/4").s("sawtooth")
  .attack(1).decay(2).sustain(.6).release(2)
  .gain(.18).room(.35).lpf(280).distort(.25)
)
