// "Skull Crusher" — Aggressive industrial beat with distorted synths and relentless percussion
setcps(140/60/4)

let kick = s("bd*4").bank("RolandTR909").distort("3:.5").gain(1.1)
let snare = s("~ cp ~ [cp cp?]").bank("RolandTR909").distort("1.5:.6").room(.05).gain(.9)
let hats = s("hh*16").bank("RolandTR909").gain(".4!3 .8").hpf(7000).pan(perlin)
let ohh = s("~ ~ ~ ~ ~ ~ ~ oh").bank("RolandTR909").gain(.5).hpf(5000)
let bass = note("c1!2 c1 [c1 eb1]").s("sawtooth").lpf(perlin.range(300,1500)).lpq(8).distort("5:.3").decay(.15).sustain(0).gain(.6)
let stab = note("[~ [c4,eb4,g4]] [~ [eb4,g4,bb4]] [~ [c4,eb4,g4]] [~ [g3,bb3,d4]]")
  .s("supersaw").lpf(2500).lpenv(5).lpd(.08).decay(.12).sustain(0).gain(.3).room(.1).delay(".15:.06:.4")

let intro = s("pink").gain(saw.slow(2).range(0,.3)).lpf(saw.slow(2).range(300,6000)).hpf(200)
  .stack(kick.struct("<[x ~ ~ ~] x*4>").lpf(saw.slow(2).range(500,4000)))

let chorus = kick.stack(snare, hats, ohh, bass, stab)

let bridge = kick.struct("x ~ [~ x] ~ x ~ [x x] ~").stack(
  snare.struct("~ ~ x ~ ~ [~ x] ~ x").gain(.7),
  hats.gain(.25).speed(.8),
  note("<c1 eb1 f1 eb1>").s("sawtooth").lpf(600).lpq(5).distort("3:.4").decay(.25).sustain(0).gain(.5)
)

let chorus2 = kick.stack(
  snare.sometimesBy(.2, x=>x.ply(2)), hats.every(4,x=>x.fast(2).gain(.5)), ohh, bass, stab,
  note("c2").s("sine").penv(36).pdecay(.06).pcurve(1).struct("x*4").gain(.4).distort("2:.4")
)

arrange([2, intro], [16, chorus], [6, bridge], [18, chorus2])
