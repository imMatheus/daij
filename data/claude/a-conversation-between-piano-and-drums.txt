// 🎹🥁 "Call & Response" — Intimate Jazz Dialogue Between Piano and Drums
setcps(0.45)

$: s("hh*8")
   .bank("RolandTR808")
   .gain(perlin.range(.02, .22))
   .pan(sine.slow(6))
   .cut(1)

$: arrange(
  [4,  silence],
  [4,  s("~ ~ bd ~, ~ ~ ~ cp").bank("RolandTR808").swing(4).gain(".45 .35").room(.5)],
  [8,  s("bd ~ [~ bd] ~, ~ cp ~ ~, [hh oh]*4").bank("RolandTR808").cut(1).swing(4).gain(perlin.range(.72,.9)).room(.35).every(4, x => x.fast(2).gain(.45))],
  [8,  s("bd(5,8), ~ [cp,sd:1] ~ [cp,sd:2], [oh hh]*4").bank("RolandTR808").cut(1).swing(4).gain(perlin.range(.8,.96)).room(.3).every(3, x => x.press())],
  [4,  s("~ ~ bd ~").bank("RolandTR808").gain(".4 .3").room(.6)],
  [4,  s("bd ~ ~ ~").bank("RolandTR808").gain(saw.slow(4).range(.4,.05)).room(.65)]
)

$: arrange(
  [4,  chord("<Cm9 Gm7 Abmaj7 Bb7>").dict('ireal').voicing()
       .s("gm_epiano1").struct("[x ~ x ~]")
       .lpf(2800).room(.6).delay(".2:.15:.3").gain(.55)],
  [4,  chord("<Cm9 Gm7 Abmaj7 Bb7>").dict('ireal').voicing()
       .s("gm_epiano1").struct("~ [x ~] ~ ~")
       .lpf(1600).room(.55).delay(".25:.15:.3").gain(.36)],
  [8,  chord("<Cm9 Gm7 Abmaj7 Bb7>").dict('ireal').voicing()
       .s("gm_epiano1").struct("[~ x]*2")
       .lpf(2200).room(.45).delay(".2:.1:.25").gain(.46)],
  [8,  chord("<Cm9 Gm7 Abmaj7 Bb7>").dict('ireal').voicing()
       .s("gm_epiano1").struct("[~ x ~ x]")
       .lpf(2400).room(.5).delay(".2:.15:.3").gain(.5)],
  [4,  chord("<Cm9 Abmaj7>/2").dict('ireal').voicing()
       .s("gm_epiano1").struct("x ~ ~ ~")
       .lpf(1200).room(.8).delay(".5:.35:.6").gain(.42)],
  [4,  chord("<Cm9>/1").dict('ireal').voicing()
       .s("gm_epiano1").struct("x ~ ~ ~")
       .lpf(sine.range(600,1400).slow(4)).room(.85)
       .gain(saw.slow(4).range(.4,.05))]
)

$: arrange(
  [4,  n("<[0 ~ 3 ~] [5 ~ 7 ~] [~ 3 ~ 5] [7 ~ 5 3]>")
       .scale("C5:dorian").s("gm_epiano1")
       .lpf(3400).room(.6).delay(".3:.2:.4")
       .gain(.5).vib(3).vibmod(.4)],
  [4,  n("<[~ ~ 0 ~] [~ 3 ~ ~] [~ ~ ~ 5] [~ 3 ~ ~]>")
       .scale("C5:dorian").s("gm_epiano1")
       .lpf(sine.range(900,2800).slow(8))
       .room(.55).delay(".35:.25:.45").gain(.3)],
  [8,  n("<[0 ~ 3 5] [~ 7 5 ~] [3 ~ 0 ~] [~ 5 7 0]>")
       .scale("C5:dorian").s("gm_epiano1")
       .lpf(sine.range(1200,3600).slow(7))
       .room(.5).delay(".25:.2:.35")
       .gain(.44).vib(3).vibmod(.5)
       .sometimes(x => x.add(note(12)))],
  [8,  n("<[~ 0 ~ 3] [5 ~ 7 ~] [3 5 ~ 0] [~ 7 5 3]>")
       .scale("C5:dorian").s("gm_epiano1")
       .lpf(sine.range(1100,3400).slow(8))
       .room(.5).delay(".25:.2:.35").gain(.46)
       .sometimes(x => x.add(note(12)))],
  [4,  n("<[0 3 5 7] [5 3 0 ~]>")
       .scale("C5:dorian").s("gm_epiano1")
       .lpf(2600).room(.7).delay(".4:.3:.5").gain(.4)],
  [4,  n("<[~ 0 ~ ~] [~ ~ 3 ~]>")
       .scale("C5:dorian").s("gm_epiano1")
       .room(.9).delay(".6:.4:.7")
       .gain(.22).lpf(sine.range(500,1600).slow(4))]
)

$: arrange(
  [4,  silence],
  [4,  note("<[c2 ~ ~ ~] [g1 ~ ~ ~] [ab1 ~ ~ ~] [bb1 ~ ~ ~]>")
       .s("gm_acoustic_bass").decay(.4).sustain(.4).room(.2).gain(.62)],
  [8,  note("<[c2 ~ c2 ~] [g1 ~ ~ g1] [ab1 ~ ab1 ~] [bb1 ~ ~ bb1]>")
       .s("gm_acoustic_bass").decay(.3).sustain(.5).room(.2).gain(.72)],
  [8,  note("<[c2 ~ c2 ~] [g1 ~ ~ g1] [ab1 ~ ab1 ~] [bb1 ~ ~ bb1]>")

  [4,  note("<c2 ~ ab1 ~>/2")
