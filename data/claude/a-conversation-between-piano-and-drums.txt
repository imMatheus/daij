// "Midnight Dialogue" — Jazz conversation between piano and drums
setcps(0.5)
samples('github:tidalcycles/dirt-samples')

const warmth = perlin.slow(9).range(1200, 4800)
const shimmer = sine.slow(7).range(2200, 6000)
const air = perlin.slow(11).range(.3, .55)

const intro =
  note("<[c4 eb4 g4 bb4] [~ ~ ab3 c4]>").s("piano")
    .att(.01).dec(.4).sus(.5).rel(1.8)
    .lpf(warmth).room(.55).size(2).delay(".25:.2:.38").gain(.38)

const chorus =
  chord("<Cm7 Fm7 Ab^7 G7>").anchor("c5").voicing().slow(4).s("piano")
    .struct("<[x ~ x ~] [~ x ~ x] [x x ~ ~] [~ ~ x x]>")
    .att(.01).dec(.32).sus(.45).rel(1)
    .lpf(warmth).hpf(110).room(air).delay(".14:.125:.2").gain(.46)
    .reset("<x ~ ~ x>")
    .stack(
      note("<[c5 ~ eb5 ~] [~ d5 ~ bb4] [ab4 ~ c5 ~] [~ g4 ~ bb4]>").slow(2).s("piano")
        .att(.01).dec(.16).sus(.18).rel(.35)
        .lpf(shimmer).room(.42).delay(".18:.2:.24").gain(.15)
        .reset("<x ~ x ~>")
    )
    .stack(
      note("<c2 ~ g1 ~ f1 ~ ab1 ~ c2 ~ eb2 ~ g1 ~ ab1 ~>").slow(2).s("piano")
        .att(.01).dec(.2).sus(.1).rel(.16)
        .lpf(750).room(.15).gain(.22)
    )
    .stack(
      s("bd").bank("RolandTR909").euclidRot(5,8,1).slow(2)
        .lpf(720).room(.14).gain(.5)
    )
    .stack(
      s("<~ sd ~ ~ ~ [sd ~] ~ sd>").bank("RolandTR909").slow(2)
        .end(.11).lpf(2200).room(.22).gain(.1)
    )
    .stack(
      s("hh*8").bank("RolandTR909").slow(2)
        .begin("<.01 .02 .01 .03>").end("<.05 .07 .06 .08>")
        .bpf(sine.slow(5).range(4500, 8800)).bpq(4)
        .gain(".024 .04 .028 .046 .026 .042 .03 .048")
        .delay(".04:.125:.1")
    )
    .stack(
      s("~ rim ~ rim ~ ~ rim ~").slow(2)
        .lpf(1300).room(.24).gain(.044)
    )

const bridge =
  s("bd").bank("RolandTR909").euclidRot(7,16,2).slow(2)
    .lpf(720).room(.16).gain(.5)
    .stack(
      s("<[sd ~ sd ~] [~ sd ~ sd] [sd sd ~ ~]>").bank("RolandTR909").slow(2)
        .end(.11).lpf(2400).room(.2).gain(.11)
    )
    .stack(
      s("hh*12").bank("RolandTR909").slow(2)
        .begin(".01 .02").end(".06 .08")
        .bpf(sine.slow(4).range(5000, 9200)).bpq(5)
        .gain("<.028 .048 .032 .05 .026 .046 .03 .052>")
    )
    .stack(
      s("~ ~ cp ~ ~ cp ~ ~ ~ cp ~ ~").slow(2)
        .lpf(1700).room(.3).gain(.055)
    )
    .stack(
      note("<[~ ~ eb5 ~] [~ ~ ~ d5] [ab4 ~ ~ ~]>").slow(2).s("piano")
        .att(.01).dec(.22).sus(.28).rel(.55)
        .lpf(sine.slow(6).range(900, 2800)).room(.48).delay(".2:.2:.3").gain(.32)
    )

arrange([2, intro], [13, chorus], [5, bridge], [15, chorus]).postgain(.94)
