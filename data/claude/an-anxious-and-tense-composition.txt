setcps(0.5)
samples('github:tidalcycles/dirt-samples')

  note("<[~ ~ ~ c2]@2 [c2 ~ [~ c2] c2]@14 [~ c2 ~ ~]@4 [c2 [~ c2] ~ c2]@15>")

  n("<~@2 [0 ~ 3 ~]*4@14 [~ 5 ~ ~]*2@4 [0 1 3 5 7 5 3 1]@15>")
