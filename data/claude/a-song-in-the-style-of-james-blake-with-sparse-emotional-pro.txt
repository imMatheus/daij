// 🎵 "All I Keep" — Sparse Emotional Electronic / James Blake-style
// At 65 BPM: intro(4) + verse(8) + chorus(8) + outro(4) = 24 cycles ≈ 88s
// Key: F minor — devastating sub, reverb-drenched keys, glitchy half-time drums

setcpm(65 / 4)

$: s("vinyl:2").gain(.08).room(1).loop(1).lpf(1800)

// -- DRUMS — mostly absent, glitchy half-time when present
$: arrange(
  [4,  silence],
  [8,  s("bd ~ ~ ~, ~ ~ [~ cp] ~")
       .bank("RolandTR808").gain(".65 .45").room(.55)
       .sometimes(x => x.speed(.82))],
  [8,  s("bd ~ ~ ~, ~ ~ [~ cp] ~, ~ ~ ~ [~ hh]")
       .bank("RolandTR808").cut(1).gain(".72 .52 .18").room(.45)
       .rarely(x => x.fast(2).gain(.2))],
  [4,  s("bd ~ ~ ~")
       .bank("RolandTR808").gain(saw.slow(4).range(.45, .06)).room(.65)]
)

// -- PIANO — sparse stabs, drenched in reverb and delay
$: arrange(
  [4,  chord("<Fm9 Dbmaj9>/2").dict('ireal').voicing()
       .s("gm_epiano1").struct("x ~ ~ ~")
       .lpf(1100).room(.88).delay(".55:.4:.65")
       .gain(.28).attack(.12)],
  [8,  chord("<Fm9 Dbmaj9 Abmaj7 Eb7>").dict('ireal').voicing()
       .s("gm_epiano1").struct("~ [x ~] ~ ~")
       .lpf(sine.range(700, 1600).slow(8))
       .room(.78).delay(".45:.3:.55")
       .gain(.35).speed(.98)],
  [8,  chord("<Fm9 Dbmaj9 Abmaj7 Eb7>").dict('ireal').voicing()
       .s("gm_epiano1").struct("[~ x] ~ ~ [~ x]")
       .lpf(sine.range(900, 2200).slow(6))
       .room(.72).delay(".38:.28:.48")
       .gain(.4).speed(.99)],
  [4,  chord("<Fm9>/1").dict('ireal').voicing()
       .s("gm_epiano1").struct("x ~ ~ ~")
       .lpf(sine.range(350, 900).slow(4))
       .room(.92).delay(".6:.45:.72")
       .gain(saw.slow(4).range(.28, .04))]
)

// -- MELODY — ghostly, high register, more silence than notes
$: arrange(
  [4,  silence],
  [8,  silence],
  [8,  n("<[~ ~ 4 ~] [~ ~ ~ ~] [~ 2 ~ ~] [~ ~ ~ 0]>")
       .scale("F5:minor:pentatonic").s("gm_epiano2")
       .lpf(2000).room(.82).delay(".55:.4:.65")
       .gain(.22).vib(4).vibmod(.55)
       .crush(14)],
  [4,  silence]
)

// -- SUB BASS — deep and devastating, the emotional anchor
$: arrange(
  [4,  note("<f1>/1").s("sine")
       .gain(.16).lpf(110)
       .attack(1.8).decay(1.2).sustain(.6).release(2.5)],
  [8,  note("<f1 db1 ab0 eb1>/4").s("sine")
       .gain(.24).lpf(125)
       .attack(.04).decay(.45).sustain(.6).release(.12)],
  [8,  note("<f1 db1 ab0 eb1>/4").s("sine")
       .gain(.28).lpf(135)
       .attack(.04).decay(.38).sustain(.68).release(.1)],
  [4,  note("<f1>/1").s("sine")
       .gain(saw.slow(4).range(.2, .03)).lpf(90)
       .attack(1.2).decay(1.5).sustain(.45).release(2.8)]
)

// -- STRINGS — barely-there pad, only chorus, like a held breath
$: arrange(
  [4,  silence],
  [8,  silence],
  [8,  chord("<Fm9 Dbmaj9>/2").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(2.2).decay(1.2).sustain(.45).release(3.5)
       .lpf(sine.range(250, 850).slow(8))
       .room(.82).gain(.14)],
  [4,  silence]
)
