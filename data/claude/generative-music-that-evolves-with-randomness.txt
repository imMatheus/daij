// 🌿 "Emergent Patterns" — Generative Ambient / Algorithmic Drift
// At 80 BPM: intro(4) + evolve1(6) + evolve2(6) + peak(6) + dissolve(4) = 26 bars ≈ 78s

setcpm(80 / 4)

$: s("vinyl:2").gain(.11).room(1).loop(1).lpf(2200)

// Generative percussion — probability-driven euclidean rhythms
$: arrange(
  [4,  s("hh(3,8)")
       .bank("RolandTR808").gain(perlin.range(.12,.28)).hpf(4500).room(.4)
       .pan(perlin.range(.3,.7))],
  [6,  s("<hh(3,8) hh(5,8)>, ~ ~ ~ [cp ~]")
       .bank("RolandTR808").gain(perlin.range(.18,.35)).hpf(4500).room(.35)
       .pan(sine.slow(6)).sometimes(x => x.gain(.08))],
  [6,  s("<hh(5,8) hh(7,8)>, bd(3,8), [~ cp ~ ~]")
       .bank("RolandTR808").gain(perlin.range(.22,.4)).hpf(3800).room(.3)
       .pan(perlin.range(.2,.8)).rarely(x => x.speed(1.5))],
  [6,  s("<hh(5,8,2) hh(7,8,3)>, bd(3,8,1), [~ cp] ~ [~ cp] ~, rimshot(2,5)")
       .bank("RolandTR808").gain(perlin.range(.25,.45)).room(.25)
       .pan(perlin.range(.15,.85)).sometimes(x => x.delay(".15:.1:.2"))
       .every(3, x => x.fast(1.5))],
  [4,  s("hh(3,8)")
       .bank("RolandTR808").gain(saw.slow(4).range(.25,.0)).hpf(5000).room(.6)
       .pan(perlin.range(.3,.7))]
)

// Drifting bass — perlin-modulated filter
$: arrange(
  [4,  note("eb1").s("sawtooth")
       .lpf(perlin.range(180,450).slow(4)).decay(.6).sustain(.4)
       .gain(.5).room(.35)],
  [6,  note("<eb1 ~ ~ ~ bb1 ~ ~ ~>/2").s("sawtooth")
       .lpf(perlin.range(200,600).slow(6)).decay(.5).sustain(.45)
       .gain(.62).room(.3).sometimes(x => x.add(note(12)).gain(.3))],
  [6,  note("<[eb1 ~ ~ bb1] [~ ~ ab1 ~]>").s("sawtooth")
       .lpf(perlin.range(250,800).slow(4)).decay(.4).sustain(.5)
       .gain(.68).room(.25).distort(.06)],
  [6,  note("<[eb1 ~ bb1 ~] [ab1 ~ gb1 ~]>").s("sawtooth")
       .lpf(perlin.range(300,1000).slow(3)).decay(.35).sustain(.55)
       .gain(perlin.range(.6,.78)).room(.2).distort(.08)
       .every(4, x => x.fast(1.5))],
  [4,  note("eb1").s("sawtooth")
       .lpf(saw.slow(4).range(600,120)).decay(.7).sustain(.3)
       .gain(saw.slow(4).range(.5,.06)).room(.45)]
)
