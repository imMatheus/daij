// "Drift State" — Generative ambient evolving through stochastic processes and perlin noise
setcps(0.5)

arrange(
  // INTRO — a pad emerges from noise
  [1, stack(
    chord("Cm").voicing().s("sawtooth").lpf(400).att(1.5).rel(3).sus(.7)
      .gain(.2).room(".8:4").orbit(1),
    s("pink").lpf(300).gain(.03).room(.5).orbit(5)
  )],
  // CHORUS 1 — layered generative textures, randomness drives melody and density
  [14, stack(
    n(irand(12).seg("<4 8 6 10>".slow(3)))
      .scale("<C4:minor:pentatonic C4:dorian>".slow(7))
      .s("<triangle sawtooth>".slow(5))
      .lpf(perlin.range(400,4500).slow(4)).lpq(perlin.range(.5,4).slow(3))
      .att(.04).dec(.3).sus(.2).rel(.5)
      .gain(perlin.range(.18,.45).slow(3))
      .delay(".5:.375:.4").room(.4).sz(2)
      .pan(rand).degradeBy(perlin.range(.1,.5).slow(6))
      .juxBy(.6,rev).orbit(2),
    n(irand(5).seg(4)).scale("C2:minor:pentatonic").s("sawtooth")
      .lpf(perlin.range(120,500).slow(9)).lpq(1.2)
      .att(.08).dec(.4).sus(.5).rel(.6)
      .gain(.4).room(.25).euclid(3,8)
      .rarely(x=>x.transpose(12)).orbit(3),
    chord("<Cm Fm Gm Eb>".slow(4)).voicing().s("sawtooth")
      .lpf(sine.range(300,1800).slow(14))
      .att(.9).dec(.3).sus(.8).rel(1.5)
      .gain(sine.range(.15,.28).slow(10))
      .room(".7:4").delay(".25:.5:.45")
      .pan(perlin.range(.3,.7).slow(6)).orbit(1),
    s("hh*8").n(irand(6)).gain(rand.range(.12,.42)).pan(rand)
      .degradeBy(perlin.range(.2,.6).slow(5))
      .sometimesBy(.3,x=>x.speed(rand.range(.5,2)))
      .delay(".15:.125:.3").hpf(perlin.range(5000,12000).slow(8)).orbit(4),
    s("pink").lpf(perlin.range(100,600).slow(11)).gain(.035).room(.5).orbit(5)
  )],
  // BRIDGE — stripped down, sparse bells over murmuring pad
  [5, stack(
    n(irand(8).seg(3)).scale("C5:lydian").s("sine")
      .att(.01).dec(1.2).sus(0).rel(1.8)
      .gain(perlin.range(.08,.22))
      .room(".9:6").delay(".6:.333:.55")
      .pan(rand).degradeBy(.45).orbit(2),
    chord("<Gm Fm>".slow(3)).voicing().s("sawtooth")
      .lpf(perlin.range(200,700).slow(5))
      .att(1.2).sus(.7).rel(2).gain(.18)
      .room(".8:5").delay(".3:.5:.5").orbit(1),
    n(irand(3).seg(2)).scale("C2:minor:pentatonic").s("triangle")
      .lpf(250).att(.15).dec(.6).sus(.3).rel(.8)
      .gain(.3).room(.4).degradeBy(.5).orbit(3),
    s("brown").lpf(perlin.range(80,300).slow(7)).gain(.04).room(.6).orbit(5)
  )],
  // CHORUS 2 — returns with intensified randomness and richer layers
  [15, stack(
    n(irand(14).seg("<8 12 6 16>".slow(3)))
      .scale("<C4:minor:pentatonic C4:dorian C4:aeolian>".slow(5))
      .s("<triangle sawtooth square>".slow(4))
      .lpf(perlin.range(500,6000).slow(3)).lpq(perlin.range(.5,5).slow(4))
      .att(.03).dec(.25).sus(.15).rel(.4)
      .gain(perlin.range(.2,.5).slow(3))
      .delay(".45:.375:.45").room(.45).sz(2)
      .pan(rand).degradeBy(perlin.range(.05,.45).slow(5))
      .juxBy(.7,rev).sometimesBy(.15,x=>x.scramble(4)).orbit(2),
    n(irand(5).seg("<4 6>".slow(3))).scale("C2:minor:pentatonic").s("sawtooth")
      .lpf(perlin.range(150,600).slow(7)).lpq(1.5)
      .att(.06).dec(.35).sus(.5).rel(.5)
      .gain(.42).room(.3).euclid("<3 5>",8)
      .rarely(x=>x.transpose(12)).orbit(3),
    chord("<Cm Fm Gm Eb Ab>".slow(5)).voicing().s("sawtooth")
      .lpf(sine.range(400,2200).slow(12))
      .att(.8).sus(.8).rel(1.5)
      .gain(sine.range(.17,.3).slow(8))
      .room(".7:4").delay(".3:.5:.5")
      .pan(perlin.range(.25,.75).slow(5)).orbit(1),
    s("hh*<8 10 12 6>").n(irand(6)).gain(rand.range(.1,.45)).pan(rand)
      .degradeBy(perlin.range(.15,.55).slow(4))
      .sometimesBy(.4,x=>x.speed(rand.range(.3,2.5)))
      .delay(".2:.125:.35").hpf(perlin.range(4000,14000).slow(7)).orbit(4),
    n(irand(7).seg(2)).scale("C5:minor:pentatonic").s("sine")
      .att(.01).dec(.6).sus(0).rel(1)
      .gain(.12).delay(".7:.3:.6").room(".8:5")
      .pan(rand).degradeBy(.6).orbit(6),
    s("pink").lpf(perlin.range(100,800).slow(9)).gain(.04).room(.5).orbit(5)
  )]
)
