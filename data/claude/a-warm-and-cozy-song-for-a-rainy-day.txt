// "Raindrops on Glass" — warm lo-fi ambient for a rainy afternoon
setcps(0.27)
samples('github:tidalcycles/dirt-samples')

stack(
  chord("<F^7 Dm7 Bb^7 Am7>").voicing()
    .s("piano").gain("<.3@2 .55@7 .35@2 .55@8>")
    .lpf(sine.range(800,2400).slow(8))
    .room(.7).size(2).delay(".3:.25:.4")
    .att(.02).release(.6)
    .reset("<x@2 x@7 x@2 x@8>"),
  note("<f2 d2 bb1 a1>").s("sawtooth")
    .lpf(280).gain("<0@2 .3@7 0@2 .3@8>")
    .att(.1).decay(.4).sustain(.5).room(.3)
    .reset("<x@2 x@7 x@2 x@8>"),
  s("hh*16").gain(perlin.range(.03,.17))
    .pan(rand).lpf(perlin.range(3000,8000).slow(5))
    .room(.8).delay(.08)
    .mask("<0@2 1@7 1@2 1@8>"),
  note("<[f3,a3,c4] [d3,f3,a3] [bb2,d3,f3] [a2,c3,e3]>")
    .s("sawtooth").lpf(sine.range(400,1100).slow(12))
    .gain("<0@2 .15@7 .08@2 .15@8>")
    .att(.5).sustain(.6).release(1).room(.9).size(3).chorus(.3)
    .reset("<x@2 x@7 x@2 x@8>"),
  n("<0 2 4 ~ [5 4] 2 ~ [0 ~]>").scale("F4:major")
    .s("piano").gain("<0@2 .4@7 .25@2 .4@8>")
    .lpf(1500).room(.65).delay(".35:.2:.5")
    .slow(2).sometimesBy(.25, x=>x.add(7)),
  s("brown").gain("<.03@2 .05@7 .04@2 .05@8>")
    .lpf(sine.range(200,500).slow(6)).room(.9).size(4)
)
