// 🌧️ "Blanket Weather" — Warm Lo-Fi Ambient Jazz / Rainy Day Comfort
// At 70 BPM: intro(4) + A(8) + B(8) + outro(4) = 24 bars ≈ 82s

setcpm(70 / 4)

$: s("vinyl:2").gain(.14).room(1).loop(1).lpf(2400)

$: s("hh*16")
   .bank("RolandTR808")
   .gain(perlin.range(.03, .22))
   .pan(perlin.range(.2, .8))
   .cut(1)
   .lpf(sine.range(2800, 5500).slow(14))

$: arrange(
  [4,  s("bd ~ ~ ~").bank("RolandTR808").gain(.48).room(.55).lpf(700)],
  [8,  s("bd ~ ~ ~, ~ [~ sd:1] ~ ~")
       .bank("RolandTR808").swing(4)
       .gain(".7 .64").room(.42)],
  [8,  s("bd ~ ~ bd, ~ [cp,sd:1] ~ ~, [~ hh]*4")
       .bank("RolandTR808").swing(4).cut(1)
       .gain(".78 .72").room(.35)
       .every(4, x => x.press())],
  [4,  s("bd ~ ~ ~")
       .bank("RolandTR808").swing(4)
       .gain(saw.slow(4).range(.48, .06)).room(.6)]
)

$: arrange(
  [4,  silence],
  [8,  note("<[a1 ~ ~ ~] [f#1 ~ ~ f#1] [d1 ~ ~ ~] [e1 ~ ~ ~]>")
       .s("gm_acoustic_bass")
       .decay(.4).sustain(.5).room(.25).gain(.72)],
  [8,  note("<[a1 ~ a1 ~] [f#1 ~ ~ f#1] [d1 ~ d1 ~] [e1 ~ ~ e1]>")
       .s("gm_acoustic_bass")
       .decay(.3).sustain(.55).room(.2).gain(.78)],
  [4,  note("<a1 ~ d1 ~>/2")
       .s("gm_acoustic_bass")
       .decay(.55).sustain(.4).room(.35).gain(.55).speed(.97)]
)

$: arrange(
  [4,  chord("<Amaj9 Dmaj7>/2").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(.9).decay(1).sustain(.7).release(2.5)
       .lpf(1000).room(.8).gain(.3)],
  [8,  chord("<Amaj9 F#m9 Dmaj9 E9>").dict('ireal').voicing()
       .s("gm_epiano1")
       .struct("~ [x ~] ~ [x ~]")
       .lpf(2000).room(.55).delay(".25:.18:.35")
       .gain(.46).speed(.98)],
  [8,  chord("<Amaj9 F#m9 Dmaj9 E9>").dict('ireal').voicing()
       .s("gm_epiano1")
       .struct("[~ x]*2")
       .lpf(2500).room(.5).delay(".2:.12:.28")
       .gain(.5)],
  [4,  chord("<Amaj9 Dmaj7>/2").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(.9).sustain(.6).release(3)
       .lpf(sine.range(400, 1100).slow(4))
       .room(.85).gain(saw.slow(4).range(.3, .06))]
)

$: arrange(
  [4,  n("<0 ~ 2 ~>/1").scale("A4:major:pentatonic")
       .s("gm_epiano2")
       .room(.8).delay(".5:.35:.6").gain(.26).lpf(1400).attack(.2)],
  [8,  n("<[~ 0 ~ 2] [4 ~ 2 ~] [~ 3 ~ 4] [2 ~ ~ 0]>")
       .scale("A5:major:pentatonic").s("gm_epiano2")
       .lpf(sine.range(900, 2800).slow(10))
       .room(.6).delay(".3:.22:.42").gain(.35)
       .rarely(x => x.add(note(12)))],
  [8,  n("<[0 ~ 2 4] [~ 5 4 ~] [2 ~ 0 ~] [~ 3 2 0]>")
       .scale("A5:major:pentatonic").s("gm_epiano2")
       .lpf(sine.range(1100, 3400).slow(8))
       .room(.55).delay(".25:.18:.35").gain(.4)
       .sometimes(x => x.add(note(12)))],
  [4,  n("<[~ 0 ~ ~] [~ ~ 2 ~] [~ 4 ~ ~] [~ ~ 0 ~]>")
       .scale("A5:major:pentatonic").s("gm_epiano2")
       .room(.85).delay(".55:.4:.65").gain(.24)
       .lpf(sine.range(500, 1600).slow(4))]
