// "Quest for the Crystal Crown" — 8-bit Chiptune Adventure Theme
// At 150 BPM, ~40 cycles ~ 64 seconds
// Key: D major — heroic, triumphant, NES-era adventure

setcpm(150 / 4)

// -- DRUMS — NES-style noise channel + kick
$: s("bd ~ bd ~, ~ cp ~ cp, [hh hh ~ hh hh hh ~ hh]*2")
   .bank("RolandTR808").cut(1)
   .gain(".85 .72 .3")
   .crush(8).room(.1)
   .every(4, x => x.struct("bd [bd bd] ~ bd, ~ cp ~ [cp cp], [hh hh ~ hh hh hh ~ hh]*2"))
   .every(7, x => x.fast(2).gain(.35))

// -- CHIP BASS — square wave, driving root movement
$: note("<[d2 ~ d2 ~] [b1 ~ ~ b1] [g1 ~ g1 ~] [a1 ~ a1 a1]>")
   .s("square")
   .lpf(600).lpq(2)
   .decay(.1).sustain(.35).release(.04)
   .gain(.36).crush(10)
   .every(4, x => x.struct("[d2 d2 ~ d2] [b1 ~ b1 ~] [g1 g1 ~ ~] [a1 ~ a1 a1]"))

// -- HERO MELODY — bright square wave lead
$: n("<[0 2 4 7] [9 7 4 2] [5 4 2 0] [2 4 5 7], [~ ~ ~ ~] [~ ~ ~ ~] [~ ~ ~ ~] [9 11 9 7]>")
   .scale("D5:major")
   .s("square")
   .lpf(4200).lpq(2)
   .decay(.07).sustain(.3).release(.03)
   .gain(.24).crush(12).room(.12)
   .every(3, x => x.rev())
   .every(5, x => x.add(note(12)).gain(.15))
   .firstOf(8, x => x.fast(2).gain(.16))

// -- ARPEGGIOS — triangle wave cascading adventure sparkle
$: n("<[0 4 7 11] [0 4 7 11] [0 2 7 11] [0 4 7 9]>*4")
   .scale("D4:major")
   .s("triangle")
   .decay(.05).sustain(.15).release(.02)
   .gain(.18).crush(10).room(.1)
   .pan(sine.slow(3))
   .every(4, x => x.iter(4))
   .every(6, x => x.rev())

// -- HARMONY PAD — triangle chords, warm bed
$: note("<[d4,fs4,a4] [b3,d4,fs4] [g3,b3,d4] [a3,cs4,e4]>/4")
   .s("triangle")
   .attack(.08).decay(.25).sustain(.45).release(.2)
   .lpf(1400).crush(12)
   .gain(.12).room(.15)

// -- COUNTER MELODY — high register fanfare
$: n("<[~ 4 ~ 7] [9 ~ 7 ~] [~ 5 ~ 4] [7 ~ 9 11]>")
   .scale("D6:major")
   .s("square")
   .lpf(5500).crush(10)
   .decay(.04).sustain(.12).release(.02)
   .gain(.1).room(.08).pan(.65)
   .delay(".08:.04:.1")
   .every(5, x => x.rev())

// -- SUB — low sine anchor
$: note("<d1 b0 g0 a0>/4")
   .s("sine")
   .attack(.02).decay(.25).sustain(.55).release(.1)
   .gain(.22).lpf(150)
