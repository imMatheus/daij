// "Pixel Quest" — 8-bit chiptune adventure theme
setcps(0.625)

const i = 4, c = 16, b = 8, f = 16

const chorusLead = n("<[0 2 4 7] [7 4 2 0] [3 5 7 5] [3 2 0 ~] [5 4 2 4] [5 7 5 ~] [4 2 4 7] [4 2 0 ~]>*2")
  .scale("D5:major").s("square").decay(.1).sustain(0).release(.04)
  .lpf(sine.range(2000,8000).slow(8)).delay(".12:.125:.2").room(.08).gain(.35)

const bridgeLead = n("<[0 2 4 5] [5 4 2 0] [5 7 5 4] [2 4 2 ~] [2 4 6 4] [2 0 2 ~] [6 4 2 4] [6 7 ~ ~]>*2")
  .scale("D5:minor").s("square").decay(.1).sustain(0).release(.04)
  .lpf(sine.range(1800,6000).slow(6)).delay(".16:.25:.3").room(.15).gain(.32)

const chorusArp = n("<[0,2,4]!2 [3,5,7]!2 [5,7,9]!2 [4,6,8]!2>")
  .scale("D4:major").s("square").decay(.05).sustain(0)
  .lpf(sine.range(1200,5000).slow(8)).gain(.15)

const bridgeArp = n("<[0,2,4]!2 [5,7,9]!2 [2,4,6]!2 [6,8,10]!2>")
  .scale("D4:minor").s("square").decay(.05).sustain(0)
  .lpf(sine.range(1000,4000).slow(6)).gain(.13)

const chorusBass = n("<0!2 3!2 5!2 4!2>").struct("x ~ x [~ x]")
  .scale("D2:major").s("triangle").decay(.15).sustain(.2)
  .lpf(sine.range(200,900).slow(6)).gain(.45)

const bridgeBass = n("<0!2 5!2 2!2 6!2>").struct("x ~ x [~ x]")
  .scale("D2:minor").s("triangle").decay(.15).sustain(.2)
  .lpf(sine.range(180,700).slow(4)).gain(.42)

const chorusDrums = s("bd ~ ~ sd, hh*8").gain(".7 .3")
  .lpf(sine.range(3000,10000).slow(8)).room(.06)

const bridgeDrums = s("bd ~ ~ sd, [hh*4 [hh oh]*2]").gain(".6 .3")
  .lpf(sine.range(2500,8000).slow(6)).room(.1)

const counter = n("<[0 4 7 4]!2 [3 7 3 7]!2 [5 2 5 2]!2 [4 0 4 0]!2>*2")
  .scale("D5:major").s("triangle").decay(.04).sustain(0).gain(.15)

$: stack(
  arrange(
    [i, n("<~ [0 2 4] [0 2 4 7]*2 [0 2 4 7 9 7 4 2]>").scale("D5:major")
      .s("square").decay(.08).sustain(0).lpf(3000).gain(.3)],
    [c, chorusLead],
    [b, bridgeLead],
    [f, chorusLead.lpf(sine.range(2200,9000).slow(8)).gain(.38)]
  ),
  arrange(
    [i, silence],
    [c, chorusArp],
    [b, bridgeArp],
    [f, chorusArp.lpf(sine.range(1400,5500).slow(8)).gain(.17)]
  ),
  arrange(
    [i, n("0").scale("D2:major").s("triangle").decay(.3).sustain(.3).gain(.4)],
    [c, chorusBass],
    [b, bridgeBass],
    [f, chorusBass.lpf(sine.range(220,1000).slow(6)).distort(".5:.9").gain(.48)]
  ),
  arrange(
    [i, s("<~ ~ hh*4 [bd ~ ~ sd, hh*8]>").gain("<.15 .15 .25 .55>")],
    [c, chorusDrums],
    [b, bridgeDrums],
    [f, s("bd [~ bd] ~ sd, hh*8").gain(".7 .3").lpf(sine.range(3500,11000).slow(8)).room(.06)]
  ),
  arrange([i, silence], [c, silence], [b, silence], [f, counter])
)
