// "Clean Getaway" — tense jazz-funk heist soundtrack, slick grooves and rising stakes
setcps(118/60/4)

samples('github:tidalcycles/dirt-samples')

s("bd ~ [~ bd] ~, ~ sd ~ sd:2, hh(5,8), [~ oh]*2")
.bank("RolandTR909").cut(1)
.gain(perlin.range(.65,.92)).room(.18)
.every(4, x => x.fast(1.5).gain(.5))
.stack(
  note("<[c2 ~ eb2 c2] [f1 ~ ab1 f1] [g1 ~ bb1 g1] [ab1 ~ g1 ~]>")
    .s("sawtooth").lpf(sine.range(200,900).slow(8))
    .decay(.15).sustain(.45).release(.05)
    .gain(.7).distort(.12).room(.1),
  chord("<Cm9 Fm9 Gm7 Abmaj7>").dict('ireal').voicing()
    .s("gm_epiano1").struct("[~ x ~ x]*2")
    .lpf(sine.range(1400,3200).slow(12)).room(.5).delay(".2:.14:.25")
    .gain(.38),
  chord("<Cm9 Fm9 Gm7 Abmaj7>").dict('ireal').voicing()
    .s("gm_synth_strings_1").attack(.6).sustain(.6).release(1.5)
    .lpf(sine.range(600,1800).slow(16)).room(.65).gain(.2),
  n("<[0 ~ 4 ~] [~ 3 ~ 7] [5 ~ 2 ~] [~ 4 7 ~]>")
    .scale("C5:minor:pentatonic").s("gm_vibraphone")
    .lpf(sine.range(1200,4800).slow(6))
    .room(.55).delay(".28:.18:.35").gain(.3)
    .sometimes(x => x.add(note(12)).gain(.18))
    .every(4, x => x.rev()),
  s("rim:2(5,16,<0 2>)").bank("RolandTR909")
    .gain(perlin.range(.12,.3)).room(.4).pan(perlin.range(.2,.8)),
  note("<c1 f1 g1 ab1>").s("sine")
    .gain(.25).lpf(90).room(.2).decay(.8).sustain(.7).release(.5)
).reset("<16 16 8 16>")
