// 🎵 "Parallel Lives" — Dual Harmonizing Songs in Relative Keys
// Song A (left): C major piano ballad | Song B (right): A minor folk
// At 84 BPM: intro(4) + merge(8) + bloom(8) + outro(4) = 24 bars ≈ 69s

setcpm(84 / 4)

// --- Shared glue: vinyl haze + sub-bass drone ---
$: s("vinyl:2").gain(.12).room(1).loop(1).lpf(2400)

$: note("<c1 a0 f0 g0>/4")
   .s("sine")
   .attack(2).decay(1).sustain(.8).release(2)
   .gain(.18).room(.5)
   .lpf(160)

// ═══════════════════════════════════════════
// SONG A — "Daydream" — C major, panned left
// ═══════════════════════════════════════════

// Song A drums: brushed 808, gentle swing

  [4,  s("bd ~ ~ ~, ~ ~ [~ hh] ~").bank("RolandTR808").gain(".42 .32").room(.4).pan(.3)],
  [8,  s("bd ~ [~ bd] ~, ~ cp ~ ~, hh*4").bank("RolandTR808").gain(perlin.range(.5,.7)).room(.3).pan(.3).swing(3).cut(1)],
  [8,  s("bd ~ bd ~, ~ [cp,sd:1] ~ cp, [hh oh hh hh]*2").bank("RolandTR808").gain(".68 .6").room(.3).pan(.3).swing(3).cut(1).every(4, x => x.fast(2).gain(.4))],
  [4,  s("bd ~ ~ ~").bank("RolandTR808").gain(saw.slow(4).range(.4,.06)).room(.55).pan(.3)]

  [8,  chord("<Cmaj7 Em7 Fmaj7 G7>").dict('ireal').voicing().s("gm_epiano1").struct("~ [x ~] ~ [x ~]").lpf(1800).room(.5).delay(".25:.15:.3").gain(.38).pan(.25)],

  [4,  chord("<Cmaj7 Fmaj7>/2").dict('ireal').voicing().s("gm_epiano1").struct("x ~ ~ ~").lpf(1100).room(.8).delay(".5:.35:.6").gain(saw.slow(4).range(.38,.05)).pan(.25)]
)

  [8,  n("<[0 2 4 ~] ~ [5 4 2 ~] ~>").scale("C5:major").s("gm_flute").lpf(sine.range(2200,5000).slow(8)).room(.55).delay(".25:.2:.35").gain(.38).vib(4).vibmod(.5).pan(.2)],
  [8,  n("<[0 ~ 4 ~] [5 ~ ~ ~] [7 ~ 5 ~] [4 ~ ~ ~]>").scale("C5:major").s("gm_flute").lpf(sine.range(2500,5500).slow(6)).room(.5).delay(".2:.15:.3").gain(.42).vib(3).vibmod(.6).pan(.2).sometimes(x => x.add(note(12)))],
  [4,  n("<[0 ~ ~ ~] ~>/1").scale("C5:major").s("gm_flute").room(.9).delay(".6:.5:.7").gain(saw.slow(4).range(.3,.05)).lpf(2000).vib(4).vibmod(.8).pan(.2)]
