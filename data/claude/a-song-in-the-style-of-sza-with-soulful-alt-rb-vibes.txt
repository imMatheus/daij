// 🎵 "Glass Garden" — Soulful Alt-R&B / SZA-inspired neo-soul

setcps(0.3)

stack(
  // soft 808s — verse/chorus alternation with trap hats
  s("<[bd ~ ~ bd, ~ cp ~ ~, hh(5,8)] [bd ~ [~ bd] ~, ~ [cp,sd] ~ [~ cp], hh*16]>")
    .bank("RolandTR808").swing(3)
    .gain(perlin.range(.65, .88))
    .room(.4).lpf(3200).cut(1)
    .every(4, x => x.fast(2).gain(.4)),

  // deep bass with filter breath
  note("<[bb1 ~ ~ bb1] [gb1 ~ ~ ~] [db2 ~ ~ db2] [ab1 ~ ~ ~]>")
    .s("gm_acoustic_bass")
    .decay(.35).sustain(.5).room(.2)
    .gain(.78).lpf(sine.range(280, 550).slow(16)),

  // neo-soul EP — rhythmic voicings
  chord("<Bbm9 Gbmaj7 Dbmaj9 Ab7>")
    .dict('ireal').voicing()
    .s("gm_epiano2")
    .struct("<[~ [x ~] ~ [x ~]] [[~ x]*2] [x ~ [~ x] ~] [[~ x]*2]>")
    .lpf(sine.range(1100, 2900).slow(12))
    .room(.55).delay(".25:.18:.38")
    .gain(.46).speed(.98)
    .sometimes(x => x.crush(14)),

  // warm string wash
  chord("<Bbm9 Gbmaj7 Dbmaj9 Ab7>")
    .dict('ireal').voicing()
    .s("gm_synth_strings_1")
    .attack(1.2).sustain(.65).release(2.5)
    .lpf(sine.range(500, 1300).slow(16))
    .room(.85).gain(.24),

  // floating pentatonic melody
  n("<[~ 0 ~ 3] [4 ~ 2 ~] [~ 1 ~ 4] [3 ~ ~ 0]>")
    .scale("Bb4:minor:pentatonic")
    .s("gm_epiano1")
    .lpf(sine.range(900, 3400).slow(10))
    .room(.6).delay(".35:.25:.5")
    .gain(.38).vib(3).vibmod(.4)
    .rarely(x => x.add(note(12))),

  // sparse flute counter-melody
  n("<[~ ~ ~ ~] [~ ~ 4 ~] [~ 3 ~ ~] [4 ~ ~ 2]>")
    .scale("Bb4:minor:pentatonic")
    .s("gm_flute")
    .room(.7).delay(".45:.3:.6")
    .gain(.26).lpf(4200)
    .vib(5).vibmod(.6)
    .attack(.05).sustain(.55).release(.4),

  // sine sub warmth
  note("<bb0 gb0 db1 ab0>/4")
    .s("sine")
    .attack(2).decay(1.5).sustain(.75).release(2.5)
    .gain(.16).room(.5).lpf(160)
)
