setcps(0.125)

note("c1").s("sine")
.attack(6).sustain(.95).release(8)
.gain(.2).room(.7).roomsize(10).lpf(100)
.stack(
  note("<c2 c2 bb1 c2>").s("sawtooth")
  .attack(5).sustain(.6).release(6)
  .gain(.04).room(.9).roomsize(10)
  .lpf(perlin.range(120,400).slow(8))
  .pan(sine.slow(13).range(.25,.75)),

  note("<[c3,g3] [ab2,eb3] [bb2,f3] [g2,d3]>").s("triangle")
  .attack(4).sustain(.5).release(5)
  .gain(.035).room(.95).roomsize(10)
  .lpf(sine.slow(20).range(250,800))
  .delay(".6:.45:.75")
  .pan(sine.slow(9).range(.2,.8)),

  chord("<Cm11 Ab^9 Bbsus4 Cm9>").dict('ireal').voicing()
  .s("gm_synth_strings_1")
  .attack(5).sustain(.65).release(6)
  .gain(sine.slow(24).range(.05,.16))
  .room(.95).roomsize(10)
  .lpf(sine.slow(28).range(250,700)),

  n("<[~ ~ 0 ~] [~ ~ ~ ~] [~ ~ ~ 3] [~ ~ ~ ~]>")
  .scale("C5:minor:pentatonic").s("gm_epiano2")
  .attack(.4).decay(1.2).sustain(.15).release(4)
  .gain(.09).room(.95).roomsize(10)
  .delay(".8:.55:.9")
  .lpf(perlin.range(500,1800).slow(5))
  .pan(perlin.range(.1,.9)),

  s("vinyl:2").gain(.06).room(1).lpf(1400).loop(1)
)
