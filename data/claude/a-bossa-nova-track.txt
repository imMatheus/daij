  [8,  s("[~ rimshot ~ rimshot] [rimshot ~ rimshot ~]")
       .bank("RolandTR808").gain(.35).room(.3).pan(sine.slow(8))],

  [16, s("[~ rimshot ~ rimshot] [rimshot ~ rimshot ~], [hh hh]*4, ~ ~ ~ [cp:1 ~]")

       .every(8, x => x.sometimes(y => y.gain(.2)))],
  [16, s("[~ rimshot ~ rimshot] [rimshot ~ rimshot ~], [hh hh]*4")
       .bank("RolandTR808").gain(".4 .35 .38 .35").room(.25).cut(1)],
  [16, s("[~ rimshot ~ rimshot] [rimshot ~ rimshot ~], [hh hh]*4, ~ ~ ~ [cp:1 ~]")

  [8,  s("[~ rimshot ~ rimshot] [rimshot ~ rimshot ~]")

  [8,  s("[~ rimshot ~ ~] [~ ~ rimshot ~]")
       .bank("RolandTR808").gain(saw.slow(8).range(.35,.08)).room(.5)]
)
