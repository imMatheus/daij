// "Aurora Veil" — Ambient ethereal soundtrack for the northern lights
setcps(0.3)
stack(
  // deep sub drone
  note("<Eb2!3 Eb2!3 Bb1!2 Eb2!2 Ab1!2 Db2 Eb2!3 Bb1!2 Eb2!2>")
    .s("sine").attack(2).release(4)
    .gain("<.15!3 .3!7 .25!3 .3!7>")
    .lpf(130).room(.6).sz(4),
  // sweeping aurora pad
  note("<[Eb3,Gb3,Bb3]!3 [Eb3,Gb3,Bb3]!3 [Bb2,Db3,F3]!2 [Eb3,Gb3,Bb3]!2 [Ab2,B2,Eb3]!2 [Db3,F3,Ab3] [Eb3,Gb3,Bb3]!3 [Bb2,Db3,F3]!2 [Eb3,Gb3,Bb3]!2>")
    .s("supersaw").detune(.15).attack(1).release(2)
    .lpf(sine.range(250,2200).slow(16)).lpq(1.5)
    .gain("<0 .06 .12 .15!5 .12!2 .1!3 .15!5 .12!2>")
    .room(.9).sz(5).delay(.35).delaytime(.33).delayfb(.5)
    .pan(sine.slow(13)),
  // shimmering arpeggios
  n("{0 2 4 7 4 2}%6".add("<0!3 0!3 3!2 0!2 -2!2 2 0!3 3!2 0!2>"))
    .scale("Eb4:minor").s("triangle")
    .attack(.1).release(1.8)
    .gain("<0!3 .1!7 .06!3 .1!7>")
    .lpf(perlin.range(600,4000)).room(.8).sz(3)
    .delay(.5).delaytime(.25).delayfb(.55)
    .pan(sine.range(.15,.85).slow(7)),
  // crystalline FM bells
  n("<~ ~ ~ 0 ~ 2 4 ~ 6 4 ~ ~ ~ 0 ~ 2 4 ~ 6 4>")
    .scale("Eb5:minor").s("sine")
    .fm(2).fmh(3.01).fmdecay(.7).fmsustain(0)
    .gain("<0!3 .06!7 0!3 .06!7>")
    .release(3).room(.95).sz(5)
    .delay(.4).delaytime(.2).delayfb(.6)
    .pan(rand)
)
