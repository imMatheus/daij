  s("cp(5,8,<0 2>) [~ cp] cp(3,8,<1 3>) ~")
    .gain(.68).room(.32).delay(".15:.1:.25")
    .pan(perlin.range(.2,.8))

  note("<[a3 c4 e4 a4 e4 c4] [bb3 d4 f4 bb4 f4 d4] [d3 f3 a3 d4 a3 f3] [e3 ab3 b3 e4 b3 ab3]>")
    .s("gm_acoustic_guitar_nylon")
    .lpf(sine.range(1200,4000).slow(10))
    .room(.5).delay(".2:.14:.32")
    .gain(.44).decay(.1).sustain(.22),
  n("<[~ ~ 4 ~] [7 ~ ~ 5] [~ 3 ~ ~] [~ ~ 0 ~]>")

  note("<a0 bb0 d1 e0>/4").s("sine")
    .attack(1.5).decay(1).sustain(.8).release(2)
    .gain(.15).lpf(125).room(.4)
