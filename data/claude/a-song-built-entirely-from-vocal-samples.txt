// 🎵 "Tongues of Glass" — Ethereal vocal-only electronica
// At 88 BPM: intro(4) + verse(8) + chorus(8) + bridge(4) + outro(4) = 28 bars ≈ 76s

setcpm(88 / 4)

// Breathy choir shimmer — vocal hi-hat texture
$: note("d6*16").s("gm_choir_aahs")
   .decay(.02).sustain(0)
   .gain(perlin.range(.03,.22))
   .pan(sine.slow(5))
   .hpf(4500).room(.3).cut(1)

// Vocal percussion — synth voice as kick + snare
$: arrange(
  [4,  note("d1 ~ ~ ~").s("gm_synth_voice").decay(.09).sustain(0).gain(.52).room(.45).lpf(320)],
  [8,  note("[d1 ~ ~ d1], [~ ~ [~ a3] ~]").s("gm_synth_voice").decay(.07).sustain(0).gain(".82 .68").room(.3).swing(3)],
  [8,  note("[d1 ~ d1 ~], [~ [~ a3] ~ [~ a3]]").s("gm_synth_voice").decay(.07).sustain(0).gain(".9 .74").room(.25).swing(3).every(4, x=>x.fast(2).gain(.4))],
  [4,  note("d1 ~ ~ ~, ~ ~ [~ a3] ~").s("gm_synth_voice").decay(.1).sustain(0).gain(".62 .5").room(.5)],
  [4,  note("d1 ~ ~ ~").s("gm_synth_voice").decay(.12).sustain(0).gain(saw.slow(4).range(.48,.06)).room(.6)]
)

// Voice oohs bass — walking low end
$: arrange(
  [4,  silence],
  [8,  note("<[d2 ~ ~ d2] [bb1 ~ ~ ~] [g1 ~ ~ g1] [a1 ~ ~ ~]>").s("gm_voice_oohs").decay(.32).sustain(.48).room(.22).gain(.72).lpf(850)],
  [8,  note("<[d2 ~ d2 ~] [bb1 ~ ~ bb1] [g1 ~ g1 ~] [a1 ~ ~ a1]>").s("gm_voice_oohs").decay(.25).sustain(.55).room(.2).gain(.78).lpf(950)],
  [4,  note("<d2 ~ bb1 ~ g1 ~ a1 ~>/2").s("gm_voice_oohs").decay(.5).sustain(.35).room(.35).gain(.55).lpf(650).speed(.97)],
  [4,  silence]
)

// Choir aahs pad chords — harmonic bed
$: arrange(
  [4,  chord("<Dm9 Am7>/2").dict('ireal').voicing().s("gm_choir_aahs").attack(.8).decay(1).sustain(.7).release(2.2).lpf(1200).room(.72).gain(.28)],
  [8,  chord("<Dm9 Bbmaj7 Gm7 A7>").dict('ireal').voicing().s("gm_choir_aahs").attack(.35).decay(.65).sustain(.7).release(1.2).lpf(sine.range(1400,2800).slow(12)).room(.56).gain(.4)],
  [8,  chord("<Dm9 Bbmaj7 Gm7 A7>").dict('ireal').voicing().s("gm_choir_aahs").attack(.22).decay(.5).sustain(.76).release(1).lpf(sine.range(1800,3400).slow(10)).room(.5).gain(.46)],
  [4,  chord("<Dm9 Bbmaj7 Gm7 A7>/2").dict('ireal').voicing().s("gm_choir_aahs").attack(.7).sustain(.6).release(2.5).lpf(sine.range(600,1500).slow(6)).room(.78).gain(.33)],
  [4,  chord("<Dm9 Am7>/2").dict('ireal').voicing().s("gm_choir_aahs").attack(1.1).sustain(.5).release(3.5).lpf(sine.range(400,950).slow(6)).room(.88).gain(saw.slow(4).range(.28,.04))]
)

// Voice oohs melody — singing pentatonic line
$: arrange(
  [4,  silence],
  [8,  n("<[~ 0 ~ 2] [4 ~ 2 ~] [~ 3 ~ 5] [4 ~ ~ 0]>").scale("D5:minor:pentatonic").s("gm_voice_oohs").lpf(sine.range(900,3000).slow(8)).room(.55).delay(".3:.2:.4").gain(.35).vib(4).vibmod(.4).rarely(x=>x.add(note(12)))],
  [8,  n("<[0 ~ 2 4] [~ 5 4 ~] [2 ~ 0 ~] [~ 4 2 0]>").scale("D5:minor:pentatonic").s("gm_voice_oohs").lpf(sine.range(1200,3800).slow(7)).room(.48).delay(".25:.18:.35").gain(.42).vib(3).vibmod(.5).sometimes(x=>x.add(note(12)))],
  [4,  n("<[~ 0 ~ ~] [~ ~ 2 ~]>").scale("D5:minor:pentatonic").s("gm_voice_oohs").room(.82).delay(".55:.4:.65").gain(.26).lpf(1400).vib(5).vibmod(.6)],
  [4,  silence]
)

// Synth voice counter-melody — chorus and bridge
$: arrange(
  [4,  silence],
  [8,  silence],
  [8,  n("<[~ ~ 4 ~] [~ 5 ~ ~] [4 ~ ~ 2] [~ ~ 0 ~]>").scale("D4:minor:pentatonic").s("gm_synth_voice").room(.62).delay(".45:.3:.55").gain(.24).lpf(2100).vib(4).vibmod(.5)],
  [4,  n("<[~ 2 ~ ~] [~ ~ 4 ~]>").scale("D4:minor:pentatonic").s("gm_synth_voice").room(.82).delay(".55:.42:.68").gain(.2).lpf(1500).vib(5).vibmod(.7)],
  [4,  silence]
)

// Deep voice oohs sub-drone — foundation
$: note("<d1 bb0 g0 a0>/4")
   .s("gm_voice_oohs")
   .attack(2.2).decay(1.2).sustain(.78).release(2.5)
   .gain(.16).room(.6)
   .lpf(260)
