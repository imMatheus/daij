// "Mouth Music" — Glitchy vocal collage from chopped speech and mouth sounds
samples('github:tidalcycles/dirt-samples')
setcps(0.5)

const intro = s("speech:4 ~ speech:6 ~")
  .slow(2)
  .begin("<.04 .1>")
  .end("<.18 .22>")
  .speed("<.9 -.72>")
  .lpf(sine.slow(5).range(700,2000))
  .room(.96)
  .delay(".46:.26:.4")
  .gain(.14)
  .stack(
    s("mouth:0")
      .striate(12)
      .slow(2)
      .hpf(sine.slow(3).range(2600,6200))
      .room(.8)
      .gain(perlin.range(.018,.048)),
    chord("<Dm9>/2").voicing()
      .s("alphabet:5")
      .attack(.12).decay(.6).sustain(.58).release(1.6)
      .lpf(sine.slow(4).range(750,1500))
      .room(.86)
      .delay(".26:.18:.28")
      .gain(.19)
  )

      .gain(.4).pan(sine.slow(5))

  .slow(2)
  .begin("<.02 .14>")
  .end("<.2 .26>")
  .speed("<-1 .8 -.88 1>")
  .room(.97).delay(".5:.36:.42")
  .lpf(sine.slow(5).range(650,1700))
  .gain(.15)
  .stack(
    s("~ mouth:4 ~ [~ mouth:1], ~ ~ mouth:5*2 ~")
      .cut(1).swing(4).hpf(280).room(.34).gain(".38 .26"),
    s("numbers:5")
      .slice(8,"0 5 2 6 1 7 3 4").fast(2)
      .room(.56).delay(".3:.18:.22")

      .gain(.09).reset("<x ~ x ~>"),
    note("<d2 ~ bb1 ~>/2")

      .decay(.42).sustain(.38).release(.38)

      .attack(.14).decay(.65).sustain(.52).release(1.4)
      .lpf(sine.slow(3).range(750,1900))
      .room(.88).delay(".42:.28:.38")

const outro = s("speech:4 ~ speech:6 ~")
  .slow(2)
  .begin("<.05 .1>")
  .end("<.2 .17>")
  .speed("<1 -.78>")
  .room(.97).delay(".48:.32:.4")
  .lpf(sine.slow(6).range(800,1900))
  .gain(saw.slow(4).range(.11,.02))
  .stack(
    s("mouth:0")
      .striate(8).slow(2)
      .hpf(sine.slow(5).range(2200,5600))
      .room(.76).gain(saw.slow(4).range(.032,.005)),
    chord("<Dm9 Gm7>/2").voicing()
      .s("alphabet:5")
      .attack(.18).decay(.8).sustain(.5).release(1.8)
      .lpf(1100).room(.88)
      .gain(saw.slow(4).range(.2,.02)),
    s("~ ~ mouth:5 ~")
      .cut(1).hpf(3200).room(.46)
      .gain(saw.slow(4).range(.12,.015))
  )

arrange(
  [2, intro],
  [13, chorus],
  [5, bridge],
  [13, chorus],
  [2, outro]
)
