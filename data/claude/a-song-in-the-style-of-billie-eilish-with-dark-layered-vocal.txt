// 🎵 "Hollow" — Dark Whisper-Pop (Billie Eilish-style)
// At 68 BPM: intro(8) + verse(16) + chorus(16) + verse2(12) + chorus2(16) + outro(8) = 76 bars ≈ 2:41

setcpm(68 / 4)

// sub-bass drone — chest-rattling darkness
$: note("<db1 ab0 bb0 gb0>/4")
   .s("sine")
   .attack(2).decay(1).sustain(.85).release(3)
   .gain(.22).room(.4)
   .lpf(100)

  [16, s("bd ~ ~ ~, ~ ~ ~ [~ sd:1]")
       .bank("RolandTR808").gain(".75 .5").room(.4).lpf(400).speed(.75)],
  [16, s("bd ~ ~ bd, ~ ~ [sd:1 ~] ~, ~ ~ ~ [~ cp:1]")
       .bank("RolandTR808").gain(".82 .6 .55").room(.35).lpf(500).speed(.78)
       .every(4, x => x.sometimes(y => y.gain(.3)))],
  [12, s("bd ~ ~ ~, ~ ~ ~ [~ sd:1]")
       .bank("RolandTR808").gain(".7 .45").room(.45).lpf(380).speed(.73)],
  [16, s("bd ~ ~ bd, ~ ~ [sd:1 ~] ~, ~ ~ ~ [~ cp:1]")
       .bank("RolandTR808").gain(".85 .65 .58").room(.35).lpf(500).speed(.78)],
  [8,  s("bd ~ ~ ~")
       .bank("RolandTR808").gain(saw.slow(8).range(.6,.05)).room(.6).lpf(300).speed(.7)]
)
