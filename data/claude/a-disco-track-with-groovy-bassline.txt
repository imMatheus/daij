// "Velvet Strobe" — Groovy disco with bouncing synth bassline and shimmering strings
samples('github:tidalcycles/dirt-samples')
setcps(120/240)

const sweep = sine.slow(6).range(700, 2800)
const glow = perlin.slow(8).range(1800, 5500)
const space = sine.slow(10).range(.18, .4)

const pump = s("bd*4").bank("RolandTR909")
  .duckorbit("2:3:4").duckattack("0.28:0.2:0.12").duckdepth("1:.72:.35").postgain(0)

const intro = stack(
  s("bd*4").bank("RolandTR909").lpf(700).gain(saw.slow(2).range(.35, .85)),
  s("hh*8").bank("RolandTR909").gain(saw.slow(2).range(.008, .028)),
  cat(
    "c2 ~ c3 ~ ~ ~ eb2 ~",
    "f1 ~ f2 ~ ~ ~ ab1 ~"
  ).note().s("sawtooth")
    .lpf(saw.slow(2).range(250, 800)).lpenv(3).lpd(.1)
    .decay(.12).sustain(.2).gain(saw.slow(2).range(.25, .65))
)

const chorus = stack(
  pump,
  cat(
    "c2 ~ c3 c2 eb2 ~ g2 c2 ~ c3 ~ eb2 f2 ~ g2 ~",
    "f1 ~ f2 f1 ab1 ~ c2 f1 ~ f2 ~ ab1 bb1 ~ c2 ~",
    "ab1 ~ ab2 ab1 c2 ~ eb2 ab1 ~ ab2 ~ c2 db2 ~ eb2 ~",
    "g1 ~ g2 g1 bb1 ~ d2 g1 ~ g2 ~ bb1 c2 ~ d2 ~"
  ).note().s("sawtooth")

    .lpf(sweep).lpenv(5).lpd(.08).lpq(3).hpf(35)
    .gain(.82).orbit(2),
  cat("c1 ~ ~ ~ ~ ~ ~ ~", "f1 ~ ~ ~ ~ ~ ~ ~", "ab1 ~ ~ ~ ~ ~ ~ ~", "g1 ~ ~ ~ ~ ~ ~ ~")
    .note().s("sine").decay(.3).sustain(.55).gain(.45).orbit(2),
  cat("[c4,eb4,g4,bb4]", "[f3,ab3,c4,eb4]", "[ab3,c4,eb4,g4]", "[g3,b3,d4,f4]")
    .note().s("gm_clavinet")
    .struct("<[~ x ~ x ~ x x ~] [~ x x ~ ~ x ~ x]>")
    .attack(.003).decay(.08).sustain(.04).release(.1)
    .lpf(perlin.range(1400, 3000).slow(5)).drive(.14)
    .room(.22).delay(.08).delaytime(.125).delayfeedback(.14)
    .gain(.22).orbit(3),
  cat("[c4,eb4,g4,bb4]", "[f3,ab3,c4,eb4]", "[ab3,c4,eb4,g4]", "[g3,b3,d4,f4]")
    .note().s("gm_synth_strings_1")
    .attack(.2).decay(.6).sustain(.62).release(1.2)
    .lpf(sine.range(1100, 3200).slow(7))
    .room(space).delay(.14).delaytime(.25).delayfeedback(.26)
    .gain(.2).orbit(4),
  cat(
    "~ g5 ~ bb5 ~ c6 bb5 ~",
    "~ ab5 ~ c6 ~ eb6 c6 ~",
    "~ eb5 ~ g5 ~ ab5 g5 ~",
    "~ d5 ~ g5 ~ b5 g5 ~"
  ).note().s("triangle")
    .attack(.01).decay(.12).sustain(.1).release(.22)
    .lpf(glow).chorus(.32).vib(5.5).vibmod(.14)
    .room(.36).delay(.22).delaytime(.25).delayfeedback(.34)
    .gain(.1).orbit(4),
  s("bd*4").bank("RolandTR909").lpf(1100).gain(.9).swing(4),
  s("cp").bank("RolandTR909")
    .struct("~ ~ ~ ~ x ~ ~ ~ ~ ~ ~ ~ x ~ ~ ~")
    .lpf(2200).room(.2).gain(.32).swing(4),
  s("hh").bank("RolandTR909")
    .euclid(13, 16)
    .begin("<.01 .02 .015 .03>").end("<.05 .06 .065 .08>")
    .bpf(glow).bpq(5)
    .delay(.08).delaytime(.125).delayfeedback(.16)
    .room(.16).gain(".026 .016 .022 .014 .024 .018 .022 .015 .026 .018 .024 .015 .026")
    .swing(4),
  s("oh").bank("RolandTR909")
    .struct("~ ~ ~ x ~ ~ ~ x ~ ~ ~ x ~ ~ ~ x")
    .begin(.04).end(.16).hpf(5800).room(.16).gain(.11).swing(4)
