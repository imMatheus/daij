// 🎵 "Loop Temple" — Psychedelic Soul with Layered Live Looping (Tash Sultana Style)
// At 88 BPM: intro(8) + verse(16) + chorus(16) + verse2(16) + chorus2(16) + bridge(8) + outro(8) = 88 bars ≈ 4:00

setcpm(88 / 4)

// Vinyl hiss — room tone, always present
$: s("vinyl:2").gain(.12).room(1).loop(1).lpf(2400)

// Percussion — beatbox-sparse to full kit, looping in layers
$: arrange(
  [8,  s("bd ~ ~ ~, ~ ~ [hh hh] ~")
       .bank("RolandTR808").gain(".55 .45").room(.45).swing(3)],
  [16, s("bd ~ [~ bd] ~, ~ sd ~ ~, hh*8")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(perlin.range(.7,.88)).room(.3)],
  [16, s("bd ~ [~ bd] bd, ~ [sd,cp] ~ [sd ~], hh*16, ~ ~ oh ~")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(".88 .82 .85 .8").room(.25)
       .every(4, x => x.press())],
  [16, s("bd ~ [~ bd] ~, ~ sd ~ ~, hh*8, ~ ~ ~ oh")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(perlin.range(.72,.88)).room(.3)
       .sometimes(x => x.gain(.4))],
  [16, s("bd ~ [~ bd] bd, ~ [sd,cp] ~ [sd ~], hh*16, ~ ~ oh ~")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(".9 .84 .88 .82").room(.25)
       .every(3, x => x.fast(2).gain(.5))],
  [8,  s("bd ~ ~ ~, ~ ~ sd ~, [hh oh]*2")
       .bank("RolandTR808").swing(4)
       .gain(".68 .58").room(.5)
       .every(2, x => x.rev())],
  [8,  s("bd ~ ~ ~")
       .bank("RolandTR808").swing(3)
       .gain(saw.slow(8).range(.55,.06)).room(.55)]
)

// Bass loop — fingerstyle groove locked to the kick
$: arrange(
  [8,  silence],
  [16, note("<[d2 ~ d2 ~] [~ f2 ~ d2] [c2 ~ ~ c2] [~ a1 ~ ~]>")
       .s("gm_acoustic_bass")
       .decay(.2).sustain(.45).release(.15)
       .gain(.82).room(.25).lpf(2800)
       .swing(3)],
  [16, note("<[d2 f2 ~ a2] [~ d2 f2 ~] [c2 ~ c2 e2] [a1 ~ c2 a1]>")
       .s("gm_acoustic_bass")
       .decay(.18).sustain(.5).release(.12)
       .gain(.86).room(.2).lpf(3200)
       .swing(3)
       .sometimes(x => x.speed(1.02))],
  [16, note("<[d2 ~ d2 ~] [~ f2 ~ d2] [c2 ~ ~ c2] [~ a1 ~ ~]>")
       .s("gm_acoustic_bass")
       .decay(.2).sustain(.45).release(.15)
       .gain(.8).room(.25).lpf(2800)
       .swing(3)
       .rarely(x => x.add(note(12)).gain(.5))],
  [16, note("<[d2 f2 ~ a2] [~ d2 f2 ~] [c2 ~ c2 e2] [a1 ~ c2 a1]>")
       .s("gm_acoustic_bass")
       .decay(.15).sustain(.5).release(.1)
       .gain(.88).room(.2).lpf(3500)
       .swing(3)
       .every(4, x => x.fast(2).gain(.45))],
  [8,  note("<d2 ~ c2 ~ a1 ~ ~ ~>/2")
       .s("gm_acoustic_bass")
       .decay(.4).sustain(.4).release(.3)
       .gain(.62).room(.4).lpf(2200)],
  [8,  silence]
)

// Rhythm guitar — clean with wah-like filter sweep, looped in
$: arrange(
  [8,  silence],
  [16, silence],
  [16, chord("<Dm9 Dm9 Cmaj7 Am7>")
       .dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("[~ x] [x ~] [~ x] [x ~]")
       .lpf(sine.range(1000,4200).slow(4))
       .gain(.42).room(.4).delay(".2:.15:.25")
       .swing(3)],
  [16, chord("<Dm9 Fmaj7 Cmaj7 Am7>")
       .dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("~ [x x] ~ [x ~]")
       .lpf(sine.range(800,3800).slow(5))
       .gain(.38).room(.4).delay(".2:.15:.25")
       .swing(3)
       .sometimes(x => x.gain(.2))],
  [16, chord("<Dm9 Fmaj7 Cmaj7 Am7>")
       .dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("[~ x]*4")
       .lpf(sine.range(1200,5000).slow(3))
       .gain(.46).room(.35).delay(".15:.12:.2")
       .swing(3)
       .every(4, x => x.fast(2).gain(.3))],
  [8,  chord("<Dm9 Am7>/2")
       .dict('ireal').voicing()
       .s("gm_electric_guitar_clean")
       .struct("x ~ ~ ~")
       .lpf(2000).room(.65).delay(".4:.3:.5")
       .gain(.34)],
  [8,  silence]
)

// Lead guitar — soulful pentatonic, heavy delay, enters late like a live loop
$: arrange(
  [8,  silence],
  [16, silence],
  [16, n("<[~ 0 ~ 2] [4 ~ 2 ~] [~ 5 ~ 4] [2 ~ 0 ~]>")
       .scale("D5:minor:pentatonic")
       .s("gm_overdriven_guitar")

       .room(.6).delay(".35:.25:.4").gain(.34)
       .vib(4).vibmod(.5)
       .swing(3)
       .rarely(x => x.add(note(12)))],
  [16, n("<[~ 0 ~ 2] [4 ~ 2 ~] [~ 5 ~ 4] [2 ~ 0 ~]>")
       .scale("D5:minor:pentatonic")
       .s("gm_overdriven_guitar")
       .lpf(sine.range(1500,4000).slow(10))
       .room(.6).delay(".35:.25:.4").gain(.32)
       .vib(4).vibmod(.5)
       .swing(3)],
  [16, n("<[0 2 4 5] [~ 7 5 ~] [4 ~ 2 4] [5 7 ~ 0]>")
       .scale("D5:minor:pentatonic")
       .s("gm_overdriven_guitar")
       .lpf(sine.range(2000,5500).slow(6))
