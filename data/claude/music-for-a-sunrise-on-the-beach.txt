// "Golden Horizon" — Warm ambient sunrise with shimmering arpeggios and soft pads
setcps(0.4)

let pad = note("<[d3,f#3,a3,d4] [a2,c#3,e3,a3] [b2,d3,f#3,b3] [g2,b2,d3,g3]>")
  .s("sawtooth").att(1.5).rel(3).sus(.8).dec(.8)
  .lpf(perlin.range(400,1800).slow(6)).lpq(1.5)
  .room(.85).sz(5).delay(.25).dfb(.4).dt(.4)
  .gain(.25).vib(2).vibmod(.06).chorus(.35)

let arp = n("0 2 4 6 4 7 4 2").every(3, rev)
  .scale("D4:major:pentatonic").s("triangle")
  .att(.01).rel(.8).sus(.1).dec(.4)
  .lpf(sine.slow(6).range(1500,6000))
  .delay(.5).dfb(.6).dt(.375)
  .room(.7).gain(.17).pan(sine.range(.2,.8).slow(5))
  .juxBy(.5, x=>x.add(7).gain(.1))

let shimmer = n("<[0,4] [2,6] [4,7] [6,2]>")
  .scale("D5:major:pentatonic").s("sine")
  .att(.3).rel(2.5).sus(.05).dec(.6)
  .lpf(perlin.range(2000,7000).slow(5))
  .room(.95).sz(8).delay(.6).dt(.33).dfb(.5)
  .gain(.06).pan(perlin.range(.1,.9))

let sub = note("<d2 a1 b1 g1>")
  .s("sine").lpf(120).gain(.28)
  .att(.8).rel(1.5)

let waves = s("pink").lpf(perlin.range(80,350).slow(8))
  .gain(.03).room(.9).sz(6)

arrange(
  [2, pad.gain(.06).lpf(600).stack(waves.gain(.015))],
  [10, pad.stack(arp).stack(shimmer).stack(sub).stack(waves)],
  [4, pad.gain(.2).lpf(sine.range(500,1600).slow(4)).stack(shimmer.slow(2).gain(.04)).stack(sub.gain(.15))],
  [12, pad.stack(arp).stack(shimmer).stack(sub).stack(waves)]
)
