// "Echoes in Time" — Contemplative canon where melody chases itself through fog
setcps(21/70)
let mel = n("0 2 3 4 3 2 -1 0").scale("D4:dorian")
arrange(
  [2, mel.slow(2).s("triangle").att(.15).rel(.8).gain(.22)
    .lpf(1200).room(.7).rsize(3).delay(".3:.2:.5")],
  [8, mel.slow(4).s("triangle").att(.03).rel(.5)
    .echoWith(3, 1, (p, i) =>
      p.transpose([12, -12, 7][i-1])
      .velocity([.7, .55, .4][i-1])
      .pan([.7, .3, .6][i-1])
      .s(i === 2 ? "triangle" : "sine")
    )
    .lpf(sine.range(700, 4500).slow(8))
    .gain(.28).room(.5).rsize(2).delay(".2:.18:.45")
    .stack(
      n("<0 3 -1 4>").scale("D2:dorian").slow(2)
        .s("sawtooth").lpf(250).lpq(2).gain(.32).rel(.3),
      s("~ cp ~ hh").gain(".07 .05").room(.4).hpf(500)
    )],
  [3, mel.slow(4).transpose(5).s("sine").att(.15).rel(1)
    .echoWith(2, 1, (p, i) =>
      p.transpose(i === 1 ? 12 : -12)
      .velocity(.5 - i * .1).pan(i === 1 ? .3 : .7)
    )
    .gain(.18).room(.85).rsize(3).delay(".4:.22:.55")
    .lpf(sine.range(500, 2500).slow(3))
    .stack(
      n("<3 0>").scale("D2:dorian").s("sine").gain(.18).lpf(200)
    )],
  [8, mel.slow(4).s("triangle").att(.03).rel(.5)
    .echoWith(3, 1, (p, i) =>
      p.transpose([12, -12, 7][i-1])
      .velocity([.75, .6, .45][i-1])
      .pan([.7, .3, .6][i-1])
      .s(i === 2 ? "triangle" : "sine")
    )
    .lpf(sine.range(600, 5500).slow(6))
    .gain(.3).room(.55).rsize(2).delay(".25:.2:.5")
    .stack(
      n("<0 3 -1 4>").scale("D2:dorian").slow(2)
        .s("sawtooth").lpf(280).lpq(2).gain(.35).rel(.3),
      s("~ cp ~ hh").gain(".08 .06").room(.4).hpf(500),
      mel.slow(8).transpose(12).s("sine").gain(.1).att(.1).rel(.6).room(.6).lpf(3000)
    )]
)
