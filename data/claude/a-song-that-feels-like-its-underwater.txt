// "Abyssal Drift" — Deep ambient underwater soundscape with muffled synths and submerged delay
setcps(0.4)

const introDrone = note("[eb1,bb1]")
  .s("sine").attack(2).release(3)
  .lpf(saw.slow(2).range(60, 150))
  .room(.92).rsize(5).rlp(1500).rdim(400)
  .gain(.3).orbit(0);

const chorusDrone = note("<eb1 [eb1 db1] eb1 bb0>".slow(2))
  .s("sine").attack(1.5).release(2.5)
  .lpf(sine.slow(12).range(100, 210))
  .room(.9).rsize(5).rlp(2000).rdim(500)
  .gain(.48).orbit(0);

const bridgeDrone = note("<eb1 gb1>".slow(2))
  .s("sine").attack(2).release(2.5)
  .lpf(130).room(.92).rsize(5).rlp(1500).rdim(400)
  .gain(.32).orbit(0);

const chorusPad = note("<[eb3,gb3,bb3] [db3,f3,ab3] [eb3,gb3,bb3] [ab2,b2,eb3]>".slow(2))
  .s("sawtooth").attack(.7).sustain(.4).release(2)
  .lpf(sine.slow(16).range(250, 500)).lpq(4)
  .phaser(1).phasersweep(1200)
  .chorus(.6).vib(.3).vibmod(.2)
  .room(.88).rsize(6).rlp(2000)
  .gain(.13).orbit(1);

const bridgePad = note("<[eb3,gb3,bb3] [db3,ab3]>".slow(2))
  .s("sawtooth").attack(1).sustain(.3).release(2.5)
  .lpf(sine.slow(8).range(200, 380)).lpq(3)
  .phaser(.5).phasersweep(800)
  .chorus(.7)
  .room(.88).rsize(6).rlp(1500)
  .gain(.06).orbit(1);

const chorusMelody = n("<[0 ~ 2 ~] [4 ~ 3 1] [2 ~ 0 ~] [~ 3 ~ 5]>")
  .scale("Eb3:minor:pentatonic").s("triangle")
  .attack(.2).release(1.2)
  .lpf(sine.slow(8).range(450, 800)).lpq(3)
  .vib(2.5).vibmod(.4)
  .delay("0.55:0.375:0.65")
  .room(.7).rsize(3)
  .gain(.2).orbit(2);

const bridgeBell = n("<2 4 3 0>")
  .scale("Eb4:minor").s("sine")
  .attack(.5).release(2)
  .fm(2).fmh(2.01).fmdecay(1).fmsustain(.2)
  .lpf(450).delay("0.6:0.5:0.7")
  .room(.8).rsize(3)
  .gain(.16).orbit(2);

const chorusBubbles = s("~ hh ~ [~ hh] ~ hh [hh ~] ~")
  .speed(rand.range(1.5, 3)).lpf(1800).pan(rand)
  .delay("0.3:0.2:0.5").room(.5)
  .gain(.1).orbit(3);

const bridgeBubbles = s("~ ~ hh ~ ~ ~ [~ hh] ~")
  .speed(rand.range(1.8, 2.5)).lpf(1200).pan(rand)
  .delay("0.4:0.25:0.6").room(.6)
  .gain(.04).orbit(3);

const chorusKick = s("bd ~ ~ ~ ~ bd ~ ~")
  .lpf(180).room(.55).rsize(4)
  .gain(.28).orbit(4);

const introShimmer = note("<eb5 ~ ~ gb5 ~ ~>".slow(3))
  .s("sine").attack(.5).release(2)
  .tremolosync(3).tremolodepth(.25)
  .lpf(sine.slow(8).range(500, 1200))
  .room(.9).delay("0.6:0.5:0.7")
  .gain(.03).orbit(5);

const shimmer = note("<eb5 ~ gb5 ~ bb5 ~ ab5 ~ gb5 ~ eb5 ~>".slow(3))
  .s("sine").attack(.3).release(1.5)
  .tremolosync(4).tremolodepth(.3)
  .lpf(sine.slow(10).range(700, 1800))
  .room(.88).delay("0.5:0.33:0.6")
  .gain(.06).orbit(5);

stack(
  arrange([2, introDrone], [10, chorusDrone], [4, bridgeDrone], [12, chorusDrone]),
  arrange([2, silence], [10, chorusPad], [4, bridgePad], [12, chorusPad]),
  arrange([2, silence], [10, chorusMelody], [4, silence], [12, chorusMelody]),
  arrange([2, silence], [10, silence], [4, bridgeBell], [12, silence]),
  arrange([2, silence], [10, chorusBubbles], [4, bridgeBubbles], [12, chorusBubbles]),
  arrange([2, silence], [10, chorusKick], [4, silence], [12, chorusKick]),
  arrange([2, introShimmer], [10, shimmer], [4, shimmer], [12, shimmer])
)
