// "Fading Light" — Melancholic ambient piano in D minor
setcps(0.3)
samples('github:tidalcycles/dirt-samples')

let arp = n("<[0 4 2 ~ 4 7 4 ~] [5 0 2 ~ 5 7 5 ~] [2 6 4 ~ 2 4 6 ~] [6 4 2 ~ 0 2 -1 ~]>")
  .scale("D4:minor").s("piano").gain(.5)
  .decay(.5).sustain(.3).release(1)
  .off(1/8, x => x.add(2).gain(.25))
  .delay(.35).delaytime(.33).delayfeedback(.4)
  .room(.6).roomsize(3)
  .lpf(perlin.range(2000, 5000).slow(6))

let pad = chord("<Dm Bb F C>").voicing()
  .s("sawtooth").gain(.15)
  .attack(.6).decay(.4).sustain(.35).release(2)
  .lpf(sine.range(500, 1400).slow(8)).lpq(2)
  .room(.7).roomsize(4)

let bass = note("<d2 bb1 f2 c2>")
  .s("triangle").gain(.25)
  .lpf(300).attack(.05).release(.6).room(.3)

let perc = s("~ sd:3 ~ ~, bd ~ ~ ~, ~@3 [~ hh]")
  .gain(.2).room(.4)
  .sometimesBy(.2, x => x.delay(.2))

let chorus = stack(arp, pad, bass, perc)

let intro = n("[0 ~ ~ 4 ~ ~ 2 ~]").scale("D4:minor").s("piano")
  .gain(.3).release(1.5)
  .delay(.5).delaytime(.4).delayfeedback(.55)
  .room(.8).roomsize(4)

let bridge = stack(
  n("<[4 ~ 7 ~ 6 ~ 4 ~] [5 ~ 7 ~ 9 ~ 7 ~] [4 ~ 2 ~ 0 ~ 2 ~]>")
    .scale("D4:minor").s("piano").gain(.4).release(1.3)
    .delay(.5).delaytime(.5).delayfeedback(.5)
    .room(.8).roomsize(4)
    .lpf(perlin.range(1500, 3000).slow(3)),
  chord("<Gm A>").voicing().s("sawtooth")
    .gain(.12).attack(.8).release(2.5)
    .lpf(600).room(.8).roomsize(4)
)

arrange(
  [1, intro],
  [8, chorus],
  [3, bridge],
  [9, chorus]
)
