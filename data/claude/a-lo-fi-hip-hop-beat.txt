// "2AM Instant Noodles" — dusty lo-fi hip hop, late night introspection
setcpm(73/4)

stack(
  s("vinyl:2").gain(.13).room(1).loop(1).lpf(2300),
  s("bd ~ ~ bd, ~ [cp,sd:1] ~ cp, hh(5,8)")
    .bank("RolandTR808").swing(4)
    .gain(perlin.range(.55,.85)).room(.35).cut(1)
    .every(4, x=>x.fast(2).gain(.4)),
  note("<[d2 ~ ~ d2] [a1 ~ ~ ~] [bb1 ~ ~ bb1] [c2 ~ ~ ~]>")
    .s("gm_acoustic_bass").decay(.3).sustain(.5)
    .room(.2).gain(.74).lpf(850),
  chord("<Dm9 Am7 Bbmaj7 C9>")
    .dict('ireal').voicing()
    .s("gm_epiano2").struct("~ [x ~] ~ [x ~]")
    .lpf(sine.range(900,2100).slow(14))
    .room(.55).delay(".28:.18:.38").gain(.46).speed(.98),
  n("<[~ 0 ~ 2] [4 ~ 3 ~] [~ 1 ~ 4] [3 ~ ~ 0]>")
    .scale("D5:minor:pentatonic")
    .s("gm_epiano2")
    .lpf(perlin.range(800,2500))
    .room(.6).delay(".4:.28:.5").gain(.3)
    .pan(sine.slow(7))
    .rarely(x=>x.add(note(12)))
    .reset("<x@3 x@5>"),
  n("<[~ ~ 3 ~] [~ 4 ~ ~] [3 ~ ~ 1] [~ ~ 0 ~]>")
    .scale("D4:minor:pentatonic")
    .s("gm_vibraphone").room(.7)
    .delay(".5:.35:.6").gain(.22).lpf(3200)
    .pan(sine.slow(9).range(.3,.7))
    .attack(.02).decay(.3).sustain(.4).release(.5),
  note("<d1 a0 bb0 c1>/4").s("sine")
    .attack(1.5).decay(1).sustain(.7).release(2)
    .gain(.17).lpf(160).room(.3)
)
