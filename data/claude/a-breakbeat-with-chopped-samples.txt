// "Chopped & Rewired" — Gritty breakbeat with sliced drum breaks and bass stabs
samples('github:tidalcycles/dirt-samples')
setcps(0.55)
let mainBreak = s("breaks165")
  .slice(16, "0 4 <2 14> [8 1] 5 <10 [12 3]> [7 11] <6 [15 13]>")
  .every(3, rev).sometimesBy(0.3, x => x.speed("<-1 0.5 2>"))
  .gain(0.85).room(0.1)
  .lpf(perlin.range(1500, 10000).slow(4))
  .juxBy(0.4, rev)
let chopLayer = s("breaks125")
  .splice(8, "<[0 3 5 2] [7 1 4 6]>")
  .gain(0.4).hpf(500)
  .delay("0.25:0.17:0.35")
  .pan(sine.range(0.3, 0.7))
let drums = s("bd:4(3,8)").gain(0.9).lpf(180)
  .stack(
    s("~ sd:3 ~ [sd:1 sd:3?]").gain(0.55).room(0.25).speed("<1 1.2>"),
    s("hh*16").gain("0.3 0.5 0.3 0.7 0.3 0.5 0.3 0.9")
      .pan(rand).hpf(6000).sometimesBy(0.2, x => x.speed(2))
  )
let bass = note("<c2 [c2 eb2] c2 [bb1 c2]>").s("sawtooth")
  .lpf(400).lpenv(4).lpd(0.15).gain(0.65)
  .decay(0.15).sustain(0)
arrange(
  [2, s("breaks165").slice(8, "0 1 2 3 4 5 6 7")
    .lpf(saw.range(200, 6000).slow(2)).room(0.4).gain(0.7)
    .stack(s("bd:4").euclid(3,8).gain(0.5).lpf(200))
  ],
  [14, mainBreak.stack(chopLayer, drums, bass)],
  [5, s("breaks165").slice(8, "0 ~ 2 ~ 4 ~ 6 ~")
    .gain(0.55).room(0.5).delay("0.4:0.25:0.5")
    .speed("<1 0.5 -1 2>/2")
    .lpf(sine.range(500, 3500).slow(2.5))
    .stack(
      s("bd:4 ~ ~ bd:4 ~ ~ bd:4 ~").gain(0.7),
      note("<c2 ~ eb2 ~>").s("sawtooth").lpf(300).gain(0.5).decay(0.3).sustain(0)
    )
  ],
  [16, mainBreak.stack(chopLayer, drums, bass,
    s("cp:4(5,8)").gain(0.35).room(0.3).delay("0.15:0.12:0.3").speed(1.5)
  )],
  [1, s("breaks165").chop(8).rev()
    .lpf(isaw.range(100, 4000)).room(0.6).gain(0.5)
  ]
)
