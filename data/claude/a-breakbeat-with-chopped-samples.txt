// "Slice Machine" — Grimy breakbeat with chopped amen slices and dusty textures
setcps(.75)

samples('github:tidalcycles/dirt-samples')

stack(
  s("breaks165").splice(8, "<0 3 [6 2] 7 [1 5] 4 [0 3] [7 6]>")
    .cut(1).gain(.72)
    .hpf(400).lpf(sine.range(3200,6800).slow(7))
    .speed("<1 1 [1.5 .75] 1>")
    .room(.18).distort(.12)
    .every(4, x => x.rev())
    .every(7, x => x.fast(2).gain(.5))
    .sometimes(x => x.crush(10)),

  s("breaks125").splice(8, "<[5 ~ 2 ~] [~ 3 ~ 7] [0 ~ 6 1] [~ 4 ~ 0]>")
    .cut(2).gain(.35)
    .hpf(1200).lpf(4200)
    .speed("<.5 1 .75 1.25>")
    .pan(sine.slow(5))
    .room(.3).delay(".15:.1:.25")
    .rarely(x => x.rev()),

  s("bd [~ bd] ~ [bd ~], ~ cp ~ [~ cp]")
    .bank("RolandTR808")
    .gain(".88 .72 .82 .68")
    .room(.15).distort(.08)
    .swing(3),

  s("[hh hh oh hh]*2")
    .bank("RolandTR808")
    .gain(perlin.range(.15,.42))
    .pan(perlin.range(.3,.7))
    .cut(3).hpf(6000),

  note("<[a1 ~ ~ a1] [~ ~ e1 ~] [g1 ~ a1 ~] [~ e1 ~ g1]>")
    .s("sawtooth")
    .lpf(sine.range(180,650).slow(11))
    .decay(.18).sustain(.4).release(.08)
    .gain(.62).room(.12).distort(.22),

  note("<a0 e0 g0 e0>/4")
    .s("sine")
    .gain(.45).lpf(90)
    .decay(1.2).sustain(.85).release(.4),

  chord("<Am7 Em7 Gm7 Em7>")
    .dict('ireal').voicing()
    .s("gm_epiano2")
    .struct("~ [x ~] ~ x")
    .lpf(sine.range(800,2400).slow(9))
    .room(.55).delay(".25:.18:.35")
    .gain(.32).crush(13),

  s("vinyl:3").gain(.09).loop(1).lpf(2200).room(.8)
).reset("<8 16 16 8>")
