// "Concrete Lung" — dark drum and bass roller, reese bass and smashed breaks
setcps(174/60/4)

s("<[bd [~ bd] ~ [bd ~], ~ sd ~ [~ sd], hh*16, ~ ~ ~ oh] [bd ~ [bd ~] [~ bd], ~ [sd ~] ~ sd, hh*16, [~ oh]*2]>")
.bank("RolandTR909").cut(1)
.gain(perlin.range(.82,.98)).room(.18)
.every(4, x => x.fast(2).gain(.5))
.every(7, x => x.press())
.stack(
  note("<[e1 ~ e1 ~] [g1 ~ ~ g1] [c1 ~ c1 ~] [b0 ~ ~ b0]>")
  .s("sawtooth").lpf(sine.range(150,850).slow(8))
  .decay(.14).sustain(.55).release(.06)
  .gain(.72).distort(.32).room(.1),
  note("<e1 g1 c1 b0>/4").s("sine")
  .lpf(100).gain(.5).decay(1.2).sustain(.85).release(.5),
  chord("<Em9 Cmaj9 Am7 Bm7>").dict('ireal').voicing()
  .s("gm_synth_strings_1")
  .attack(.8).sustain(.6).release(1.5)
  .lpf(sine.range(400,1500).slow(16)).room(.75).gain(.22),
  n("<[~ 0 ~ ~] [~ ~ 3 ~] [~ ~ ~ 5] [2 ~ ~ ~]>")
  .scale("E5:minor:pentatonic").s("triangle")
  .lpf(sine.range(900,3000).slow(12))
  .room(.55).delay(".35:.25:.45").gain(.2)
  .attack(.01).decay(.15).sustain(.25).release(.2)
  .rarely(x => x.add(note(12)).gain(.1))
)
