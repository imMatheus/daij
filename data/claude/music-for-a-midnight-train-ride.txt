// "Midnight Express" — Nocturnal dub electronica for a midnight train ride
samples('github:tidalcycles/dirt-samples')
setcps(80/240)

let chPad = cat(
  "[bb3,db4,f4] ~ [ab3,c4,f4] ~",
  "[bb3,db4,gb4] ~ [bb3,eb4,gb4] ~",
  "[bb3,db4,f4] ~ [bb3,db4,gb4] ~",
  "[ab3,c4,f4] ~ [bb3,eb4,gb4] ~"
).note()
let brPad = cat(
  "[eb3,gb3,bb3] ~ [f3,ab3,c4] ~",
  "[gb3,bb3,db4] ~ [ab3,c4,f4] ~",
  "[eb3,gb3,bb3] ~ [db3,f3,ab3] ~"
).note()

let chBass = cat(
  "bb1 ~ f1 ~ bb1 ~ f1 ab1",
  "gb1 ~ db2 ~ eb1 ~ gb1 ~",
  "bb1 ~ f1 ~ gb1 ~ db1 ~",
  "f1 ~ c2 ~ eb1 ~ bb0 ~"
).note()
let brBass = cat(
  "eb1 ~ ~ f1 ~ ~ bb0 ~",
  "gb1 ~ ~ db1 ~ ~ f1 ~",
  "eb1 ~ ~ bb0 ~ ~ f1 ~"
).note()

let chLead = cat(
  "~ bb4 db5 f5 ab5 ~ f5 db5",
  "~ gb4 bb4 db5 gb5 ~ eb5 db5",
  "~ bb4 db5 eb5 ab5 ~ f5 eb5",
  "~ ab4 c5 f5 ab5 ~ f5 db5"
).note()
let brLead = cat(
  "~ ~ eb5 ~ f5 ~ ab5 ~",
  "~ gb5 ~ ab5 ~ f5 ~ eb5",
  "~ ~ db5 ~ eb5 ~ f5 ~"
).note()

let chKit = cat(
  "bd hh [~ hh] bd:3 hh ~ bd hh",
  "bd hh bd:3 hh ~ hh bd hh",
  "[bd lt] hh bd hh ~ hh bd hh",
  "bd hh bd:3 [hh hh] ~ hh [bd lt] hh"
)
let brKit = cat(
  "~ hh ~ ~ rim ~ hh ~",
  "~ ~ lt ~ ~ hh ~ ~",
  "~ hh ~ ~ rim ~ ~ ~"
)

arrange(
  [1, "[bb3,db4,f4] ~ ~ ~".note()],
  [9, chPad],
  [3, brPad],
  [9, chPad],
  [1, silence]
)
  .s("supersaw").detune(0.12)
  .attack(0.25).release(1.5)
  .lpf(sine.slow(14).range(600, 1800))
  .room(0.72)
  .delay(0.35).delaytime(0.375).delayfeedback(0.7)
  .gain(0.18)
  .stack(
    arrange(
      [1, "bb1 ~ ~ ~ ~ ~ ~ ~".note()],
      [9, chBass],
      [3, brBass],
      [9, chBass],
      [1, silence]
    )
      .s("sawtooth")
      .attack(0.01).release(0.3)
      .lpf(sine.slow(10).range(140, 400))
      .lpenv(1.5).lpd(0.2)
      .room(0.1)
      .gain(0.3),
    arrange(
      [1, s("~ ~ ~ lt ~ ~ ~ ~")],
      [9, chKit.s()],
      [3, brKit.s()],
      [9, chKit.s()],
      [1, s("~")]
    )
      .lpf(sine.slow(11).range(800, 3000))
      .room(0.2)
      .gain(0.3)
      .reset("<x ~ ~ ~ ~ ~ ~ ~>"),
    arrange(
      [1, s("hh*4")],
      [9, s("hh*8")],
      [3, s("hh").euclid(5, 8).fast(2)],
      [9, s("hh*8")],
      [1, s("~")]
    )
      .begin(0.82).end(0.95)
      .speed("<1 0.5 1 -0.5>")
      .bpf(sine.slow(9).range(3200, 7200))
      .bpq(7)
      .pan(sine.range(0.3, 0.7).slow(5))
      .room(0.55)
      .delay(0.12).delaytime(0.125).delayfeedback(0.45)
      .gain(0.07),
    arrange(
      [1, silence],
      [9, chLead],
      [3, brLead],
      [9, chLead],
      [1, silence]
    )
      .s("triangle")
      .attack(0.02).release(0.4)
      .vib(3).vibmod(0.12)
      .lpf(sine.slow(12).range(1200, 3500))
      .delay(0.38).delaytime(0.25).delayfeedback(0.62)
      .room(0.42)
      .gain(0.14)
      .reset("<x ~ ~ x ~ ~ ~ ~>"),
    arrange(
      [1, s("pink*2")],
      [9, s("pink*8")],
      [3, s("pink*4")],
      [9, s("pink*8")],
      [1, s("pink*2")]
    )
      .speed("<1 0.5 1 -0.5>")
      .bpf(sine.slow(16).range(400, 1200))
      .bpq(6)
      .room(0.85)
      .delay(0.18).delaytime(0.5).delayfeedback(0.5)
      .gain(0.035)
  )
  .postgain(1.04)
