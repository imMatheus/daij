// "Coin Rush" — Chiptune retro arcade soundtrack
setcps(0.5)

let intro = stack(
  n("[~ ~ [~ 7] 12] [9 7 4 [0 2]]").scale("C4:minor:pentatonic").s("square")
    .decay(.08).sustain(0).lpf(saw.slow(2).range(800,5000)).gain(.35)
    .delay(".2:.15:.4"),
  s("[~ ~ ~ hh] [~ hh ~ hh]").bank("RolandTR707").gain(.25),
  n("[~ ~] [~ 0]").scale("C2:minor:pentatonic").s("sawtooth")
    .lpf(300).decay(.15).sustain(.2).gain(.4)
)

let chorus = stack(
  n("[0 0 [~ 4] 7] [9 ~ 7 4] [0 0 [~ 4] 7] [<12 9> ~ <9 [9 12]> ~]")
    .scale("C4:minor:pentatonic").s("square")
    .decay(.08).sustain(0).lpf(3500).gain(.35)
    .delay(".12:.125:.3")
    .off(1/8, x=>x.add(7).gain(.18))
    .every(4, rev),
  n("[0 2 4 7 9 7 4 2]*2").scale("C5:minor:pentatonic").s("square")
    .decay(.04).sustain(0).lpf(perlin.range(1500,5000))
    .gain(.12).pan(sine),
  n("0 ~ [0 -3] [~ 0]").scale("C2:minor:pentatonic").s("sawtooth")
    .lpf(500).lpenv(4).lpd(.12).decay(.2).sustain(.3).gain(.5),
  s("bd [~ bd:1] sd [~ sd],hh*8").bank("RolandTR707")
    .gain(".7 .5 1 .5, .35 .25 .45 .25 .35 .25 .45 .3"),
  note("c6").s("sine").fm(12).fmh(2.5)
    .fmdecay(.04).fmsustain(0).decay(.05).sustain(0)
    .gain(.12).struct("~ ~ ~ ~ ~ ~ [~ x] ~")
)

let bridge = stack(
  n("[4 7 9 ~] [12 9 7 ~] [4 7 [~ 9] 7] [4 ~ ~ ~]")
    .scale("C4:minor:pentatonic").s("square")
    .decay(.1).sustain(0).lpf(2000).gain(.3).room(.2),
  note("[c5 ~ e5 ~] [g5 ~ c6 ~]").s("sine")
    .fm(6).fmh(3).fmdecay(.06).fmsustain(0)
    .decay(.08).sustain(0).gain(.2),
  n("[2 ~ 4 0]").scale("C2:minor:pentatonic").s("sawtooth")
    .lpf(400).lpenv(3).lpd(.1).decay(.2).sustain(.3).gain(.45),
  s("bd ~ sd ~,hh*4").bank("RolandTR707").gain(.5)
)

arrange([2, intro], [14, chorus], [5, bridge], [14, chorus])
