// 🕹️ "Quarter Muncher" — Retro arcade chiptune brawler
setcps(150/60/4)

stack(
  s("<bd [~ bd] sd ~> <bd bd [~ sd] [bd sd]>, hh(5,8), ~ ~ ~ <oh ~>")
    .bank("RolandTR808").crush(8)
    .gain(".85 .7").room(.15)
    .every(4, x => x.fast(2).gain(.4)),
  note("<[c2 ~ c2 c3] [eb2 ~ eb2 eb3] [f2 ~ f2 f3] [g2 ~ g2 g3]>")
    .s("square").crush(6).gain(.55)
    .decay(.12).sustain(.3).lpf(1300),
  note("<[c4,eb4,g4] [eb4,g4,bb4] [f4,ab4,c5] [g4,bb4,d5]>")
    .s("square").struct("~ x ~ x").crush(5)
    .gain(.32).decay(.2).sustain(.4)
    .lpf(sine.range(1600,3200).slow(8)).room(.2),
  n("<[0 2 4 7] [7 4 2 0] [2 5 7 9] [12 9 7 5]>")
    .scale("C5:minor:pentatonic").s("triangle").crush(5)
    .gain(.42).decay(.1).sustain(.35)
    .lpf(3800).room(.2).delay(".12:.08:.15")
    .sometimes(x => x.add(note(12)).gain(.25)),
  n("[0 4 7 11]*4").scale("C6:minor:pentatonic")
    .s("square").crush(4).gain(.14)
    .lpf(sine.range(3000,5500).slow(6))
    .room(.15).pan(sine.slow(2))
    .every(3, x => x.rev()),
  s("square").note(perlin.range(72,96).fast(8))
    .crush(3).gain(.06).lpf(2000)
    .struct("~ ~ ~ [x ~]")
    .room(.3).delay(".3:.2:.4"),
  note("<c1 eb1 f1 g1>").s("sine")
    .gain(.16).lpf(120).room(.3)
)
