// "Temporal Inversion" — Experimental ambient built from reversed and pitch-shifted textures
samples('github:tidalcycles/dirt-samples')
setcps(20/70)
arrange(
  // INTRO — ghostly reversed textures emerging from silence
  [1, s("space:1").speed("-0.5").chop(4).rev()
    .lpf(800).room(.9).sz(3).delay(".5:.15:.7").gain(.2)
    .stack(
      s("chin:0 ~ chin:2 ~").speed("<-1.2 -0.7>")
        .gain(.12).room(.85).delay(".45:.2:.65").lpf(1400)
    )],
  // CHORUS — reversed breaks, pitch-shifted piano, backward hats
  [8, s("breaks165").fit().slice(8, "7 5 <6 3> 1 [4 0] 2 6 3")
    .speed("<-0.8 -1.2 1.5 -0.6>").jux(rev)
    .lpf(perlin.range(700,3500).slow(4)).room(.4).delay(".3:.12:.5").gain(.33)
    .stack(
      s("piano:3 ~ piano:7 ~, ~ piano:5 ~ piano:1")
        .speed("<-1 -0.7 -1.5 -0.8>").chop(6).rev()
        .lpf(2200).gain(.2).room(.65).delay(".35:.18:.5"),
      note("<c2 ab1 f1 bb1>").s("sawtooth")
        .lpf(300).gain(.28).release(.2),
      s("hh*8").speed("-1 -1.5 -0.7 -2 -1.3 -0.5 -1.8 -1")
        .gain(".14 .08").room(.2).lpf(5500).pan(perlin.range(.2,.8))
    )],
  // BRIDGE — sparse eerie reversed space
  [3, s("space:3").speed("-0.35").chop(6).rev()
    .lpf(900).room(.95).sz(4).delay(".55:.2:.7").gain(.16)
    .stack(
      s("arpy:0 ~ arpy:3 ~").speed("<-0.6 -1.4>")
        .gain(.12).room(.8).delay(".5:.12:.6").lpf(1800)
    )],
  // FINAL CHORUS — reversed landscape returns with added vocal layer
  [8, s("breaks165").fit().slice(8, "7 5 <6 3> 1 [4 0] 2 6 3")
    .speed("<-1 -0.6 1.3 -1.5>").jux(rev)
    .lpf(perlin.range(600,4200).slow(3)).room(.5).delay(".35:.15:.55").gain(.3)
    .stack(
      s("piano:3 ~ piano:7 ~, ~ piano:5 ~ piano:1")
        .speed("<-1.5 -0.8 -1 -0.6>").chop(6).rev()
        .lpf(1900).gain(.18).room(.7).delay(".4:.2:.55"),
      note("<c2 ab1 f1 bb1>").s("sawtooth")
        .lpf(280).gain(.26).release(.25),
      s("hh*8").speed("-1.5 -1 -2 -0.7 -1 -1.8 -0.5 -1.3")
        .gain(".12 .07").room(.25).lpf(4500).pan(perlin.range(.2,.8)),
      s("mouth:0 ~ mouth:3 ~").speed("<-0.4 -0.6 -0.3 -0.8>")
        .chop(8).rev().gain(.1).room(.85).delay(".5:.2:.65").lpf(2000)
    )]
)
