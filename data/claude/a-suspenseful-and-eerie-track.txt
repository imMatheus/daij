// "The Watcher" — Suspenseful eerie ambient with dissonant drones and spectral textures
setcps(0.4)
samples('github:tidalcycles/dirt-samples')

const creep = perlin.slow(12).range(250, 1000)
const drift = sine.slow(14).range(0.12, 0.35)

const intro =
  note("[c2,gb2]")
    .s("sawtooth")
    .attack(1.5).release(2).sustain(.3)
    .lpf(saw.slow(1).range(80, 350)).lpq(4)
    .room(.9).rsize(6)
    .gain(saw.slow(1).range(0, .22))
    .stack(
      s("~ ~ ~ metal:3")
        .speed(.4).gain(.07)
        .room(.95).rsize(8)
        .delay("0.7:0.3:0.8")
    )

const chorus =
  note("<[c1 ~] [~ db1]>")
    .s("sawtooth")
    .lpf(sine.slow(10).range(60, 220)).lpq(6)
    .gain(.35).release(.5)
    .room(.5).rsize(3)
    .distort("0.8:0.4")
    .stack(
      note("<[c3,gb3,bb3] [db3,e3,ab3]>/2")
        .s("triangle")
        .attack(1).release(2).sustain(.2)
        .lpf(creep).lpq(2)
        .room(.9).rsize(5)
        .delay("0.45:0.25:0.7")
        .gain(.15)
    )
    .stack(
      s("[~ metal:2] ~ [~ ~ metal:5 ~] ~")
        .speed("<.6 -.4 .3 -.7>")
        .gain(.11).hpf(2500)
        .room(.85).delay("0.6:0.2:0.8")
    )
    .stack(
      note("~ [gb5 ~] ~ [~ b5]")
        .s("sine").gain(.07)
        .vib("5:2").release(1.2)
        .room(.8).delay("0.5:0.35:0.65")
    )
    .stack(
      s("bd:4").euclid(3,16)
        .gain(.28).lpf(140)
        .room(.4).distort("1.5:0.3")
    )
    .stack(
      sound("brown").gain(drift)
        .lpf(perlin.slow(8).range(150, 550))
        .hpf(50).room(.6)
    )

const bridge =
  note("c1!4")
    .s("sine").lpf(100)
    .gain(.4).release(.3)
    .tremolo(1.5).room(.7).rsize(5)
    .stack(
      s("[~ hh:3] ~ ~ ~")
        .gain(.05).speed(.3)
        .room(.95).rsize(7)
        .delay("0.8:0.35:0.85")
    )
    .stack(
      note("~ ~ ~ [~ gb6]")
        .s("triangle").gain(.04)
        .release(2.5).room(.9).rsize(6)
        .vib("8:4")
    )

arrange(
  [1, intro],
  [12, chorus],
  [3, bridge],
  [12, chorus]
).postgain(.85)
