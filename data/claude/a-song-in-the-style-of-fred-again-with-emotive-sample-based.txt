setcps(132/60/4)

stack(
  s("bd*4, ~ cp ~ cp, [hh hh ~ hh hh ~ hh ~]*2, <~ ~ ~ oh(3,8)>")
    .bank("RolandTR909").cut(1)
    .gain(".92 .58 .24 .16")
    .room(.18).swing(3)
    .every(8, x => x.press()),

  s("rim:2(5,16)")
    .bank("RolandTR909")
    .gain(perlin.range(.08,.26))
    .pan(sine.slow(5)).room(.45)
    .delay(".12:.08:.18"),

  chord("<Ebm9 Bmaj9 Gbmaj7 Db>").dict('ireal').voicing()
    .s("gm_epiano1").struct("<[x ~ ~ ~] [~ x ~ ~] [x ~ ~ x] [~ ~ x ~]>")
    .lpf(sine.range(700, 2600).slow(8))
    .room(.78).delay(".38:.28:.5")
    .gain(.36).speed(.97)
    .sometimes(x => x.crush(13)),

  chord("<Ebm9 Bmaj9 Gbmaj7 Db>").dict('ireal').voicing()
    .s("gm_synth_strings_1")
    .attack(.7).sustain(.7).release(2)
    .lpf(sine.range(350, 1300).slow(16))
    .room(.72).gain(.19),

  n("<[~ ~ 4 ~] [7 ~ ~ 4] [~ 2 ~ ~] [4 ~ 7 ~]>")
    .scale("Eb5:minor:pentatonic").s("gm_epiano2")
    .lpf(perlin.range(1600, 4800))
    .room(.82).delay(".42:.32:.58")
    .gain(.24).crush(14)
    .rarely(x => x.add(note(12))),

  note("<[eb2 ~ ~ eb2] [~ ~ b1 ~] [gb1 ~ gb1 ~] [~ db2 ~ ~]>")
    .s("sawtooth")
    .lpf(sine.range(120, 520).slow(12))
    .decay(.2).sustain(.45).release(.06)
    .gain(.55).distort(.14).room(.1),

  note("<eb1 b0 gb0 db1>/4").s("sine")
    .gain(.26).lpf(115)
    .decay(.9).sustain(.8).release(.15),

  s("vinyl:2").gain(.07).loop(1).lpf(1600).room(.65)
).reset("<8 16 16 8>")
