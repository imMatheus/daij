// "Two Tongues" — Conversational call and response between two melodic voices
setcps(0.32)
samples('github:tidalcycles/dirt-samples')

let callA = note("<[D5 F5 A5 ~] ~ [G5 F5 E5 D5] ~ [A5 ~ F5 D5] ~ [D5 E5 F5 A5] ~>")
  .s("triangle").gain(.3)
  .att(.03).dec(.22).sus(.4).rel(.55)
  .vib(5).vibmod(.1)
  .lpf(sine.range(2800, 5500).slow(6))
  .room(.3).delay(.15).dt(.4).dfb(.35)
  .pan(.28)

let respB = note("<~ [A4 C5 D5 C5] ~ [D5 C5 A4 G4] ~ [E5 D5 C5 A4] ~ [A4 C5 D5 F5]>")
  .s("sawtooth").gain(.22)
  .att(.05).dec(.28).sus(.3).rel(.6)
  .lpf(perlin.range(1500, 3500).slow(5))
  .lpq(1.2)
  .room(.25).delay(.18).dt(.35).dfb(.4)
  .pan(.72)

let bass = note("<[D2 ~ ~ A2] [D2 ~ ~ E2] [G2 ~ ~ D2] [G2 ~ ~ A2] [F2 ~ ~ C3] [F2 ~ ~ G2] [Bb2 ~ ~ A2] [D2 ~ ~ A2]>")
  .s("sawtooth").lpf(450).lpq(2)
  .gain(.42).att(.01).dec(.2).sus(.6).rel(.15)
  .room(.1).pan(.45)

let drums = stack(
  s("hh*4").gain(".18 .28 .18 .28").room(.06),
  s("~ [~ sd] ~ [sd ~]").gain(.08).room(.12),
  s("<[bd ~ ~ ~] [bd ~ ~ bd]>").gain(.2)
)

let chorus = stack(callA, respB, bass, drums)

let intro = note("<[D5 ~ A5 ~] [F5 ~ D5 ~]>")
  .s("triangle").gain(.25)
  .att(.05).dec(.3).sus(.3).rel(.8)
  .vib(4).vibmod(.08)
  .lpf(3000)
  .room(.45).sz(2).delay(.25).dt(.5).dfb(.4)
  .pan(.35)

let bridge = stack(
  note("<[D5 F5 ~ A5] [G5 ~ F5 E5] [A5 G5 F5 D5] [E5 ~ D5 ~]>")
    .s("triangle").gain(.25)
    .att(.04).dec(.25).sus(.35).rel(.6)
    .vib(5).vibmod(.12)
    .lpf(3500).room(.35).delay(.2).dt(.4).dfb(.38).pan(.35),
  note("<[A4 C5 ~ D5] [E5 ~ D5 C5] [F5 E5 D5 A4] [C5 ~ A4 ~]>")
    .s("sawtooth").gain(.2)
    .att(.06).dec(.3).sus(.3).rel(.65)
    .lpf(2200).lpq(1.5)
    .room(.3).delay(.15).dt(.35).dfb(.35).pan(.65),
  note("<[Bb2 ~ D3 ~] [A2 ~ C3 ~] [G2 ~ Bb2 ~] [A2 ~ E2 ~]>")
    .s("sawtooth").lpf(500).lpq(2)
    .gain(.42).att(.01).dec(.2).sus(.6).rel(.15)
    .room(.12).pan(.45),
  s("hh*4").gain(".15 .25 .15 .25").room(.08),
  s("~ sd ~ ~").gain(.06).room(.15)
)

arrange(
  [2, intro],
  [8, chorus],
  [4, bridge],
  [8, chorus]
)
