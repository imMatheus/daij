// "Pocket Funk" — greasy slap bass groove with clavinet stabs and syncopated 909s
setcps(100/60/4)

stack(
  s("<bd ~ [~ bd] ~, ~ cp ~ cp> <bd [~ bd] ~ bd, ~ [cp cp] ~ cp>")
    .bank("RolandTR909").swing(3)
    .gain(perlin.range(.82,.96)).room(.25),
  s("hh(13,16), <~ ~ ~ oh ~ ~ oh ~>")
    .bank("RolandTR909").cut(1)
    .gain(perlin.range(.22,.52))
    .pan(sine.slow(7)).room(.18),
  note("<[e2 ~ e3 g2] [~ a2 ~ e2] [e2 ~ g2 ~] [a2 b2 a2 ~] [e2 g2 e3 ~] [a2 ~ g2 e2] [e2 ~ g2 a2] [b2 ~ a2 g2]>")
    .s("gm_slap_bass_1")
    .attack(.005).decay(.16).sustain(.35).release(.08)
    .lpf(sine.range(1800,4200).slow(8)).hpf(100)
    .distort(.08).gain(.88).room(.18),
  chord("<Em9 Am7 <D9 Em9> Bm7>")
    .dict('ireal').voicing()
    .s("gm_clavinet")
    .struct("<[~ x ~ x]*2 [~ x] [x ~] [~ x] [x ~]>")
    .lpf(sine.range(2200,5200).slow(5))
    .gain(.48).room(.28).delay(".1:.06:.16").swing(3),
  chord("<Em7 Am7 D7 Bm7>")
    .dict('ireal').voicing()
    .s("gm_electric_guitar_muted")
    .struct("~ [x x] ~ [x ~]")
    .lpf(sine.range(900,5000).slow(3))
    .gain(.36).room(.28).swing(3),
  n("<[~ 0 2 ~] [4 ~ 2 0] [~ 4 5 ~] [7 ~ 5 4]>")
    .scale("E5:dorian")
    .s("gm_epiano2")
    .lpf(sine.range(1200,3600).slow(10))
    .room(.45).delay(".22:.15:.28").gain(.32)
    .sometimes(x => x.add(note(12))),
  note("<e1 a1 d1 b1>/4")
    .s("sine").gain(.18).lpf(140)
    .attack(1.2).sustain(.7).release(2)
)
