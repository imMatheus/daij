// 🎵 "Slap City Strut" — Funk Groove with Slap Bass
// At 108 BPM: intro(8) + verse(16) + chorus(16) + verse2(16) + chorus2(16) + bridge(8) + outro(8) = 88 bars ≈ 3:15

setcpm(108 / 4)

// Drum groove
$: arrange(
  [8,  s("bd ~ ~ ~, ~ ~ hh ~")
       .bank("RolandTR909").swing(3).gain(".8 .7").room(.3)],
  [16, s("bd ~ [~ bd] ~, ~ cp ~ cp, hh*8, ~ ~ ~ oh")
       .bank("RolandTR909").cut(1).swing(3)
       .gain(perlin.range(.78,.95))
       .room(.3)
       .every(4, x => x.press())],
  [16, s("bd [~ bd] ~ bd, ~ cp ~ [cp cp], hh*16, ~ ~ oh ~")
       .bank("RolandTR909").cut(1).swing(3)
       .gain(".92 .85 .9 .85")
       .room(.25)
       .every(3, x => x.fast(2).gain(.5))],
  [16, s("bd ~ [~ bd] ~, ~ cp ~ cp, hh*8, ~ ~ ~ oh")
       .bank("RolandTR909").cut(1).swing(3)
       .gain(perlin.range(.78,.95))
       .room(.3)
       .sometimes(x => x.press())],
  [16, s("bd [~ bd] ~ bd, ~ cp ~ [cp cp], hh*16, ~ ~ oh ~")
       .bank("RolandTR909").cut(1).swing(3)
       .gain(".92 .85 .9 .85")
       .room(.25)
       .every(4, x => x.fast(2).gain(.5))],
  [8,  s("bd ~ ~ bd, ~ cp ~ ~, [hh oh]*2")
       .bank("RolandTR909").cut(1).swing(4)
       .gain(".85 .75").room(.4)
       .every(2, x => x.rev())],
  [8,  s("bd ~ ~ ~, ~ ~ cp ~")
       .bank("RolandTR909").swing(4)
       .gain(saw.slow(8).range(.7,.1)).room(.5)]
)

// Slap bass — percussive funk line
$: arrange(
  [8,  silence],
  [16, note("<[e2 ~ e2 g2] [~ a2 ~ e2] [e2 ~ g2 ~] [a2 ~ e2 ~]>")
       .s("gm_slap_bass_1")
       .decay(.15).sustain(.3).release(.1)
       .gain(.88).room(.2)
       .lpf(3500)
       .every(4, x => x.add(note(12)).gain(.55))],
  [16, note("<[e2 g2 ~ a2] [b2 ~ a2 g2] [e2 ~ e2 g2] [a2 b2 ~ e2]>")
       .s("gm_slap_bass_1")
       .decay(.12).sustain(.35).release(.1)
       .gain(.92).room(.2)
       .lpf(4000)
       .sometimes(x => x.speed(1.02))],
  [16, note("<[e2 ~ e2 g2] [~ a2 ~ e2] [e2 ~ g2 ~] [a2 ~ e2 ~]>")
       .s("gm_slap_bass_1")
       .decay(.15).sustain(.3).release(.1)
       .gain(.88).room(.2)
       .lpf(3500)
       .rarely(x => x.add(note(12)).gain(.5))],
  [16, note("<[e2 g2 ~ a2] [b2 ~ a2 g2] [e2 ~ e2 g2] [a2 b2 ~ e2]>")
       .s("gm_slap_bass_1")
       .decay(.12).sustain(.35).release(.1)
       .gain(.92).room(.2)
       .lpf(4000)
       .every(3, x => x.fast(2).gain(.5))],
  [8,  note("<[e2 ~ ~ g2] [~ a2 ~ ~] [e2 ~ ~ ~] [g2 ~ ~ ~]>")
       .s("gm_slap_bass_1")
       .decay(.25).sustain(.4).release(.2)
       .gain(.75).room(.35).lpf(2800)],
  [8,  note("<e2 ~ ~ ~>/1")
       .s("gm_slap_bass_1")
       .decay(.5).sustain(.3).release(.5)
       .gain(saw.slow(8).range(.7,.1)).room(.4)]
)
