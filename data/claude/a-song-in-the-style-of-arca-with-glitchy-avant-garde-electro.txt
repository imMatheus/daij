// 🔮 "Nonbinary Flesh" — Glitchy avant-garde electronics (Arca-style)
// ≈ 1:10

setcps(.45)

stack(
  // unstable sub drone
  note("<db1 [~ ab0] bb0 [e1 ~]>")
    .s("sine").lpf(95)
    .attack(.6).sustain(.9).release(1.5)
    .gain(.3).room(.25)
    .speed(perlin.range(.96,1.04)),

  // broken dembow-industrial percussion
  s("<bd(3,8) bd(5,8) bd(3,8) [bd ~ ~ bd]>, <[~ cp:3 ~ ~] [~ [cp:3 sd:2] ~ cp:3] [~ cp:3 ~ [~ sd:2]] [~ ~ cp:3 ~]>, <[hh oh hh ~]*2 [oh hh]*3 hh(5,8) hh*4>")
    .bank("RolandTR909").cut(1)
    .gain(perlin.range(.5,.95))
    .crush(sine.range(5,14).slow(7))
    .room(.2).distort(.15)
    .every(3, x => x.fast(2).crush(3))
    .every(7, x => x.rev())
    .sometimes(x => x.speed(perlin.range(.3,2.5)))
    .pan(perlin.range(.15,.85)),

  // glitch dust — scattered metallic particles
  s("hh*<16 32 8 32>").bank("RolandTR909")
    .gain(perlin.range(.01,.13))
    .crush(perlin.range(2,5))
    .speed(perlin.range(.05,6))
    .pan(perlin.range(0,1)).room(.5),

  // mutant saw stabs — detuned and aggressive
  note("<[db2 ~ ~ db2] [~ ab1 e2 ~] [bb1 ~ ~ bb1] [e2 ~ db2 ~]>")
    .s("sawtooth")
    .lpf(sine.range(180,2200).slow(5))
    .decay(.12).sustain(.5)
    .gain(.5).room(.15)
    .distort(.4).crush(8)
    .speed(perlin.range(.9,1.1))
    .every(4, x => x.fast(2).gain(.28).crush(4)),

  // spectral pad — eerie breathing texture
  chord("<Dbm9 Abm7 Emaj7 Bb7b9>").dict('ireal').voicing()
    .s("square")
    .struct("<[x ~ x ~] [~ x ~ ~] [x ~ ~ x] [~ ~ x ~]>")
    .lpf(sine.range(350,2500).slow(9))
    .room(.65).delay(".2:.15:.3")
    .gain(.24).crush(12)
    .attack(.03).decay(.2).sustain(.25).release(.3)
    .pan(sine.slow(5)),

  // alien melody fragments
  n("<[~ 0 ~ 3] [5 ~ ~ ~] [~ ~ 2 ~] [~ 6 ~ ~]>")
    .scale("Db5:phrygian")
    .s("triangle")
    .lpf(sine.range(500,3800).slow(7))
    .room(.55).delay(".35:.25:.5")
    .gain(.22).crush(10)
    .pan(perlin.range(0,1))
    .speed(perlin.range(.82,1.18))
    .rarely(x => x.add(note(7)).fast(2))
)
