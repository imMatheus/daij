// 🤖 "Genesis Protocol" — Glitch-Industrial / Cybernetic Awakening
setcps(0.45)

stack(
  // power core hum
  note("c1").s("sawtooth")
    .lpf(saw.slow(30).range(50,500))
    .gain(saw.slow(22).range(.03,.28))
    .room(.5).roomsize(5)
    .attack(1).decay(2).sustain(.6),

  // machine heartbeat
  s("bd(3,8)")
    .bank("RolandTR909")
    .gain(saw.slow(28).range(.06,.88))
    .lpf(saw.slow(28).range(200,2500))
    .room(.3).distort(saw.slow(30).range(0,.1)),

  // mechanical clicks and servo noise
  s("hh*8")
    .bank("RolandTR808")
    .gain(perlin.range(.04,.3))
    .pan(perlin.range(0,1))
    .crush(saw.slow(24).range(4,14))
    .lpf(sine.range(1200,5500).slow(7))
    .cut(1).room(.4),

  // snare system activates
  s("~ cp ~ <~ [cp cp]>")
    .bank("RolandTR909")
    .gain(saw.slow(30).range(0,.6))
    .room(.35).delay(".2:.12:.25"),

  // circuit sparks — low register arp
  n("<[~ 0 ~ ~] [~ ~ 2 ~] [0 ~ 3 ~] [2 ~ 5 ~]>")
    .scale("C3:minor").s("square").crush(8)
    .lpf(saw.slow(26).range(200,2800))
    .gain(saw.slow(26).range(0,.3))
    .room(.5).delay(".3:.2:.35")
    .decay(.15).sustain(.25),

  // consciousness emerging — melody
  n("<[~ ~ 0 ~] [~ 2 ~ 3] [5 ~ 3 ~] [~ 7 ~ 5]>")
    .scale("C5:minor:pentatonic").s("triangle")
    .lpf(saw.slow(32).range(400,4500))
    .gain(saw.slow(32).range(0,.25))
    .room(.6).delay(".25:.18:.45")
    .decay(.1).sustain(.2).release(.4)
    .sometimes(x => x.add(note(12)).gain(.12)),

  // awareness expanding — pad
  note("<c3 eb3 f3 g3>/4").s("sawtooth")
    .attack(2).decay(1.5).sustain(.5).release(2)
    .lpf(saw.slow(30).range(150,1800))
    .gain(saw.slow(30).range(0,.2))
    .room(.7).roomsize(4),

  // diagnostic pings
  n("<[~ ~ ~ 0] [~ ~ ~ ~] [~ ~ ~ 2] [~ ~ ~ ~]>")
    .scale("C6:minor").s("sine")
    .gain(saw.slow(32).range(0,.16))
    .room(.8).delay(".5:.3:.6")
    .lpf(3500).decay(.05).sustain(.1).release(.3)
)
