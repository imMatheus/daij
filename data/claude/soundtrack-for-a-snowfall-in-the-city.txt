// "First Snow" — Ambient Cinematics (~1:20)
// At 60 BPM: intro(4) + A(8) + B(4) + outro(4) = 20 cycles ~ 1:20
// Key: Db major — Dbmaj7, Abm9, Gbmaj7, Ebm7

setcpm(60 / 4)

// -- SNOW TEXTURE — layered noise + vinyl hiss
$: s("white").gain(.06).lpf(1800).hpf(600).room(1).loop(1)
$: s("vinyl:2").gain(.08).room(1).loop(1).lpf(1400)

// -- SPARSE PIANO — crystalline chords through heavy reverb
$: arrange(
  [4, chord("<Dbmaj7 Gbmaj7>/2").dict('ireal').voicing()
     .s("gm_epiano1")
     .struct("x ~ ~ ~")
     .lpf(1400).room(.85).roomsize(6)
     .delay(".5:.4:.65").gain(.32)
     .attack(.2)],
  [8, chord("<Dbmaj7 Abm9 Gbmaj7 Ebm7>").dict('ireal').voicing()
     .s("gm_epiano1")
     .struct("~ [x ~] ~ ~")
     .lpf(2000).room(.8).roomsize(5)
     .delay(".4:.3:.55").gain(.4)],
  [4, chord("<Dbmaj7 Gbmaj7>/2").dict('ireal').voicing()
     .s("gm_epiano1")
     .struct("x ~ ~ ~")
     .lpf(sine.range(500,1400).slow(4)).room(.9).roomsize(6)
     .delay(".6:.5:.7").gain(.28)
     .attack(.3)],
  [4, chord("<Dbmaj7>/1").dict('ireal').voicing()
     .s("gm_epiano1")
     .struct("x ~ ~ ~")
     .lpf(sine.range(300,900).slow(4)).room(.95).roomsize(8)
     .delay(".7:.5:.75").gain(saw.slow(4).range(.25,.04))
     .attack(.4)]
)

// -- GENTLE DRUMS — brushed and sparse, only in A and B
$: arrange(
  [4,  silence],
  [8,  s("bd ~ ~ ~, ~ ~ ~ ~, [hh ~]*4")
       .bank("RolandTR808").cut(1).swing(3)
       .gain(perlin.range(.4,.58)).room(.5)],
  [4,  s("bd ~ ~ ~, ~ ~ ~ ~, [hh ~]*4")
       .bank("RolandTR808").cut(1)
       .gain(perlin.range(.25,.42)).room(.65)],
  [4,  silence]
)

// -- FLOATING MELODY — vibraphone like snowflakes landing
$: arrange(
  [4,  silence],
  [8,  n("<[~ 0 ~ ~] [~ ~ 2 ~] [~ ~ ~ 4] [~ 2 ~ ~]>")
       .scale("Db4:major:pentatonic").s("gm_vibraphone")
       .lpf(sine.range(1200,3500).slow(8))
       .room(.85).roomsize(5)
       .delay(".5:.4:.65").gain(.22)
       .vib(2).vibmod(.3)],
  [4,  n("<[~ 0 ~ ~] [~ ~ ~ ~] [~ ~ 2 ~] [~ ~ ~ ~]>")
       .scale("Db4:major:pentatonic").s("gm_vibraphone")
       .lpf(1500).room(.9).roomsize(6)
       .delay(".6:.5:.7").gain(.18)
       .vib(2).vibmod(.3)],
  [4,  silence]
)

// -- HIGH SHIMMER — crystalline sparkles, very sparse
$: arrange(
  [4,  silence],
  [8,  n("<[0 ~ ~ ~] [~ ~ ~ 2] [~ ~ ~ ~] [~ 4 ~ ~]>")
       .scale("Db5:major:pentatonic").s("gm_vibraphone")
       .room(.9).roomsize(6)
       .delay(".6:.5:.7").gain(.14)
       .lpf(sine.range(2000,5000).slow(6))],
  [4,  silence],
  [4,  silence]
)

// -- DEEP BASS — city hum underneath the snow
$: arrange(
  [4,  note("<db1>/1").s("sine")
       .attack(2).decay(1.5).sustain(.7).release(3)
       .gain(.16).room(.6).lpf(200)],
  [8,  note("<db1 ab1 gb1 eb1>/4").s("sine")
       .attack(1.5).decay(1).sustain(.7).release(2)
       .gain(.18).room(.5).lpf(200)],
  [4,  note("<db1 gb1>/2").s("sine")
       .attack(2).decay(1.5).sustain(.6).release(3)
       .gain(.14).room(.65).lpf(180)],
  [4,  note("<db1>/1").s("sine")
       .attack(3).decay(2).sustain(.5).release(4)
       .gain(saw.slow(4).range(.14,.02)).room(.7).lpf(160)]
)

// -- STRING PAD — warm blanket against the cold
$: arrange(
  [4,  silence],
  [8,  chord("<Dbmaj7 Abm9 Gbmaj7 Ebm7>").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(1.5).decay(1).sustain(.7).release(2.5)
       .lpf(sine.range(400,1000).slow(8))
       .room(.75).gain(.18)],
  [4,  chord("<Dbmaj7 Gbmaj7>/2").dict('ireal').voicing()
       .s("gm_synth_strings_1")
       .attack(2).decay(1.5).sustain(.6).release(3)
       .lpf(sine.range(300,700).slow(6))
       .room(.85).gain(.14)],
  [4,  silence]
)
