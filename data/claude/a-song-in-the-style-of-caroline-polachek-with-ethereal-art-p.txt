// "Liquid Mirror" ‚Äî Ethereal Art-Pop √Ý la Caroline Polachek (~1:07)
// intro(4) + verse(8) + chorus(8) + bridge(4) + outro(6) = 30 cycles at 108 BPM

setcpm(108 / 4)

// -- BREATH TEXTURE ‚Äî airy presence throughout
$: s("white").gain(.04).lpf(sine.range(800,2200).slow(12)).room(1).hpf(600)

// -- CRYSTAL PAD ‚Äî shimmering sawtooth drone, wide and ethereal
$: arrange(
  [4,  note("<[c4,eb4,g4]>/1").s("sawtooth")
       .attack(2).decay(1.5).sustain(.5).release(3)
       .lpf(sine.range(300,900).slow(4)).gain(.1).room(.8)],
  [8,  note("<[c4,eb4,g4] [ab3,c4,eb4] [eb4,g4,bb4] [g3,bb3,d4]>/4").s("sawtooth")
       .attack(.8).decay(.6).sustain(.55).release(1.5)
       .lpf(sine.range(500,1600).slow(8)).gain(.14).room(.65)],
  [8,  note("<[c4,eb4,g4] [f3,ab3,c4] [ab3,c4,eb4] [g3,b3,d4]>/4").s("sawtooth")
       .attack(.5).decay(.5).sustain(.6).release(1)
       .lpf(sine.range(700,2200).slow(6)).gain(.16).room(.55)],
  [4,  note("<[ab3,c4,eb4] [g3,bb3,d4]>/2").s("sawtooth")
       .attack(1.5).decay(1).sustain(.4).release(2.5)
       .lpf(sine.range(300,800).slow(4)).gain(.11).room(.8)],
  [6,  note("<[c4,eb4,g4]>/1").s("sawtooth")
       .attack(1.5).decay(1.5).sustain(.35).release(3)
       .lpf(sine.range(200,600).slow(6)).gain(saw.slow(6).range(.13,.02)).room(.85)]
)

// -- VOCAL MELODY ‚Äî breathy flute simulating Polachek's swooping melisma
$: arrange(
  [4,  silence],
  [8,  n("<[~ 0 ~ 3] [5 ~ 4 ~] [~ 2 ~ 0] [~ ~ 3 ~]>")
       .scale("C5:minor").s("gm_flute")
       .lpf(sine.range(1200,3500).slow(8))
       .room(.7).delay(".4:.3:.5").gain(.35)
       .vib(5).vibmod(.6).attack(.08)],
  [8,  n("<[0 2 4 7] [~ 5 4 2] [7 ~ 5 ~] [4 2 0 ~]>")
       .scale("C5:minor").s("gm_flute")
       .lpf(sine.range(1500,4500).slow(6))
       .room(.6).delay(".3:.2:.4").gain(.4)
       .vib(4).vibmod(.7)
       .sometimes(x => x.add(note(12)).gain(.2))],
  [4,  n("<[~ 7 ~ 5] [~ ~ 4 ~]>")
       .scale("C5:minor").s("gm_flute")
       .room(.85).delay(".6:.5:.7").gain(.28)
       .lpf(sine.range(800,2000).slow(4))
       .vib(6).vibmod(.8)],
  [6,  n("<[0 ~ 2 ~] [~ 4 ~ ~]>")
       .scale("C5:minor").s("gm_flute")
       .room(.9).delay(".5:.4:.65").gain(saw.slow(6).range(.32,.05))
       .lpf(sine.range(600,1400).slow(6))
       .vib(5).vibmod(.6)]
)

// -- GLASS ARPEGGIOS ‚Äî crystalline epiano shimmer
$: arrange(
  [4,  n("<[0 ~ 4 ~]>").scale("C5:minor").s("gm_epiano1")
       .room(.75).delay(".5:.4:.6").gain(.2).lpf(1600).attack(.15)],
  [8,  n("<[0 ~ 2 4] [~ 5 ~ 2] [4 ~ 0 ~] [2 ~ 4 5]>").scale("C5:minor").s("gm_epiano1")
       .lpf(sine.range(1000,2800).slow(8)).room(.6).delay(".3:.2:.4")
       .gain(.28).pan(sine.slow(3))],
  [8,  n("<[0 2 4 7] [5 4 2 0]>*2").scale("C5:minor").s("gm_epiano1")
       .lpf(sine.range(1200,3500).slow(6)).room(.5).delay(".25:.15:.3")
       .gain(.32).pan(sine.slow(2))
       .rarely(x => x.add(note(12)))],
  [4,  n("<[0 ~ ~ 4] [~ ~ 2 ~]>").scale("C5:minor").s("gm_epiano1")
       .room(.8).delay(".55:.45:.65").gain(.2).lpf(1200)],
  [6,  n("<[0 ~ 4 ~] [2 ~ ~ 0]>").scale("C6:minor").s("gm_epiano1")
       .lpf(sine.range(500,1200).slow(6)).room(.8).delay(".5:.4:.6")
       .gain(saw.slow(6).range(.25,.04)).pan(sine.slow(4))]
)

// -- DRUMS ‚Äî crisp restrained art-pop beat
$: arrange(
  [4,  silence],
  [8,  s("bd ~ ~ ~, ~ [cp ~] ~ ~, [~ hh]*4").bank("RolandTR909").cut(1)
       .gain(perlin.range(.5,.7)).room(.3)],
  [8,  s("bd ~ [~ bd] ~, ~ cp ~ [~ cp], hh*8").bank("RolandTR909").cut(1)
       .gain(perlin.range(.6,.85)).room(.25)
       .every(4, x => x.fast(2).gain(.4))],
  [4,  s("~ ~ ~ ~, ~ ~ [cp ~] ~").bank("RolandTR909")
       .gain(perlin.range(.3,.45)).room(.5)],
  [6,  s("bd ~ ~ ~, ~ ~ ~ ~, [~ hh]*2").bank("RolandTR909").cut(1)
       .gain(saw.slow(6).range(.6,.1)).room(.4)]
)

// -- BASS ‚Äî deep sine pulse, art-pop warmth
$: arrange(
  [4,  silence],
  [8,  note("<[c2 ~ ~ ~] [ab1 ~ ~ ~] [eb2 ~ ~ ~] [g1 ~ ~ ~]>")
       .s("sine").gain(.45).lpf(200).attack(.05).decay(.3).sustain(.6)],
  [8,  note("<[c2 ~ c2 ~] [f1 ~ ~ f1] [ab1 ~ ab1 ~] [g1 ~ ~ g1]>")
       .s("sine").gain(.5).lpf(220).attack(.03).decay(.25).sustain(.65)
       .room(.15)],
  [4,  note("<c2 ~ ab1 ~>/2").s("sine")
       .gain(.35).lpf(180).decay(.5).sustain(.4)],
  [6,  silence]
)

// -- HIGH GHOST MELODY ‚Äî vibraphone doubling in chorus + outro
$: arrange(
  [4,  silence],
  [8,  silence],
  [8,  n("<[~ 0 ~ 4] [7 ~ 5 ~] [~ 4 2 ~] [0 ~ ~ ~]>")
       .scale("C6:minor").s("gm_vibraphone")
       .room(.8).delay(".5:.4:.6").gain(.18)
       .lpf(sine.range(2000,5000).slow(8))
       .pan(sine.slow(3))],
  [4,  silence],
  [6,  n("<[~ 0 ~ ~] [~ ~ 4 ~]>")
       .scale("C6:minor").s("gm_vibraphone")
       .room(.9).roomsize(8).delay(".7:.5:.8")
       .gain(saw.slow(6).range(.15,.03))
       .lpf(sine.range(1000,3000).slow(6))]
)
