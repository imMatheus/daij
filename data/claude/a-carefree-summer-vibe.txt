// "Endless August" — Breezy tropical summer groove with warm synths and laid-back beats
setcps(0.55)
samples('github:tidalcycles/dirt-samples')
stack(
  chord("<D Bm G A>").voicing()
    .s("sawtooth").attack(.3).decay(.5).sustain(.6).release(.8)
    .lpf(sine.range(600,2200).slow(16))
    .chorus(.5).room(.4).gain(.25),
  s("bd [~ bd] ~ bd, ~ sd ~ [sd ~]")
    .bank("RolandTR909").gain(".8 .65")
    .room(.08)
    .mask("<~!4 x!13 ~!5 x!13 ~!3>"),
  s("[~ hh]*4").gain(".4 .65 .45 .75 .4 .7 .45 .8")
    .swingBy(1/6,4).pan(rand.range(.3,.7))
    .sometimesBy(.15,x=>x.speed(1.2))
    .mask("<~!2 x!33 ~!3>"),
  s("~ ~ ~ rim").gain(.28).room(.3)
    .mask("<~!4 x!13 ~!5 x!13 ~!3>"),
  note("<[d2 ~ d2 ~] [~ d2 ~ a1] [g1 ~ g1 ~] [a1 ~ a1 ~]>")
    .s("sawtooth").lpf(500).lpq(1.5)
    .decay(.1).sustain(.5).release(.08)
    .gain(.55)
    .mask("<~!4 x!13 [~ x]!5 x!13 ~!3>"),
  n("<[0 ~ 2 4] [7 4 2 ~] [0 2 ~ 7] [9 7 4 2]>")
    .scale("D4:major:pentatonic")
    .s("triangle").decay(.12).sustain(.1).release(.35)
    .delay(.3).delaytime(3/8).delayfeedback(.3)
    .lpf(perlin.range(2500,5500)).room(.2).gain(.4)
    .every(4,x=>x.rev())
    .mask("<~!5 x!12 ~!5 x!12 ~!4>"),
  n("<[4 ~ 7 ~] [9 ~ 4 ~]>*2").scale("D5:major:pentatonic")
    .s("sine").decay(.05).sustain(0).release(.1)
    .gain(.17).pan(sine.range(.2,.8).slow(5))
    .delay(.5).delayfeedback(.45).delaytime(3/16)
    .mask("<~!7 x!10 ~!5 x!10 ~!6>")
)
