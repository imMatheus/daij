// 🎷 "Smoke & Velvet" — Late-Night Jazz Ballad with Walking Bass
// ~70 BPM, Db major — smoky Rhodes, brushed drums, acoustic bass walks the changes

setcps(.28)

$: s("vinyl:2").gain(.11).room(1).loop(1).lpf(2200)

// brushed drums — gentle swing
$: s("bd ~ ~ ~, ~ [~ sd:1] ~ ~, [hh hh oh hh]*2")
  .bank("RolandTR808")
  .swing(4)
  .gain(".6 .0 .0 .0, .0 .55 .0 .0, .18 .12 .2 .12")
  .room(.5).lpf(3200)
  .sometimes(x => x.gain(.25))

// walking bass — quarter notes through Dbmaj7 Bbm7 Ebm7 Ab7
$: note("<[db2 f2 ab2 bb2] [bb1 db2 f2 ab2] [eb2 gb2 bb2 db3] [ab1 c2 eb2 gb2]>")
  .s("gm_acoustic_bass")
  .decay(.3).sustain(.5).room(.25).gain(.78)
  .swing(4)
  .rarely(x => x.add(note(12)).gain(.3))

// Rhodes comping — syncopated jazz voicings
$: chord("<Dbmaj9 Bbm9 Ebm7 Ab13>")
  .dict('ireal').voicing()
  .s("gm_epiano1")
  .struct("~ [x ~] ~ [~ x]")
  .lpf(sine.range(1400, 2800).slow(12))
  .room(.55).delay(".25:.18:.35")
  .gain(.45).speed(.98)
  .swing(4)
  .every(4, x => x.struct("[x ~] ~ [~ x] ~").gain(.38))

// melody — pentatonic phrases like a muted trumpet
$: n("<[~ 0 ~ 4] [2 ~ ~ 5] [~ 4 2 ~] [0 ~ ~ ~]>")
  .scale("Db5:major:pentatonic")
  .s("gm_epiano2")
  .lpf(sine.range(900, 3200).slow(10))
  .room(.6).delay(".35:.25:.5")
  .gain(.38).swing(4)
  .vib(3).vibmod(.5)
  .sometimes(x => x.add(note(12)).gain(.22))

// strings pad — warm bed underneath
$: chord("<Dbmaj9 Bbm9 Ebm7 Ab13>")
  .dict('ireal').voicing()
  .s("gm_synth_strings_1")
  .attack(.8).sustain(.65).release(2.5)
  .lpf(sine.range(400, 1200).slow(8))
  .room(.8).gain(.24)

// sub drone
$: note("<db1 bb0 eb1 ab0>/4")
  .s("sine")
  .attack(2).decay(1).sustain(.8).release(2)
  .gain(.16).room(.5).lpf(160)
