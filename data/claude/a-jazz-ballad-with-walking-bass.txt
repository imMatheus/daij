// "Midnight in Blue" — Smoky jazz ballad with walking bass
setcps(0.3)
samples('github:tidalcycles/dirt-samples')

addVoicings('jazz', {
  7: ['3M 7m 9M 12P 15P', '7m 10M 13M 16M 19P'],
  '^7': ['3M 6M 9M 12P 14M', '7M 10M 13M 16M 19P'],
  m7: ['8P 11P 14m 17m 19P', '5P 8P 11P 14m 17m'],
}, ['C3', 'C6'])

let bass = note("<[F2 A2 C3 Db3] [D2 F2 A2 F#2] [G2 Bb2 D3 B2] [C2 E2 G2 Ab2] [A2 C3 E3 C#3] [D2 E2 F2 F#2] [G2 A2 Bb2 B2] [C2 E2 G2 E2]>")
  .s("sawtooth").lpf(500).lpq(2)
  .gain(.5).att(.01).dec(.2).sus(.65).rel(.12)
  .room(.15).pan(.42)

let comp = "<F^7 Dm7 Gm7 C7 Am7 Dm7 Gm7 C7>".voicings('jazz').note()
  .s("piano").gain(.28)
  .struct("<[~ x ~ ~] [~ [x ~] ~ x] [~ ~ [~ x] ~] [x ~ ~ [~ x]] [~ x ~ ~] [~ ~ x ~] [~ [~ x] ~ ~] [x ~ ~ x]>")
  .room(.3).delay(.12).dt(.3).dfb(.3)
  .lpf(perlin.range(1800, 4500).slow(5))

let melody = note("<[~ A4 ~ C5] [F5 E5 D5 ~] [~ Bb4 ~ A4] [G4 ~ ~ ~] [~ C5 ~ E5] [D5 ~ C5 A4] [Bb4 ~ D5 ~] [C5 ~ ~ ~]>")
  .s("triangle").gain(.18)
  .att(.06).dec(.35).sus(.3).rel(.7)
  .vib(4).vibmod(.15)
  .lpf(sine.range(2000, 3500).slow(6))
  .room(.35).sz(2).delay(.2).dt(.4).dfb(.4)
  .pan(.58)

let drums = stack(
  s("hh*4").gain(".2 .3 .2 .3").room(.08),
  s("~ [~ sd] ~ [sd ~]").gain(.08).room(.12),
  s("<[bd ~ ~ ~] [bd ~ ~ bd]>").gain(.2)
)

let chorus = stack(bass, comp, melody, drums)

let intro = stack(
  note("<[F2 ~ ~ A2] [D2 ~ ~ E2]>")
    .s("sawtooth").lpf(400).gain(.35)
    .att(.02).dec(.3).sus(.5).rel(.4).room(.25),
  "<F^7 Dm7>".voicings('jazz').note()
    .s("piano").gain(.15)
    .struct("~ ~ ~ x")
    .room(.55).delay(.35).dt(.45).dfb(.45)
)

let bridge = stack(
  note("<[Bb2 D3 F3 A2] [Bb2 Db3 F3 E2] [A2 C3 E3 G#2] [G2 B2 D3 F#2]>")
    .s("sawtooth").lpf(500).lpq(2)
    .gain(.5).att(.01).dec(.2).sus(.65).rel(.12)
    .room(.15).pan(.42),
  "<Bb^7 Bbm7 Am7 G7>".voicings('jazz').note()
    .s("piano").gain(.3)
    .struct("[x ~] ~ [~ x] ~")
    .room(.4).delay(.18).dt(.35).dfb(.38),
  note("<[D5 ~ F5 ~] [Db5 ~ Bb4 ~] [C5 E5 ~ D5] [D5 ~ B4 ~]>")
    .s("triangle").gain(.15)
    .att(.08).dec(.4).sus(.25).rel(.9)
    .vib(3).vibmod(.12)
    .lpf(2500).room(.45).sz(3)
    .delay(.22).dt(.5).dfb(.4).pan(.58),
  s("hh*4").gain(".15 .28 .15 .28").room(.1),
  s("~ sd ~ ~").gain(.06).room(.15)
)

arrange(
  [2, intro],
  [8, chorus],
  [4, bridge],
  [8, chorus]
)
