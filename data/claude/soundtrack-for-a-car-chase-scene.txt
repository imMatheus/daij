// "Burnout Protocol" — high-octane breakbeat chase, screeching synths and relentless percussion
setcps(168/60/4)

samples('github:tidalcycles/dirt-samples')

stack(
  s("<[bd [~ bd] bd [bd ~], ~ sd ~ sd:2, hh*16, ~ oh ~ oh] [bd*2 [~ bd] [bd ~] bd, ~ [sd sd] ~ sd, hh*16, oh(3,8)]>")
    .bank("RolandTR909").cut(1)
    .gain(perlin.range(.78,.98)).room(.15)
    .every(4, x => x.fast(2).gain(.55))
    .every(7, x => x.rev()),
  s("breaks165").splice(8, "<[0 3 6 2] [7 1 5 4] [0 ~ 3 7] [6 2 ~ 1]>")
    .cut(2).gain(.38)
    .hpf(800).lpf(sine.range(2200,5800).slow(5))
    .speed("<1 1.5 .75 2>")
    .sometimes(x => x.crush(8)).room(.2),
  note("<[e1 ~ e1 ~] [~ g1 ~ e1] [a1 ~ e1 ~] [~ b0 ~ e1]>")
    .s("sawtooth").lpf(sine.range(180,1200).slow(6))
    .decay(.12).sustain(.5).release(.06)
    .gain(.75).distort(.35).room(.1),
  note("<e0 g0 a0 b0>/4").s("sine")
    .gain(.4).lpf(80).decay(1.5).sustain(.85).release(.5),
  note("<[e3,g3,b3] [a3,c4,e4] [g3,b3,d4] [b3,d4,f4]>")
    .s("square").struct("[x ~ x ~]*2")
    .lpf(sine.range(800,4200).slow(7))
    .gain(.38).distort(.3).room(.25).crush(9)
    .every(3, x => x.fast(2).gain(.22)),
  n("<[0 ~ 4 7] [~ 10 7 3] [5 ~ 7 ~] [10 7 ~ 3]>")
    .scale("E5:minor:pentatonic").s("sawtooth")
    .lpf(sine.range(1500,5500).slow(4))
    .gain(.28).distort(.2).room(.4).delay(".2:.18:.3")
    .decay(.08).sustain(.2)
    .every(3, x => x.rev())
    .sometimes(x => x.add(note(12)).gain(.15)),
  note("e2*16").s("sawtooth")
    .gain(perlin.range(.05,.2))
    .lpf(sine.range(100,600).slow(3))
    .distort(.6).room(.12),
  s("cp(5,16,<0 2 4>)").bank("RolandTR909")
    .gain(perlin.range(.15,.35)).room(.5).pan(perlin.range(.2,.8))
    .delay(".15:.12:.2")
).reset("<16 16 8 16>")
