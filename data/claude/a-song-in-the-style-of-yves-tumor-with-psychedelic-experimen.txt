// 🌀 "Gospel for No One" — Psychedelic Experimental Rock (Yves Tumor style)
samples('github:tidalcycles/dirt-samples')
setcps(.45)

stack(
  s("bd [~ bd] <~ bd> ~, ~ [sd:3 ~] ~ <sd:3 [sd:3 cp]>, [hh hh oh hh]*2")
    .bank("RolandTR909").cut(1)
    .gain(perlin.range(.6,.95)).room(.3).distort(.15)
    .swing(2).every(4, x=>x.press())
    .sometimes(x=>x.speed(sine.range(.8,1.3)))
    .reset("<x x x x x x x x x x x x x@4>"),

  note("<[eb2 ~ ~ eb2] [bb1 ~ bb1 ~] [gb1 ~ ~ gb1] [ab1 ~ ab1 ~]>")
    .s("sawtooth").lpf(sine.range(250,700).slow(6))
    .gain(.72).distort(.35).room(.2).decay(.25).sustain(.55)
    .reset("<x x x x x x x x x x x x x@4>"),

  chord("<Ebm7 Bbm7 Gb^7 Ab7>").dict('ireal').voicing()
    .s("gm_electric_guitar_clean")
    .struct("<[x ~ [~ x] ~] [x ~ x ~] [[x x] ~ x ~] [x ~ [~ x] x]>")
    .lpf(sine.range(1400,4200).slow(8)).room(.55)
    .delay(".25:.18:.4").gain(.5).distort(.42).crush(13)
    .every(3, x=>x.crush(6).gain(.65))
    .reset("<x x x x x x x x x x x x x@4>"),

  n("<[~ 0 ~ 3] [5 ~ 7 ~] [~ 6 3 ~] [7 5 ~ 0]>")
    .scale("Eb5:phrygian")
    .s("gm_electric_guitar_clean")
    .lpf(sine.range(1200,5000).slow(7))
    .room(.6).delay(".35:.25:.55").gain(.4).distort(.32)
    .vib(5).vibmod(.8)
    .sometimes(x=>x.add(note(12)))
    .rarely(x=>x.rev())
    .reset("<~ ~ ~ ~ x x x x x x x x ~@4>"),

  n("<[~ ~ 5 ~] [7 ~ ~ 3] [~ 6 ~ ~] [~ ~ 7 5]>")
    .scale("Eb4:phrygian").s("gm_epiano2")
    .room(.75).delay(".45:.35:.6").gain(.28)
    .lpf(2600).crush(10).distort(.25)
    .pan(sine.slow(5))
    .reset("<~ ~ ~ ~ ~ ~ ~ ~ x x x x ~@4>"),

  note("eb1").s("sine").attack(1.5).sustain(.8).release(2)
    .gain(.2).room(.6).lpf(sine.range(100,350).slow(10))
    .distort(.15),

  s("<~ ~ ~ ~ ~ ~ ~ ~ [noise:2 ~] [~ noise:2] ~ ~ ~@4>")
    .gain(perlin.range(.03,.12)).room(.9).roomsize(8)
    .lpf(sine.range(400,2200).slow(3)).crush(6)
    .pan(perlin.range(0,1))
).reset("<x@12 x@4>")
