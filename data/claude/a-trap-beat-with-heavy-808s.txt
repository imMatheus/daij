// "808 Murk" — Dark trap beat with crushing 808 bass and rolling hi-hats
setcpm(35)
arrange(
  // INTRO — sparse hats and filtered 808 rumble
  [2, s("hh*8").bank("RolandTR808")
    .gain(".4 .22 .55 .22 .4 .22 .7 .22")
    .lpf(perlin.range(3000,7000))
    .stack(
      note("[f1 ~]*2").s("sine").penv(24).pdec(.05).pcurve(1)
      .decay(.35).sustain(0).gain(.6).lpf(280).distort("1:.3")
    )],
  // CHORUS — full trap arrangement
  [15, note("<[f1 ~] [~ f1] [~ ~ f1 ~] [f1 ~ eb1 ~]>")
    .s("sine").penv(24).pdec(.06).pcurve(1).decay(.9).sustain(0)
    .gain(1.4).lpf(300).distort("2.5:.35")
    .stack(
      s("~ cp ~ cp").bank("RolandTR808").gain(.85).room(.12).delay(.08),
      s("hh*16").bank("RolandTR808")
      .gain(".5 .22 .65 .22 .5 .22 1 .22 .5 .22 .65 .22 .5 .28 1 .38")
      .sometimesBy(.12, x=>x.ply(2))
      .lpf(perlin.range(4000,11000)),
      s("~ ~ ~ ~ ~ ~ oh ~").bank("RolandTR808").gain(.38).delay(.18).room(.1),
      note("<[~ c4 ~ eb4] [~ ~ f4 ~] [~ c4 ~ ~] [ab3 ~ ~ ~]>")
      .s("sawtooth").lpf(700).lpenv(3).lpd(.15).lpq(2)
      .gain(.25).delay(".3:.18:.5").room(.2).decay(.12).sustain(0)
    )],
  // BRIDGE — stripped back, 808 breathes
  [5, note("<[f1 ~ ~ ~] [~ ~ eb1 ~]>")
    .s("sine").penv(24).pdec(.1).pcurve(1).decay(1.4).sustain(0)
    .gain(1.1).lpf(200).distort("1.5:.25")
    .stack(
      s("hh*8").bank("RolandTR808").gain(".28 .16 .42 .16 .28 .16 .5 .16")
      .lpf(perlin.range(2000,5500)),
      note("<[c4 ~ eb4 ~] [f4 ~ ~ ab4]>")
      .s("sawtooth").lpf(450).lpenv(2).lpd(.25).lpq(3)
      .gain(.16).delay(".4:.22:.6").room(.45).decay(.18).sustain(0)
    )],
  // CHORUS 2 — full beat with extra energy and hat rolls
  [15, note("<[f1 ~] [~ f1] [~ ~ f1 ~] [f1 ~ eb1 ~]>")
    .s("sine").penv(24).pdec(.06).pcurve(1).decay(.9).sustain(0)
    .gain(1.5).lpf(350).distort("3:.35")
    .stack(
      s("~ cp ~ cp").bank("RolandTR808").gain(.9).room(.12).delay(.08),
      s("hh*16").bank("RolandTR808")
      .gain(".5 .22 .65 .22 .5 .22 1 .22 .5 .22 .65 .22 .5 .28 1 .38")
      .sometimesBy(.22, x=>x.ply(3))
      .lpf(perlin.range(5000,13000)),
      s("~ ~ ~ ~ ~ ~ oh ~").bank("RolandTR808").gain(.45).delay(.15).room(.1),
      note("<[~ c4 ~ eb4] [~ ~ f4 ~] [~ c4 ~ ~] [ab3 ~ ~ ~]>")
      .s("sawtooth").lpf(850).lpenv(3).lpd(.15).lpq(2)
      .gain(.3).delay(".28:.16:.5").room(.2).decay(.12).sustain(0),
      s("~ ~ ~ ~ [~ lt] ~ ~ ~").bank("RolandTR808").gain(.38).room(.22)
    )],
  // OUTRO — hats and 808 fade to nothing
  [4, s("hh*8").bank("RolandTR808")
    .gain(isaw.range(.06,.42).slow(4))
    .lpf(isaw.range(800,7000).slow(4))
    .stack(
      note("f1").s("sine").penv(24).pdec(.08).pcurve(1)
      .decay(1.6).sustain(0)
      .gain(isaw.range(.12,.85).slow(4)).lpf(180).distort("1.5:.25")
    )]
)
