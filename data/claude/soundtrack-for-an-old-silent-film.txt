// "The Projectionist's Waltz" — Sepia-toned melodrama for flickering celluloid
samples('github:tidalcycles/dirt-samples')
setcps(0.3)

const projector = perlin.slow(10).range(2000, 5500)
const filmgrain = perlin.slow(8).range(.005, .018)
const sepia = sine.slow(12).range(900, 2400)

const intro =
  note("g4 ~ d5 ~ bb4 a4 ~ ~").s("piano")
    .attack(.01).decay(.5).sustain(.18).release(1.8)
    .lpf(1600).room(.75).roomsize(3).vib(2).vibmod(.08)
    .delay(".35:.33:.45").gain(.38)
    .stack(
      s("~ hh:3 ~").gain(.04).lpf(2200).room(.3)
    )
    .stack(
      s("white").gain(filmgrain).bpf(4000).bpq(3).room(.4)
    )

const chorus =
  chord("<Gm D7 Gm Eb Cm D7 Gm D7>").dict('ireal').anchor("d5").voicing().slow(8).s("piano")
    .struct("<[x ~ x [~ x]] [x x ~ x] [x ~ x x] [~ x x ~] [x ~ x [x ~]] [x x ~ x] [x ~ [x x] x] [x x ~ [~ x]]>")
    .attack(.006).decay(.22).sustain(.12).release(.28)
    .lpf(sepia).room(.5).delay(".07:.125:.12").gain(.56).vib(1.5).vibmod(.05)
    .stack(
      note("<d5 g5 bb5 d5 a5 d5 f#5 a5 d5 g5 bb5 g5 bb5 eb5 g5 bb5 g5 c5 eb5 c5 a4 d5 f#5 a5 d5 g5 bb5 d5 a5 d5 f#5 d5>").slow(8).s("piano")
        .attack(.002).decay(.12).sustain(0).release(.1)
        .lpf(3200).room(.25).gain(.2)
        .reset("<x ~ ~ x ~ x ~ ~>")
    )
    .stack(
      note("<g1 d2 g2 d2 d2 a2 d3 a2 g1 d2 g2 d2 eb2 bb2 eb3 bb2 c2 g2 c3 g2 d2 a2 d3 a2 g1 d2 g2 d2 d2 a2 d3 a2>").slow(8).s("gm_acoustic_bass")
        .attack(.01).decay(.16).sustain(.05).release(.06)
        .lpf(sine.slow(9).range(400, 1200)).room(.14).gain(.62)
    )
    .stack(
      chord("<Gm D7 Gm Eb Cm D7 Gm D7>").dict('ireal').anchor("d6").voicing().slow(8).s("gm_string_ensemble_1")
        .attack(.35).decay(.5).sustain(.65).release(1.4)
        .lpf(sine.slow(11).range(900, 1800)).room(.72).gain(.15)
    )
    .stack(
      note("<~ [d5 g5] ~ [a5 d5] ~ [d5 g5] ~ [g5 bb5] ~ [c5 eb5] ~ [a4 d5] ~ [d5 g5] ~ [a5 f#5]>").slow(8).s("folkharp")
        .attack(.01).decay(.1).sustain(0).release(.2)
        .lpf(4400).room(.6).delay(".15:.125:.2").gain(.1)
        .reset("<x ~ x ~>")
    )
    .stack(
      s("~ hh:3 ~ hh:3").gain(sine.slow(4).range(.02, .05)).lpf(2000).room(.2)
    )
    .stack(
      s("white").gain(filmgrain).bpf(projector).bpq(2).room(.38)
    )

const bridge =
  chord("<Cm F7 Bb>").dict('ireal').anchor("d5").voicing().slow(3).s("piano")
    .struct("<[x ~ ~ x] [~ x x ~] [x ~ x [~ x]]>")
    .attack(.02).decay(.42).sustain(.25).release(1.9)
    .lpf(sine.slow(5).range(700, 1500)).room(.82).delay(".28:.25:.38").gain(.4)
    .vib(2).vibmod(.1)
    .stack(
      note("<c2 g2 c3 g2 f2 c3 f3 c2 bb1 f2 bb2 f2>").slow(3).s("gm_acoustic_bass")
        .decay(.2).sustain(.06).release(.08)
        .lpf(sine.slow(7).range(300, 700)).room(.18).gain(.52)
    )
    .stack(
      note("<eb5 c5 g4 ~ a4 f4 c4 ~ d5 bb4 f4 ~>").slow(3).s("gm_oboe")
        .attack(.05).decay(.22).sustain(.5).release(.45)
        .lpf(sine.slow(5).range(1200, 2600))
        .room(.7).delay(".32:.33:.38").gain(.14).vib(5).vibmod(.18)
    )
    .stack(
      chord("<Cm F7 Bb>").dict('ireal').anchor("d6").voicing().slow(3).s("gm_string_ensemble_1")
        .attack(.55).decay(.7).sustain(.72).release(2.2)
        .lpf(1100).room(.88).gain(.16)
    )
    .stack(
      s("white").gain(perlin.slow(4).range(.01, .028)).bpf(sine.slow(5).range(2400, 5200)).bpq(2).room(.45)
    )

arrange([1, intro], [8, chorus], [3, bridge], [8, chorus]).postgain(.96)
