// "Obsidian Descent" — Dark mysterious ambient drone
setcps(0.3)

const darkChord = "<[c#3,e3,g#3,b3] [a2,c#3,e3,g3] [f#2,a2,c#3,e3] [g#2,b2,d#3,f#3]>"
const bridgeChord = "<[d3,f3,ab3,b3] [bb2,db3,f3,ab3]>"
const dread = perlin.slow(14).range(300, 1600)
const murk = sine.slow(10).range(0.3, 0.7)

const intro =
  note("<[c#3,e3,g#3]>")
    .s("sawtooth")
    .attack(2).release(3).sustain(.4)
    .lpf(sine.slow(6).range(200, 800)).lpq(3)
    .room(.85).rsize(6).delay(.3).delayfb(.55)
    .gain(.15)
    .stack(
      note("<g#5 ~ ~ b5>")
        .s("sine").attack(.4).release(2.5)
        .gain(.06).delay(.7).delayfb(.7).delayt(.5)
        .room(.95).rsize(8)
    )

const chorus =
  note(darkChord)
    .slow(2)
    .s("sawtooth")
    .attack(.8).release(1.8).sustain(.35)
    .lpf(dread).lpq(2.5)
    .room(murk).rsize(5).delay(.2).delayfb(.5)
    .gain(.32)
    .stack(
      note("<c#2 a1 f#1 g#1>")
        .slow(2)
        .s("sine").lpf(120).gain(.45)
        .attack(.3).release(1.5)
    )
    .stack(
      note("<g#4 ~ b4 ~ e4 ~ d#4 ~>")
        .slow(2)
        .s("triangle")
        .attack(.05).release(.9)
        .lpf(sine.slow(7).range(800, 3500))
        .delay(.55).delayfb(.65).delayt(.375)
        .room(.6).gain(.14)
        .reset("<x ~ ~ x>")
    )
    .stack(
      s("bd").euclid(3,8)
        .slow(2)
        .gain(.4).lpf(200)
        .room(.3).distort("1.5:.3")
    )
    .stack(
      s("~ cp ~ ~, ~ ~ ~ [~ hh]")
        .slow(2)
        .gain(.12).room(.5).hpf(2000)
        .sometimesBy(.3, x => x.delay(.4))
    )
    .stack(
      sound("brown").gain(.06)
        .lpf(perlin.slow(8).range(100, 600))
        .room(.7)
    )

const bridge =
  note(bridgeChord)
    .slow(2)
    .s("sawtooth")
    .attack(1.2).release(2.5).sustain(.3)
    .lpf(sine.slow(5).range(400, 1200)).lpq(3.5)
    .hpf(perlin.slow(6).range(80, 400))
    .room(.9).rsize(7).delay(.4).delayfb(.6).delayt(.5)
    .gain(.25)
    .stack(
      note("<f5 ~ ab5 ~ db5 ~ b4 ~>")
        .slow(2)
        .s("sine")
        .attack(.15).release(1.8)
        .delay(.65).delayfb(.72).delayt(.33)
        .room(.85).gain(.1)
    )
    .stack(
      s("~ ~ [~ lt] ~")
        .slow(2)
        .gain(.15).lpf(700).room(.6)
    )

arrange(
  [1, intro],
  [8, chorus],
  [3, bridge],
  [8, chorus.lpf(perlin.slow(6).range(400, 2000))]
).postgain(.88)
