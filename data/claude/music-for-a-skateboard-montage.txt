// "Kickflip Sunset" — Punk-rock skateboard montage with driving drums and distorted riffs
setcps(0.55)
samples('github:tidalcycles/dirt-samples')

note("[e2 e2 [e2 g2] a2]*2")
.s("sawtooth")
.lpf(saw.range(250,1600).slow(4))
.distort("3:.35").decay(.18).sustain(0)
.gain(.65).room(.08)
.stack(
  s("bd [~ bd:4] <~ bd:4> [bd bd], ~ [~ cp] ~ cp, hh*8")
  .bank("RolandTR909")
  .gain(".85!4, 1 .8 1 .9, .4 .6 .4 .65 .4 .6 .4 .8")
  .swingBy(1/6, 4)
  .every(4, x => x.ply(2))
  .room(.08)
  .mask("<~!2 1!14 ~!6 1!14 ~!2>"),
  note("<[e4 g4 b4 a4] [g4 e4 d4 b3] [a4 b4 e4 g4] [b4 a4 g4 e4]>*2")
  .s("supersaw").gain(.35)
  .lpf(perlin.range(1200,5000).slow(5))
  .delay(.2).delaytime(.17).delayfeedback(.3)
  .room(.15).juxBy(.4, rev)
  .mask("<~!2 1!14 ~!6 1!14 ~!2>"),
  note("<[e3,b3,e4] [g3,d4,g4] [d3,a3,d4] [a3,e4,a4]>")
  .struct("[x ~ x ~]*2")
  .s("sawtooth").gain(.3)
  .distort("2:.3").lpf(1400).room(.1)
  .mask("<~!2 1!14 ~!6 1!14 ~!2>"),
  s("breaks165").slice(8,"<[0 1 2 3 4 5 6 7] [7 3 5 1 6 0 2 4] [0 6 3 7 1 5 4 2]>")
  .gain(.45).room(.3).delay(.35).delayfeedback(.45)
  .lpf(perlin.range(800,6000).slow(3))
  .mask("<~!16 1!6 ~!16>"),
  note("[e3,b3,g4]").s("sawtooth")
  .lpf(saw.range(80,2500).slow(2))
  .attack(.8).release(1.5).gain(.35).room(.35)
  .mask("<1!2 ~!36>")
)
