// 🎵 "Fuzz Riot" — Raw garage rock with blown-out amps and pounding drums
setcps(150/60/4)

$: s("vinyl:2").gain(.09).room(1).loop(1).lpf(2800)

$: stack(
  s("bd ~ bd ~, ~ sd ~ sd, hh*8, ~ ~ ~ oh")
    .bank("RolandTR909").cut(1).swing(2)
    .gain(perlin.range(.84,.97))
    .room(.3).distort(.12)
    .every(4, x => x.fast(2).gain(.6)),
  note("<[a1 a1 ~ a1] [a1 ~ a1 ~] [d2 d2 ~ d2] [e2 ~ e2 ~]>")
    .s("sawtooth")
    .lpf(sine.range(700,1100).slow(8))
    .decay(.18).sustain(.55)
    .distort(.32).gain(.72).room(.18),
  chord("<Am Am Dm Em>").dict('ireal').voicing()
    .s("gm_electric_guitar_clean")
    .struct("[x ~ x x] [x ~ x ~]")
    .hpf(220).lpf(3500)
    .distort(.42).crush(10)
    .room(.28).gain(.58).speed(.98)
    .every(3, x => x.press()),
  n("<[~ 0 ~ 3] [5 ~ 3 ~] [~ 0 ~ 5] [3 ~ ~ 0]>")
    .scale("A4:minor:pentatonic")
    .s("gm_electric_guitar_clean")
    .lpf(sine.range(1600,4500).slow(6))
    .room(.5).delay(".18:.12:.25")
    .distort(.35).gain(.4)
    .sometimes(x => x.add(note(12))),
  note("<a4 e4>/4").s("sawtooth")
    .attack(.4).sustain(.8).release(.8)
    .lpf(sine.range(500,1800).slow(6))
    .distort(.5).crush(8)
    .room(.7).roomsize(5)
    .gain(.14).pan(sine.slow(3)),
  note("<a1 d2 a1 e2>/4").s("sine")
    .attack(1.5).sustain(.7).release(2)
    .gain(.17).room(.4).lpf(130)
).reset(16)
