// "Basement Fury" — Raw garage rock with fuzzy riffs and pounding drums
setcpm(150/4)
samples('github:tidalcycles/dirt-samples')
arrange(
  // INTRO — hi-hats build tension, muffled guitar stabs
  [2, s("hh*8").gain(".35 .2 .5 .2 .35 .2 .65 .2").room(.15)
    .stack(s("~ ~ ~ cp").gain(.5).room(.3))
    .stack(note("[e3,b3]*2").s("sawtooth").distort("2:.2").lpf(700).gain(.25).room(.1))
  ],
  // CHORUS — full garage band, E-G-A-G power chord riff
  [16, note("<[e3,b3,e4] [g3,d4,g4] [a3,e4,a4] [g3,d4,g4]>*8")
    .s("supersaw").detune(.15).distort("5:.3")
    .lpf(perlin.range(1500,3200)).hpf(120)
    .decay(.1).sustain(.85).release(.05).gain(.45).room(.06)
    .stack(
      note("<e1 g1 a1 g1>*4").s("sawtooth")
      .lpf(400).distort("2:.35").gain(.6).decay(.08).sustain(.9),
      s("bd ~ [~ bd] ~, ~ sd ~ sd, hh*8")
      .gain(".8,.9,.38").room(.08).swingBy(1/8,4)
    )
  ],
  // BRIDGE — stripped back, moody D-C churn
  [6, note("<[d3,a3,d4] [c3,g3,c4]>*4")
    .s("square").distort("3:.25").lpf(1400)
    .decay(.18).sustain(.65).release(.1).gain(.38).room(.18).delay(.12)
    .stack(
      note("<d1 c1>*2").s("sawtooth")
      .lpf(280).distort("1.5:.3").gain(.5).decay(.1).sustain(.85),
      s("bd ~ ~ ~, ~ sd ~ ~, hh*4").gain(".65,.75,.32").room(.1)
    )
  ],
  // CHORUS 2 — cranked louder, snare fills, open hats
  [18, note("<[e3,b3,e4] [g3,d4,g4] [a3,e4,a4] [g3,d4,g4]>*8")
    .s("supersaw").detune(.22).distort("6:.3")
    .lpf(perlin.range(1800,3800)).hpf(120)
    .decay(.1).sustain(.85).release(.05).gain(.5).room(.08)
    .stack(
      note("<e1 g1 a1 g1>*4").s("sawtooth")
      .lpf(450).distort("2.5:.35").gain(.65).decay(.08).sustain(.9),
      s("bd ~ [~ bd] ~, ~ sd ~ [sd sd], hh!7 oh")
      .gain(".85,.95,.4").room(.1).swingBy(1/8,4)
      .sometimesBy(.15, x => x.ply(2))
    )
  ]
)
