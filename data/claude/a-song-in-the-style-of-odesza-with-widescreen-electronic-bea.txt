// 🎵 "Golden Hour" — Widescreen Electronic / ODESZA-style Anthemic Beats
// At 84 BPM: intro(4) + verse(8) + chorus(12) + outro(4) = 28 bars ≈ 80s

setcpm(84 / 4)

$: s("vinyl:2").gain(.1).room(1).loop(1).lpf(2600)

// Drums — cinematic 808 with swing and space
$: arrange(
  [4,  s("[hh hh]*4").bank("RolandTR808").gain(perlin.range(.04,.16)).room(.75).lpf(sine.range(1200,3000).slow(4)).pan(perlin.range(.3,.7))],
  [8,  s("bd ~ [~ bd] ~, ~ cp ~ ~, [hh hh oh hh]*2").bank("RolandTR808").cut(1).swing(3).gain(perlin.range(.68,.9)).room(.42)],
  [12, s("bd [~ bd] ~ bd, ~ cp ~ cp, [oh hh hh hh]*2, ~ ~ ~ [sd:1 sd:1]").bank("RolandTR808").cut(1).swing(3).gain(".9 .83 .87 .85").room(.35).every(4, x => x.press())],
  [4,  s("bd ~ ~ ~, ~ ~ cp ~").bank("RolandTR808").swing(4).gain(saw.slow(4).range(.6,.05)).room(.7)]
)

// Deep bass — Bb minor foundation
$: arrange(
  [4,  silence],
  [8,  note("<[bb1 ~ ~ bb1] [gb1 ~ ~ ~] [db2 ~ ~ db2] [ab1 ~ ~ ~]>").s("gm_acoustic_bass").decay(.3).sustain(.5).room(.2).gain(.76)],
  [12, note("<[bb1 ~ bb1 ~] [gb1 ~ ~ gb1] [db2 ~ db2 ~] [ab1 ~ ~ ab1]>").s("gm_acoustic_bass").decay(.25).sustain(.55).room(.2).gain(.84)],
  [4,  note("<bb1 ~ db2 ~>/2").s("sine").lpf(100).gain(saw.slow(4).range(.42,.04)).decay(.8).sustain(.3).room(.3)]
)

  [8,  chord("<Bbm9 Gbmaj7 Db Ab>").dict('ireal').voicing().s("gm_synth_strings_1").attack(.6).decay(.8).sustain(.72).release(2).lpf(2000).room(.72).roomsize(5).gain(.4).pan(sine.slow(9))],
  [12, chord("<Bbm9 Gbmaj7 Db Ab>").dict('ireal').voicing().s("gm_synth_strings_1").attack(.4).decay(.6).sustain(.76).release(1.5).lpf(2600).room(.65).roomsize(4).gain(.47).pan(sine.slow(7))],
  [4,  chord("<Bbm9 Gbmaj7>/2").dict('ireal').voicing().s("gm_synth_strings_1").attack(1.5).sustain(.55).release(4).lpf(sine.range(280,800).slow(4)).room(.92).roomsize(9).gain(saw.slow(4).range(.3,.04))]
)

// Pluck synth — ODESZA signature staccato chords
$: arrange(
  [4,  silence],
  [8,  chord("<Bbm9 Gbmaj7 Db Ab>").dict('ireal').voicing().s("gm_epiano1").struct("[~ x] [x ~] [~ x] [x ~]").lpf(2200).room(.45).delay(".25:.15:.3").gain(.44).speed(.99)],
  [12, chord("<Bbm9 Gbmaj7 Db Ab>").dict('ireal').voicing().s("gm_epiano1").struct("[x ~ x ~]*2").lpf(2700).room(.4).delay(".2:.12:.25").gain(.5).speed(.99).every(4, x => x.fast(2).gain(.28))],
  [4,  chord("<Bbm9 Gbmaj7>/2").dict('ireal').voicing().s("gm_epiano1").struct("x ~ ~ ~").lpf(1100).room(.78).delay(".5:.35:.6").gain(.25).speed(.96).crush(13)]

// Lead melody — soaring minor pentatonic with delay trails
$: arrange(
  [4,  silence],
  [8,  n("<[~ 0 ~ 2] [4 ~ 3 ~] [~ 5 3 ~] [2 ~ 0 ~]>").scale("Bb4:minor:pentatonic").s("gm_epiano2").lpf(sine.range(900,3200).slow(10)).room(.6).delay(".35:.25:.45").gain(.35).vib(3).vibmod(.4).rarely(x => x.add(note(12)))],
  [12, n("<[0 2 4 5] [~ 7 ~ 5] [4 5 7 ~] [~ 5 4 0]>").scale("Bb4:minor:pentatonic").s("gm_epiano2").lpf(sine.range(1300,4000).slow(7)).room(.55).delay(".3:.2:.4").gain(.43).vib(4).vibmod(.5).sometimes(x => x.add(note(12)))],
