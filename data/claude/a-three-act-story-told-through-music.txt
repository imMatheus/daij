// "The Arc" — Three-act cinematic narrative: awakening, storm, resolution
setcps(20/70)
arrange(
  // INTRO — a lone voice emerges from silence
  [2, stack(
    n("~ ~ 0 ~ ~ ~ 4 ~").scale("D4:dorian").s("triangle")
      .lpf(800).rel(1).room(.6).gain(.3),
    n("[0,4]").scale("D3:dorian").s("sawtooth")
      .lpf(400).att(1.5).rel(2).gain(.1)
  )],
  // ACT I / CHORUS — the theme awakens, D dorian, hopeful but shadowed
  [12, stack(
    n("0 2 4 ~ 3 2 0 ~").scale("D4:dorian").s("triangle")
      .lpf(sine.range(900,3500).slow(8)).rel(.35)
      .delay(".25:.125:.45").room(.3).gain(.5),
    n("<[0,2,4] [3,5,7] [-1,2,4] [0,3,5]>").scale("D3:dorian").s("sawtooth")
      .lpf(perlin.range(500,1400).slow(4)).att(.5).rel(1.2).gain(.18),
    s("bd(3,8), ~ sd ~ ~, [~ hh]*4").gain(.4).room(.12),
    n("0 ~ ~ 3 ~ ~ 0 ~").scale("D2:dorian").s("triangle")
      .lpf(250).gain(.35).rel(.15)
  )],
  // ACT II / BRIDGE — the storm, Bb phrygian, dark and relentless
  [6, stack(
    n("[0 1] [3 4] [5 ~] [4 2]").scale("Bb3:phrygian").s("sawtooth")
      .lpf(sine.range(500,6000).slow(3)).lpq(5).rel(.1)
      .distort("2:.3").delay(".15:.1:.6").gain(.4),
    n("[0,2,4] ~ [1,3,5] ~").scale("Bb2:phrygian").s("square")
      .lpf(350).att(.08).rel(.5).gain(.25),
    s("bd [bd ~] [~ bd] bd, [sd:3 ~ sd:3 ~]*2, hh(7,16)").gain(.55).hpf(150),
    n("0 ~ 3 ~ 0 ~ [3 5] ~").scale("Bb1:phrygian").s("sawtooth")
      .lpf(160).gain(.5).rel(.06)
  )],
  // ACT III / CHORUS RETURN — resolution, D major, bright and triumphant
  [12, stack(
    n("0 2 4 ~ 3 2 0 ~").scale("D4:major").s("triangle")
      .lpf(sine.range(1500,7000).slow(6)).rel(.5)
      .delay(".3:.15:.4").room(.4).gain(.45),
    n("0 ~ 4 ~ 3 ~ 0 ~").scale("D5:major").s("triangle")
      .lpf(4000).rel(.3).delay(.35).gain(.2),
    n("<[0,2,4] [2,4,6] [-1,2,4] [0,3,5]>").scale("D3:major").s("sawtooth")
      .lpf(perlin.range(800,2500).slow(4)).att(.3).rel(1.5).gain(.2),
    s("bd(3,8), ~ sd ~ ~, hh(5,8), ~ ~ ~ ~ ~ ~ oh ~").gain(.4).room(.18),
    n("0 ~ 4 ~ 2 ~ [0 5] ~").scale("D2:major").s("sawtooth")
      .lpf(400).gain(.4).rel(.1)
  )],
  // OUTRO — echoes dissolve into stillness
  [3, n("[0,2,4,7]").scale("D4:major").s("triangle")
    .lpf(600).rel(3).room(.7).delay(".4:.2:.5").gain(.3).att(.8)]
)
