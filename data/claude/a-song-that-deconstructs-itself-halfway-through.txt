// "Unraveling" — An electronic piece that deconstructs itself halfway through
setcps(0.5)

arrange(
  [2, note("d4 ~ [~ f4] ~").s("sine").gain(.3).lpf(1800)
    .room(.4).delay(".3:.18:.5")
    .stack(s("~ ~ ~ [~ hh]").gain(.15))],
  [13, note("<[d4 f4 a4 g4] [c4 e4 g4 a4] [bb3 d4 f4 a4] [a3 c4 e4 g4]>")
    .s("sawtooth").lpf(2200).lpq(2).gain(.3)
    .room(.15).delay(".2:.125:.45")
    .off(1/8, x=>x.add(12).gain(.12).lpf(3000))
    .stack(
      note("<[d3,f3,a3] [c3,e3,g3] [bb2,d3,f3] [a2,c3,e3]>")
        .s("sawtooth").lpf(1200).gain(.18).room(.2).release(.4),
      note("<d2 c2 bb1 a1>").s("sawtooth").lpf(500).gain(.45).lpenv(2).lpd(.15),
      s("bd ~ [~ bd] ~, ~ sd ~ sd, [~ hh]*4").gain(".65 .5").room(.08))],
  [4, note("<[d4 f4 a4 g4] [c4 e4 g4 a4]>")
    .s("sawtooth").lpf(1400).gain(.25).room(.35)
    .delay(".3:.15:.55").sometimesBy(.25, x=>x.crush(8)).juxBy(.3, rev)
    .stack(
      note("<[d3,f3,a3] [bb2,d3,f3]>").s("sawtooth").lpf(800).gain(.15).room(.4),
      note("<d2 bb1>").s("sawtooth").lpf(350).gain(.35),
      s("bd ~ [~ bd] ~, ~ sd ~ sd, hh*8").gain(".55 .4").degradeBy(.2))],
  [16, note("<[d4 f4 a4 g4] [c4 e4 g4 a4] [bb3 d4 f4 a4] [a3 c4 e4 g4]>")
    .s("sawtooth").gain(.25)
    .lpf(perlin.range(200, 3500).slow(3))
    .crush("<16!4 10!4 6!4 4!4>")
    .room(perlin.range(.2, .85).slow(5))
    .delay(".45:.2:.65")
    .degradeBy("<0!4 .15!4 .35!4 .6!4>")
    .sometimesBy("<.1 .2 .4 .6>", x=>x.speed("<1 -1 .5 2>"))
    .bite(4, "<0 1 2 3 [3 2] [1 0] [0 3 1 2] [3 3 0 0]>")
    .juxBy("<0 .3 .6 1>", rev)
    .stack(
      note("<[d3,f3,a3] [c3,e3,g3] [bb2,d3,f3] [a2,c3,e3]>")
        .s("sawtooth").gain(.15)
        .lpf("<1600!4 900!4 400!4 180!4>")
        .room("<.25 .45 .65 .85>")
        .degradeBy("<0!4 .2!4 .45!4 .7!4>")
        .iter("<1 2 3 4>"),
      note("<d2 c2 bb1 a1>").s("sawtooth")
        .lpf("<450!4 280!4 140!4 60!4>")
        .gain("<.4!4 .3!4 .18!4 .05!4>")
        .sometimesBy(.35, x=>x.hurry("<1 2 4 .5>")),
      s("bd ~ [~ bd] ~, ~ sd ~ sd, hh*8")
        .gain("<.6!4 .45!4 .25!4 .1!4>")
        .degradeBy("<.1!4 .35!4 .55!4 .8!4>")
        .crush("<16!4 10!4 5!4 3!4>")
        .scramble("<1 2 4 8>"))]
)
