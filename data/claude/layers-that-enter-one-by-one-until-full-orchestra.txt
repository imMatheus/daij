// 🎵 "The Gathering" — Orchestral Crescendo, layers building to full tutti
// At 80 BPM: solo(4) + duo(4) + trio(4) + quartet(4) + quintet(4) + tutti(6) + resolve(2) = 28 bars ≈ 84s

setcpm(80 / 4)

$: s("vinyl:2").gain(.1).room(1).loop(1).lpf(2400)

// Layer 1 — Strings pad (present from the start, the harmonic bed)
$: arrange(
  [4,  chord("<Dmaj7 Gmaj7>/2").dict('ireal').voicing()
       .s("gm_synth_strings_1").attack(1.2).decay(.8).sustain(.7).release(2)

  [4,  chord("<Dmaj7 Gmaj7 Bm7 A7>").dict('ireal').voicing()
       .s("gm_synth_strings_1").attack(.9).decay(.6).sustain(.7).release(1.5)
       .lpf(1400).room(.7).gain(.28)],
  [4,  chord("<Dmaj7 Gmaj7 Bm7 A7>").dict('ireal').voicing()
       .s("gm_synth_strings_1").attack(.7).decay(.5).sustain(.75).release(1.2)
       .lpf(1800).room(.6).gain(.32)],
  [4,  chord("<Dmaj7 Gmaj7 Bm7 A7>").dict('ireal').voicing()
       .s("gm_synth_strings_1").attack(.5).decay(.4).sustain(.8).release(1)
       .lpf(2200).room(.55).gain(.36)],
  [4,  chord("<Dmaj7 Gmaj7 Bm7 A7>").dict('ireal').voicing()
       .s("gm_synth_strings_1").attack(.4).decay(.4).sustain(.8).release(.8)
       .lpf(2600).room(.5).gain(.4)],
  [6,  chord("<Dmaj7 Gmaj7 Bm7 A7>").dict('ireal').voicing()
       .s("gm_synth_strings_1").attack(.3).decay(.3).sustain(.85).release(.8)
       .lpf(3000).room(.45).gain(.45)],
  [2,  chord("<Dmaj7>").dict('ireal').voicing()
       .s("gm_synth_strings_1").attack(1).sustain(.6).release(3)
       .lpf(sine.range(500,1200).slow(2)).room(.8).gain(saw.slow(2).range(.35,.05))]
)

// Layer 2 — Cello (enters bar 5, grounding the harmony)
$: arrange(
  [4,  silence],
  [4,  note("<d2 ~ ~ ~ g1 ~ ~ ~>/2")
       .s("gm_cello").attack(.15).decay(.4).sustain(.5).release(.6)
       .room(.35).gain(.45)],
  [4,  note("<[d2 ~ d2 ~] [g1 ~ g1 ~] [b1 ~ b1 ~] [a1 ~ a1 ~]>")
       .s("gm_cello").attack(.1).decay(.3).sustain(.55).release(.5)
       .room(.3).gain(.52)],
  [4,  note("<[d2 ~ a2 d2] [g1 ~ d2 g1] [b1 ~ fs2 b1] [a1 ~ e2 a1]>")
       .s("gm_cello").decay(.25).sustain(.6).room(.25).gain(.58)],
  [4,  note("<[d2 ~ a2 d2] [g1 ~ d2 g1] [b1 ~ fs2 b1] [a1 ~ e2 a1]>")
       .s("gm_cello").decay(.2).sustain(.6).room(.25).gain(.65)],
  [6,  note("<[d2 a2 d2 a2] [g1 d2 g1 d2] [b1 fs2 b1 fs2] [a1 e2 a1 e2]>")
       .s("gm_cello").decay(.15).sustain(.65).room(.2).gain(.72)],
  [2,  note("<d2>/1")
       .s("gm_cello").attack(.5).decay(.8).sustain(.4).release(2)
       .room(.5).gain(saw.slow(2).range(.5,.08))]

  [4,  n("<[~ 0 ~ 2] [4 ~ 5 ~] [~ 4 ~ 2] [0 ~ ~ ~]>")

       .lpf(1800).room(.5).gain(.28)],
  [4,  chord("<D G Bm A>").dict('ireal').voicing()
       .s("gm_french_horn")
       .struct("[x ~ x ~]")
       .attack(.2).decay(.4).sustain(.65).release(.6)
       .lpf(2200).room(.45).gain(.34)],
  [6,  chord("<D G Bm A>").dict('ireal').voicing()
       .s("gm_french_horn")
       .struct("[x ~ x ~]")
       .attack(.15).decay(.35).sustain(.7).release(.5)

  [2,  chord("<D>").dict('ireal').voicing()
       .s("gm_french_horn")
