// 🎵 "Planet Her Frequency" — Pop / Hip-Hop / Retro Funk à la Doja Cat
// At 104 BPM: intro(8) + verse1(16) + chorus(16) + verse2(16) + chorus2(16) + bridge(8) + outro(8) = 88 bars ≈ 3:23

setcpm(104 / 4)

// vinyl texture
$: s("vinyl:2").gain(.12).room(1).loop(1).lpf(3200)

// hi-hats — bouncy trap-pop pattern
$: s("hh*16")
   .bank("RolandTR808")
   .gain(perlin.range(.1, .48))
   .pan(sine.slow(5))
   .cut(1)
   .every(4, x => x.struct("hh*32").gain(.2))

// drums — 808 pop-trap foundation
$: arrange(
  [8,  s("bd ~ ~ ~, ~ ~ cp ~").bank("RolandTR808").swing(3).gain(.7).room(.3)],
  [16, s("bd ~ ~ bd, ~ [cp,sd:1] ~ cp, [~ hh]*4").bank("RolandTR808").swing(3).gain(".9 .82").room(.25).every(4, x => x.fast(2).gain(.5))],
  [16, s("bd ~ bd ~, ~ [cp,sd:1] ~ [cp,sd:1], [hh oh hh hh]*2").bank("RolandTR808").cut(1).swing(3).gain(".95 .88").room(.2).every(3, x => x.press())],
  [16, s("bd ~ ~ bd, ~ [cp,sd:1] ~ cp, [~ hh]*4").bank("RolandTR808").swing(3).gain(".9 .82").room(.25).sometimes(x => x.gain(.3))],
  [16, s("bd ~ bd ~, ~ [cp,sd:1] ~ [cp,sd:1], [hh oh hh hh]*2").bank("RolandTR808").cut(1).swing(3).gain(".95 .88").room(.2).every(4, x => x.fast(2).gain(.5))],
  [8,  s("bd ~ ~ ~, ~ cp ~ ~, [oh ~ ~ ~]*2").bank("RolandTR808").swing(4).gain(".75 .7").room(.45)],
  [8,  s("bd ~ ~ ~").bank("RolandTR808").swing(4).gain(saw.slow(8).range(.65, .08)).room(.5)]
)

// bass — bouncy 808 sub bass
$: arrange(
  [8,  silence],
  [16, note("<[f1 ~ ~ f1] [~ ab1 ~ ~] [bb1 ~ ~ bb1] [c2 ~ ~ ~]>").s("sawtooth").lpf(280).gain(.82).decay(.4).sustain(.5).room(.15).distort(.08)],
  [16, note("<[f1 ~ f1 ~] [ab1 ~ ~ ab1] [bb1 ~ bb1 ~] [c2 ~ ~ c2]>").s("sawtooth").lpf(320).gain(.88).decay(.35).sustain(.55).room(.15).distort(.1)],
  [16, note("<[f1 ~ ~ f1] [~ ab1 ~ ~] [bb1 ~ ~ bb1] [c2 ~ ~ ~]>").s("sawtooth").lpf(280).gain(.82).decay(.4).sustain(.5).room(.15).distort(.08)],
  [16, note("<[f1 ~ f1 ~] [ab1 ~ ~ ab1] [bb1 ~ bb1 ~] [c2 ~ ~ c2]>").s("sawtooth").lpf(320).gain(.88).decay(.35).sustain(.55).room(.15).distort(.1)],
  [8,  note("<f1 ~ ab1 ~ bb1 ~ c2 ~>/2").s("sawtooth").lpf(240).gain(.6).decay(.6).sustain(.35).room(.3)],
  [8,  silence]
)

// chords — retro funk keys with pop sheen
$: arrange(

  [16, chord("<Fm9 Abmaj7 Bbm7 C7b9>").dict('ireal').voicing().s("gm_clavinet").struct("[~ x] [x ~] [~ x] [x ~]").lpf(3500).room(.35).delay(".15:.1:.2").gain(.48).speed(.98)],
  [16, chord("<Fm9 Abmaj7 Bbm7 C7b9>").dict('ireal').voicing().s("gm_epiano2").struct("[x ~ x ~]*2").lpf(4000).room(.3).delay(".2:.125:.25").gain(.52).speed(.99).every(4, x => x.crush(10))],

  [16, chord("<Fm9 Abmaj7 Bbm7 C7b9>").dict('ireal').voicing().s("gm_epiano2").struct("[x ~ x ~]*2").lpf(4000).room(.3).delay(".2:.125:.25").gain(.52).speed(.99).every(4, x => x.crush(10))],
  [8,  chord("<Fm9 Abmaj7 Bbm7 C7b9>/2").dict('ireal').voicing().s("gm_epiano1").struct("x ~ ~ ~").lpf(1800).room(.7).delay(".4:.3:.5").gain(.4).crush(12)],
  [8,  chord("<Fm9 Ab^7>/2").dict('ireal').voicing().s("gm_synth_strings_1").attack(.8).sustain(.5).release(3).lpf(sine.range(500,1400).slow(8)).room(.8).gain(saw.slow(8).range(.3,.05))]
