setcps(0.3)

note("<[d3,f3,a3,c4] [a2,c3,e3,g3] [bb2,d3,f3,a3] [g2,bb2,d3,f3]>/2")
.s("sine").att(2).rel(4).sus(.75).dec(1.5)
.lpf(perlin.range(500,1800).slow(7))
.vib(.3).vibmod(.1)
.room(.9).sz(5).delay(.2).dt(.45).dfb(.3)
.gain(sine.range(.12,.28).slow(16))
.orbit(1)
.stack(
  n("<~ 4 ~ [2 ~] ~ 7 [~ 3] ~>")
  .off(1/6, x=>x.add(2).gain(.06))
  .scale("D5:minor:pentatonic").s("triangle")
  .degradeBy(.15)
  .att(.01).rel(1).sus(.04).dec(.4)
  .lpf(sine.range(3000,8000).slow(5))
  .room(.95).sz(7).delay(.7).dt(.28).dfb(.5)
  .gain(.09).pan(perlin.range(.05,.95))
  .mask("<~!2 1!7 ~!3 1!7 ~!2>")
  .orbit(2),
  note("<d1 d1 a1 bb1>/4")
  .s("sine").att(3).rel(5).sus(.85)
  .lpf(130).gain(.17).room(.3).sz(2)
  .orbit(3),
  s("pink").lpf(perlin.range(60,300).slow(9))
  .gain("<.015!2 .03!7 .05!3 .03!7 .015!2>")
  .room(.8).sz(5)
  .orbit(4),
  n("<0 ~ 2 ~ 4 7 ~ 4>")
  .scale("D4:dorian").s("sine")
  .att(.15).rel(2).sus(.06).dec(.6)
  .lpf(perlin.range(1200,4500).slow(5))
  .room(.92).sz(6).delay(.5).dt(.35).dfb(.4)
  .gain(.05).pan(sine.range(.2,.8).slow(4))
  .mask("<~!2 1!7 ~!3 1!7 ~!2>")
  .orbit(5),
  note("<d6 ~ a5 ~ f5 ~ a5 ~>/2")
  .s("sine").att(1).rel(3).sus(.3).dec(1)
  .lpf(5000).gain(.02).room(.95).sz(8)
  .delay(.6).dt(.5).dfb(.4)
  .pan(cosine.range(.3,.7).slow(6))
  .mask("<~!2 1!7 ~!3 1!7 ~!2>")
  .orbit(6)
