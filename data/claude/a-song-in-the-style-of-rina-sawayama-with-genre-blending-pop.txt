  .every(4, x => x.struct("[bd bd] cp [bd ~] [cp bd]").gain(.8))

  note("<[e1 ~ e1 ~] [c1 ~ ~ c1] [g1 ~ g1 ~] [d1 ~ d1 d1]>")

  chord("<Em Cmaj7 G D>").dict('ireal').voicing()

    .lpf(sine.range(400,3800).slow(12)).detune(4)
    .room(.55).gain(.32).delay(".08:.06:.12")
    .jux(rev),
  chord("<Em Cmaj7 G D>").dict('ireal').voicing()

  n("<[0 4 7 11 7 4 0 4] [0 3 7 11 7 3 0 3] [0 4 7 11 7 4 0 4] [0 2 7 9 7 2 0 2]>")

    .room(.5).delay(".25:.18:.4").pan(sine.slow(3))

  n("<[~ 0 ~ 4] [3 ~ 7 ~] [~ 5 ~ 2] [7 ~ 4 ~]>")
    .scale("E5:minor:pentatonic").s("square")
    .lpf(perlin.range(3000,8000)).gain(.22)
    .crush(12).room(.6).delay(".3:.2:.5")
    .pan(perlin.range(.2,.8))
    .rarely(x => x.fast(2).gain(.12)),
  chord("<Em Cmaj7 G D>").dict('ireal').voicing()
    .s("gm_synth_strings_1").attack(.4).sustain(.75).release(2)
    .lpf(sine.range(600,4500).slow(16))
    .room(.7).roomsize(4).gain(.28),
  n("<[4 ~ ~ 7] [~ 3 ~ ~] [7 ~ 4 ~] [~ ~ 2 ~]>")

    .lpf(sine.range(1200,5000).slow(10))
    .room(.45).delay(".4:.3:.55").gain(.24)
    .attack(.02).release(.4)
    .sometimes(x => x.speed(2).gain(.12)),
