// "Goodnight, Dissolving" — Lullaby melody gently unraveling into static and distortion
samples('github:tidalcycles/dirt-samples')
setcps(24/70)

const intro = 1
const chorus = 10
const bridge = 4
const finalChorus = 9

const decay = saw.slow(24)
const clarity = isaw.slow(24)

const melody = note("<[eb5 bb4 g4 ~ eb5 bb4 g4 ~] [f5 c5 ab4 ~ f5 c5 ab4 ~] [eb5 bb4 g4 ~ d5 bb4 g4 ~] [c5 ab4 f4 ~ bb4 ab4 f4 ~]>")
  .s("triangle").att(.005).dec(.15).sus(.1).rel(.6)
  .fm(decay.range(0, 6)).fmh(decay.range(1, 3.14)).fmdecay(decay.range(.5, .02))
  .vib(decay.range(0, 14)).vibmod(decay.range(0, 5))
  .lpf(clarity.rangex(600, 12000)).distort(decay.range(0, 9))
  .postgain(clarity.range(.2, .5))
  .delay(".25:.33:.4").room(.5).gain(.4)

const pad = note("<[eb3,g3,bb3] [f3,ab3,c4] [eb3,g3,bb3] [ab3,c4,eb4]>")
  .s("sawtooth").att(.4).dec(.3).sus(.4).rel(1.2)
  .lpf(clarity.rangex(250, 2800)).lpq(decay.range(0, 10))
  .distort(decay.range(0, 7)).postgain(clarity.range(.12, .3))
  .room(.4).chorus(.2).gain(.2)

const bass = note("<eb2 f2 eb2 ab2>")
  .s("sine").att(.01).dec(.3).sus(.5).rel(.4)
  .lpf(clarity.rangex(50, 350))
  .distort(decay.range(0, 5)).postgain(clarity.range(.2, .45))
  .gain(.3)

const noise = s("pink").att(.1).rel(1.5)
  .lpf(decay.rangex(300, 14000)).hpf(decay.rangex(40, 2400))
  .distort(decay.range(0, 7)).postgain(.25)
  .delay(".4:.125:.7").room(.7).gain(decay.range(0, .22))

const glitch = s("hh").euclid(3,8)
  .speed(decay.range(1, 5)).hpf(decay.rangex(2000, 9000))
  .distort(decay.range(0, 10)).postgain(.25)
  .delay(".3:.125:.6").gain(decay.range(0, .12))

const introMelody = note("[eb5 ~ ~ bb4 ~ ~ g4 ~]")
  .s("triangle").att(.01).dec(.2).sus(.08).rel(1.2)
  .lpf(10000).delay(".4:.4:.55").room(.7).gain(.3)

const introPad = note("[eb3,g3,bb3]")
  .s("sine").att(.8).rel(2.5).lpf(2000).room(.6).gain(.1)

const bridgePad = note("<[e3,ab3,bb3] [f3,a3,c4] [eb3,gb3,bb3] [ab3,b3,d4]>")
  .s("sawtooth").att(.05).dec(.2).sus(.3).rel(.6)
  .lpf(700).lpq(8).distort("5:.35:fold")
  .delay(".5:.125:.75").room(.6).gain(.14)

const bridgeMelody = note("<[eb5 e5 f5 e5 eb5 d5 c5 b4] [c5 db5 d5 eb5 e5 f5 gb5 g5]>")
  .fast(2).s("pulse").att(.005).dec(.08).sus(.04).rel(.1)
  .lpf(1100).distort("6:.35:sinefold")
  .delay(".4:.125:.65").room(.3).gain(.1)

const bridgeNoise = s("pink").struct("x [~ x] x [x x]")
  .att(.01).rel(.4).bpf(3200).bpq(10)
  .distort("8:.3:diode").room(.8).gain(.07)

arrange([intro, introMelody], [chorus, melody], [bridge, bridgeMelody], [finalChorus, melody])
  .stack(arrange([intro, introPad], [chorus, pad], [bridge, bridgePad], [finalChorus, pad]))
  .stack(arrange([intro, silence], [chorus, bass], [bridge, bass], [finalChorus, bass]))
  .stack(arrange([intro, silence], [chorus, noise], [bridge, bridgeNoise], [finalChorus, noise]))
  .stack(arrange([intro, silence], [chorus, glitch], [bridge, silence], [finalChorus, glitch]))
  .postgain(.85)
