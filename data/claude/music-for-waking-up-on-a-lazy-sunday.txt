// "Sunday Morning Light" — Warm lo-fi jazz chords and gentle textures for a lazy sunday wake-up
setcps(0.3)

stack(
  // soft pad — warm bed throughout
  chord("<F^7 Am7 Dm7 Bb^7>").voicing()
  .s("supersaw").gain(.1).detune(.12)
  .lpf(sine.range(400,1100).slow(8))
  .attack(.5).release(1.5).room(.7).size(3),

  // piano chords — enter after intro
  chord("<F^7 Am7 Dm7 Bb^7>").voicing()
  .s("piano").gain(.5)
  .lpf(perlin.range(1400,3200).slow(4))
  .attack(.04).release(.5)
  .room(.5).size(2).delay(".18:.375:.4")
  .mask("<~ 1@8 ~@3 1@8 ~>"),

  // warm bass root notes
  note("<f2 a2 d2 bb1>").s("triangle")
  .gain(.3).lpf(500).release(.3).room(.3)
  .mask("<~ 1@8 ~@3 1@8 ~>"),

  // gentle fm bell melody
  n("<[0 ~ 2] [4 2 ~] [~ 3 5] [2 ~ 4]>")
  .scale("F4:major").s("sine")
  .fm(1.5).fmh(2).fmdecay(.25).fmsustain(0)
  .gain(.28).attack(.06).release(.7)
  .delay(".2:.5:.45").room(.4)
  .lpf(perlin.range(1200,4500).slow(6))
  .mask("<~@2 1@7 1@3 1@7 ~@2>"),

  // soft shuffled drums
  s("bd:3 ~ [~ bd:3] ~, ~ sd:1 ~ sd:1, [hh:1 hh:2]*4")
  .gain(".4!2 .38 .42")
  .lpf(2000).room(.2)
  .mask("<~ 1@8 ~@3 1@8 ~>")
  .swingBy(1/6, 4),

  // vinyl hiss warmth
  s("white").gain(.03).lpf(700).hpf(250).room(.15)
)
