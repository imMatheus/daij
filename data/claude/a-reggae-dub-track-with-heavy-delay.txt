// "Dubwise Pressure" — Deep reggae dub with heavy delay and spring reverb
samples('github:tidalcycles/dirt-samples')
setcps(70/60/4)

stack(
  // One-drop kick on beat 3
  s("~ ~ bd:4 ~")
    .gain(.9).lpf(150)
    .mask("<0 1!19>"),

  // Rimshot with massive dub delay (dotted 8th)
  s("~ cp:3 ~ cp:3")
    .gain(.5).hpf(500)
    .delay(.85).delaysync(3/16).delayfeedback(.7)
    .room(.45).size(2.5)
    .mask("<0 1!8 0!3 1!8>"),

  // Offbeat hats with perlin-swept filter
  s("[~ hh]*4")
    .gain(".3 .5 .3 .5 .3 .5 .3 .5")
    .lpf(perlin.range(3500, 8000).slow(5))
    .delay(.12).delaysync(3/16).delayfeedback(.25)
    .mask("<0!2 1!7 0!3 1!8>"),

  // Deep dub bass — Gm to Cm
  note("<[g1 ~ ~ g1 ~ [g1 bb1] ~ ~] [g1 ~ ~ g1 ~ [bb1 g1] ~ ~] [c2 ~ ~ c2 ~ [c2 eb2] ~ ~] [c2 ~ ~ [eb2 c2] ~ c2 ~ ~]>")
    .s("sawtooth")
    .lpf(350).lpq(4).lpenv(3).lpd(.1)
    .decay(.12).sustain(.8).release(.08)
    .gain(.75)
    .mask("<0 1!19>"),

  // Offbeat organ skank with echo
  note("<[g3,bb3,d4]!2 [c4,eb4,g4]!2>")
    .struct("~ x ~ x ~ x ~ x")
    .s("sawtooth").gain(.16)
    .decay(.04).sustain(0).lpf(2000)
    .delay(.65).delaysync(3/16).delayfeedback(.6)
    .room(.3).size(1.5)
    .mask("<0!2 1!7 0!3 1!8>"),

  // Dub siren — bridge only
  note("eb5 ~ ~ ~ d5 ~ ~ ~").slow(2)
    .s("sine").gain(.1)
    .vib("5:3")
    .delay(.9).delaysync(3/8).delayfeedback(.8)
    .room(.7).size(4).rlp(2000)
    .mask("<0!9 1!3 0!8>"),

  // Intro pad swell
  note("[g2,bb2,d3,f3]")
    .s("sawtooth").gain(.1)
    .attack(1.5).release(2.5)
    .lpf(700).lpq(2)
    .room(.65).size(3)
    .delay(.5).delaysync(1/4).delayfeedback(.55)
    .mask("<1 0!19>")
)
