// "Whispered Vows" — Romantic ambient ballad with ethereal pads in F major
setcps(0.38)

let chorusMel = n("<[4 5 4 2] [3 2 1 ~] [0 2 4 5] [4 ~ 3 2]>")
  .scale("F4:major").s("sine")
  .release(1).gain(.45).vib(4).vibmod(.15)
  .lpf(sine.slow(12).range(1500, 3500))
  .delay(.25).delaytime(.2).delayfeedback(.5)
  .room(.6).roomsize(2)

let chorusPad = chord("<F Dm Bb C>").voicing()
  .s("triangle").gain(.28)
  .attack(.3).release(1.5)
  .lpf(1500).lpq(1)
  .room(.7).roomsize(3)
  .slow(2)

let chorusArp = n("<[0 2 4 2] [5 0 2 0] [3 5 0 5] [4 0 2 0]>")
  .scale("F5:major").s("sine")
  .gain(.12).release(.6)
  .delay(.4).delaytime(.3).delayfeedback(.6)
  .room(.5)

let chorusBass = note("<f2 d2 bb1 c2>")
  .s("triangle").gain(.25)
  .release(1.2).lpf(600)
  .room(.3).slow(2)

let chorus = stack(chorusMel, chorusPad, chorusArp, chorusBass)

let intro = stack(
  n("~ 4 ~ 2").scale("F4:major").s("sine")
    .gain(.15).release(2)
    .delay(.3).delaytime(.2).delayfeedback(.4)
    .room(.8).roomsize(3).lpf(1200),
  chord("F").voicing().s("triangle")
    .gain(.1).attack(.8).release(2)
    .lpf(900).room(.8).roomsize(3)
)

let bridge = stack(
  n("<[5 6 7 6] [4 5 6 ~] [3 4 5 4] [2 1 ~ ~]>")
    .scale("F4:major").s("sine")
    .gain(.4).release(1.2).vib(5).vibmod(.2)
    .lpf(2000)
    .delay(.3).delaytime(.25).delayfeedback(.55)
    .room(.7).roomsize(3),
  chord("<Gm Am Bb C>").voicing().s("triangle")
    .gain(.25).attack(.4).release(1.8)
    .lpf(1200).room(.8).roomsize(3),
  note("<g2 a2 bb2 c3>").s("triangle")
    .gain(.22).release(1.2).lpf(500).room(.3)
)

arrange(
  [2, intro],
  [10, chorus],
  [4, bridge],
  [10, chorus]
)
