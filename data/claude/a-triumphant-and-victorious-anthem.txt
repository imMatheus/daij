// "Rise Victorious" — Triumphant cinematic anthem with soaring synths and pounding drums
setcps(140/60/4)

let pad = note("<[d3,f#3,a3,d4] [a2,c#3,e3,a3] [b2,d3,f#3,b3] [g2,b2,d3,g3]>")
  .s("supersaw").detune(.2).room(.3).release(.3)
let bass = note("<d2 a1 b1 g1>").s("sawtooth").lpf(400).gain(.6)
let drums = s("bd [~ bd] [~ bd] ~, ~ sd ~ sd, hh*8")
  .gain(".9 .75 .85 .75").room(.08)
let lead = n("<[0 2 4 7] [7 4 2 0] [0 2 4 9] [9 7 4 2]>*2")
  .scale("D4:major").s("sawtooth").gain(.25).delay(.2).room(.15)

arrange(
  [2, pad.lpf(saw.range(200,2500).slow(2)).gain(.35).delay(.3)
    .stack(note("d1").s("sine").decay(1).sustain(0).gain(.6))],
  [16, pad.lpf(perlin.range(1500,5000)).delay(.12).gain(.55)
    .stack(bass, drums, lead.lpf(sine.range(2000,6000).slow(4)))],
  [6, note("<[e3,g3,b3,e4] [g2,b2,d3,g3] [a2,c#3,e3,a3]!2>")
    .s("supersaw").detune(.12).lpf(saw.range(600,3500).slow(3))
    .room(.4).delay(.25).gain(.4)
    .stack(
      note("<e2 g1 a1 a1>").s("sawtooth").lpf(300).gain(.5),
      s("bd ~ [~ bd] ~, ~ sd ~ ~, hh*4").gain(.65)
    )],
  [16, pad.lpf(perlin.range(2000,6500)).delay(.15).gain(.6)
    .stack(
      bass.gain(.65),
      s("bd [~ bd] [~ bd] bd, ~ sd ~ sd, [hh oh]*2 hh*4")
        .gain(".9 .75 .85 .8").room(.1),
      lead.s("supersaw").detune(.08)
        .lpf(sine.range(2500,7000).slow(4)).gain(.3).room(.2)
        .juxBy(.4, rev)
    )]
)
