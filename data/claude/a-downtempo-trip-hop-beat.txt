// "Midnight Underpass" — Downtempo Trip-Hop
// 82 BPM ~ 60s

setcpm(82 / 4)

samples('github:tidalcycles/dirt-samples')

$: s("bd ~ [~ bd] ~, ~ ~ sd ~, hh(5,8)")
   .bank("RolandTR808")
   .swing(.18)
   .gain(".88 .72 .8 .7")
   .room(.4)
   .cut(1)
   .reset("<8 16 16 8>")
   .when("<0 1 1 0>/48", x => x.s("bd ~ [~ bd] ~, ~ ~ [sd,cp] ~, [hh oh hh hh]*2").bank("RolandTR808").gain(".92 .8").room(.3).every(4, y => y.fast(2).gain(.45)))

$: note("<[d1 ~ ~ d1] [f1 ~ ~ ~] [bb0 ~ ~ bb0] [a1 ~ g1 ~]>")
   .s("gm_acoustic_bass")
   .decay(.3).sustain(.5).room(.25).gain(.82)
   .reset("<8 16 16 8>")
   .when("<0 1 1 0>/48", x => x.note("<[d1 ~ d1 ~] [f1 ~ ~ f1] [bb0 ~ bb0 ~] [a1 ~ ~ g1]>").s("gm_acoustic_bass").decay(.22).sustain(.6).gain(.86))

$: chord("<Dm9 Fmaj7 Bbmaj7 A7b9>")
   .dict('ireal').voicing()
   .s("gm_epiano2")
   .struct("~ [x ~] ~ [x ~]")
   .lpf(sine.range(1200,2400).slow(16))
   .room(.55).delay(".3:.2:.4")
   .gain(.46).speed(.98)
   .reset("<8 16 16 8>")
   .when("<0 0 0 1>/48", x => x.struct("x ~ ~ ~").lpf(1100).room(.75).delay(".5:.4:.55").gain(.38).crush(12))

$: n("<[~ 0 ~ 3] [4 ~ 2 ~] [~ 5 ~ 4] [2 ~ ~ 0]>")
   .scale("D4:minor:pentatonic")
   .s("gm_epiano1")
   .lpf(sine.range(800,2800).slow(10))
   .room(.6).delay(".35:.25:.5")
   .gain(.34)
   .rarely(x => x.add(note(12)))
   .reset("<8 16 16 8>")
   .when("<1 0 0 0>/48", x => silence)

$: chord("<Dm9 Bbmaj7>/2")
   .dict('ireal').voicing()
   .s("gm_synth_strings_1")
   .attack(.9).decay(1).sustain(.7).release(2.5)
   .lpf(sine.range(500,1200).slow(12))
   .room(.8).gain(.28)

$: note("<d1 f1 bb0 a1>/4")
   .s("sine")
   .attack(2).decay(1).sustain(.75).release(2)
   .gain(.18).room(.55)
   .lpf(160)

$: s("hh*8")
   .bank("RolandTR808")
   .gain(perlin.range(.05,.32))
   .pan(sine.slow(6))
   .cut(2)
