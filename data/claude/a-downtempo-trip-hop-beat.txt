// "Concrete Ghosts" — Downtempo trip hop with heavy breaks and dark atmospherics
samples('github:tidalcycles/dirt-samples')
setcps(78/60/4)

const padLfo = sine.slow(16).range(300, 1600)
const ghostLfo = perlin.slow(8).range(1200, 3800)

const intro =
  s("pink").gain(.04).lpf(1800).hpf(200).orbit(5).stack(
    note("[c3,eb3,g3,bb3]").s("sawtooth")
    .attack(1.2).release(3).decay(2).sustain(.15)
    .lpf(sine.range(200, 900).slow(8)).lpq(.2)
    .gain(.1).room(.9).rsize(6).delay(.6).delaytime(.5).delayfeedback(.6).orbit(3)
  )

const chorus =
  s("[bd ~] [~ <~ bd>] [~ [~ bd]] ~, ~ [~ sd] ~ sd, hh(5,8)")
  .bank('RolandTR808').swingBy(1/6, 4)
  .gain(".85 .6 .8 .5, 1 [.4 1] .7 1, .3 .5 .3 .5 .35")
  .room(.1).rsize(2).delay(.1).delaytime(.375).delayfeedback(.2).orbit(1)
  .stack(
    note("<[c1 ~] [~ c1] [eb1 ~] [~ c1]>")
    .s("sawtooth").lpf(140).lpq(6).gain(.8)
    .attack(.005).decay(.3).sustain(.55).release(.08).orbit(2),
    note("<[c3,eb3,g3,bb3] [ab2,c3,eb3,g3]>".slow(2))
    .s("sawtooth").attack(.8).release(3).decay(1.5).sustain(.2)
    .lpf(padLfo).lpq(.25).gain(.12)
    .room(.8).rsize(5).delay(.55).delaytime(.5).delayfeedback(.5).orbit(3),
    note("~ [g4 ~] ~ [eb4 ~]".slow(2))
    .s("piano").gain(.18).room(.5).delay(.4).delaytime(.333).delayfeedback(.5)
    .lpf(ghostLfo).pan(sine.range(.25,.75).slow(5)).orbit(4),
    s("pink").gain(.03).lpf(2500).hpf(300).orbit(5)
  )

const bridge =
  s("bd ~ ~ ~, ~ ~ [~ sd] ~, hh(3,8)")
  .bank('RolandTR808').swingBy(1/6, 4)
  .gain(".65 .4 .4 .4, .7 .4 [.3 .7] .4, .2 .3 .2")
  .room(.2).rsize(3).delay(.2).delaytime(.5).delayfeedback(.35).orbit(1)
  .stack(
    note("<c1 ~ eb1 ~>".slow(2))
    .s("sawtooth").lpf(120).lpq(5).gain(.6)
    .attack(.005).decay(.4).sustain(.4).release(.1).orbit(2),
    note("[ab2,c3,eb3,g3]").s("sawtooth")
    .attack(1).release(4).decay(2).sustain(.15)
    .lpf(sine.range(200,1000).slow(8)).lpq(.2)
    .gain(.09).room(.85).rsize(6).delay(.6).delaytime(.5).delayfeedback(.55).orbit(3),
    note("<[eb4 ~] [~ d4] ~ [bb3 ~]>".slow(2))
    .s("piano").gain(.13).room(.6).delay(.5).delaytime(.333).delayfeedback(.55)
    .lpf(perlin.range(800,2500).slow(6)).pan(sine.range(.2,.8).slow(4)).orbit(4)
  )

arrange([1, intro], [9, chorus], [4, bridge], [9, chorus]).postgain(.9)
