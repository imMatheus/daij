// "Midnight Chrome" — neon-soaked synthwave racing through a digital sunset
setcps(120/60/4)

s("bd*4").bank("RolandTR909").gain(1.1).room(.15).lpf(900)
.stack(
  s("~ sd:3 ~ sd:3").bank("RolandTR909").gain(.78).room(.55).roomsize(3).delay(".06:.04:.08"),
  s("hh(13,16)").bank("RolandTR909").gain(perlin.range(.12,.38)).pan(sine.slow(5)).cut(1).lpf(sine.range(4000,9000).slow(7)),
  s("[~ oh]*2").bank("RolandTR909").gain(.18).room(.35).cut(1),
  note("<[a1 ~ a1 ~] [a1 ~ ~ a1] [f1 ~ f1 ~] [g1 ~ ~ g1]>")
    .s("sawtooth").lpf(sine.range(100,500).slow(16)).gain(.55)
    .decay(.18).sustain(.45).release(.06).distort(.1),
  n("<[0 4 7 12 7 4 0 4] [0 5 8 12 8 5 0 5] [0 4 7 12 7 4 0 4] [0 4 7 11 7 4 0 4]>")
    .scale("A3:minor").s("sawtooth").decay(.06).sustain(.15).release(.04)
    .lpf(sine.range(1400,5000).slow(6)).gain(.32).room(.45).delay(".22:.16:.3")
    .pan(sine.slow(4)),
  chord("<Am9 Fmaj7 C6 G>").dict('ireal').voicing()
    .s("sawtooth").attack(.5).decay(.8).sustain(.55).release(1.2)
    .lpf(sine.range(500,1600).slow(14)).room(.65).gain(.22)
    .jux(rev),
  n("<[~ 0 ~ 4] [2 ~ 7 ~] [~ 5 ~ 3] [7 ~ 4 ~]>")
    .scale("A5:minor:pentatonic").s("square").decay(.12).sustain(.35).release(.15)
    .lpf(sine.range(1800,5500).slow(8)).room(.5).delay(".35:.24:.48")
    .gain(.3).pan(sine.slow(6)),
  note("a0").s("sine").gain(.16).lpf(110).attack(2).sustain(.8).release(2.5)
)
.reset("<x x x [x x]>".slow(8))
