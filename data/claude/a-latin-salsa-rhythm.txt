// 🎵 "Sabor de la Noche" — Latin Salsa Dura
// At 108 BPM: intro(4) + verso(8) + coro(8) + mambo(4) + outro(4) = 28 bars ≈ 1:02

setcpm(108 / 4)

// son clave 3-2 — the heartbeat of salsa
$: s("<[cp [~ cp] ~ cp] [~ cp cp ~]>")
   .bank("RolandTR808").speed(1.5).hpf(3500).gain(.42).room(.25)

// campana — cowbell riding the groove
$: arrange(
  [4, silence],
  [8, s("[~ cowbell:0 ~ cowbell:0]*2").bank("RolandTR808").speed(1.25).hpf(4200).gain(.28).room(.2).pan(.6)],
  [8, s("[cowbell:0 cowbell:0]*4").bank("RolandTR808").speed(1.25).hpf(4200).gain(.32).room(.15).pan(.6).every(4, x => x.struct("[x ~ x x] [x x ~ x]").gain(.22))],
  [4, s("[~ cowbell:0]*4").bank("RolandTR808").speed(1.25).hpf(4200).gain(.22).room(.3).pan(.6)],
  [4, silence]
)

// timbales and congas
$: arrange(
  [4, s("[~ hh]*4, bd ~ ~ bd").bank("RolandTR808").gain(.68).room(.28).swing(3)],
  [8, s("[~ hh]*4, bd ~ [~ bd] ~, ~ cp ~ cp").bank("RolandTR808").gain(".84 .76 .8 .76").room(.22).swing(3)],
  [8, s("[hh hh]*4, bd [~ bd] [bd ~] ~, ~ [cp sd:1] ~ [cp sd:1], [oh ~ ~ ~]*2").bank("RolandTR808").cut(1).gain(".88 .8 .85 .78").room(.18).swing(3).every(4, x => x.fast(2).gain(.4))],
  [4, s("[~ hh]*4, bd ~ ~ ~, ~ ~ cp ~").bank("RolandTR808").gain(.62).room(.38).swing(3)],
  [4, s("bd ~ ~ ~").bank("RolandTR808").gain(saw.slow(4).range(.55,.08)).room(.45)]
)

// bass — tumbao in G minor
$: arrange(
  [4, note("<[~ ~ g2 ~] [~ ~ d2 ~]>/2").s("gm_acoustic_bass").decay(.22).sustain(.42).room(.18).gain(.58)],
  [8, note("<[~ ~ g2 g1] [~ ~ c2 c1] [~ ~ eb2 eb1] [d2 ~ ~ d1]>").s("gm_acoustic_bass").decay(.25).sustain(.5).room(.18).gain(.8).swing(3)],
  [8, note("<[g1 ~ g2 g1] [c1 ~ c2 ~] [eb1 ~ eb2 eb1] [d1 ~ ~ d2]>").s("gm_acoustic_bass").decay(.2).sustain(.55).room(.18).gain(.85).swing(3)],
  [4, note("<g1 ~ d1 ~>/2").s("gm_acoustic_bass").decay(.5).sustain(.3).room(.32).gain(.52)],
  [4, silence]
)

// piano montuno — driving syncopation
$: arrange(
  [4, chord("<Gm7 D7>/2").dict('ireal').voicing().s("gm_epiano1").struct("~ x ~ x").lpf(2600).room(.38).gain(.4)],
  [8, chord("<Gm7 Cm7 Ebmaj7 D7>").dict('ireal').voicing().s("gm_epiano1").struct("[~ x] [x ~] [~ x] [x ~]").lpf(3200).room(.32).delay(".1:.08:.15").gain(.55).swing(3)],
  [8, chord("<Gm7 Cm7 Ebmaj7 D7>").dict('ireal').voicing().s("gm_epiano1").struct("[~ x x ~] [~ x x ~]").lpf(3800).room(.28).delay(".12:.1:.18").gain(.62).swing(3).every(4, x => x.fast(2).gain(.3))],
  [4, chord("<Gm7 D7>/2").dict('ireal').voicing().s("gm_epiano1").struct("~ x ~ ~").lpf(2000).room(.55).delay(".3:.2:.4").gain(.36)],
  [4, chord("<Gm7>").dict('ireal').voicing().s("gm_epiano1").struct("x ~ ~ ~").lpf(1400).room(.7).gain(saw.slow(4).range(.34,.05))]
)

// brass section — trombone-style riffs
$: arrange(
  [4, silence],
  [8, n("<[0 ~ ~ 2] [~ 4 ~ ~] [2 ~ 0 ~] [~ ~ 4 2]>").scale("G4:minor:pentatonic").s("gm_synth_strings_1").lpf(4000).room(.38).gain(.36).attack(.05).decay(.22).sustain(.58).release(.32)],
  [8, n("<[0 2 4 5] [~ 4 2 ~] [5 ~ 4 2] [4 2 0 ~]>").scale("G4:minor:pentatonic").s("gm_synth_strings_1").lpf(4600).room(.32).gain(.44).attack(.03).decay(.15).sustain(.65).release(.25).sometimes(x => x.add(note(12)))],
  [4, n("<[0 ~ 2 ~] [4 ~ 5 ~]>/2").scale("G4:minor:pentatonic").s("gm_synth_strings_1").lpf(3000).room(.55).delay(".4:.3:.5").gain(.3).attack(.08)],
  [4, silence]
)

// sub bass warmth

   .s("sine").attack(1.2).decay(1).sustain(.7).release(1.8)
