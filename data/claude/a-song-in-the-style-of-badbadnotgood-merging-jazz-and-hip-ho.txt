// "Smoke Break" — BADBADNOTGOOD-style jazz-hop, basement session grit

setcps(86/60/4)

stack(
  s("vinyl:2").gain(.11).room(1).loop(1).lpf(2100),
  s("bd ~ [~ bd] ~, ~ [sd,cp] ~ [~ sd:1], [hh hh oh hh]*2")
    .bank("RolandTR808").cut(1).swing(5)
    .gain(perlin.range(.72,.94)).room(.28)
    .every(4, x => x.press())
    .sometimes(x => x.s("bd [~ bd] ~ bd, ~ [sd,cp] [~ sd:1] [cp ~], [oh hh hh oh hh hh]*2")),
  note("<[bb1 ~ ~ bb1] [eb2 ~ ~ ~] [ab1 ~ ~ ab1] [f1 ~ ~ ~] [bb1 ~ bb1 ~] [eb2 ~ eb2 ~] [ab1 ~ f1 ~] [f1 ~ c2 ~]>")
    .s("gm_acoustic_bass").decay(.28).sustain(.55)
    .room(.2).gain(.82).lpf(1000)
    .sometimes(x => x.add(note(12)).gain(.4)),
  chord("<Bbm9 Eb13 Abmaj7 F7b9>")
    .dict('ireal').voicing()
    .s("gm_epiano1")
    .struct("~ [x ~] ~ [~ x]")
    .lpf(sine.range(1100,2600).slow(11))
    .room(.48).delay(".2:.14:.3").gain(.5).speed(.98)
    .every(4, x => x.struct("[x x]*2").gain(.38)),
  n("<[~ 0 3 ~] [5 ~ ~ 7] [~ 5 3 ~] [2 ~ 0 ~] [0 ~ 3 5] [~ 7 ~ 9] [7 5 ~ 3] [~ 5 3 0]>")
    .scale("Bb4:dorian")
    .s("gm_epiano2")
    .lpf(sine.range(1000,3500).slow(9))
    .room(.55).delay(".32:.22:.42").gain(.36)
    .vib(4).vibmod(.5)
    .rarely(x => x.add(note(12)))
    .pan(sine.slow(7)),
  n("<[~ ~ 5 ~] [~ 7 ~ ~] [5 ~ ~ 3] [~ ~ 0 ~]>")
    .scale("Bb3:dorian")
    .s("gm_vibraphone").room(.65)
    .delay(".45:.3:.55").gain(.24).lpf(3400)
    .pan(sine.slow(8).range(.3,.7))
    .attack(.02).decay(.3).sustain(.4).release(.5),
  note("<bb0 eb1 ab0 f1>/4").s("sine")
    .attack(2).decay(1).sustain(.8).release(2)
    .gain(.17).room(.4).lpf(165)
)
