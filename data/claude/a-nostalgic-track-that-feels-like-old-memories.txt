// "Fading Photographs" — Nostalgic lo-fi ambient with warm arpeggios and hazy reverb
setcps(0.27)
samples('github:tidalcycles/dirt-samples')

let arp = note("<[d4 f#4 a4 f#4] [b3 d4 f#4 d4] [g3 b3 d4 b3] [a3 c#4 e4 c#4]>*2")
  .s("piano").gain(.55)
  .decay(.5).sustain(.3).release(.9)
  .off(1/8, x => x.transpose(12).gain(.2))
  .delay(.4).delaytime(.375).delayfeedback(.45)
  .room(.7).sz(2)
  .lpf(perlin.range(500, 3200).slow(7))

let pad = note("<[d3,f#3,a3] [b2,d3,f#3] [g2,b2,d3] [a2,c#3,e3]>")
  .s("sawtooth").gain(.15)
  .attack(.8).decay(1.2).sustain(.2).release(2.5)
  .lpf(sine.range(200, 900).slow(19)).lpq(2)
  .room(.9).sz(4).chorus(.35)

let bass = note("<d2 b1 g1 a1>").s("sine")
  .lpf(300).gain(.4)
  .decay(.5).sustain(.6).release(.5)

let perc = s("[~ hh]*4").gain(.22)
  .lpf(3000).room(.25).delay(.15)

let melody = n("0 2 4 [7 ~] ~ 4 2 ~").scale("D5:major")
  .s("sine").attack(.02).decay(.7).sustain(.1).release(1.5)
  .gain(.28).room(.6)
  .delay(.35).delaytime(.3).delayfeedback(.55)
  .lpf(perlin.range(1200, 4000).slow(6))

let chorus = stack(arp, pad, bass, perc, melody)

let intro = stack(
  note("[d4 ~ ~ f#4 ~ ~ a4 ~]").s("piano")
    .gain(.3).release(1.5)
    .delay(.5).delaytime(.4).delayfeedback(.55)
    .room(.8).sz(4).lpf(1500),
  pad.gain(.08)
)

let bridge = stack(
  n("[~ 4] 2 [0 ~] -2").slow(1.5).scale("D4:minor:pentatonic")
    .s("triangle").attack(.15).decay(.8).sustain(.25).release(2)
    .gain(.25).room(.85)
    .delay(.5).delaytime(.5).delayfeedback(.6)
    .lpf(1100),
  pad,
  bass.gain(.25)
)

arrange(
  [1, intro],
  [7, chorus],
  [3, bridge],
  [8, chorus]
)
