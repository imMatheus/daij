setcps(68/60/4)

chord("<Ebm9 Gbmaj7 Abm7 Bb7sus4>").dict('ireal').voicing()
  .s("gm_epiano2").struct("~ [x ~] ~ [x ~]").swing(4)
  .attack(.06).decay(1.6).sustain(.3).release(3)
  .lpf(sine.range(700,2200).slow(16)).room(.6)
  .delay(".3:.2:.45").gain(.35).speed(.97)
.stack(
  n("<[~ 0 ~ ~] [~ ~ 2 ~] [4 ~ ~ 3] [~ 1 ~ ~]>")
    .scale("Eb4:minor:pentatonic").s("gm_epiano1")
    .lpf(perlin.range(800,3000).slow(1)).room(.7)
    .delay(".45:.3:.6").gain(.24).pan(sine.slow(8))
    .rarely(x=>x.add(note(12)).gain(.12)),
  n("<[~ ~ 3 ~] [4 ~ ~ ~] [~ 2 ~ ~] [~ ~ ~ 0]>/2")
    .scale("Eb5:minor:pentatonic").s("gm_vibraphone")
    .lpf(2000).room(.8).delay(".5:.35:.65")
    .gain(.16).pan(perlin.range(.2,.8)),
  note("<[eb1 ~ ~ ~] [gb1 ~ ~ gb1] [ab1 ~ ~ ~] [bb1 ~ ~ ~]>")
    .s("gm_acoustic_bass").decay(.35).sustain(.5)
    .room(.2).gain(.65).lpf(380),
  note("<eb0 gb0 ab0 bb0>/4").s("sine")
    .attack(2.5).sustain(.75).release(3)
    .gain(.14).room(.45).lpf(130),
  s("bd ~ ~ ~, ~ [~ sd:1] ~ ~")
    .bank("RolandTR808").swing(4)
    .gain(".62 .55").room(.4).lpf(2500)
    .every(8, x=>x.struct("bd ~ ~ bd, ~ [~ sd:1] ~ [~ cp]")),
  s("hh(5,8)").bank("RolandTR808")
    .gain(perlin.range(.03,.18)).pan(sine.slow(7))
    .cut(1).lpf(sine.range(2800,5000).slow(12)),
  s("vinyl:2").gain(.1).room(1).loop(1).lpf(1800)
)
