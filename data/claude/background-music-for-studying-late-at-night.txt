// "Late Night Pages" — Lo-fi ambient study beat, warm and drowsy
setcps(75/4/60)
samples('github:tidalcycles/dirt-samples')
stack(
  // warm analog pad — drifts throughout the whole piece
  chord("<Dm7 G7 C^7 Am7>").voicing()
    .s("sawtooth").gain(.22).attack(.5).release(1.2)
    .lpf(perlin.range(400,1000).slow(8))
    .room(.75).size(3).delay(".15:.3:.55").orbit(1),
  // gentle piano melody — chorus sections only
  n("<[0 ~ 2 4] [3 2 ~ 0] [~ 4 3 ~] [2 ~ 0 4]>")
    .scale("C4:major:pentatonic").sometimesBy(.3, x=>x.add(5))
    .s("piano").gain(.28)
    .lpf(perlin.range(2000,4500).slow(6))
    .room(.45).delay(".2:.18:.45")
    .mask("<0 1!8 0!3 1!8 0!2>").orbit(2),
  // sub bass following roots
  note("<d2 g1 c2 a1>").s("triangle")
    .gain(.4).lpf(300).decay(.35).sustain(.75).release(.25)
    .mask("<0 1!8 0!3 1!8 0!2>").orbit(3),
  // ethereal FM bells — bridge only
  n("4 ~ 2 ~ 3 ~ 0 ~").scale("C5:major:pentatonic")
    .s("sine").fm(2).fmdecay(.4).fmsustain(0)
    .gain(.15).attack(.1).release(.8)
    .room(.7).delay(".35:.2:.6")
    .mask("<0 0!8 1!3 0!8 0!2>").orbit(4),
  // lo-fi drums with lazy swing
  stack(
    s("bd ~ [~ bd] ~").gain(.45),
    s("~ [~ sd:3] ~ sd:3").gain(.28).lpf(3000),
    s("hh*8").gain(".22 .15 .25 .15 .22 .15 .25 .18").lpf(5500)
  ).swingBy(1/6, 4).room(.12)
    .mask("<0 1!8 0!3 1!8 0!2>").orbit(0),
  // vinyl hiss — constant soft texture
  s("pink").gain(.035).lpf(2200).hpf(500)
)
